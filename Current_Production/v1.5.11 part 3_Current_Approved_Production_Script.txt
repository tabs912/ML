



function rebuildChangedPMRsOnExistingMaster_(currentDemo, currentMaster, reportSheet, monthParts, runFullProcessing, timing) {
  const changedInfo = getChangedPMRsFromMonthlyChangeReport_(reportSheet);
  const changedPMRs = Array.from(changedInfo.keys());
  markRuntimeStep_(timing, "Update - read PMRs from Monthly Change Report (" + changedPMRs.length + " PMRs)");


  if (changedPMRs.length === 0) {
    return 0;
  }


  const result = updateMasterListFromMonthlyChangeActions_(currentDemo, currentMaster, reportSheet, monthParts, changedInfo, timing);
  const affectedPMRFilter = result.affectedPMRs || new Set(changedPMRs);
  markRuntimeStep_(timing, "Update - build affected PMR filter (" + affectedPMRFilter.size + " PMRs)");


  if (runFullProcessing) {
    processMasterListFull_(currentMaster, monthParts, affectedPMRFilter, timing);
  } else {
    processMasterListDataOnly_(currentMaster, affectedPMRFilter, timing);
    markRuntimeStep_(timing, "Update - process affected PMRs data-only");
    formatMasterListSheet_(currentMaster);
    markRuntimeStep_(timing, "Update - format Master List");
    groupMasterListRowsByPMR_(currentMaster);
    markRuntimeStep_(timing, "Update - group/sort Master List rows");
    hideRowsWithBlankDOB_(currentMaster);
    markRuntimeStep_(timing, "Update - hide non-primary rows");
  }


  removeDisenrolledPMRBlocksFromMasterUsingDemoP_(currentMaster, currentDemo);
  markRuntimeStep_(timing, "Update - remove disenrolled PMR blocks after update actions");


  return changedPMRs.length;
}


function updateMasterListFromMonthlyChangeActions_(currentDemo, currentMaster, reportSheet, monthParts, changedInfo, timing) {
  const enrollmentPMRs = getPMRsForMonthlyChangeSections_(changedInfo, ["Enrollments"]);
  const disenrollmentPMRs = getPMRsForMonthlyChangeSections_(changedInfo, ["Disenrollments"]);
  const primaryUpdatePMRs = getPMRsForMonthlyChangeSections_(changedInfo, ["Demographic Changes", "Caseload Changes", "Banner Changes"]);
  const contactChangePMRs = getPMRsForMonthlyChangeSections_(changedInfo, ["Contact Changes"]);
  const affectedPMRs = new Set();


  disenrollmentPMRs.forEach(pmr => affectedPMRs.add(pmr));
  enrollmentPMRs.forEach(pmr => affectedPMRs.add(pmr));
  primaryUpdatePMRs.forEach(pmr => affectedPMRs.add(pmr));
  contactChangePMRs.forEach(pmr => affectedPMRs.add(pmr));


  deletePMRBlocksFromMasterListBySet_(currentMaster, disenrollmentPMRs, reportSheet, monthParts, "Disenrollments - remove PMR block");
  markRuntimeStep_(timing, "Update - remove Disenrollment PMR blocks (" + disenrollmentPMRs.size + " PMRs)");


  deletePMRBlocksFromMasterListBySet_(currentMaster, enrollmentPMRs, reportSheet, monthParts, "Enrollments - replace/add PMR block");
  const addedRows = copyChangedPMRsFromDemoPToMasterList_(currentDemo, currentMaster, enrollmentPMRs, monthParts);
  markRuntimeStep_(timing, "Update - add Enrollment PMR blocks (" + enrollmentPMRs.size + " PMRs, " + addedRows + " rows)");


  const primaryResult = updatePrimaryRowsFromDemoPForPMRs_(currentDemo, currentMaster, primaryUpdatePMRs, monthParts);
  markRuntimeStep_(timing, "Update - primary row updates (" + primaryResult.rowsUpdated + " rows, " + primaryResult.columnsUpdated + " columns)");


  const contactResult = mergeSecondaryRowsFromDemoPForPMRs_(currentDemo, currentMaster, contactChangePMRs, monthParts);
  markRuntimeStep_(timing, "Update - contact row merge (" + contactResult.rowsUpdated + " updates, " + contactResult.rowsAdded + " adds, " + contactResult.rowsRemoved + " removes)");


  return {
    affectedPMRs: affectedPMRs,
    enrollmentPMRs: enrollmentPMRs,
    disenrollmentPMRs: disenrollmentPMRs,
    primaryUpdatePMRs: primaryUpdatePMRs,
    contactChangePMRs: contactChangePMRs,
    primaryResult: primaryResult,
    contactResult: contactResult
  };
}


function getPMRsForMonthlyChangeSections_(changedInfo, sectionsToMatch) {
  const targets = new Set(sectionsToMatch || []);
  const result = new Set();


  (changedInfo || new Map()).forEach((sections, pmr) => {
    if (!sections) return;
    const hasMatch = Array.from(sections).some(section => targets.has(section));
    if (hasMatch) result.add(pmr);
  });


  return result;
}


function deletePMRBlocksFromMasterListBySet_(masterSheet, pmrSet, reportSheet, monthParts, reasonText) {
  if (!masterSheet || !pmrSet || pmrSet.size === 0) return 0;


  const rowsByPMR = buildRowsByPMR_(masterSheet, HEADER_ROW, DATA_START_ROW);
  const headers = getHeaders_(masterSheet, HEADER_ROW);
  const logRows = [];
  const rowsToDelete = [];
  const timestamp = new Date();


  pmrSet.forEach(pmr => {
    const items = rowsByPMR.get(pmr) || [];
    items.forEach(item => {
      logRows.push([
        timestamp,
        monthParts.firstDay,
        pmr,
        reasonText || "Monthly Change action",
        masterSheet.getName(),
        item.rowNumber,
        reportSheet ? reportSheet.getName() : "",
        JSON.stringify(rowObjectFromHeaders_(headers, item.values))
      ]);
      rowsToDelete.push(item.rowNumber);
    });
  });


  appendMasterListChangeLog_(logRows);
  deleteRowNumberBatches_(masterSheet, rowsToDelete);
  clearSheetRuntimeCachesForSheet_(masterSheet);
  return rowsToDelete.length;
}


function updatePrimaryRowsFromDemoPForPMRs_(demoSheet, masterSheet, pmrSet, monthParts) {
  const result = { rowsUpdated: 0, columnsUpdated: 0, missingInMaster: 0, missingInDemo: 0 };
  if (!demoSheet || !masterSheet || !pmrSet || pmrSet.size === 0) return result;


  const demoData = getDataValues_(demoSheet, HEADER_ROW, DATA_START_ROW);
  const masterData = getDataValues_(masterSheet, HEADER_ROW, DATA_START_ROW);
  const demoBlocks = buildMergeRowsByPMRFromData_(demoData);
  const masterBlocks = buildMergeRowsByPMRFromData_(masterData);
  const masterHeaders = masterData.headers;
  const masterHeaderMap = masterData.headerMap;
  const sourceSheetName = demoSheet.getName();


  pmrSet.forEach(pmr => {
    const demoPrimary = getPrimaryMergeRowItem_(demoBlocks.get(pmr) || [], demoData.headerMap);
    const masterPrimary = getPrimaryMergeRowItem_(masterBlocks.get(pmr) || [], masterHeaderMap);


    if (!demoPrimary) {
      result.missingInDemo++;
      return;
    }
    if (!masterPrimary) {
      result.missingInMaster++;
      return;
    }


    const changedColumns = getPrimaryRowChangedColumnDetails_(
      masterPrimary.values,
      masterHeaderMap,
      demoPrimary.values,
      demoData.headerMap
    ).map(item => item.column);


    if (changedColumns.length === 0) return;


    setMasterRowColumnsFromDemoRow_(
      masterSheet,
      masterPrimary.rowNumber,
      masterHeaders,
      masterHeaderMap,
      demoPrimary.values,
      demoData.headerMap,
      changedColumns,
      monthParts,
      sourceSheetName,
      "Primary row updated from Demo P"
    );


    result.rowsUpdated++;
    result.columnsUpdated += changedColumns.length;
  });


  clearSheetRuntimeCachesForSheet_(masterSheet);
  return result;
}


function mergeSecondaryRowsFromDemoPForPMRs_(demoSheet, masterSheet, pmrSet, monthParts) {
  const result = { rowsUpdated: 0, rowsAdded: 0, rowsRemoved: 0, missingInDemo: 0 };
  if (!demoSheet || !masterSheet || !pmrSet || pmrSet.size === 0) return result;


  const demoData = getDataValues_(demoSheet, HEADER_ROW, DATA_START_ROW);
  let masterData = getDataValues_(masterSheet, HEADER_ROW, DATA_START_ROW);
  const demoBlocks = buildMergeRowsByPMRFromData_(demoData);
  let masterBlocks = buildMergeRowsByPMRFromData_(masterData);
  const masterHeaders = masterData.headers;
  const masterHeaderMap = masterData.headerMap;
  const sourceSheetName = demoSheet.getName();
  const rowsToDelete = [];
  const rowsToAppend = [];


  pmrSet.forEach(pmr => {
    const demoRows = demoBlocks.get(pmr) || [];
    const masterRows = masterBlocks.get(pmr) || [];
    if (!demoRows.length) {
      result.missingInDemo++;
      return;
    }


    const demoKeys = buildSecondaryMergeKeyMapForRows_(demoRows, demoData.headerMap, pmr);
    const masterKeys = buildSecondaryMergeKeyMapForRows_(masterRows, masterHeaderMap, pmr);
    const allKeys = new Set([].concat(Array.from(demoKeys.keys()), Array.from(masterKeys.keys())));


    allKeys.forEach(key => {
      const demoItem = demoKeys.get(key);
      const masterItem = masterKeys.get(key);


      if (demoItem && masterItem) {
        const changedFields = getMergeAuditChangedFields_(masterItem.values, masterHeaderMap, demoItem.values, demoData.headerMap);
        if (changedFields.length > 0) {
          setMasterRowColumnsFromDemoRow_(
            masterSheet,
            masterItem.rowNumber,
            masterHeaders,
            masterHeaderMap,
            demoItem.values,
            demoData.headerMap,
            changedFields,
            monthParts,
            sourceSheetName,
            "Contact row updated from Demo P"
          );
          result.rowsUpdated++;
        }
      } else if (demoItem && !masterItem) {
        rowsToAppend.push(buildMappedMasterRowFromDemoRow_(
          demoItem.values,
          demoData.headers,
          demoData.headerMap,
          masterHeaders,
          masterHeaderMap,
          monthParts,
          sourceSheetName,
          "Contact row added from Demo P"
        ));
        result.rowsAdded++;
      } else if (!demoItem && masterItem) {
        rowsToDelete.push(masterItem.rowNumber);
        result.rowsRemoved++;
      }
    });
  });


  if (rowsToAppend.length) {
    const appendRow = Math.max(masterSheet.getLastRow() + 1, DATA_START_ROW);
    masterSheet.getRange(appendRow, 1, rowsToAppend.length, masterHeaders.length).setValues(rowsToAppend);
  }


  if (rowsToDelete.length) {
    deleteRowNumberBatches_(masterSheet, rowsToDelete);
  }


  clearSheetRuntimeCachesForSheet_(masterSheet);
  return result;
}


function buildMappedMasterRowFromDemoRow_(demoRow, demoHeaders, demoHeaderMap, masterHeaders, masterHeaderMap, monthParts, sourceSheetName, statusText) {
  const output = new Array(masterHeaders.length).fill("");


  demoHeaders.forEach((header, sourceIdx) => {
    if (!header) return;
    const targetIdx = masterHeaderMap[header];
    if (targetIdx !== undefined) output[targetIdx] = demoRow[sourceIdx];
  });


  applyMasterListUpdateMetadataToRow_(output, masterHeaderMap, monthParts, sourceSheetName, statusText);
  return output;
}


function setMasterRowColumnsFromDemoRow_(masterSheet, rowNumber, masterHeaders, masterHeaderMap, demoRow, demoHeaderMap, headersToUpdate, monthParts, sourceSheetName, statusText) {
  if (!masterSheet || !rowNumber || rowNumber < DATA_START_ROW) return;
  const rowValues = masterSheet.getRange(rowNumber, 1, 1, masterHeaders.length).getValues()[0];


  (headersToUpdate || []).forEach(header => {
    const masterIdx = masterHeaderMap[header];
    const demoIdx = demoHeaderMap[header];
    if (masterIdx === undefined || demoIdx === undefined) return;
    rowValues[masterIdx] = demoRow[demoIdx];
  });


  applyMasterListUpdateMetadataToRow_(rowValues, masterHeaderMap, monthParts, sourceSheetName, statusText);
  masterSheet.getRange(rowNumber, 1, 1, masterHeaders.length).setValues([rowValues]);
}


function applyMasterListUpdateMetadataToRow_(rowValues, headerMap, monthParts, sourceSheetName, statusText) {
  const statusIdx = headerMap["Master List Update Status"];
  const monthIdx = headerMap["Master List Update Month"];
  const sourceIdx = headerMap["Master List Source Sheet"];


  if (statusIdx !== undefined) rowValues[statusIdx] = statusText || "Updated from Demo P";
  if (monthIdx !== undefined) rowValues[monthIdx] = monthParts ? monthParts.firstDay : new Date();
  if (sourceIdx !== undefined) rowValues[sourceIdx] = sourceSheetName || "";
}

function isSystemOrTestingSheet_(sheet) {
  if (!sheet) return false;


  const sheetName = String(sheet.getName() || "").trim();
  const lower = sheetName.toLowerCase();
  const explicitNames = getSystemAndTestingSheetNames_();


  // Do not treat Template - sheets as system sheets. Templates have their own
  // separate show/hide menu actions and triggers.
  return explicitNames.has(sheetName) ||
    lower.indexOf("framework test") !== -1 ||
    lower.indexOf("template validation") !== -1 ||
    lower.indexOf("framework health check") !== -1 ||
    lower.indexOf("dashboard audit") !== -1 ||
    lower.endsWith(" audit") ||
    lower.endsWith(" validation") ||
    lower.endsWith(" diagnostics") ||
    lower.endsWith(" health check") ||
    lower.indexOf("timing") !== -1;
}


function hideSystemSheets_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let hiddenCount = 0;


  ss.getSheets().forEach(sheet => {
    if (!isSystemOrTestingSheet_(sheet)) return;


    try {
      if (!sheet.isSheetHidden()) {
        sheet.hideSheet();
        hiddenCount++;
      }
    } catch (err) {
      Logger.log("Could not hide system/testing sheet " + sheet.getName() + ": " + err.message);
    }
  });


  return hiddenCount;
}

function formatSortOrderDisplayValues_(sheet) {
  sheet = sheet || SpreadsheetApp.getActiveSheet();


  const headers = getHeaders_(sheet, HEADER_ROW);
  const sortIdx = headers.indexOf("Sort Order");


  if (sortIdx === -1) return;


  const lastRow = sheet.getLastRow();
  if (lastRow < DATA_START_ROW) return;


  const range = sheet.getRange(DATA_START_ROW, sortIdx + 1, lastRow - DATA_START_ROW + 1, 1);
  const values = range.getValues();


  const output = values.map(row => {
    const value = row[0];
    const num = Number(value);


    if (isNaN(num) || value === "" || value === null || value === undefined) return [value];


    const whole = Math.floor(num);
    const detail = Math.round((num - whole) * 1000);


    if (detail <= 0) return [String(whole)];
    return [whole + "." + detail];
  });


  range;
  range.setValues(output);
}

function buildUpdateMasterListMergeAuditForMonth_(monthParts) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const currentDemo = getCurrentDemoPSheet_(monthParts);
  const sourceMaster = getPreviousMasterListSheet_(monthParts);


  if (!currentDemo) {
    notify_("Current month Demo P was not found. Format Demo P first.");
    return null;
  }


  if (!sourceMaster) {
    notify_("Previous month Master List was not found for merge audit.");
    return null;
  }


  const reportSheet = getMonthlySheetByPrefixAndDate_(
    ss,
    MONTHLY_CHANGE_REPORT_PREFIX,
    monthParts.firstDay,
    monthParts.lastDay
  );


  if (!reportSheet) {
    notify_("Monthly Change Report was not found. Create Monthly Change Report first.");
    return null;
  }


  const changedInfo = getChangedPMRsFromMonthlyChangeReport_(reportSheet);
  const contactChangePMRs = Array.from(changedInfo.keys())
    .filter(pmr => changedInfo.get(pmr) && changedInfo.get(pmr).has("Contact Changes"))
    .sort();


  const demoData = getDataValues_(currentDemo, HEADER_ROW, DATA_START_ROW);
  const masterData = getDataValues_(sourceMaster, HEADER_ROW, DATA_START_ROW);


  const demoBlocks = buildMergeRowsByPMRFromData_(demoData);
  const masterBlocks = buildMergeRowsByPMRFromData_(masterData);


  const summaryRows = [];
  const detailRows = [];


  contactChangePMRs.forEach(pmr => {
    const demoRows = demoBlocks.get(pmr) || [];
    const masterRows = masterBlocks.get(pmr) || [];
    const participantName = getMergeAuditParticipantName_(demoRows, demoData.headerMap, masterRows, masterData.headerMap);


    const demoKeys = buildSecondaryMergeKeyMapForRows_(demoRows, demoData.headerMap, pmr);
    const masterKeys = buildSecondaryMergeKeyMapForRows_(masterRows, masterData.headerMap, pmr);


    const contactKeys = new Set(
      [].concat(Array.from(demoKeys.keys()), Array.from(masterKeys.keys()))
    );


    let matchedCount = 0;
    let updateCount = 0;
    let addCount = 0;
    let removeCount = 0;
    let noActionCount = 0;


    contactKeys.forEach(key => {
      const demoItem = demoKeys.get(key);
      const masterItem = masterKeys.get(key);
      const masterSummary = masterItem ? buildMergeAuditContactSummary_(masterItem.values, masterData.headerMap) : "";
      const demoSummary = demoItem ? buildMergeAuditContactSummary_(demoItem.values, demoData.headerMap) : "";


      let auditAction = "";
      let differenceFound = "";
      let recommendedAction = "";


      if (demoItem && masterItem) {
        matchedCount++;
        const changedFields = getMergeAuditChangedFields_(masterItem.values, masterData.headerMap, demoItem.values, demoData.headerMap);


        if (changedFields.length > 0) {
          auditAction = "Update";
          differenceFound = "Contact exists in both; changed fields: " + changedFields.join(", ");
          recommendedAction = "Update existing Master List contact row";
          updateCount++;
        } else {
          auditAction = "No action";
          differenceFound = "Contact exists in both; no compared field differences";
          recommendedAction = "No Master List contact row change needed";
          noActionCount++;
        }
      } else if (demoItem && !masterItem) {
        auditAction = "Add";
        differenceFound = "Contact exists in Demo P only";
        recommendedAction = "Add new contact row to Master List";
        addCount++;
      } else if (!demoItem && masterItem) {
        auditAction = "Remove";
        differenceFound = "Contact exists in previous Master List only";
        recommendedAction = "Remove contact row from Master List";
        removeCount++;
      }


      detailRows.push([
        pmr,
        participantName.lastName,
        participantName.firstName,
        auditAction,
        differenceFound,
        recommendedAction,
        masterItem ? masterItem.rowNumber : "",
        demoItem ? demoItem.rowNumber : "",
        "Secondary/Contact",
        "Yes",
        masterSummary,
        demoSummary,
        key
      ]);
    });


    let recommendation = "No contact row change needed";
    if (addCount || removeCount || updateCount) {
      const pieces = [];
      if (updateCount) pieces.push(updateCount + " update");
      if (addCount) pieces.push(addCount + " add");
      if (removeCount) pieces.push(removeCount + " remove");
      recommendation = pieces.join(", ");
    }


    summaryRows.push([
      pmr,
      participantName.lastName,
      participantName.firstName,
      masterRows.length,
      demoRows.length,
      matchedCount,
      updateCount,
      addCount,
      removeCount,
      noActionCount,
      recommendation
    ]);
  });


  const auditName = "Update Master List Merge Audit";
  let audit = ss.getSheetByName(auditName);
  if (audit) ss.deleteSheet(audit);
  audit = ss.insertSheet(auditName);


  audit.getRange("A1").setValue(auditName);
  audit.getRange("A2").setValue("Date");
  audit.getRange("B2").setValue(monthParts.firstDay);
  audit.getRange("C2").setValue("to");
  audit.getRange("D2").setValue(monthParts.lastDay);
  audit.getRange("A3").setValue("Audit Scope");
  audit.getRange("B3").setValue("Contact Changes only - secondary rows only");
  audit.getRange("C3").setValue("Source Master List");
  audit.getRange("D3").setValue(sourceMaster.getName());
  audit.getRange("E3").setValue("Source Demo P");
  audit.getRange("F3").setValue(currentDemo.getName());


  const summaryHeaders = [
    "PMR #",
    "Last Name",
    "First Name",
    "Master Row Count",
    "Demo P Row Count",
    "Contacts Matched",
    "Contacts To Update",
    "Contacts To Add",
    "Contacts To Remove",
    "Contacts With No Action",
    "Merge Recommendation"
  ];


  audit.getRange(5, 1, 1, summaryHeaders.length).setValues([summaryHeaders]);
  if (summaryRows.length) {
    audit.getRange(6, 1, summaryRows.length, summaryHeaders.length).setValues(summaryRows);
  } else {
    audit.getRange(6, 1).setValue("No PMRs were found in the Contact Changes section of the Monthly Change Report.");
  }


  const detailStart = Math.max(8 + summaryRows.length, 8);
  audit.getRange(detailStart, 1).setValue("Contact Change Details");
  const detailHeaders = [
    "PMR #",
    "Last Name",
    "First Name",
    "Audit Action",
    "Difference Found",
    "Recommended Master List Action",
    "Master Row #",
    "Demo P Row #",
    "Row Type",
    "Secondary Row Only",
    "Master Contact Summary",
    "Demo P Contact Summary",
    "Contact Match Key"
  ];


  audit.getRange(detailStart + 1, 1, 1, detailHeaders.length).setValues([detailHeaders]);
  if (detailRows.length) {
    audit.getRange(detailStart + 2, 1, detailRows.length, detailHeaders.length).setValues(detailRows);
  }


  applyStandardFormatting_(audit, {
    sheetType: SHEET_TYPE.MASTER_LIST_CHANGE_LOG,
    title: auditName,
    startDate: monthParts.firstDay,
    endDate: monthParts.lastDay,
    renameSheet: false
  });
// v1.1.0 removed orphaned legacy formatting statement.
  return audit;
}




function buildUpdateMasterListPrimaryRowAuditForMonth_(monthParts) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const currentDemo = getCurrentDemoPSheet_(monthParts);
  const sourceMaster = getPreviousMasterListSheet_(monthParts);


  if (!currentDemo || !sourceMaster) return null;


  const reportSheet = getMonthlySheetByPrefixAndDate_(
    ss,
    MONTHLY_CHANGE_REPORT_PREFIX,
    monthParts.firstDay,
    monthParts.lastDay

function getPrimaryMergeRowItem_(rows, headerMap) {
  const primaryIdx = headerMap ? headerMap["Primary PMR Row"] : undefined;


  if (primaryIdx !== undefined) {
    const primaryItem = (rows || []).find(item => {
      const row = item.values || [];
      const value = String(row[primaryIdx] || "").trim().toLowerCase();
      return value === "yes" || value === "true" || value === "primary";
    });


    if (primaryItem) return primaryItem;
  }


  return (rows || [])[0] || null;
}

function buildMergeRowsByPMRFromData_(data) {
  const result = new Map();
  if (!data || !data.values || !data.values.length) return result;


  const pmrIdx = getPMRIndex_(data.headerMap);
  if (pmrIdx === -1) return result;


  data.values.forEach((row, offset) => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr) return;
    if (!result.has(pmr)) result.set(pmr, []);
    result.get(pmr).push({
      rowNumber: DATA_START_ROW + offset,
      values: row,
      blockOffset: result.get(pmr).length
    });
  });


  return result;
}




function buildSecondaryMergeKeyMapForRows_(rows, headerMap, pmr) {
  const result = new Map();
  const primaryIdx = headerMap ? headerMap["Primary PMR Row"] : undefined;


  (rows || []).forEach((item, idx) => {
    const row = item.values || [];
    const primaryValue = primaryIdx === undefined ? "" : String(row[primaryIdx] || "").trim().toLowerCase();
    const isPrimary = primaryValue === "yes" || primaryValue === "true" || primaryValue === "primary";


    if (isPrimary) return;
    if (primaryIdx === undefined && idx === 0) return;


    const key = buildContactMergeRowKey_(row, headerMap, pmr, idx + 1);


    if (!result.has(key)) {
      result.set(key, item);
    } else {
      const duplicateKey = key + "|DUP" + item.rowNumber;
      result.set(duplicateKey, item);
    }
  });


  return result;
}


function buildMergeKeyMapForRows_(rows, headerMap, pmr) {
  const result = new Map();
  const primaryIdx = headerMap["Primary PMR Row"];


  (rows || []).forEach((item, idx) => {
    const row = item.values || [];
    const isPrimary = primaryIdx !== undefined && String(row[primaryIdx] || "").trim().toLowerCase() === "yes";
    let key;


    if (isPrimary || idx === 0) {
      key = pmr + "|PRIMARY";
    } else {
      key = buildContactMergeRowKey_(row, headerMap, pmr, idx + 1);
    }


    if (!result.has(key)) {
      result.set(key, item);
    } else {
      const duplicateKey = key + "|DUP" + item.rowNumber;
      result.set(duplicateKey, item);
    }
  });


  return result;
}


function buildContactMergeRowKey_(row, headerMap, pmr, fallbackPosition) {
  const parts = [
    pmr,
    "CONTACT",
    getMergeRowValue_(row, headerMap, ["Contact - Last Name"]),
    getMergeRowValue_(row, headerMap, ["Contact - First Name"]),
    getMergeRowValue_(row, headerMap, ["Relationship"]),
    getMergeRowValue_(row, headerMap, ["Type of Contact"]),
    getMergeRowValue_(row, headerMap, ["AD1 - Phone", "Phone Number", "Phone 1 - Value"])
  ];


  const cleanParts = parts.map(normalizeKeyPart_);
  const hasContactIdentity = cleanParts.slice(2).some(Boolean);


  if (!hasContactIdentity) {
    return pmr + "|CONTACT|ROW" + String(fallbackPosition || "");
  }


  return cleanParts.join("|");
}


function getMergeRowValue_(row, headerMap, possibleHeaders) {
  for (let i = 0; i < possibleHeaders.length; i++) {
    const idx = headerMap[possibleHeaders[i]];
    if (idx !== undefined) return row[idx];
  }
  return "";
}


function updateMasterListForMonth_(monthParts) {
  if (!monthParts) return null;


  const timing = startRuntimeTiming_("Update Master List", monthParts);
  const hiddenSheetNamesAtStart = captureHiddenSheetNames_();
  let currentMaster = null;
  let rebuiltCount = 0;


  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();


    markRuntimeStep_(timing, "Start");
    const currentDemo = getCurrentDemoPSheet_(monthParts);
    const previousDemo = getPreviousDemoPSheet_(monthParts);
    const previousMaster = getPreviousMasterListSheet_(monthParts);
    markRuntimeStep_(timing, "Locate monthly source sheets");


    if (!currentDemo) {
      notify_("Current month Demo P was not found. Format Demo P first.");
      markRuntimeStep_(timing, "Stopped - missing current Demo P");
      return null;
    }


    if (!previousMaster) {
      notify_("Previous month Master List was not found. Create the previous month Master List first.");
      markRuntimeStep_(timing, "Stopped - missing previous Master List");
      return null;
    }


    currentMaster = copyPreviousMasterListToCurrentMonth_(previousMaster, monthParts);
    markRuntimeStep_(timing, "Copy previous Master List to current month");


    let reportSheet;
    try {
      reportSheet = getOrBuildMonthlyChangeReport_(ss, monthParts, currentDemo, previousDemo);
      markRuntimeStep_(timing, "Get or build Monthly Change Report");
    } catch (err) {
      markRuntimeStep_(timing, "Stopped - Monthly Change Report error");
      notify_(err.message);
      return null;
    }


    if (reportSheet) {
      rebuiltCount = rebuildChangedPMRsOnExistingMaster_(
        currentDemo,
        currentMaster,
        reportSheet,
        monthParts,
        true,
        timing
      );
      markRuntimeStep_(timing, "Rebuild changed PMRs on copied Master List");
    } else {
      processMasterListFull_(currentMaster, monthParts, null);
      markRuntimeStep_(timing, "No change report rows - full current-month processing");
    }


    finalizeWorkflowAfterCreateOrUpdate_();
    markRuntimeStep_(timing, "Finalize workflow after create/update");


    hidePreviousMonthSheets_(monthParts);
    markRuntimeStep_(timing, "Hide previous month sheets");


    currentMaster.activate();


    removeDisenrolledPMRBlocksFromMasterUsingDemoP_(currentMaster, currentDemo);
    markRuntimeStep_(timing, "Remove disenrolled PMR blocks");


    prepareMasterListSortOrderBeforeFormatting_(currentMaster);
    markRuntimeStep_(timing, "Prepare Sort Order before formatting");


    applyMasterListTemplateModeUpdateDisplay_(currentMaster);
    markRuntimeStep_(timing, "Apply template-mode update display settings");


    applyFinalMasterListColorAndDisplay_(currentMaster);
    markRuntimeStep_(timing, "Apply final Sort Order coloring and display lock");


    autoHidePreviousMonthSheetsAfterWorkflow_(monthParts);
    markRuntimeStep_(timing, "Auto-hide previous month sheets");


    enforceMasterListFinalDisplay_(currentMaster);
    markRuntimeStep_(timing, "Final Master List display enforcement");


    restorePreviouslyHiddenSheets_(hiddenSheetNamesAtStart, currentMaster);
    markRuntimeStep_(timing, "Restore previously hidden sheets");


    writeRuntimeTimingReport_(timing);
    hideSystemSheets_();


    notify_(
      "Update Master List complete.\\n\\n" +
      `New Master List: ${currentMaster.getName()}\\n` +
      `Changed PMRs rebuilt: ${rebuiltCount}\\n` +
      `Runtime: ${formatSeconds_((new Date().getTime() - timing.startMs) / 1000)}\\n` +
      "Previous month sheets hidden."
    );


    return currentMaster;
  } catch (err) {
    markRuntimeStep_(timing, "ERROR - " + err.message);
    try {
      writeRuntimeTimingReport_(timing);
    } catch (reportErr) {
      Logger.log("Could not write timing report: " + reportErr.message);
    }
    throw err;
  }
}

function getChangedPMRsFromMonthlyChangeReport_(reportSheet) {
  const result = new Map();
  const lastRow = reportSheet.getLastRow();
  const lastCol = reportSheet.getLastColumn();


  if (lastRow < 1 || lastCol < 1) return result;


  const sectionTitles = ["Enrollments", "Disenrollments", "Demographic Changes", "Caseload Changes", "Banner Changes", "Contact Changes"];


  let foundSectionLayout = false;


  sectionTitles.forEach(section => {
    const titleRow = findMonthlyChangeSectionTitleRow_(reportSheet, section);
    if (titleRow === -1) return;


    foundSectionLayout = true;


    const headerRow = titleRow + 2;
    const dataStartRow = headerRow + 1;
    const nextTitleRow = findNextMonthlyChangeSectionTitleRow_(reportSheet, titleRow);
    const dataEndRow = nextTitleRow === -1 ? lastRow : nextTitleRow - 1;


    if (dataEndRow < dataStartRow) return;


    const headers = reportSheet.getRange(headerRow, 1, 1, lastCol)
      .getValues()[0]
      .map(h => String(h || "").trim());
    const headerMap = buildHeaderIndexMap_(headers);
    const pmrIdx = getPMRIndex_(headerMap);


    if (pmrIdx === -1) return;


    const values = reportSheet.getRange(dataStartRow, 1, dataEndRow - dataStartRow + 1, lastCol).getValues();


    values.forEach(row => {
      const pmr = normalizePMR_(row[pmrIdx]);
      if (!pmr) return;


      if (!result.has(pmr)) result.set(pmr, new Set());
      result.get(pmr).add(section);
    });
  });


  if (foundSectionLayout) return result;


  if (lastRow < 5) return result;


  const headers = reportSheet.getRange(4, 1, 1, lastCol)
    .getValues()[0]
    .map(h => String(h || "").trim());
  const headerMap = buildHeaderIndexMap_(headers);


  const sectionIdx = headerMap["Section"];
  const pmrIdx = headerMap["Participant PMR#"];


  if (sectionIdx === undefined || pmrIdx === undefined) return result;


  const values = reportSheet.getRange(5, 1, lastRow - 4, lastCol).getValues();


  values.forEach(row => {
    const pmr = normalizePMR_(row[pmrIdx]);
    const section = String(row[sectionIdx] || "").trim();


    if (!pmr) return;


    if (!result.has(pmr)) result.set(pmr, new Set());
    if (section) result.get(pmr).add(section);
  });


  return result;
}

function copyChangedPMRsFromDemoPToMasterList_(demoSheet, masterSheet, pmrsToCopy, monthParts) {
  if (!demoSheet || !masterSheet || !pmrsToCopy || pmrsToCopy.size === 0) return 0;


  const demoLastRow = demoSheet.getLastRow();
  const demoLastCol = demoSheet.getLastColumn();


  if (demoLastRow < DATA_START_ROW || demoLastCol < 1) return 0;


  const demoHeaders = getHeaders_(demoSheet, HEADER_ROW);
  const demoHeaderMap = buildHeaderIndexMap_(demoHeaders);
  const masterHeaders = getHeaders_(masterSheet, HEADER_ROW);
  const masterHeaderMap = buildHeaderIndexMap_(masterHeaders);
  const pmrIdx = getPMRIndex_(demoHeaderMap);


  if (pmrIdx === -1) throw new Error("Demo P is missing Participant PMR# / PMR # column.");


  const demoValues = demoSheet.getRange(DATA_START_ROW, 1, demoLastRow - DATA_START_ROW + 1, demoLastCol).getValues();
  const disenrolledPMRs = buildDisenrolledPMRSetFromDemoPValues_(demoValues, demoHeaderMap);
  const exclusionPMRs = loadDisenrolledExclusionPMRsForPart3_();
  exclusionPMRs.forEach(pmr => disenrolledPMRs.add(pmr));


  const output = [];


  demoValues.forEach(row => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr) return;
    if (!pmrsToCopy.has(pmr)) return;
    if (disenrolledPMRs.has(pmr)) return;


    const out = new Array(masterHeaders.length).fill("");
    demoHeaders.forEach((header, sourceIdx) => {
      if (!header) return;
      const targetIdx = masterHeaderMap[header];
      if (targetIdx !== undefined) out[targetIdx] = row[sourceIdx];
    });


    const statusIdx = masterHeaderMap["Master List Update Status"];
    const monthIdx = masterHeaderMap["Master List Update Month"];
    const sourceIdx = masterHeaderMap["Master List Source Sheet"];


    if (statusIdx !== undefined) out[statusIdx] = "Updated from Demo P";
    if (monthIdx !== undefined) out[monthIdx] = monthParts.firstDay;
    if (sourceIdx !== undefined) out[sourceIdx] = demoSheet.getName();


    output.push(out);
  });


  if (output.length === 0) return 0;


  const appendRow = Math.max(masterSheet.getLastRow() + 1, DATA_START_ROW);
  masterSheet.getRange(appendRow, 1, output.length, masterHeaders.length).setValues(output);


  return output.length;
}

function shouldProcessRowByPMR_(row, headerMap, pmrFilter) {


  const primaryIdx = headerMap["Primary PMR Row"];


  if (primaryIdx !== undefined) {
    if (!isPrimaryPMRRowValue_(row[primaryIdx])) return false;
  } else {
    const dobIdx = getDOBIndex_(headerMap);
    if (dobIdx !== -1 && normalizeCompareValue_(row[dobIdx]) === "") return false;
  }


  if (!pmrFilter || pmrFilter.size === 0) return true;


  const pmrIdx = getPMRIndex_(headerMap);
  if (pmrIdx === -1) return false;


  const pmr = normalizePMR_(row[pmrIdx]);
  return pmrFilter.has(pmr);
}

function syncRowsFromSourceMap_(masterSheet, sourceMap, config, pmrFilter) {
  const data = getDataValues_(masterSheet, HEADER_ROW, DATA_START_ROW);


  if (!data.values.length) return;


  const headers = data.headers;
  const headerMap = data.headerMap;
  const values = data.values;


  const fields = normalizeSyncFieldPairs_(config.fields);


  values.forEach(row => {
    if (!shouldProcessRowByPMR_(row, headerMap, pmrFilter)) return;


    const key = config.masterKeyHeaders
      .map(header => normalizeKeyPart_(row[headerMap[header]]))
      .join("|||");


    if (key.replace(/\|/g, "") === "") return;
    if (!sourceMap.has(key)) return;


    const source = sourceMap.get(key);


    fields.forEach(pair => {
      const destHeader = pair[0];
      const sourceHeader = pair[1];


      const destIdx = headerMap[destHeader];
      if (destIdx === undefined) return;


      if (!Object.prototype.hasOwnProperty.call(source, sourceHeader)) return;


      row[destIdx] = source[sourceHeader];
    });
  });


  if (data.range) data.range.setValues(values);
}




function normalizeSyncFieldPairs_(fields) {
  return fields.map(item => {
    if (Array.isArray(item)) return item;
    return [item, item];
  });
}


function buildSourceMapBySingleKeyForPart5_(sheet, headerRow, dataStartRow, keyHeader) {
  headerRow = headerRow || HEADER_ROW;
  dataStartRow = dataStartRow || DATA_START_ROW;


  const data = getDataValues_(sheet, headerRow, dataStartRow);
  const map = new Map();


  if (!data.values.length) return map;


  const headers = data.headers;
  const headerMap = data.headerMap;
  const keyIdx = headerMap[keyHeader];


  if (keyIdx === undefined) return map;


  data.values.forEach(row => {
    const key = normalizeKeyPart_(row[keyIdx]);
    if (!key) return;


    const record = {};
    headers.forEach((header, idx) => {
      if (header) record[header] = row[idx];
    });


    map.set(key, record);
  });


  return map;
}




function buildSourceMapByCompositeKeyForPart5_(sheet, headerRow, dataStartRow, keyHeaders) {

    headers.forEach((header, idx) => {
      if (header) record[header] = row[idx];
    });


    map.set(key, record);
  });


  return map;
}

  const dataRange = sheet.getRange(DATA_START_ROW, 1, lastRow - DATA_START_ROW + 1, sheet.getLastColumn());

    "Phone 3 - Value",

    "Phone 4 - Value"
  ], HEADER_ROW);


  const headers = getHeaders_(sheet, HEADER_ROW);
  const headerMap = buildHeaderIndexMap_(headers);


  const phoneIdx = headerMap["Phone Number"];
  if (phoneIdx === undefined) return;

      if (valueIdx !== undefined) row[valueIdx] = phone;
    });

function buildContactEntriesByPMR_(data, headers, headerMap) {
  const pmrIdx = getPMRIndex_(headerMap);
  if (pmrIdx === -1) return new Map();


  const rank = {
    "guardian": 1,
    "mdpoa": 2,
    "power of attorney": 3,
    "emergency contact #1": 4,
    "emergency contact #2": 5,
    "emergency contact #3": 6,
    "caregiver": 7,
    "spouse": 8,
    "lives with": 9,
    "direct ppt responsibility": 10,
    "family": 11,
    "family contact": 12,
    "follow up contact": 13,
    "friend": 14,
    "healthcare proxy": 15,
    "hos contact": 16,
    "mailing contact #1": 17,
    "mailing contact #2": 18,
    "next of kin": 19,
    "others": 20,
    "partner": 21,
    "attorney/agent": 22,
    "other": 23,
    "other financial": 24,
    "guarantor": 25
  };


  const contactFirstIdx = headerMap["Contact - First Name"];
  const contactLastIdx = headerMap["Contact - Last Name"];
  const participantFirstIdx = headerMap["First Name"];
  const participantLastIdx = headerMap["Last Name"];
  const relationshipIdx = headerMap["Relationship"];
  const typeIdx = headerMap["Type of Contact"];
  const languageIdx = headerMap["Contact - Primary Language"];


  const phoneIndexes = [
    headerMap["AD1 - Phone"],
    headerMap["AD2 - Phone"],
    headerMap["AD3 - Phone"]
  ].filter(idx => idx !== undefined && idx !== -1);


  const byPMR = new Map();


  data.forEach(row => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr) return;


    const contactFirst = contactFirstIdx !== undefined ? String(row[contactFirstIdx] || "").trim() : "";
    const contactLast = contactLastIdx !== undefined ? String(row[contactLastIdx] || "").trim() : "";
    if (!contactFirst && !contactLast) return;


    const participantFirst = participantFirstIdx !== undefined ? String(row[participantFirstIdx] || "").trim() : "";
    const participantLast = participantLastIdx !== undefined ? String(row[participantLastIdx] || "").trim() : "";


    if (
      contactFirst.toLowerCase() === participantFirst.toLowerCase() &&
      contactLast.toLowerCase() === participantLast.toLowerCase()
    ) {
      return;
    }


    if (!byPMR.has(pmr)) byPMR.set(pmr, new Map());


    const contactKey = `${contactFirst.toLowerCase()}|${contactLast.toLowerCase()}`;
    const contactMap = byPMR.get(pmr);


    if (!contactMap.has(contactKey)) {
      contactMap.set(contactKey, {
        first: contactFirst,
        last: contactLast,
        types: new Set(),
        relationships: new Set(),
        phones: new Set(),
        language: "",
        rank: 99
      });
    }


    const entry = contactMap.get(contactKey);


    const relationship = relationshipIdx !== undefined ? String(row[relationshipIdx] || "").trim() : "";
    const type = typeIdx !== undefined ? String(row[typeIdx] || "").trim() : "";
    const language = languageIdx !== undefined ? String(row[languageIdx] || "").trim() : "";


    if (relationship) {
      entry.relationships.add(relationship);
      const relationshipRank = rank[relationship.toLowerCase()] || 99;
      if (relationshipRank < entry.rank) entry.rank = relationshipRank;
    }


    if (type) entry.types.add(type);


    if (language && language.toLowerCase() !== "english") {
      entry.language = language;
    }


    phoneIndexes.forEach(phoneIdx => {
      if (phoneIdx === undefined || phoneIdx === -1) return;
      if (!row[phoneIdx]) return;


      if (isExpiredADPhoneCell_(row, headers, phoneIdx)) return;


      entry.phones.add(String(row[phoneIdx]).trim());
    });
  });


  return byPMR;
}


function formatContactEntryForMasterList_(entry) {
  if (!entry) return "";


  const name = `${entry.first || ""} ${entry.last || ""}`.trim();
  const relationships = Array.from(entry.relationships).join(", ");
  const phones = Array.from(entry.phones).slice(0, 4).join(", ");
  const types = Array.from(entry.types).join(", ");


  let output = `${name} - ${relationships} - ${phones}---${types}`.trim();


  if (entry.language) {
    output += ` Language--${entry.language}`;
  }


  return output;
}


function writePMRContactsToParticipantRows_(sheet, data, headers, headerMap, pmrFilter) {
  const pmrIdx = getPMRIndex_(headerMap);
  if (pmrIdx === -1) return;


  const contactColumns = [1, 2, 3, 4, 5, 6, 7, 8].map(n => headerMap[`Contact - ${n}`]);
  const contactSummaryIdx = headerMap["Contact - Summary"];
  const bannerSummaryIdx = headerMap["Banner Summary"];
  const contactNotesIdx = headerMap["Contact - Notes"];


  const contactMapByPMR = buildContactEntriesByPMR_(data, headers, headerMap);


  data.forEach((row) => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr) return;


    if (pmrFilter && pmrFilter.size > 0 && !pmrFilter.has(pmr)) return;


    const entriesMap = contactMapByPMR.get(pmr);
    const sortedEntries = entriesMap
      ? Array.from(entriesMap.values()).sort((a, b) => a.rank - b.rank)
      : [];


    const summaryParts = [];


    if (bannerSummaryIdx !== undefined && row[bannerSummaryIdx]) {
      summaryParts.push(String(row[bannerSummaryIdx]).trim());
    }


    contactColumns.forEach((colIdx, i) => {
      if (colIdx === undefined || colIdx === -1) return;


      const entryText = sortedEntries[i] ? formatContactEntryForMasterList_(sortedEntries[i]) : "";
      row[colIdx] = entryText;


      if (entryText) summaryParts.push(entryText);
    });


    if (contactNotesIdx !== undefined && row[contactNotesIdx]) {
      summaryParts.push(String(row[contactNotesIdx]).trim());
    }


    if (contactSummaryIdx !== undefined) {
      row[contactSummaryIdx] = summaryParts.join("\n");
    }
  });
}

function buildContactMapByPMR_(data, headerMap, pmrFilter) {
  const contactMap = new Map();


  const pmrIdx = getPMRIndex_(headerMap);
  const firstIdx = headerMap["First Name"];
  const lastIdx = headerMap["Last Name"];
  const contactFirstIdx = headerMap["Contact - First Name"];
  const contactLastIdx = headerMap["Contact - Last Name"];
  const relIdx = headerMap["Relationship"];
  const typeIdx = headerMap["Type of Contact"];
  const langIdx = headerMap["Contact - Primary Language"];


  const phoneIdxs = [
    headerMap["AD1 - Phone"],
    headerMap["AD2 - Phone"],
    headerMap["AD3 - Phone"]
  ].filter(idx => idx !== undefined);


  if (
    pmrIdx === -1 ||
    firstIdx === undefined ||
    lastIdx === undefined ||
    contactFirstIdx === undefined ||
    contactLastIdx === undefined
  ) {
    return contactMap;
  }


  data.forEach(row => {
    if (!shouldProcessRowByPMR_(row, headerMap, pmrFilter)) return;


    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr) return;


    const participantFirst = String(row[firstIdx] || "").trim();
    const participantLast = String(row[lastIdx] || "").trim();
    const contactFirst = String(row[contactFirstIdx] || "").trim();
    const contactLast = String(row[contactLastIdx] || "").trim();


    if (!contactFirst && !contactLast) return;


    if (
      contactFirst.toLowerCase() === participantFirst.toLowerCase() &&
      contactLast.toLowerCase() === participantLast.toLowerCase()
    ) {
      return;
    }


    if (!contactMap.has(pmr)) contactMap.set(pmr, new Map());


    const perParticipant = contactMap.get(pmr);
    const contactKey = `${contactFirst.toLowerCase()}|${contactLast.toLowerCase()}`;


    if (!perParticipant.has(contactKey)) {
      perParticipant.set(contactKey, {
        first: contactFirst,
        last: contactLast,
        types: new Set(),
        relationships: new Set(),
        phones: new Set(),
        language: "",
        rank: 999
      });
    }


    const contact = perParticipant.get(contactKey);


    const relationship = relIdx !== undefined ? String(row[relIdx] || "").trim() : "";
    const type = typeIdx !== undefined ? String(row[typeIdx] || "").trim() : "";
    const language = langIdx !== undefined ? String(row[langIdx] || "").trim() : "";


    if (relationship) {
      contact.relationships.add(capitalizeFirstLetter_(relationship));
      const rank = CONTACT_RANK_ORDER[relationship.toLowerCase()] || 999;
      if (rank < contact.rank) contact.rank = rank;
    }


    if (type) contact.types.add(capitalizeFirstLetter_(type));


    if (language && language.toLowerCase() !== "english") {
      contact.language = language;
    }


    phoneIdxs.forEach(idx => {
      const phone = String(row[idx] || "").trim();
      if (phone) contact.phones.add(phone);
    });
  });


  return contactMap;
}


function writeRankedContactsToRows_(data, headerMap, contactMap, pmrFilter) {
  const pmrIdx = getPMRIndex_(headerMap);


  const contactColIdxs = [1, 2, 3, 4, 5, 6, 7, 8]
    .map(num => headerMap[`Contact - ${num}`]);


  if (pmrIdx === -1) return;


  data.forEach(row => {
    if (!shouldProcessRowByPMR_(row, headerMap, pmrFilter)) return;


    const pmr = normalizePMR_(row[pmrIdx]);
    const contactsForPMR = contactMap.get(pmr);


    contactColIdxs.forEach(idx => {
      if (idx !== undefined) row[idx] = "";
    });


    if (!contactsForPMR) return;


    const sorted = Array.from(contactsForPMR.values())
      .sort((a, b) => a.rank - b.rank || `${a.last} ${a.first}`.localeCompare(`${b.last} ${b.first}`));


    for (let i = 0; i < 8; i++) {
      const idx = contactColIdxs[i];
      if (idx === undefined) continue;


      const contact = sorted[i];
      if (!contact) {
        row[idx] = "";
        continue;
      }


      const phoneText = Array.from(contact.phones).slice(0, 4).join(", ");
      const relationshipText = Array.from(contact.relationships).join(", ");
      const typeText = Array.from(contact.types).join(", ");


      let text = `${contact.first} ${contact.last}`.trim();


      if (relationshipText) text += ` - ${relationshipText}`;
      if (phoneText) text += ` - ${phoneText}`;
      if (typeText) text += `---${typeText}`;
      if (contact.language) text += ` Language--${contact.language}`;


      row[idx] = text;
    }
  });
}

    const parts = [];


    if (bannerIdx !== undefined && String(row[bannerIdx] || "").trim()) {
      parts.push(String(row[bannerIdx]).trim());
    }


    contactColIdxs.forEach(idx => {
      const value = String(row[idx] || "").trim();
      if (value) parts.push(value);
    });


    if (contactNotesIdx !== undefined && String(row[contactNotesIdx] || "").trim()) {
      parts.push(String(row[contactNotesIdx]).trim());
    }


    row[summaryIdx] = parts.join("\n");
  });
}


function combineNotesSummary(sheet, pmrFilter) {
  sheet = sheet || getActiveMasterListOrSheet_();


  const lastRow = sheet.getLastRow();
  if (lastRow < DATA_START_ROW) return;


  ensureHeaders_(sheet, ["Notes"], HEADER_ROW);


  const headers = getHeaders_(sheet, HEADER_ROW);
  const headerMap = buildHeaderIndexMap_(headers);


  const notesIdx = headerMap["Notes"];
  if (notesIdx === undefined) return;


  const sourceHeaders = [
    "Banner Summary",
    "Contact - Summary",
    "Additional Important Information"
  ];


  const dataRange = sheet.getRange(DATA_START_ROW, 1, lastRow - DATA_START_ROW + 1, sheet.getLastColumn());
  const data = dataRange.getValues();


  data.forEach(row => {
    if (!shouldProcessRowByPMR_(row, headerMap, pmrFilter)) return;


    const parts = [];


    sourceHeaders.forEach(header => {
      const idx = headerMap[header];
      if (idx === undefined) return;


      const value = String(row[idx] || "").trim();
      if (value) parts.push(value);
    });


    row[notesIdx] = parts.join("\n\n");
  });


  dataRange.setValues(data);
}

function collectMasterListHealthCheckRows_(masterSheet) {
  const outputRows = [["Check", "Status", "Count", "Quality Notes"]];

  if (!masterSheet) {
    outputRows.push(["Master List Sheet", "FAIL", 0, "Master List sheet was not found."]);
    return outputRows;
  }

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const data = getDataValues_(masterSheet, HEADER_ROW, DATA_START_ROW);
  const rows = data.values || [];
  const headerMap = data.headerMap || {};
  const pmrIdx = getPMRIndex_(headerMap);
  const bannerIdx = headerMap["Banner Summary"];

  const carePlanHeaders = [
    "Participant Name",
    "IDT Meeting Date",
    "Care Plan Start Date",
    "Enrollment Date",
    "Last Care Plan",
    "Next Care Plan Due",
    "CP Type"
  ];

  const detailRows = [];
  const byPMR = new Map();
  const duplicatePMRs = [];
  let rowsMissingPMR = 0;
  let rowsMissingBannerSummary = 0;
  let rowsMissingCarePlanFields = 0;

  rows.forEach(function(row, offset) {
    const rowNumber = DATA_START_ROW + offset;
    const pmr = pmrIdx === -1 ? "" : normalizePMR_(row[pmrIdx]);

    if (!pmr) {
      rowsMissingPMR++;
      detailRows.push(["Rows Missing PMR Detail", "FAIL", rowNumber, "Participant PMR# / PMR # is blank."]);
      return;
    }

    if (!byPMR.has(pmr)) byPMR.set(pmr, []);
    byPMR.get(pmr).push({ rowNumber: rowNumber, row: row });

    if (bannerIdx !== undefined && String(row[bannerIdx] || "").trim() === "") {
      rowsMissingBannerSummary++;
      detailRows.push(["Missing Banner Summary", "WARNING", pmr, "Master List row " + rowNumber + " has no Banner Summary."]);
    }

    const missingCareFields = carePlanHeaders.filter(function(header) {
      const idx = headerMap[header];
      return idx !== undefined && String(row[idx] || "").trim() === "";
    });

    if (missingCareFields.length > 0) {
      rowsMissingCarePlanFields++;
      detailRows.push(["Missing Care Plan Field(s)", "WARNING", pmr, missingCareFields.join(", ")]);
    }
  });

  byPMR.forEach(function(items, pmr) {
    if (items.length > 1) {
      duplicatePMRs.push(pmr);
      detailRows.push(["Duplicate PMR", "WARNING", pmr, "Rows: " + items.map(function(item) { return item.rowNumber; }).join(", ")]);
    }
  });

  const masterSuffix = getWorkflowMonthSuffixFromSheet_(masterSheet);
  const demoSheet = masterSuffix
    ? findMonthlyOutputSheetByPrefixAndSuffix_(ss, DEMO_P_PREFIX, masterSuffix)
    : getNewestFormattedMonthlySheetByPrefix_(ss, DEMO_P_PREFIX);

  let demoPrimaryPMRs = new Set();
  if (demoSheet) {
    const demoData = getDataValues_(demoSheet, HEADER_ROW, DATA_START_ROW);
    const demoPMRIdx = getPMRIndex_(demoData.headerMap || {});
    const demoPrimaryIdx = (demoData.headerMap || {})["Primary PMR Row"];
    (demoData.values || []).forEach(function(row) {
      if (demoPrimaryIdx !== undefined && String(row[demoPrimaryIdx] || "").trim().toLowerCase() !== "yes") return;
      const pmr = demoPMRIdx === -1 ? "" : normalizePMR_(row[demoPMRIdx]);
      if (pmr) demoPrimaryPMRs.add(pmr);
    });
  }

  const masterPMRs = new Set(Array.from(byPMR.keys()));
  const demoMasterMatches = countSetMatches_(masterPMRs, demoPrimaryPMRs);

  outputRows.push(["Master List Sheet", "PASS", 1, masterSheet.getName()]);
  outputRows.push(["Master List Month", masterSuffix ? "PASS" : "WARNING", masterSuffix || "Unknown", "Primary PMR validation is not performed here; Primary PMR is owned by Demo P Section M."]);
  outputRows.push(["Total Master List Rows", "INFO", rows.length, masterSheet.getName()]);
  outputRows.push(["Unique PMRs", "INFO", byPMR.size, ""]);
  outputRows.push(["Rows Missing PMR", rowsMissingPMR === 0 ? "PASS" : "FAIL", rowsMissingPMR, ""]);
  outputRows.push(["Duplicate PMRs", duplicatePMRs.length === 0 ? "PASS" : "WARNING", duplicatePMRs.length, ""]);
  outputRows.push(["Rows Missing Banner Summary", rowsMissingBannerSummary === 0 ? "PASS" : "WARNING", rowsMissingBannerSummary, ""]);
  outputRows.push(["Rows Missing Care Plan Fields", rowsMissingCarePlanFields === 0 ? "PASS" : "WARNING", rowsMissingCarePlanFields, ""]);
  outputRows.push(["Master List / Demo P Reconciliation", demoSheet && demoMasterMatches === masterPMRs.size ? "PASS" : "WARNING", demoMasterMatches, "Master PMRs: " + masterPMRs.size + "; Demo P primary PMRs: " + demoPrimaryPMRs.size + (demoSheet ? "; Demo P: " + demoSheet.getName() : "; Demo P not found")]);

  if (detailRows.length) {
    outputRows.push(["Detail", "INFO", detailRows.length, "Detailed issues follow."]);
    detailRows.slice(0, 100).forEach(function(row) {
      outputRows.push(row);
    });
  }

  return outputRows;
}

function runMonthlyChangeValidation() {
  const monthParts = promptForLockedYearReportMonth_("Monthly Change Validation");
  if (!monthParts) return;


  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const reportSheet = getMonthlySheetByPrefixAndDate_(ss, MONTHLY_CHANGE_REPORT_PREFIX, monthParts.firstDay, monthParts.lastDay);


  if (!reportSheet) {
    notify_("Monthly Change Report was not found for the selected month.");
    return;
  }


  const sections = ["Enrollments", "Disenrollments", "Demographic Changes", "Caseload Changes", "Contact Changes"];
  const summaryRows = [];
  const detailRows = [];


  sections.forEach(section => {
    const titleRow = findMonthlyChangeSectionTitleRow_(reportSheet, section);
    if (titleRow === -1) {
      summaryRows.push([section, "FAIL", 0, "Section title not found."]);
      detailRows.push(["FAIL", section, "Missing section title."]);
      return;
    }


    const headerRow = titleRow + 2;
    const dataStartRow = headerRow + 1;
    const nextTitleRow = findNextMonthlyChangeSectionTitleRow_(reportSheet, titleRow);
    const dataEndRow = nextTitleRow === -1 ? reportSheet.getLastRow() : nextTitleRow - 1;


    if (dataEndRow < dataStartRow) {
      summaryRows.push([section, "PASS", 0, "No rows in section."]);
      return;
    }


    const lastCol = Math.max(reportSheet.getLastColumn(), 1);
    const headers = reportSheet.getRange(headerRow, 1, 1, lastCol).getValues()[0].map(h => String(h || "").trim());
    const headerMap = buildHeaderIndexMap_(headers);
    const pmrIdx = getPMRIndex_(headerMap);


    const values = reportSheet.getRange(dataStartRow, 1, dataEndRow - dataStartRow + 1, lastCol).getValues();
    const nonBlankRows = values.filter(row => row.some(value => String(value || "").trim() !== ""));
    const uniquePMRs = new Set();
    let missingPMR = 0;


    nonBlankRows.forEach((row, idx) => {
      const rowNumber = dataStartRow + idx;
      const pmr = pmrIdx === -1 ? "" : normalizePMR_(row[pmrIdx]);
      if (!pmr) {
        missingPMR++;
        detailRows.push(["WARNING", section, "Missing PMR on row " + rowNumber]);
      } else {
        uniquePMRs.add(pmr);
      }
    });


    summaryRows.push([section, missingPMR === 0 ? "PASS" : "WARNING", nonBlankRows.length, "Unique PMRs: " + uniquePMRs.size + "; Missing PMRs: " + missingPMR]);
  });


  writeDiagnosticReport_(
    "Monthly Change Validation",
    "Monthly Change Validation",
    monthParts,
    ["Section", "Status", "Rows", "Details"],
    summaryRows,
    ["Status", "Section", "Details"],
    detailRows
  );


  notify_("Monthly Change Validation complete.");
}


function getDiagnosticSourceKeyInfo_(sheet, headerRow, dataStartRow, keyHeader, usePMR) {
  const result = {
    rowsRead: 0,
    keys: new Set(),
    missingHeaders: []
  };


  if (!sheet) {
    result.missingHeaders.push("Source sheet not found");
    return result;
  }


  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();


  if (lastRow < dataStartRow || lastCol < 1) return result;


  const headers = getHeaders_(sheet, headerRow);
  const headerMap = buildHeaderIndexMap_(headers);
  const keyIdx = headerMap[keyHeader];


  if (keyIdx === undefined) {
    result.missingHeaders.push(keyHeader);
    return result;
  }


  const values = sheet.getRange(dataStartRow, 1, lastRow - dataStartRow + 1, lastCol).getValues();
  result.rowsRead = values.length;


  values.forEach(row => {
    const key = usePMR ? normalizePMR_(row[keyIdx]) : normalizeKeyPart_(row[keyIdx]);
    if (key) result.keys.add(key);
  });


  return result;
}


function countSetMatches_(leftSet, rightSet) {
  let count = 0;
  leftSet.forEach(value => {
    if (rightSet.has(value)) count++;
  });
  return count;
}


function writeDiagnosticReport_(sheetName, title, monthParts, summaryHeaders, summaryRows, detailHeaders, detailRows) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(sheetName);


  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
  } else {
    sheet.clear();
  }


  sheet;


  sheet.getRange("A1").setValue(title);
  sheet.getRange("A2").setValue("Date");
  sheet.getRange("B2").setValue(monthParts.firstDay);
  sheet.getRange("C2").setValue("to");
  sheet.getRange("D2").setValue(monthParts.lastDay);


  let row = 4;
  sheet.getRange(row, 1, 1, summaryHeaders.length).setValues([summaryHeaders]);
  sheet.getRange(row, 1, 1, summaryHeaders.length);
  row++;


  if (summaryRows && summaryRows.length > 0) {
    sheet.getRange(row, 1, summaryRows.length, summaryHeaders.length).setValues(summaryRows);
    row += summaryRows.length;
  }


  row += 2;
  sheet.getRange(row, 1).setValue("Details");
  row++;
  sheet.getRange(row, 1, 1, detailHeaders.length).setValues([detailHeaders]);
  sheet.getRange(row, 1, 1, detailHeaders.length);
  row++;


  if (detailRows && detailRows.length > 0) {
    sheet.getRange(row, 1, detailRows.length, detailHeaders.length).setValues(detailRows);
  } else {
    sheet.getRange(row, 1).setValue("No issues found.");
  }


  const lastRow = Math.max(sheet.getLastRow(), 1);
  const lastCol = Math.max(sheet.getLastColumn(), 1);


  sheet.getRange(1, 1, lastRow, lastCol);


  sheet;
  applyStandardRowHeights_(sheet);


  for (let col = 1; col <= lastCol; col++) {
    sheet;
  }


  if (sheet.getFilter()) sheet;
  sheet.getRange(4, 1, Math.max(lastRow - 3, 1), lastCol);
}

function startRuntimeTiming_(processName, monthParts) {
  const now = new Date().getTime();
  return {

function markRuntimeStep_(timing, label, details) {
  if (!timing) return;


  const now = new Date().getTime();
  const stepSeconds = (now - timing.lastMs) / 1000;
  const totalSeconds = (now - timing.startMs) / 1000;
  const severity = getRuntimeTimingSeverity_(stepSeconds);
  const cleanLabel = String(label || "Step");
  const cleanDetails = details === undefined || details === null ? "" : String(details);


  timing.steps.push({
    step: cleanLabel,
    seconds: stepSeconds,
    totalSeconds: totalSeconds,
    severity: severity,
    details: cleanDetails,
    timestamp: new Date()
  });


  if (severity !== "OK") {
    timing.warnings.push(cleanLabel + " - " + severity + " - " + formatSeconds_(stepSeconds));
  }


  timing.lastMs = now;
  logRuntimeTiming_(timing.processName, cleanLabel, stepSeconds, totalSeconds, severity, cleanDetails);
}

function findFrameworkTimingSectionRow_(sheet, sectionTitle) {
  if (!sheet) return 0;
  const target = String(sectionTitle || "").trim();
  const lastRow = sheet.getLastRow();
  if (lastRow < 1) return 0;

  const values = sheet.getRange(1, 1, lastRow, 1).getDisplayValues();
  for (let i = 0; i < values.length; i++) {
    if (String(values[i][0] || "").trim() === target) return i + 1;
  }
  return 0;
}

function findNextFrameworkTimingSectionRow_(sheet, startRow) {
  if (!sheet) return 0;
  const lastRow = sheet.getLastRow();
  if (startRow > lastRow) return 0;

  const values = sheet.getRange(startRow, 1, lastRow - startRow + 1, 1).getDisplayValues();
  for (let i = 0; i < values.length; i++) {
    const value = String(values[i][0] || "").trim();
    if (/^SECTION [A-Z] - /.test(value)) return startRow + i;
  }
  return 0;
}

function collectExistingFrameworkTimingSectionBlocks_(sheet) {
  const blocks = {};
  if (!sheet || sheet.getLastRow() < 1) return blocks;

  getFrameworkTimingSectionRegistry_().forEach(function(section) {
    const titleRow = findFrameworkTimingSectionRow_(sheet, section.title);
    if (!titleRow) return;

    const nextRow = findNextFrameworkTimingSectionRow_(sheet, titleRow + 1);
    const endRow = nextRow ? nextRow - 1 : sheet.getLastRow();
    const rowCount = Math.max(1, endRow - titleRow + 1);
    const values = sheet.getRange(titleRow, 1, rowCount, 6).getValues();
    blocks[section.id] = trimTrailingBlankRows_(values);
  });

  return blocks;
}

function normalizeFrameworkTimingSectionBlock_(section, block) {
  /*
   * v1.5.11:
   * Framework Timing sections use title/header template rows only.
   * Removed the generated "Rows / Status / Notes" summary rows.
   */
  let rows = trimTrailingBlankRows_(block || []);
  if (!rows.length || String(rows[0][0] || "").trim() !== section.title) {
    rows = buildDefaultFrameworkTimingSectionBlock_(section);
  }

  const dataRows = [];
  for (let i = 2; i < rows.length; i++) {
    const joined = (rows[i] || []).join(" ").toLowerCase();
    if (joined.indexOf("runtime") !== -1 || joined.indexOf("threshold") !== -1 || joined.indexOf("expected value") !== -1 || joined.indexOf("recommendation") !== -1) {
      for (let r = i + 1; r < rows.length; r++) {
        if (rowHasAnyValue_(rows[r])) dataRows.push(rows[r]);
      }
      break;
    }
  }
  if (!dataRows.length) dataRows.push(["Enter Data here"]);

  const output = [[section.title], ["Last Updated", "Date and time"], section.headers].concat(dataRows);
  return output.map(function(row) {
    return normalizeSectionRowForWidth_(row, 6);
  });
}

function rebuildFrameworkTimingReportShellCompact_(sheet) {
  if (!sheet) return;

  const requiredCols = 6;
  if (sheet.getMaxColumns() < requiredCols) {
    sheet.insertColumnsAfter(sheet.getMaxColumns(), requiredCols - sheet.getMaxColumns());
  }

  const rows = [];
  rows.push(normalizeSectionRowForWidth_(["Report Timing Framework - v" + MASTER_LIST_MERGE_ML_VERSION + " - Framework Timing Report"], requiredCols));

  getFrameworkTimingSectionRegistry_().forEach(function(section) {
    const block = buildDefaultFrameworkTimingSectionBlock_(section);
    block.forEach(function(row) {
      rows.push(normalizeSectionRowForWidth_(row, requiredCols));
    });
  });

  if (sheet.getMaxRows() < rows.length) {
    sheet.insertRowsAfter(sheet.getMaxRows(), rows.length - sheet.getMaxRows());
  }

  const clearRows = Math.max(sheet.getLastRow(), rows.length, 1);
  sheet.getRange(1, 1, clearRows, requiredCols).clearContent().breakApart();
  sheet.getRange(1, 1, rows.length, requiredCols).setValues(rows);

  try {
    sheet.getRange(1, 1, 1, requiredCols).merge()
      .setFontFamily("Arial")
      .setFontSize(12)
      .setFontWeight("bold")
      .setBackground(RFF_SYSTEM_SHEET_TITLE_COLOR);
    sheet.setFrozenRows(1);
  } catch (err) {
    Logger.log("Framework Timing compact shell formatting skipped: " + err.message);
  }

  styleFrameworkTimingReport_(sheet);
}


function hasFrameworkTimingReportShell_(sheet) {
  if (!sheet) return false;

  const expectedTitle = "Report Timing Framework - v" + MASTER_LIST_MERGE_ML_VERSION + " - Framework Timing Report";
  const actualTitle = String(sheet.getRange(1, 1).getDisplayValue() || "").trim();
  if (actualTitle !== expectedTitle) return false;

  const sections = getFrameworkTimingSectionRegistry_();
  for (let i = 0; i < sections.length; i++) {
    if (!findFrameworkTimingSectionRow_(sheet, sections[i].title)) return false;
  }

  return true;
}

function initializeFrameworkTimingSheet_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(getFrameworkTimingReportSheetName_());

  if (!sheet) {
    sheet = ss.insertSheet(getFrameworkTimingReportSheetName_());
  } else {
    showSheetIfNeeded_(sheet);
  }

  if (!hasFrameworkTimingReportShell_(sheet)) {
    rebuildFrameworkTimingReportShellCompact_(sheet);
  } else {
    styleFrameworkTimingReport_(sheet);
  }

  return sheet;
}


function ensureFrameworkTimingReport_() {
  return initializeFrameworkTimingSheet_();
}

function styleFrameworkTimingReport_(sheet) {
  if (!sheet) return;

  const widths = [300, 300, 500, 350, 350, 350];
  for (let i = 0; i < widths.length; i++) {
    try {
      if (sheet.getColumnWidth(i + 1) !== widths[i]) sheet.setColumnWidth(i + 1, widths[i]);
    } catch (err) {
      Logger.log("Framework Timing width skipped for column " + (i + 1) + ": " + err.message);
    }
  }

  const lastRow = Math.max(sheet.getLastRow(), 1);
  const width = widths.length;

  try {
    sheet.getRange(1, 1, lastRow, width)
      .setFontFamily("Arial")
      .setFontSize(10)
      .setVerticalAlignment("top")
      .setHorizontalAlignment("left")
      .setBorder(true, true, true, true, true, true, RFF_SYSTEM_SHEET_BORDER_COLOR, SpreadsheetApp.BorderStyle.DOTTED);

    sheet.getRange(1, 1, lastRow, 1).setWrapStrategy(SpreadsheetApp.WrapStrategy.OVERFLOW);
    [2, 3, 4, 5, 6].forEach(function(col) {
      sheet.getRange(1, col, lastRow, 1).setWrapStrategy(SpreadsheetApp.WrapStrategy.WRAP);
    });

    sheet.getRange(1, 1, 1, width)
      .setFontWeight("bold")
      .setFontSize(12)
      .setBackground(RFF_SYSTEM_SHEET_TITLE_COLOR);

    getFrameworkTimingSectionRegistry_().forEach(function(section) {
      const row = findFrameworkTimingSectionRow_(sheet, section.title);
      if (!row) return;

      sheet.getRange(row, 1, 1, width)
        .setFontWeight("bold")
        .setFontStyle("normal")
        .setFontSize(10)
        .setBackground(RFF_SYSTEM_SHEET_SECTION_COLOR);

      sheet.getRange(row + 1, 1, 1, width)
        .setFontWeight("normal")
        .setFontStyle("italic")
        .setFontSize(10)
        .setBackground(RFF_SYSTEM_SHEET_SUBHEADER_COLOR);

      const headerRange = sheet.getRange(row + 2, 1, 1, width);
      const headerValues = headerRange.getDisplayValues()[0];
      headerRange
        .setFontWeight("bold")
        .setFontStyle("normal")
        .setFontSize(10)
        .setBackgrounds([headerValues.map(function(value) {
          return String(value || "").trim() === "" ? RFF_SYSTEM_SHEET_UNUSED_COLOR : RFF_SYSTEM_SHEET_SUBHEADER_COLOR;
        })]);

      const nextSectionRow = findNextFrameworkTimingSectionRow_(sheet, row + 1);
      const dataStartRow = row + 3;
      const dataEndRow = nextSectionRow ? nextSectionRow - 1 : lastRow;
      const dataRowCount = Math.max(0, dataEndRow - dataStartRow + 1);
      if (dataRowCount > 0) {
        sheet.getRange(dataStartRow, 1, dataRowCount, width)
          .setFontWeight("normal")
          .setFontStyle("normal")
          .setBackground("#FFFFFF");
      }
    });

    sheet.setFrozenRows(1);
  } catch (err) {
    Logger.log("Framework Timing styling skipped: " + err.message);
  }
}

function getFrameworkTimingSectionForId_(sectionId) {
  const id = String(sectionId || "").trim().toUpperCase();
  const matches = getFrameworkTimingSectionRegistry_().filter(function(section) {
    return section.id === id || section.title === sectionId;
  });
  return matches.length ? matches[0] : null;
}

function replaceFrameworkTimingSectionRows_(sheet, sectionId, rows) {
  if (!sheet) return;

  const section = getFrameworkTimingSectionForId_(sectionId);
  if (!section) throw new Error("Unknown Framework Timing Report section: " + sectionId);

  if (!hasFrameworkTimingReportShell_(sheet)) {
    initializeFrameworkTimingSheet_();
  }

  const width = section.headers.length;
  const titleRow = findFrameworkTimingSectionRow_(sheet, section.title);
  if (!titleRow) {
    throw new Error("Framework Timing section not found: " + section.title);
  }

  const normalized = (rows || []).map(function(row) {
    const out = (row || []).slice(0, width);
    while (out.length < width) out.push("");
    return out;
  });

  const nextRow = findNextFrameworkTimingSectionRow_(sheet, titleRow + 1);
  const existingBlockRows = nextRow ? nextRow - titleRow : Math.max(1, sheet.getLastRow() - titleRow + 1);
  const desiredBlockRows = Math.max(4, normalized.length + 3);

  if (existingBlockRows < desiredBlockRows) {
    const insertAt = nextRow || (titleRow + existingBlockRows);
    sheet.insertRowsBefore(insertAt, desiredBlockRows - existingBlockRows);
  } else if (existingBlockRows > desiredBlockRows && nextRow) {
    sheet.deleteRows(titleRow + desiredBlockRows, existingBlockRows - desiredBlockRows);
  }

  sheet.getRange(titleRow, 1, desiredBlockRows, 6).clearContent();
  sheet.getRange(titleRow, 1, 1, 6).setValues([normalizeSectionRowForWidth_([section.title], 6)]);
  sheet.getRange(titleRow + 1, 1, 1, 2).setValues([["Last Updated", new Date()]]);
  sheet.getRange(titleRow + 2, 1, 1, width).setValues([section.headers]);

  if (normalized.length) {
    sheet.getRange(titleRow + 3, 1, normalized.length, width).setValues(normalized);
  } else {
    sheet.getRange(titleRow + 3, 1).setValue("Enter Data here");
  }

  sheet.getRange(titleRow + 1, 2).setNumberFormat("mm/dd/yyyy hh:mm:ss");
  styleFrameworkTimingReport_(sheet);
}

  if (text.indexOf("template") !== -1) return 45;

}

}

function getFrameworkTimingDetailRows_(sheet) {
  if (!sheet || sheet.getLastRow() < 1) return [];
  const section = getFrameworkTimingSectionForId_("C");
  if (!section) return [];

  const titleRow = findFrameworkTimingSectionRow_(sheet, section.title);
  if (!titleRow) return [];

  const dataStart = titleRow + 3;
  const nextRow = findNextFrameworkTimingSectionRow_(sheet, titleRow + 1);
  const endRow = nextRow ? nextRow - 1 : sheet.getLastRow();
  const dataRows = Math.max(0, endRow - dataStart + 1);

  return values.filter(function(row) {
    return row.some(function(value) {
      return String(value === null || value === undefined ? "" : value).trim() !== "" && String(value).trim() !== "Enter Data here";
    });
  });
}





function buildFrameworkTimingProcessSummaryRows_(detailRows) {
  const processMap = {};

  (detailRows || []).forEach(function(row) {
    const process = String(row[1] || "").trim();
    if (!process) return;

    const stepSeconds = Number(row[3]) || 0;
    const severity = String(row[4] || "OK").trim() || "OK";
    const timestamp = row[0] || "";

    if (!processMap[process]) {
      processMap[process] = {
        process: process,
        steps: 0,
        runtime: 0,
        slowSteps: 0,
        worstSeverity: "OK",
        lastTimestamp: timestamp
      };
    }

    const item = processMap[process];
    item.steps += 1;
    item.runtime += stepSeconds;
    item.lastTimestamp = timestamp || item.lastTimestamp;
    if (severity !== "OK") item.slowSteps += 1;
    item.worstSeverity = worseTimingSeverity_(item.worstSeverity, severity);
  });

  return Object.keys(processMap).sort().map(function(process) {
    const item = processMap[process];
    const benchmark = getFrameworkTimingBenchmarkForProcess_(process);
    const variance = benchmark ? item.runtime - benchmark : "";
    return [
      process,
      Number(item.runtime.toFixed(3)),
      item.worstSeverity === "OK" ? "PASS" : item.worstSeverity,
      benchmark || "",
      variance === "" ? "" : Number(variance.toFixed(3)),
      item.slowSteps ? item.slowSteps + " slow/bottleneck/critical step(s)" : "Within target"
    ];
  });
}



function buildFrameworkTimingIssueRows_(detailRows) {
  return (detailRows || []).filter(function(row) {
    const severity = String(row[4] || "").trim().toUpperCase();
    return severity === "SLOW" || severity === "BOTTLENECK" || severity === "CRITICAL";
  }).map(function(row) {
    const recommendation = getPerformanceRecommendationForTimingStep_(String(row[1] || ""), String(row[2] || ""), Number(row[3]) || 0, String(row[4] || ""));
    return [row[4], row[1], row[3], getFrameworkTimingThresholdForSeverity_(String(row[4] || "")), row[2], recommendation];
  });
}



function buildFrameworkTimingRecommendationRows_(issueRows) {
  const seen = {};
  const rows = [];

  (issueRows || []).forEach(function(row) {
    const process = String(row[1] || "").trim();
    const priority = String(row[0] || "").trim();
    const recommendation = String(row[5] || "").trim();
    const key = process + "|" + priority + "|" + recommendation;
    if (!process || seen[key]) return;
    seen[key] = true;
    rows.push([process, row[4] || "Performance issue", priority, recommendation, priority, "Open"]);
  });

  return rows;
}
