function runMonthlyUpdate() {
  const monthParts = promptForLockedYearReportMonth_("Monthly Update");
  if (!monthParts) return;


  const messages = [];
  messages.push("Automatic monthly-sheet hiding is disabled in v1.3.5; archive monthly sheets separately.");


  try {
    const reportSheet = buildMonthlyChangeReportFromDemoPForMonth_(monthParts);
    messages.push("Monthly Change Report: " + (reportSheet ? reportSheet.getName() : "No report created"));
  } catch (err) {
    notify_("Monthly Update stopped during Monthly Change Report: " + err.message);
    throw err;
  }


  try {
    const masterSheet = updateMasterListForMonth_(monthParts);
    messages.push("Master List Update: " + (masterSheet ? masterSheet.getName() : "Not completed"));
  } catch (err) {
    notify_("Monthly Update stopped during Update Master List: " + err.message);
    throw err;
  }


  try {
    createIndexSheet();
  } catch (err) {
    Logger.log("Monthly Update final index refresh skipped: " + err.message);
  }


  notify_("Monthly Update complete.\n\n" + messages.join("\n"));
}

function hidePreviousMonthSheets_(monthParts) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();


  const currentMonthStart = new Date(
    monthParts.firstDay.getFullYear(),
    monthParts.firstDay.getMonth(),
    1
  );


  activateSafeSheetBeforeHiding_(ss);


  const hiddenNames = [];
  const notHiddenNames = [];


  ss.getSheets().forEach(sheet => {
    const sheetName = String(sheet.getName() || "").trim();


    if (isSystemSheetName_(sheetName)) return;


    const sheetMonthStart = getSheetMonthStartFromName_(sheetName);
    if (!sheetMonthStart) return;


    if (sheetMonthStart.getTime() >= currentMonthStart.getTime()) return;


    hideSheetSafely_(sheet, hiddenNames, notHiddenNames);
  });


  hideBannerReportForPreviousMonth_(ss, monthParts, hiddenNames, notHiddenNames);
  hideSystemSheets_();


  return {
    hiddenNames: hiddenNames,
    notHiddenNames: notHiddenNames
  };
}

function activateSafeSheetBeforeHiding_(ss) {
  const sheets = ss.getSheets();


  const safeSheet =
    ss.getSheetByName("Index") ||
    sheets.find(sheet => {
      const name = String(sheet.getName() || "").trim();
      return !isSystemSheetName_(name) &&
        name.toLowerCase().indexOf("banner") === -1 &&
        !sheet.isSheetHidden();
    }) ||
    sheets.find(sheet => !sheet.isSheetHidden());


  if (safeSheet) {
    ss.setActiveSheet(safeSheet);
  }
}


function getSheetMonthStartFromName_(sheetName) {
  const match = String(sheetName || "").trim().match(/(\d{1,2})\.(\d{1,2})\.(\d{2}|\d{4})/);
  if (!match) return null;


  let year = Number(match[3]);
  if (year < 100) year += 2000;


  const month = Number(match[1]) - 1;
  const day = Number(match[2]);
  const date = new Date(year, month, day);


  if (
    date.getFullYear() !== year ||
    date.getMonth() !== month ||
    date.getDate() !== day
  ) {
    return null;
  }


  return new Date(year, month, 1);
}


function hideBannerReportForPreviousMonth_(ss, monthParts, hiddenNames, notHiddenNames) {
  const previousStartText = formatDateForSheetName_(monthParts.previousMonthFirstDay);
  const previousEndText = formatDateForSheetName_(monthParts.previousMonthLastDay);
  const previousRangeText = previousStartText + " - " + previousEndText;


  ss.getSheets().forEach(sheet => {
    const sheetName = String(sheet.getName() || "").trim();
    const lower = sheetName.toLowerCase();


    if (isSystemSheetName_(sheetName)) return;
    if (lower.indexOf("banner") === -1) return;
    if (sheetName.indexOf(previousRangeText) === -1) return;
    if (hiddenNames.indexOf(sheetName) !== -1) return;


    hideSheetSafely_(sheet, hiddenNames, notHiddenNames);
  });
}


function hideSheetSafely_(sheet, hiddenNames, notHiddenNames) {
  const sheetName = String(sheet.getName() || "").trim();


  try {
    sheet.hideSheet();


    if (sheet.isSheetHidden()) {
      hiddenNames.push(sheetName);
    } else {
      notHiddenNames.push(sheetName + " - hideSheet ran but sheet is still visible");
    }
  } catch (err) {
    notHiddenNames.push(sheetName + " - " + err.message);
    Logger.log("Could not hide sheet " + sheetName + ": " + err.message);
  }
}

    sheetName === DEMO_P_TEMPLATE_SHEET
  );
}

    headersToHide = headersToHide.concat(getMasterListAdditionalHiddenHeaders_());

  return hideColumnsByHeaders_(sheet, headersToHide);
}

}


function hideColumnNumberBatches_(sheet, columnNumbers) {
  if (!columnNumbers || columnNumbers.length === 0) return;


  const sorted = Array.from(new Set(columnNumbers)).sort((a, b) => a - b);
  let start = sorted[0];
  let previous = sorted[0];


  for (let i = 1; i < sorted.length; i++) {
    const current = sorted[i];


    if (current === previous + 1) {
      previous = current;
      continue;
    }


    sheet.hideColumns(start, previous - start + 1);
    start = current;
    previous = current;
  }


  sheet.hideColumns(start, previous - start + 1);
}

function promptForReportMonth_(title) {
  const ui = SpreadsheetApp.getUi();
  const response = ui.prompt(
    title || "Report Month",
    "Enter any date in the report month. Example: 05/01/26",
    ui.ButtonSet.OK_CANCEL
  );


  if (response.getSelectedButton() !== ui.Button.OK) return null;


  const reportDate = coerceToValidDate_(response.getResponseText());
  if (!reportDate) {
    notify_("Invalid date. Please enter a date like 05/01/26.");
    return null;
  }


  return getMonthDateParts_(reportDate);
}


function promptForLockedYearReportMonth_(title) {
  const ui = SpreadsheetApp.getUi();


  const monthResponse = ui.prompt(
    title || "Report Month",
    "Enter report month only as MM. Example: 05 for May.\n\nFebruary through November use the locked workflow year.\nDecember and January allow manual year input.",
    ui.ButtonSet.OK_CANCEL
  );


  if (monthResponse.getSelectedButton() !== ui.Button.OK) return null;


  const rawMonth = String(monthResponse.getResponseText() || "").trim();


  if (/^\d{1,2}[\/.\-]\d{1,2}[\/.\-](\d{2}|\d{4})$/.test(rawMonth)) {
    const reportDate = coerceToValidDate_(rawMonth);
    if (!reportDate) {
      notify_("Invalid date. Please enter a date like 05/01/26.");
      return null;
    }
    return getMonthDateParts_(reportDate);
  }


  if (!/^\d{1,2}$/.test(rawMonth)) {
    notify_("Invalid month. Enter a month from 01 through 12, or enter a full date like 05/01/26.");
    return null;
  }


  const month = Number(rawMonth);


  if (month < 1 || month > 12) {
    notify_("Invalid month. Enter a month from 01 through 12.");
    return null;
  }


  let year = getLockedWorkflowYearForMonthOnlyPrompt_();


  if (month === 12 || month === 1) {
    const yearResponse = ui.prompt(
      title || "Report Year",
      `Enter report year for ${String(month).padStart(2, "0")}/01. Example: ${month === 1 ? 2027 : 2026}`,
      ui.ButtonSet.OK_CANCEL
    );


    if (yearResponse.getSelectedButton() !== ui.Button.OK) return null;


    const yearText = String(yearResponse.getResponseText() || "").trim();


    if (!/^\d{4}$/.test(yearText)) {
      notify_("Invalid year. Please enter a 4-digit year such as 2026.");
      return null;
    }


    year = Number(yearText);


    if (year < 2000 || year > 2099) {
      notify_("Invalid year. Please enter a year from 2000 through 2099.");
      return null;
    }
  }


  return getMonthDateParts_(createLocalDateOnly_(year, month - 1, 1));
}

  const currentMonth = today.getMonth() + 1;


  if (currentYear < 2027) {

function getMonthDateParts_(date) {

  const previousMonthFirstDay = new Date(d.getFullYear(), d.getMonth() - 1, 1);

  };
}


function coerceToValidDate_(value) {
  if (value instanceof Date && !isNaN(value.getTime())) {
    return isReasonableReportDate_(value) ? value : null;
  }


  if (value === null || value === undefined || value === "") return null;


  if (typeof value === "number") {
    if (value < 30000) return null;
    return spreadsheetSerialDateToLocalDate_(value);
  }


  const text = String(value).trim();
  if (!text) return null;


  const match = text.match(/^(\d{1,2})[\/.\-](\d{1,2})[\/.\-](\d{2}|\d{4})$/);
  if (match) {
    const month = Number(match[1]) - 1;
    const day = Number(match[2]);
    let year = Number(match[3]);
    if (year < 100) year += 2000;


    const d = new Date(year, month, day);
    if (d.getFullYear() !== year || d.getMonth() !== month || d.getDate() !== day) return null;
    return isReasonableReportDate_(d) ? d : null;
  }


  const d = new Date(text);
  return !isNaN(d.getTime()) && isReasonableReportDate_(d) ? d : null;
}


function spreadsheetSerialDateToLocalDate_(serialValue) {
  if (serialValue === null || serialValue === undefined || serialValue === "") return null;


  const serial = Number(serialValue);
  if (isNaN(serial) || serial < 1) return null;


  const wholeDays = Math.floor(serial);
  const fractional = serial - wholeDays;


  const date = new Date(1899, 11, 30);
  date.setDate(date.getDate() + wholeDays);


  if (fractional > 0) {
    const millisInDay = Math.round(fractional * 24 * 60 * 60 * 1000);
    date.setMilliseconds(date.getMilliseconds() + millisInDay);
  }


  return isReasonableReportDate_(date) ? date : null;
}


function isReasonableReportDate_(date) {
  const year = date.getFullYear();
  return year >= 2000 && year <= 2099;
}

function isDateDisplayInReportWindow_(value, startDate, endDate) {

function convertMonthlyChangeReportDateValues_(rowValues, reportHeaders) {
  const output = rowValues.slice();


  GLOBAL_DATE_FORMAT_HEADERS.forEach(headerName => {
    const idx = reportHeaders.indexOf(headerName);
    if (idx === -1 || idx >= output.length) return;


    const value = output[idx];


    if (value instanceof Date && !isNaN(value.getTime())) return;


    if (typeof value === "number" && !isNaN(value) && value > 20000) {
      const convertedDate = spreadsheetSerialDateToLocalDate_(value);
      if (convertedDate) output[idx] = convertedDate;
      return;
    }


    const parsed = coerceToValidDate_(value);
    if (parsed) output[idx] = parsed;
  });


  return output;
}

    output[changeIdx] = changedColumns && changedColumns.size > 0
      ? Array.from(changedColumns).sort().join(", ")
      : "";
  }


  return convertMonthlyChangeReportDateValues_(output, reportHeaders);
}

function parseStandardMonthlySheetDateFromName_(sheetName, prefix) {
  const name = String(sheetName || "").trim();
  const prefixText = String(prefix || "").trim();
  if (!prefixText) return null;


  const escapedPrefix = prefixText.replace(/[.*+?^${}()|[\]\\]/g, "\\$&");
  const re = new RegExp("^" + escapedPrefix + "\\s+(\\d{1,2})\\.(\\d{2})(?:\\s|$|\\()", "i");
  const match = name.match(re);
  if (!match) return null;


  const month = Number(match[1]);
  const year = 2000 + Number(match[2]);
  if (month < 1 || month > 12 || year < 2000 || year > 2099) return null;


  return new Date(year, month - 1, 1);
}


function getNewestFormattedMonthlySheetByPrefix_(ss, prefix) {
  if (!ss || !prefix) return null;


  const sheets = ss.getSheets();
  const matches = [];


  sheets.forEach(sheet => {
    const parsed = parseStandardMonthlySheetDateFromName_(sheet.getName(), prefix);
    if (!parsed) return;
    matches.push({
      sheet,
      key: monthKey_(parsed),
      index: sheet.getIndex ? sheet.getIndex() : 0
    });
  });


  if (!matches.length) return null;


  matches.sort((a, b) => {
    if (a.key !== b.key) return a.key < b.key ? 1 : -1;
    return b.index - a.index;
  });


  return matches[0].sheet;
}


function setUniqueSheetName_(sheet, desiredName) {
  if (!sheet || !desiredName) return desiredName;


  const ss = sheet.getParent();
  let name = desiredName;
  let counter = 2;


  if (sheet.getName && sheet.getName() === desiredName) {
    return desiredName;
  }


  let existing = ss.getSheetByName(name);
  while (existing && existing.getSheetId && existing.getSheetId() !== sheet.getSheetId()) {
    name = `${desiredName} (${counter})`;
    counter++;
    existing = ss.getSheetByName(name);
  }


  if (sheet.getName() !== name) {
    sheet.setName(name);
    clearMonthlySheetLookupCache_();
    clearSheetRuntimeCachesForSheet_(sheet);
  }


  return name;
}


/**
 * Fast monthly output rename helper.
 * Deletes/replaces an existing monthly output with the same name before renaming,
 * then clears sheet lookup caches. This helper is required by Format Raw Data,
 * Demo P, Master List, and other monthly output functions.
 */
function setMonthlySheetNameFast_(sheet, desiredName) {
  if (!sheet || !desiredName) return desiredName;


  const ss = sheet.getParent();
  if (sheet.getName && sheet.getName() === desiredName) return desiredName;


  const existing = ss.getSheetByName(desiredName);
  if (existing && existing.getSheetId && existing.getSheetId() !== sheet.getSheetId()) {
    ss.deleteSheet(existing);
  }


  if (sheet.getName() !== desiredName) {
    sheet.setName(desiredName);
    clearMonthlySheetLookupCache_();
    clearSheetRuntimeCachesForSheet_(sheet);
  }


  return desiredName;
}

  // v1.4.2: all production sync sources use the standardized monthly output name Prefix mm.yy.
  if (!ss || !prefix) return null;


  const cacheKey = getMonthlySheetLookupCacheKey_(ss, prefix, firstDay, lastDay);
  const cachedSheet = ML_MONTHLY_SHEET_LOOKUP_CACHE_[cacheKey];
  if (cachedSheet) {
    try {
      if (cachedSheet.getName()) return cachedSheet;
    } catch (err) {
      delete ML_MONTHLY_SHEET_LOOKUP_CACHE_[cacheKey];
    }
  }


  const standardName = buildStandardMonthlySheetName_(prefix, firstDay);
  let sheet = ss.getSheetByName(standardName);
  if (sheet) {
    ML_MONTHLY_SHEET_LOOKUP_CACHE_[cacheKey] = sheet;
    return sheet;
  }


  sheet = getNewestFormattedMonthlySheetByPrefix_(ss, prefix);
  if (sheet) {
    ML_MONTHLY_SHEET_LOOKUP_CACHE_[cacheKey] = sheet;
    return sheet;
  }


  return null;
}

function buildRowsByPMR_(sheet, headerRow, dataStartRow) {
  headerRow = headerRow || HEADER_ROW;
  dataStartRow = dataStartRow || DATA_START_ROW;


  const data = getDataValues_(sheet, headerRow, dataStartRow);
  const result = new Map();


  if (!data.values.length) return result;


  const pmrIdx = getPMRIndex_(data.headerMap);


  if (pmrIdx === -1) throw new Error(`${sheet.getName()} is missing Participant PMR# / PMR # column.`);


  data.values.forEach((row, offset) => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr) return;


    if (!result.has(pmr)) result.set(pmr, []);
    result.get(pmr).push({
      rowNumber: dataStartRow + offset,
      values: row
    });
  });


  return result;
}






function getDataValues_(sheet, headerRow, dataStartRow) {
  headerRow = headerRow || HEADER_ROW;
  dataStartRow = dataStartRow || DATA_START_ROW;


  const dimensions = getSheetDimensions_(sheet);
  const lastRow = dimensions.lastRow;
  const lastCol = dimensions.lastCol;
  const headers = getHeaders_(sheet, headerRow);
  const headerMap = getHeaderMap_(sheet, headerRow);


  if (lastRow < dataStartRow || lastCol < 1) {
    return {
      headers: headers,
      headerMap: headerMap,
      values: [],
      range: null,
      lastRow: lastRow,
      lastCol: lastCol
    };
  }


  const range = sheet.getRange(dataStartRow, 1, lastRow - dataStartRow + 1, lastCol);


  return {
    headers: headers,
    headerMap: headerMap,
    values: range.getValues(),
    range: range,
    lastRow: lastRow,
    lastCol: lastCol
  };
}






function deleteRowNumberBatches_(sheet, rowNumbers) {
  if (!rowNumbers || rowNumbers.length === 0) return;


  const sorted = Array.from(new Set(rowNumbers)).sort((a, b) => b - a);


  let start = sorted[0];
  let end = sorted[0];


  for (let i = 1; i < sorted.length; i++) {
    const current = sorted[i];


    if (current === end - 1) {
      end = current;
      continue;
    }


    sheet.deleteRows(end, start - end + 1);
    start = current;
    end = current;
  }


  sheet.deleteRows(end, start - end + 1);
}

  }


  return text
    .toLowerCase()
    .replace(/\s+/g, " ");
}

function assignSortOrderByParticipant_(sheet) {
  sheet = sheet || SpreadsheetApp.getActiveSheet();


  const primaryCol = ensurePrimaryPMRRowColumn_(sheet);
  const sortCol = ensureSortOrderColumn_(sheet);
  if (sortCol < 1) {
    notify_("Sort Order skipped: Sort Order column is not defined in Dashboard Sheet Headers.");
    return;
  }
  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();


  if (lastRow < DATA_START_ROW || lastCol < 1) return;


  const headers = getHeaders_(sheet, HEADER_ROW);
  const headerMap = buildHeaderIndexMap_(headers);
  const pmrIdx = getPMRIndex_(headerMap);
  const primaryIdx = headerMap["Primary PMR Row"];


  if (pmrIdx === -1) {
    notify_("Sort Order skipped: Participant PMR# / PMR # column not found.");
    return;
  }


  const sortIdx = sortCol - 1;
  const dataRange = sheet.getRange(DATA_START_ROW, 1, lastRow - DATA_START_ROW + 1, lastCol);
  const values = dataRange.getValues();


  const blankOrNoPMRRows = [];
  values.forEach((row, idx) => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (pmr) return;
    if (primaryIdx !== undefined) row[primaryIdx] = "";
    row[sortIdx] = "";
    blankOrNoPMRRows.push({ row: row, originalIndex: idx });
  });


  const blocks = buildParticipantBlocksForSortOrder_(values, headerMap, pmrIdx);


  blocks.forEach(block => {
    assignPrimaryRowForBlock_(block, headerMap, primaryIdx);
  });


  blocks.sort((a, b) => {
    const aLast = normalizeKeyPart_(a.lastName);
    const bLast = normalizeKeyPart_(b.lastName);
    if (aLast !== bLast) return aLast < bLast ? -1 : 1;


    const aFirst = normalizeKeyPart_(a.firstName);
    const bFirst = normalizeKeyPart_(b.firstName);
    if (aFirst !== bFirst) return aFirst < bFirst ? -1 : 1;


    const aPMR = normalizePMR_(a.pmr);
    const bPMR = normalizePMR_(b.pmr);
    if (aPMR !== bPMR) return aPMR < bPMR ? -1 : 1;


    return 0;
  });


  let participantNumber = 1;
  const outputRows = [];


  blocks.forEach(block => {
    block.rows.sort((a, b) => {
      const aPrimary = isPrimaryPMRRowValue_(a.row[primaryIdx]);
      const bPrimary = isPrimaryPMRRowValue_(b.row[primaryIdx]);


      if (aPrimary !== bPrimary) return aPrimary ? -1 : 1;


      return a.originalIndex - b.originalIndex;
    });


    block.rows.forEach((item, idx) => {
      item.row[sortIdx] = idx === 0 ? participantNumber : Number((participantNumber + (idx / 1000)).toFixed(3));
      outputRows.push(item.row);
    });


    participantNumber++;
  });


  blankOrNoPMRRows.sort((a, b) => a.originalIndex - b.originalIndex);
  blankOrNoPMRRows.forEach(item => outputRows.push(item.row));


  if (outputRows.length > 0) {
    dataRange.setValues(outputRows);
    sheet.getRange(DATA_START_ROW, sortCol, outputRows.length, 1);
    sheet.getRange(DATA_START_ROW, primaryCol, outputRows.length, 1);
  }
}

function assignPrimaryPMRRows_(sheet) {
  sheet = sheet || SpreadsheetApp.getActiveSheet();


  ensurePrimaryPMRRowColumn_(sheet);


  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();
  if (lastRow < DATA_START_ROW || lastCol < 1) return;


  const headers = getHeaders_(sheet, HEADER_ROW);
  const headerMap = buildHeaderIndexMap_(headers);
  const pmrIdx = getPMRIndex_(headerMap);

  if (pmrIdx === -1 || primaryIdx === undefined) return;


  const dataRange = sheet.getRange(DATA_START_ROW, 1, lastRow - DATA_START_ROW + 1, lastCol);
  const values = dataRange.getValues();


  values.forEach(row => {
    if (!normalizePMR_(row[pmrIdx])) row[primaryIdx] = "";
  });


  const blocks = buildParticipantBlocksForSortOrder_(values, headerMap, pmrIdx);


  blocks.forEach(block => assignPrimaryRowForBlock_(block, headerMap, primaryIdx));


  dataRange.setValues(values);
}
function assignPrimaryRowForBlock_(block, headerMap, primaryIdx) {
  if (!block || !block.rows || block.rows.length === 0 || primaryIdx === undefined) return;


  let primaryItem = block.rows.find(item => isPrimaryPMRRowValue_(item.row[primaryIdx]));


  if (!primaryItem) {
    primaryItem = block.rows.slice().sort((a, b) => {
      const scoreA = getPrimaryRowScoreForSortOrder_(a.row, headerMap);
      const scoreB = getPrimaryRowScoreForSortOrder_(b.row, headerMap);


      if (scoreA !== scoreB) return scoreB - scoreA;
      return a.originalIndex - b.originalIndex;
    })[0];
  }


  block.rows.forEach(item => {
    item.row[primaryIdx] = item === primaryItem ? "Yes" : "";
  });
}

  return text === "yes" || text === "y" || text === "true" || text === "primary" || text === "1";
}

function buildParticipantBlocksForSortOrder_(values, headerMap, pmrIdx) {
  const lastNameIdx = headerMap["Last Name"];
  const firstNameIdx = headerMap["First Name"];


  const blocksByPMR = new Map();


  values.forEach((row, idx) => {
    const pmr = normalizePMR_(row[pmrIdx]);


    if (!pmr) return;


    if (!blocksByPMR.has(pmr)) {
      blocksByPMR.set(pmr, {
        pmr: pmr,
        lastName: lastNameIdx !== undefined ? row[lastNameIdx] : "",
        firstName: firstNameIdx !== undefined ? row[firstNameIdx] : "",
        rows: []
      });
    }


    const block = blocksByPMR.get(pmr);


    if (!String(block.lastName || "").trim() && lastNameIdx !== undefined && String(row[lastNameIdx] || "").trim()) {
      block.lastName = row[lastNameIdx];
    }


    if (!String(block.firstName || "").trim() && firstNameIdx !== undefined && String(row[firstNameIdx] || "").trim()) {
      block.firstName = row[firstNameIdx];
    }


    block.rows.push({ row: row, originalIndex: idx });
  });


  return Array.from(blocksByPMR.values());
}
function getPrimaryRowScoreForSortOrder_(row, headerMap) {
  let score = 0;


  const checks = [
    ["Date of Birth", 1000],
    ["Enrollment Date", 250],
    ["Capitation Date", 200],
    ["Address Line 1", 100],
    ["Phone Number", 50],
    ["Primary Language", 25],
    ["Participant Name", 20],
    ["PMR #", 10]
  ];


  checks.forEach(pair => {
    const idx = headerMap[pair[0]];
    if (idx !== undefined && normalizeCompareValue_(row[idx]) !== "") {
      score += pair[1];
    }
  });


  return score;
}

function applyIndexSheetRowFills_(sheet) {
  if (!sheet) return;


  const lastRow = sheet.getLastRow();
  const lastCol = INDEX_HEADERS.length;


  if (lastRow < 2) return;


  const indexValues = sheet.getRange(2, 1, lastRow - 1, INDEX_HEADERS.length).getDisplayValues();


  const rowItems = indexValues.map(row => {
    return {
      monthText: String(row[0] || "").trim(),
      sheetName: String(row[1] || "").trim(),
      sheetTypeText: String(row[3] || "").trim(),
      date: parseIndexMonthDate_(row[0]) || extractFirstDateFromSheetName_(row[1])
    };
  });


  const newestDate = getNewestMonthlySheetDate_(rowItems);


  const backgrounds = rowItems.map(item => {
    const standardType = getStandardSheetTypeFromOrganizationType_(item.sheetTypeText);
    const baseColor = standardType ? SHEET_BASE_COLORS[standardType] : null;


    if (!baseColor) {
      return new Array(lastCol).fill("#ffffff");
    }


    const theme = getThemeColorsFromBase_(baseColor);
    const isCurrent = item.date && newestDate && item.date.getTime() === newestDate.getTime();


    const fillColor = isCurrent ? theme.level3 : theme.level2;


    return new Array(lastCol).fill(fillColor);
  });


  sheet.getRange(2, 1, backgrounds.length, lastCol);
}


function organizeSheetTabs_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const categorized = ss.getSheets().map(sheet => {
    const name = sheet.getName();


    return {
      sheet: sheet,
      name: name,
      type: getSheetTypeForOrganization_(name),
      date: extractFirstDateFromSheetName_(name)
    };
  });


  const newestDate = getNewestMonthlySheetDate_(categorized);


  const indexItems = [];
  const currentSheets = [];
  const referenceSheets = [];
  const olderSheets = [];


  categorized.forEach(item => {
    if (item.name === "Index") {
      indexItems.push(item);
      return;
    }


    if (isReferenceSheetForOrganization_(item.name)) {
      referenceSheets.push(item);
      return;
    }


    if (item.date && newestDate && item.date.getTime() === newestDate.getTime()) {
      currentSheets.push(item);
    } else {
      olderSheets.push(item);
    }
  });


  indexItems.sort(compareSheetOrganization_);
  currentSheets.sort(compareSheetOrganization_);
  referenceSheets.sort(compareSheetOrganization_);
  olderSheets.sort(compareOlderSheetOrganization_);


  const ordered = indexItems.concat(currentSheets).concat(referenceSheets).concat(olderSheets);


  ordered.forEach((item, idx) => {
    ss.setActiveSheet(item.sheet);
    ss.moveActiveSheet(idx + 1);
  });
}


function getSheetTypeForOrganization_(name) {
  const lower = String(name || "").toLowerCase();


  if (lower === "index") return "Index";
  if (lower.indexOf("master list change log") !== -1 || lower.indexOf("change log") !== -1) return "Master List Change Log";
  if (lower.indexOf("disenrollment exclusion") !== -1 || lower.indexOf("disenrolled exclusion") !== -1) return "Disenrolled Exclusion List";
  if (lower.indexOf("monthly change") !== -1) return "Monthly Change Report";
  if (lower.indexOf("master list") !== -1) return "Master List";
  if (lower.indexOf("demo p") !== -1) return "Demo P";
  if (lower.indexOf("care plan due") !== -1) return "Care Plan Due Date Report";
  if (lower.indexOf("unlock") !== -1) return "Unlocked Care Plan Report";
  if (lower.indexOf("banner") !== -1) return "Banner Report";


  return "Other";
}


function isReferenceSheetForOrganization_(name) {
  const lower = String(name || "").toLowerCase();


  return (
    lower.indexOf("change log") !== -1 ||
    lower.indexOf("disenrollment exclusion") !== -1 ||
    lower.indexOf("disenrolled exclusion") !== -1
  );
}


function compareSheetOrganization_(a, b) {
  const aIndex = SHEET_TAB_ORDER.indexOf(a.type);
  const bIndex = SHEET_TAB_ORDER.indexOf(b.type);
  const aOrder = aIndex === -1 ? 999 : aIndex;
  const bOrder = bIndex === -1 ? 999 : bIndex;


  if (aOrder !== bOrder) return aOrder - bOrder;


  return String(a.name).localeCompare(String(b.name));
}


function compareOlderSheetOrganization_(a, b) {
  if (a.date && b.date && a.date.getTime() !== b.date.getTime()) {
    return b.date.getTime() - a.date.getTime();
  }


  return compareSheetOrganization_(a, b);
}


function getNewestMonthlySheetDate_(items) {
  const dates = items
    .map(item => item.date)
    .filter(Boolean)
    .sort((a, b) => b.getTime() - a.getTime());


  return dates.length ? dates[0] : null;
}

function formatRawData() {
  const sheet = SpreadsheetApp.getActiveSheet();
  const monthParts = promptForLockedYearReportMonth_("Format Raw Data");
  if (!monthParts) return;


  const timing = startRuntimeTiming_("Format Raw Data", monthParts);


  try {
    const dashboard = loadDashboardConfig_();
    const sheetDef = getSheetDefinitionByTypeOrNull_(dashboard, SHEET_TYPE.RAW_DATA) ||
      getSheetDefinitionByType_(dashboard, SHEET_TYPE.DEMO_P);
    const behavior = dashboard.behaviors[sheetDef.sheetType] ||
      dashboard.behaviors[SHEET_TYPE.DEMO_P] ||
      getDefaultBehavior_();


    // v1.4.8: Raw Data is a dashboard/template sheet type, but the active
    // imported source may still arrive with headers on row 1. Insert the
    // standard title rows once so headers land on row 4 and data starts row 5.
    ensureRawDataHeaderRows_(sheet);
    markRuntimeStep_(timing, "Ensure Raw Data title/header row structure");


    let headers = getRawDataCurrentHeadersOrDefaults_(sheet);
    headers = prepareRawDataSourceSheetForDashboardFormat_(sheet, headers);
    markRuntimeStep_(timing, "Ensure Raw Data approved sync columns exist");


    applyDashboardTemplateFormattingToActiveReportSheet_(
      sheet,
      dashboard,
      sheetDef,
      headers,
      behavior,
      monthParts,
      monthParts.lastDay,
      "",
      timing
    );
    markRuntimeStep_(timing, "Apply Raw Data dashboard template formatting");


    try {
      sheet.getRange("A1").setValue("Raw Data");
    } catch (titleErr) {
      Logger.log("Could not set Raw Data title: " + titleErr.message);
    }
    markRuntimeStep_(timing, "Set Raw Data title");


    markRuntimeStep_(timing, "Preserve Raw Data current headers - no column removal");


    processRawDataApprovedSyncColumns_(sheet, monthParts);
    markRuntimeStep_(timing, "Populate Raw Data approved sync columns: Primary PMR Row + Banner fields");


    setMonthlySheetNameFast_(sheet, buildRawDataSheetName_(monthParts));
    markRuntimeStep_(timing, "Rename sheet to " + sheet.getName());


    lockFinalOutputRowHeights_(sheet, "Raw Data");
    markRuntimeStep_(timing, "Lock Raw Data final row heights");


    clearSheetRuntimeCachesForSheet_(sheet);
    writeRuntimeTimingReport_(timing);


    notify_(
      "Raw Data formatted only.\n" +
      "Sheet renamed to: " + sheet.getName() + "\n" +
      "A1 set to Raw Data.\n" +
      "Headers were preserved and moved to row 4 when needed.\n" +
      "Only approved Raw Data sync columns are appended: Primary PMR Row and Banner fields.\n" +
      "Run Process Demo P next when ready to create the Demo P working sheet."
    );


    return sheet;
  } catch (err) {
    markRuntimeStep_(timing, "ERROR - " + err.message);
    writeRuntimeTimingReport_(timing);
    notify_("Format Raw Data stopped: " + err.message + "\nTiming log updated: " + FORMAT_DEMO_P_TIMING_SHEET);
    throw err;
  }
}


function ensureRawDataHeaderRows_(sheet) {
  if (!sheet) return;


  const row4Headers = sheet.getRange(HEADER_ROW, 1, 1, Math.max(sheet.getLastColumn(), 1)).getValues()[0];
  if (rowLooksLikeParticipantHeader_(row4Headers)) return;


  const row1Headers = sheet.getRange(1, 1, 1, Math.max(sheet.getLastColumn(), 1)).getValues()[0];
  if (rowLooksLikeParticipantHeader_(row1Headers)) {
    sheet.insertRowsBefore(1, HEADER_ROW - 1);
    clearSheetRuntimeCachesForSheet_(sheet);
    return;
  }


  // If neither row clearly looks like the source header, do not destructively
  // move rows. The existing formatter will surface missing-header errors later.
}


function rowLooksLikeParticipantHeader_(row) {
  const normalized = (row || []).map(function(value) {
    return normalizeHeader_(value);
  });
  const required = ["Last Name", "First Name", "Participant PMR#"].map(normalizeHeader_);
  return required.every(function(header) {
    return normalized.indexOf(header) !== -1;
  });
}


function getRawDataCurrentHeadersOrDefaults_(sheet) {
  const current = getHeaders_(sheet, HEADER_ROW)
    .map(function(header) { return String(header || "").trim(); })
    .filter(function(header) { return header !== ""; });
  return current.length ? current : getRawDataDemoPSourceHeaders_();
}




function enforceDemoPStrictDashboardSchema_(sheet) {
  /*
   * v1.4.24:
   * This function no longer inserts or deletes columns. Dashboard Section C is
   * the column authority; Template Header Audit reports drift.
   */
  if (!sheet) return;


  const dashboard = loadDashboardConfig_();
  const expectedHeaders = getHeadersForSheetType_(dashboard, SHEET_TYPE.DEMO_P)
    .map(normalizeHeader_);
  const expectedSet = {};
  expectedHeaders.forEach(function(header) {
    if (header) expectedSet[header] = true;
  });


  const currentHeaders = getHeaders_(sheet, HEADER_ROW).map(normalizeHeader_);
  const currentSet = {};
  currentHeaders.forEach(function(header) {
    if (header) currentSet[header] = true;
  });


  const missing = expectedHeaders.filter(function(header) {
    return header && !currentSet[header];
  });
  const unexpected = currentHeaders.filter(function(header) {
    return header && !expectedSet[header];
  });


  if (missing.length || unexpected.length) {
    Logger.log(
      "Demo P header drift detected. Missing: " +
      missing.join(", ") +
      " Unexpected: " +
      unexpected.join(", ")
    );
  }


  clearSheetRuntimeCachesForSheet_(sheet);
}


function processDemoP() {
  const monthParts = promptForLockedYearReportMonth_("Process Demo P");
  if (!monthParts) return;


  const timing = startRuntimeTiming_("Process Demo P", monthParts);
  let movedResult = {
    pmrsRemoved: 0,
    rowsRemoved: 0,
    rowsCopied: 0
  };


  try {
    const buildResult = getOrCreateDemoPProcessingSheet_(monthParts, timing);
    const sheet = buildResult && buildResult.sheet ? buildResult.sheet : buildResult;


    if (buildResult && buildResult.movedResult) {
      movedResult = buildResult.movedResult;
    }


    ensureStandardTitleRows_(sheet);
    markRuntimeStep_(timing, "Prepare Demo P title/header rows");


    enforceDemoPStrictDashboardSchema_(sheet);
    clearSheetRuntimeCachesForSheet_(sheet);
    markRuntimeStep_(timing, "Enforce Demo P strict dashboard schema and clear caches");


    if (!buildResult || !buildResult.sourceWasRawData) {
      movedResult = moveDisenrolledPMRsFromDemoPToExclusion_(sheet, monthParts);
      markRuntimeStep_(
        timing,
        "Move disenrolled PMRs to exclusion list - PMRs removed: " + movedResult.pmrsRemoved +
          ", rows copied: " + movedResult.rowsCopied +
          ", rows removed: " + movedResult.rowsRemoved
      );


      processDemoPAsWorkingSource_(sheet, monthParts, timing);
    } else {
      markRuntimeStep_(
        timing,
        "Disenrolled exclusion built from Raw Data - PMRs excluded from Demo P: " + movedResult.pmrsRemoved +
          ", rows copied: " + movedResult.rowsCopied
      );


      if (buildResult.preProcessedWorkingSource) {
        markRuntimeStep_(timing, "Demo P Working Source processing skipped - already completed in optimized Raw Data single-pass build");
      } else {
        processDemoPAsWorkingSource_(sheet, monthParts, timing);
      }
    }


    if (buildResult && buildResult.createdFromDashboardTemplate) {
      markRuntimeStep_(timing, "Apply Demo P dashboard template formatting skipped - template copy already formatted");
    } else {
      applyDemoPTemplateToSheet_(sheet, monthParts);
      markRuntimeStep_(timing, "Apply Demo P dashboard template formatting");
    }


    setUniqueSheetName_(
      sheet,
      buildMonthlySheetNameNoDashAfterPrefix_(
        "Demo P",
        monthParts.firstDay,
        monthParts.lastDay
      )
    );
    markRuntimeStep_(timing, "Rename processed Demo P sheet");


    lockFinalOutputRowHeights_(sheet, "Demo P");
    markRuntimeStep_(timing, "Lock Demo P final row heights");


    writeRuntimeTimingReport_(timing);


    notify_(
      "Demo P processed as the working source.\n" +
      "Raw Data preserved when source sheet was Raw Data.\n" +
      "Demo P contains active participants only when built from Raw Data.\n" +
      "Disenrolled PMRs excluded from Demo P: " + movedResult.pmrsRemoved + "\n" +
      "Disenrolled primary rows copied to exclusion list: " + movedResult.rowsCopied + "\n" +
      "Contact, phone, address, language, notes, banner, and Primary PMR fields were processed on Demo P.\n" +
      "Care Plan Due and Unlocked Care Plan remain Master List sync functions.\n" +
      "Master List can now be created from Primary PMR rows only."
    );


    return sheet;
  } catch (err) {
    markRuntimeStep_(timing, "ERROR - " + err.message);
    writeRuntimeTimingReport_(timing);
    notify_("Process Demo P stopped: " + err.message + "\nTiming log updated: " + FORMAT_DEMO_P_TIMING_SHEET);
    throw err;
  }
}

function markPrimaryPMRRowsForSequentialData_(data) {
  if (!data || !data.values || !data.values.length || !data.headerMap) return;


  const headerMap = data.headerMap;
  const pmrIdx = getPMRIndex_(headerMap);
  const primaryIdx = headerMap["Primary PMR Row"];


  if (pmrIdx === -1 || primaryIdx === undefined) return;


  let previousPMR = "";


  data.values.forEach(row => {
    const pmr = normalizePMR_(row[pmrIdx]);


    if (!pmr) {
      row[primaryIdx] = "";
      return;
    }


    row[primaryIdx] = pmr !== previousPMR ? "Yes" : "";
    previousPMR = pmr;
  });
}




function assignPrimaryPMRRowsInData_(data) {
  if (!data || !data.values || !data.values.length || !data.headerMap) return;


  const headerMap = data.headerMap;
  const pmrIdx = getPMRIndex_(headerMap);
  const primaryIdx = headerMap["Primary PMR Row"];


  if (pmrIdx === -1 || primaryIdx === undefined) return;


  data.values.forEach(row => {
    if (!normalizePMR_(row[pmrIdx])) row[primaryIdx] = "";
  });


  const blocks = buildParticipantBlocksForSortOrder_(data.values, headerMap, pmrIdx);
  blocks.forEach(block => assignPrimaryRowForBlock_(block, headerMap, primaryIdx));
}

function getOrCreateDemoPProcessingSheet_(monthParts, timing) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const active = ss.getActiveSheet();
  const rawName = buildRawDataSheetName_(monthParts);
  const targetName = buildMonthlySheetNameNoDashAfterPrefix_("Demo P", monthParts.firstDay, monthParts.lastDay);


  if (!active) {
    throw new Error("No active sheet found. Select the Raw Data sheet or existing Demo P sheet before running Process Demo P.");
  }


  const activeName = String(active.getName() || "").trim();
  const activeLower = activeName.toLowerCase();


  if (activeName === targetName || activeLower.indexOf("demo p") === 0) {
    markRuntimeStep_(timing, "Using active Demo P sheet: " + activeName);
    return {
      sheet: active,
      sourceWasRawData: false,
      movedResult: {
        pmrsRemoved: 0,
        rowsRemoved: 0,
        rowsCopied: 0
      },
      preProcessedWorkingSource: false
    };
  }


  let rawSheet = null;


  if (activeName === rawName || activeLower.indexOf("raw data") === 0) {
    rawSheet = active;
  } else {
    rawSheet = ss.getSheetByName(rawName);
  }


  if (!rawSheet) {
    throw new Error("Raw Data sheet was not found for " + Utilities.formatDate(monthParts.firstDay, Session.getScriptTimeZone(), "MM.yy") + ". Run Format Raw Data first.");
  }


  deleteSheetIfExistsForDemoPProcess_(ss, targetName, rawSheet.getName());


  const buildResult = createActiveDemoPFromRawData_(rawSheet, targetName, monthParts, timing);
  const demoSheet = buildResult && buildResult.sheet ? buildResult.sheet : buildResult;


  ss.setActiveSheet(demoSheet);
  clearSheetRuntimeCachesForSheet_(demoSheet);
  markRuntimeStep_(timing, "Active-only processed Demo P built from protected Raw Data: " + targetName);


  return {
    sheet: demoSheet,
    sourceWasRawData: true,
    createdFromDashboardTemplate: true,
    preProcessedWorkingSource: !!(buildResult && buildResult.preProcessedWorkingSource),
    movedResult: (buildResult && buildResult.movedResult) ? buildResult.movedResult : getLastRawDataDisenrolledBuildResult_()
  };
}






function deleteSheetIfExistsForDemoPProcess_(ss, sheetName, protectedSheetName) {
  const existing = ss.getSheetByName(sheetName);
  if (!existing) return;


  if (existing.getName() === protectedSheetName) {
    throw new Error("Cannot replace Demo P processing sheet because it is also the protected Raw Data sheet.");
  }


  ss.deleteSheet(existing);

function createDisenrolledList() {
  const monthParts = promptForLockedYearReportMonth_("Create Disenrolled List");
  if (!monthParts) return;


  const timing = startRuntimeTiming_("Create Disenrolled List", monthParts);


  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const rawName = buildRawDataSheetName_(monthParts);
    const rawSheet = ss.getSheetByName(rawName);


    if (!rawSheet) {
      throw new Error("Raw Data sheet was not found for " + Utilities.formatDate(monthParts.firstDay, Session.getScriptTimeZone(), "MM.yy") + ". Expected sheet: " + rawName);
    }
    markRuntimeStep_(timing, "Locate Raw Data source sheet: " + rawName);


    const rawData = getDataValues_(rawSheet, HEADER_ROW, DATA_START_ROW);
    if (!rawData.values.length) {
      throw new Error("Raw Data sheet has no participant rows: " + rawName);
    }
    markRuntimeStep_(timing, "Read Raw Data values");


    const split = splitRawDataRowsIntoActiveAndDisenrolled_(rawData);
    markRuntimeStep_(timing, "Identify disenrolled PMRs by sequential block scan");


    const rowsCopied = appendDisenrolledRowsFromRawDataToExclusion_(
      rawSheet,
      rawData.headers,
      split.disenrolledRowsByPMR,
      monthParts
    );
    markRuntimeStep_(timing, "Write Disenrolled Exclusion primary rows only: " + rowsCopied);


    const result = {
      pmrsRemoved: split.disenrolledPMRs.size,
      rowsRemoved: 0,
      rowsCopied: rowsCopied,
      primaryRowsOnly: true
    };
    setLastRawDataDisenrolledBuildResult_(result);


    writeRuntimeTimingReport_(timing);


    notify_(
      "Disenrolled List created from " + rawName + " as " + buildDatedDisenrolledOutputName_(monthParts) + ".\n" +
      "PMRs identified: " + split.disenrolledPMRs.size + "\n" +
      "Primary rows written: " + rowsCopied + "\n" +
      "Process Demo P no longer rebuilds this list automatically."
    );


    return result;
  } catch (err) {
    markRuntimeStep_(timing, "ERROR - " + err.message);
    writeRuntimeTimingReport_(timing);
    notify_("Create Disenrolled List stopped: " + err.message);
    throw err;
  }
}


function createDisenrollmentList() {
  return createDisenrolledList();
}


function createActiveDemoPFromRawData_(rawSheet, targetName, monthParts, timing) {
  const rawData = getDataValues_(rawSheet, HEADER_ROW, DATA_START_ROW);

  if (!rawData.values.length) {
    throw new Error("Raw Data has no participant rows to process.");
  }

  const pmrIdx = getPMRIndex_(rawData.headerMap);
  if (pmrIdx === -1) {
    throw new Error("Raw Data is missing Participant PMR# / PMR # column.");
  }

  const disenrolledPMRSet = loadDisenrolledPMRSetForMonth_(monthParts);
  const split = splitRawDataRowsIntoActiveAndDisenrolled_(rawData, disenrolledPMRSet);
  const movedResult = {
    pmrsRemoved: split.disenrolledPMRs.size,
    rowsRemoved: 0,
    rowsCopied: split.disenrolledRows.length,
    primaryRowsOnly: true
  };

  setLastRawDataDisenrolledBuildResult_(movedResult);

  movedResult.outputRowsWritten = 0;
  markRuntimeStep_(timing, "Disenrolled PMRs identified and excluded from Demo P - " +
    (disenrolledPMRSet && disenrolledPMRSet.size ? "used Disenrolled List PMR set" : "used Raw Data disenrollment fields") +
    " | Rows Read: " + rawData.values.length +
    "; Rows Processed: " + split.activeRows.length +
    "; Rows Excluded: " + split.disenrolledRows.length);

  const dashboard = loadDashboardConfig_();
  const sheetDef = getSheetDefinitionByType_(dashboard, SHEET_TYPE.DEMO_P);
  const headers = getHeadersForSheetType_(dashboard, SHEET_TYPE.DEMO_P);
  const outputRows = mapRowsToHeaders_(split.activeRows, rawData.headers, headers);

  const workingData = {
    headers: headers,
    headerMap: buildHeaderIndexMap_(headers),
    values: outputRows,
    range: null
  };

  populateDemoPUpdateColumns_(workingData, monthParts, rawSheet.getName(), "Created");
  markRuntimeStep_(timing, "Demo P Working Source - populate Demo P update tracking columns | Rows Processed: " + workingData.values.length + "; Write Mode: in-memory only");

  const demoSheet = createOutputSheetFromDashboardTemplate_(
    SHEET_TYPE.DEMO_P,
    targetName,
    workingData.values,
    monthParts.firstDay,
    monthParts.lastDay
  );
  markRuntimeStep_(timing, "Demo P Working Source - copy active Raw Data rows to Demo P template | Rows Read: " + rawData.values.length + "; Rows Written: " + workingData.values.length + "; Write Mode: single output write");

  try {
    demoSheet.getRange("A1").setValue(sheetDef.reportTitle || "Demo P");
  } catch (err) {
    Logger.log("Could not set Demo P title: " + err.message);
  }

  clearSheetRuntimeCachesForSheet_(demoSheet);

  processDemoPAsWorkingSource_(demoSheet, monthParts, timing);

  return {
    sheet: demoSheet,
    movedResult: movedResult,
    createdFromDashboardTemplate: true,
    sourceWasRawData: true,
    preProcessedWorkingSource: true
  };
}





function populateDemoPUpdateColumns_(data, monthParts, sourceSheetName, status) {
  if (!data || !data.headerMap || !data.values) return;
  const statusIdx = data.headerMap["Demo P Update Status"];
  const monthIdx = data.headerMap["Demo P Update Month"];
  const sourceIdx = data.headerMap["Demo P Source Sheet"];
  if (statusIdx === undefined && monthIdx === undefined && sourceIdx === undefined) return;


  const monthText = monthParts && monthParts.firstDay
    ? Utilities.formatDate(monthParts.firstDay, Session.getScriptTimeZone(), "MM.yy")
    : "";
  data.values.forEach(function(row) {

    if (monthIdx !== undefined) row[monthIdx] = monthText;
    if (sourceIdx !== undefined) row[sourceIdx] = sourceSheetName || "";
  });
}

function splitRawDataRowsIntoActiveAndDisenrolled_(rawData, disenrolledPMRSet) {
  const headerMap = rawData.headerMap;
  const pmrIdx = getPMRIndex_(headerMap);
  const statusIdx = headerMap["Enrollment Status"];
  const disenrollmentDateIdx = findHeaderIndex_(headerMap, ["Disenrollment Effective Date", "Disenrollment Date", "Date of Death"]);
  const useExternalDisenrolledSet = !!(disenrolledPMRSet && disenrolledPMRSet.size);


  const activeRows = [];
  const disenrolledRows = [];
  const disenrolledRowsByPMR = new Map();
  const disenrolledPMRs = new Set();


  const values = rawData.values || [];
  let i = 0;


  while (i < values.length) {
    const firstRow = values[i];
    const pmr = normalizePMR_(firstRow[pmrIdx]);


    if (!pmr) {
      i++;
      continue;
    }


    const block = [];
    let blockIsDisenrolled = useExternalDisenrolledSet && disenrolledPMRSet.has(pmr);


    while (i < values.length && normalizePMR_(values[i][pmrIdx]) === pmr) {
      const row = values[i];
      block.push(row);


      if (!useExternalDisenrolledSet) {
        const status = statusIdx === undefined ? "" : String(row[statusIdx] || "").trim().toLowerCase();
        const disenrollmentValue = disenrollmentDateIdx === -1 ? "" : row[disenrollmentDateIdx];


        if (status === "disenrolled" || normalizeCompareValue_(disenrollmentValue) !== "") {
          blockIsDisenrolled = true;
        }
      }


      i++;
    }


    if (blockIsDisenrolled) {
      disenrolledPMRs.add(pmr);
      const primaryRow = block[0];
      disenrolledRows.push(primaryRow);
      disenrolledRowsByPMR.set(pmr, [primaryRow]);
    } else {
      for (let b = 0; b < block.length; b++) {
        activeRows.push(block[b]);
      }
    }
  }


  return {
    activeRows: activeRows,
    disenrolledRows: disenrolledRows,
    disenrolledRowsByPMR: disenrolledRowsByPMR,
    disenrolledPMRs: disenrolledPMRs
  };
}
