

  return "Review this slow step and confirm it is necessary for the governed dashboard/template path.";
}


function worseTimingSeverity_(current, next) {
  const rank = { "OK": 0, "PASS": 0, "INFO": 0, "WARNING": 1, "SLOW": 2, "BOTTLENECK": 3, "CRITICAL": 4 };
  const c = rank[String(current || "OK").toUpperCase()] || 0;
  const n = rank[String(next || "OK").toUpperCase()] || 0;
  return n > c ? String(next || "OK") : String(current || "OK");
}


function appendRuntimeTimingToFrameworkTimingReport_(timing) {
  if (!timing || !timing.steps || timing.steps.length === 0) return;

  try {
    const sheet = ensureFrameworkTimingReport_();
    const existing = getFrameworkTimingDetailRows_(sheet);

    const rows = timing.steps.map(function(step) {
      return [
        step.timestamp || new Date(),
        step.processName || timing.processName || "",
        step.stepName || step.step || "",
        Number(step.stepSeconds || step.seconds || 0),
        step.severity || getRuntimeTimingSeverity_(Number(step.stepSeconds || step.seconds || 0)),
        step.details || ""
      ];
    });

    const combined = existing.concat(rows).slice(-500);
    replaceFrameworkTimingSectionRows_(sheet, "C", combined);
  } catch (err) {
    Logger.log("[WARNING] Framework Timing Report append skipped | " + err.message);
  }
}






function formatSeconds_(seconds) {
  const value = Number(seconds || 0);
  if (value < 60) return value.toFixed(2) + " sec";


  const minutes = Math.floor(value / 60);
  const remaining = value - (minutes * 60);
  return minutes + " min " + remaining.toFixed(1) + " sec";
}

/**
 * v1.1.0 helper: creates an output from a verified dashboard template and pastes values only.
 * This is the approved Template First path. It does not perform manual formatting.
 */




function setValuesInRowChunks_(sheet, startRow, startCol, values, width, chunkSize) {
  if (!sheet || !values || !values.length) return 0;


  width = width || (values[0] ? values[0].length : 1);
  chunkSize = Math.max(250, Number(chunkSize) || 2000);


  let written = 0;
  for (let offset = 0; offset < values.length; offset += chunkSize) {
    const chunk = values.slice(offset, offset + chunkSize).map(function(row) {
      if (row.length === width) return row;
      const out = row.slice(0, width);
      while (out.length < width) out.push("");
      return out;
    });
    sheet.getRange(startRow + offset, startCol, chunk.length, width).setValues(chunk);
    written += chunk.length;
  }


  return written;
}






function createOutputSheetFromDashboardTemplate_(sheetType, desiredName, rawValues, startDate, endDate) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const dashboard = loadDashboardConfig_();
  const sheetDef = getSheetDefinitionByType_(dashboard, sheetType);
  const template = ss.getSheetByName(sheetDef.templateName);

  if (!template) {
    throw new Error("Template not found: " + sheetDef.templateName + ". Run Create / Refresh All Templates first.");
  }

  const output = template.copyTo(ss);
  output.showSheet();
  setUniqueSheetName_(output, desiredName || sheetDef.reportTitle);

  const headers = getHeadersForSheetType_(dashboard, sheetType);
  const governedWidth = Math.max(headers.length, 1);

  const titleUpdates = [];
  if (startDate) titleUpdates.push(["B2", startDate]);
  if (endDate) titleUpdates.push(["D2", endDate]);
  titleUpdates.forEach(function(update) {
    output.getRange(update[0]).setValue(update[1]);
  });

  const values = rawValues || [];
  let rowsWritten = 0;

  if (values.length) {
    const requiredRows = DATA_START_ROW + values.length - 1;

    if (output.getMaxRows() < requiredRows) {
      output.insertRowsAfter(output.getMaxRows(), requiredRows - output.getMaxRows());
    }

    const normalized = values.map(function(row) {
      const out = (row || []).slice(0, governedWidth);
      while (out.length < governedWidth) out.push("");
      return out;
    });

    const cellCount = normalized.length * governedWidth;
    if (cellCount <= 350000) {
      output.getRange(DATA_START_ROW, 1, normalized.length, governedWidth).setValues(normalized);
    } else {
      setValuesInRowChunks_(output, DATA_START_ROW, 1, normalized, governedWidth, 2000);
    }
    rowsWritten = normalized.length;
  }

  trimOutputSheetToDataSize_(output, values.length || 0, sheetType);
  applyFinalRowHeightLockForSheetType_(output, sheetType);
  clearSheetRuntimeCachesForSheet_(output);

  try {
    output.getRange("A1").setNote(
      "Rows Written: " + rowsWritten + "\\n" +
      "Output Sheet Type: " + sheetType + "\\n" +
      "v1.5 output created from dashboard template"
    );
  } catch (err) {
    Logger.log("Output note skipped for " + output.getName() + ": " + err.message);
  }

  return output;
}



function shouldApplyFinalRowHeightLockForSheetType_(sheetType) {
  return [
    SHEET_TYPE.RAW_DATA,
    SHEET_TYPE.DEMO_P,
    SHEET_TYPE.MASTER_LIST,
    SHEET_TYPE.DISENROLLED_EXCLUSION
  ].indexOf(sheetType) !== -1;
}


function applyFinalRowHeightLockForSheetType_(sheet, sheetType) {
  if (!shouldApplyFinalRowHeightLockForSheetType_(sheetType)) return;
  applyFinalRowHeightLock_(sheet, 25);
}



function applyFinalRowHeightLock_(sheet, height) {
  /*
   * v1.5:
   * Force final 25px row-height only on populated data rows after output processing.
   * This is intentionally not used by template creation or reports that do not expand rows.
   */
  if (!sheet) return;

  const rowHeight = height || 25;
  const lastRow = Math.max(sheet.getLastRow ? sheet.getLastRow() : 0, DATA_START_ROW - 1);
  const numRows = lastRow - DATA_START_ROW + 1;
  if (numRows <= 0) return;

  try {
    if (typeof sheet.setRowHeightsForced === "function") {
      sheet.setRowHeightsForced(DATA_START_ROW, numRows, rowHeight);
    } else {
      sheet.setRowHeights(DATA_START_ROW, numRows, rowHeight);
    }
  } catch (err) {
    Logger.log("Final fixed row-height lock skipped for " + sheet.getName() + ": " + err.message);
  }
}





function trimOutputSheetToDataSize_(sheet, dataRowCount, sheetType) {
  if (!sheet) return;


  const bufferByType = {};
  bufferByType[SHEET_TYPE.RAW_DATA] = 150;
  bufferByType[SHEET_TYPE.DEMO_P] = 150;
  bufferByType[SHEET_TYPE.MASTER_LIST] = 100;
  bufferByType[SHEET_TYPE.DISENROLLED_EXCLUSION] = 50;


  const bufferRows = bufferByType[sheetType] !== undefined ? bufferByType[sheetType] : 100;
  const dataRows = Math.max(0, Number(dataRowCount || 0));
  const desiredRows = Math.max(DATA_START_ROW + dataRows - 1 + bufferRows, DATA_START_ROW + 25);
  const maxRows = sheet.getMaxRows();


  if (maxRows > desiredRows) {
    try {
      sheet.deleteRows(desiredRows + 1, maxRows - desiredRows);
    } catch (err) {
      Logger.log("Right-sized row trim skipped for " + sheet.getName() + ": " + err.message);
    }
  }
}

function collectFormatDashboardSheetDefinitionVerificationRows_(dashboardSheet) {

function collectFormatDashboardSheetHeaderVerificationRows_(dashboardSheet) {

    sectionRows = readDashboardSectionRows_(dashboardSheet, RFF_SECTION_HEADERS);
  } catch (err) {
    rows.push(["Dashboard Section C", RFF_SECTION_HEADERS, "Present", "Missing", "FAIL", "Critical", err.message]);

  }

  const headersByType = {};
  sectionRows.forEach(function(row) {

  });

}



function collectFormatDashboardColumnDefinitionVerificationRows_(dashboardSheet) {
  const rows = [["Sheet Type", "Header Name", "Definition Check", "Actual Value", "Status", "Severity", "Quality Notes"]];
  let headerRows = [];
  let columnRows = [];

  try {
    headerRows = readDashboardSectionRows_(dashboardSheet, RFF_SECTION_HEADERS);
  } catch (err) {
    rows.push(["Dashboard Section C", RFF_SECTION_HEADERS, "Section Present", "Missing", "FAIL", "Critical", err.message]);
    return rows;
  }

  try {
    columnRows = readDashboardSectionRows_(dashboardSheet, RFF_SECTION_COLUMNS);
  } catch (err) {
    rows.push(["Dashboard Section D", RFF_SECTION_COLUMNS, "Section Present", "Missing", "FAIL", "Critical", err.message]);
    return rows;
  }

  const requiredHeaders = {};
  headerRows.forEach(function(row) {
    const sheetType = String(row[0] || "").trim();
    const header = String(row[2] || "").trim();
    if (header) requiredHeaders[header] = sheetType || "";
  });

  const columnMap = {};
  columnRows.forEach(function(row) {
    const header = String(row[0] || "").trim();
    if (header) columnMap[header] = row;
  });

  const checks = [
    { label: "Width", index: 1 },
    { label: "Alignment", index: 6 },
    { label: "Wrap", index: 5 },
    { label: "Number Format", index: 8 },
    { label: "Visibility", index: 4 }
  ];

  Object.keys(requiredHeaders).sort().forEach(function(header) {
    const sheetType = requiredHeaders[header];
    const row = columnMap[header];

    if (!row) {
      rows.push([sheetType, header, "Column Definition Row", "Missing", "MISSING", "Critical", "Header exists in Section C but not Section D."]);
      return;
    }

    checks.forEach(function(check) {
      const value = row[check.index];
      const required = check.label === "Number Format" ? toBool_(row[3]) : true;
      const hasValue = !required || String(value === null || value === undefined ? "" : value).trim() !== "";
      rows.push([sheetType, header, check.label, value || "", hasValue ? "PASS" : "MISSING", hasValue ? "Info" : "Warning", hasValue ? "OK" : "Required column formatting value is blank."]);
    });
  });

  return rows;
}

function enforceDashboardSheetDefinitionGovernance_(sheet) {
  /*
   * v1.4.33:
   * Keep approved row-count/test-row governance synchronized even when Setup
   * restores saved dashboard defaults from an older version.
   */
  if (!sheet) return false;


  const values = sheet.getDataRange().getValues();
  let sectionRow = -1;
  let nextSectionRow = values.length;


  for (let i = 0; i < values.length; i++) {
    if (String(values[i][0] || "").trim() === RFF_SECTION_SHEETS) {
      sectionRow = i;
      break;
    }
  }
  if (sectionRow === -1) return false;


  for (let i = sectionRow + 1; i < values.length; i++) {
    if (String(values[i][0] || "").trim().indexOf("SECTION ") === 0) {
      nextSectionRow = i;
      break;
    }
  }


  let changed = false;
  for (let r = sectionRow + 2; r < nextSectionRow; r++) {
    const sheetType = normalizeHeader_(values[r][0]);
    if (sheetType === RFF_SHEET_TYPES.MONTHLY_CHANGE || sheetType === RFF_SHEET_TYPES.DISENROLLED || sheetType === RFF_SHEET_TYPES.MASTER_LIST_CHANGE_LOG) {
      // Columns are 1-based in the sheet: Template Row Count H, Minimum Rows J, Buffer K, Test Rows L, Use Test Rows M.
      if (sheetType === RFF_SHEET_TYPES.DISENROLLED && Number(values[r][7]) !== 1000) {
        sheet.getRange(r + 1, 8).setValue(1000);
        changed = true;
      }
      if (Number(values[r][11]) !== 1000) {
        sheet.getRange(r + 1, 12).setValue(1000);
        changed = true;
      }
      if (String(values[r][12] || "").toUpperCase() !== "FALSE") {
        sheet.getRange(r + 1, 13).setValue("FALSE");
        changed = true;
      }
    }
  }


  if (changed) clearDashboardConfigCache_();
  return changed;
}































/**
 * v1.4.9:
 * Keeps saved Dashboard edits, but safely adds any new script-required dashboard
 * rows introduced after the saved defaults were created. This prevents an older
 * saved dashboard from hiding new sheet types such as Raw Data.

function rebuildDashboardHeaderSectionRows_(sheet, timing) {
  const bounds = getDashboardSectionBounds_(sheet, RFF_SECTION_HEADERS);
  if (!bounds) return false;


  const defaultSets = getDefaultHeaderSets_();
  const rows = [];
  getProductionSheetTypeOrder_().forEach(function(sheetType) {
    const headers = defaultSets[sheetType] || [];
    headers.forEach(function(item, index) {
      rows.push([sheetType, index + 1, item.header, item.source || ""]);
    });
  });


  const existingCount = Math.max(bounds.dataRowCount, 0);
  let existingRows = [];
  if (existingCount > 0) {
    existingRows = sheet.getRange(bounds.dataStartRow, 1, existingCount, 4).getValues();
  }


  if (dashboardRowsEqual_(existingRows, rows)) {
    return false;
  }


  if (existingCount > 0) {
    sheet.deleteRows(bounds.dataStartRow, existingCount);
  }


  if (rows.length) {
    sheet.insertRowsBefore(bounds.dataStartRow, rows.length);
    sheet.getRange(bounds.dataStartRow, 1, rows.length, 4).setValues(rows);
  }


  if (timing) {
    markFrameworkStep_(timing, "Rebuilt dashboard Section C Sheet Headers from current Appendix C mapping");
  }
  clearDashboardConfigCache_();
  return true;
}


function dashboardRowsEqual_(a, b) {
  if ((a || []).length !== (b || []).length) return false;
  for (let r = 0; r < b.length; r++) {
    for (let c = 0; c < b[r].length; c++) {
      if (normalizeText_(a[r] && a[r][c]) !== normalizeText_(b[r][c])) return false;
    }
  }
  return true;
}


function ensureDashboardRowExistsByFirstColumn_(sheet, sectionName, firstColumnValue, rowsToInsert, timing, label) {
  if (!rowsToInsert || rowsToInsert.length === 0) return false;


  const bounds = getDashboardSectionBounds_(sheet, sectionName);
  if (!bounds) return false;


  const existingRowCount = Math.max(bounds.dataRowCount, 0);
  let exists = false;


  if (existingRowCount > 0) {
    const values = sheet.getRange(bounds.dataStartRow, 1, existingRowCount, 1).getValues();
    exists = values.some(function(row) {
      return normalizeText_(row[0]) === normalizeText_(firstColumnValue);
    });
  }


  if (exists) return false;


  // Insert at the first blank row inside the section if a spacer exists.
  // This prevents Raw Data migration rows from being placed after the section
  // spacer rows and visually separating them from the rest of Section B/C/E.
  let insertAt = findFirstBlankDashboardRowInSection_(sheet, bounds);
  if (!insertAt) {
    insertAt = bounds.insertBeforeRow;
  }


  if (insertAt <= sheet.getMaxRows()) {
    sheet.insertRowsBefore(insertAt, rowsToInsert.length);
  } else {
    sheet.insertRowsAfter(sheet.getMaxRows(), rowsToInsert.length);
  }


  sheet.getRange(insertAt, 1, rowsToInsert.length, rowsToInsert[0].length).setValues(rowsToInsert);


  if (timing) {
    markFrameworkStep_(timing, "Migrated dashboard defaults: added " + label);
  }
  return true;
}


function findFirstBlankDashboardRowInSection_(sheet, bounds) {
  if (!bounds || bounds.nextSectionRow <= bounds.dataStartRow) return null;


  const rowCount = bounds.nextSectionRow - bounds.dataStartRow;
  const colCount = Math.max(sheet.getLastColumn(), 1);
  const values = sheet.getRange(bounds.dataStartRow, 1, rowCount, colCount).getValues();


  for (let i = 0; i < values.length; i++) {
    const isBlank = values[i].every(function(cell) {
      return normalizeText_(cell) === "";
    });
    if (isBlank) {
      return bounds.dataStartRow + i;
    }
  }


  return null;
}


function compactDashboardSectionDataRows_(sheet, sectionName, timing, label) {
  const bounds = getDashboardSectionBounds_(sheet, sectionName);
  if (!bounds || bounds.nextSectionRow <= bounds.dataStartRow) return false;


  const colCount = Math.max(sheet.getLastColumn(), 1);
  const rowsToCheck = bounds.nextSectionRow - bounds.dataStartRow;
  const values = sheet.getRange(bounds.dataStartRow, 1, rowsToCheck, colCount).getValues();


  const blankRows = [];
  values.forEach(function(row, index) {
    const isBlank = row.every(function(cell) {
      return normalizeText_(cell) === "";
    });
    if (isBlank) {
      blankRows.push(bounds.dataStartRow + index);
    }
  });


  if (blankRows.length === 0) return false;


  // Delete from bottom to top so row numbers remain valid.
  for (let i = blankRows.length - 1; i >= 0; i--) {
    sheet.deleteRow(blankRows[i]);
  }


  if (timing) {
    markFrameworkStep_(timing, "Compacted dashboard migration spacing: " + label);
  }


  return true;
}




function getDashboardSectionBounds_(sheet, sectionName) {
  const lastRow = Math.max(sheet.getLastRow(), 1);
  const maxRows = Math.max(sheet.getMaxRows(), lastRow);
  const values = sheet.getRange(1, 1, lastRow, 1).getValues().map(function(row) {
    return normalizeText_(row[0]);
  });


  const target = normalizeText_(sectionName);
  let sectionRow = -1;


  for (let i = 0; i < values.length; i++) {
    if (values[i] === target) {
      sectionRow = i + 1;
      break;
    }
  }


  if (sectionRow === -1) return null;


  const sectionNames = [
    RFF_SECTION_GLOBAL,
    RFF_SECTION_SHEETS,
    RFF_SECTION_HEADERS,
    RFF_SECTION_COLUMNS,
    RFF_SECTION_BEHAVIORS
  ].map(function(name) {
    return normalizeText_(name);
  });


  let nextSectionRow = lastRow + 1;
  for (let r = sectionRow + 1; r <= lastRow; r++) {
    if (sectionNames.indexOf(values[r - 1]) !== -1) {
      nextSectionRow = r;
      break;
    }
  }


  const headerRow = sectionRow + 1;
  const dataStartRow = sectionRow + 2;


  let lastDataRow = dataStartRow - 1;
  for (let r = dataStartRow; r < nextSectionRow; r++) {
    if (normalizeText_(values[r - 1]) !== "") {
      lastDataRow = r;
    }
  }


  const dataRowCount = Math.max(lastDataRow - dataStartRow + 1, 0);
  let insertBeforeRow = lastDataRow + 1;


  // Keep inserted rows inside the current section.  If the section has a blank
  // spacer before the next section, insert before that spacer.  If there is no
  // spacer, insert directly before the next section heading.
  if (insertBeforeRow < dataStartRow) {
    insertBeforeRow = dataStartRow;
  }
  if (insertBeforeRow >= nextSectionRow) {
    insertBeforeRow = nextSectionRow;
  }


  return {
    sectionRow: sectionRow,
    headerRow: headerRow,
    dataStartRow: dataStartRow,
    dataRowCount: dataRowCount,
    lastDataRow: lastDataRow,
    nextSectionRow: nextSectionRow,
    insertBeforeRow: insertBeforeRow,
    maxRows: maxRows
  };
}

function sortSheetDefinitionsByProductionOrder_(sheetDefinitions) {
  const order = getProductionSheetTypeOrder_();
  const orderMap = {};
  order.forEach(function(sheetType, index) {
    orderMap[normalizeKey_(sheetType)] = index;
  });


  return (sheetDefinitions || []).slice().sort(function(a, b) {
    const aOrder = orderMap[normalizeKey_(a && a.sheetType)] !== undefined ? orderMap[normalizeKey_(a.sheetType)] : 999;
    const bOrder = orderMap[normalizeKey_(b && b.sheetType)] !== undefined ? orderMap[normalizeKey_(b.sheetType)] : 999;
    if (aOrder !== bOrder) return aOrder - bOrder;
    return String(a && a.sheetType || "").localeCompare(String(b && b.sheetType || ""));
  });
}

    markFrameworkStep_(timing, "Dashboard loaded");


    const results = auditTemplateHeaders_(dashboard);
    saveDashboardQualityRowsForTemplateHeaderAudit_(results);
    markFrameworkStep_(timing, "Dashboard Quality Section B saved");


    const failed = results.filter(function(item) {
      return item.status !== "PASS";
    });


    return results;
  });
}


function auditTemplateHeaders_(dashboard) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const results = [];


  sortSheetDefinitionsByProductionOrder_(dashboard.sheetDefinitions).forEach(function(sheetDef) {
    const expectedHeaders = getHeadersForSheetType_(dashboard, sheetDef.sheetType).map(normalizeHeader_).filter(Boolean);
    const expectedSet = {};
    expectedHeaders.forEach(function(header) {
      expectedSet[header] = true;
    });


    const sheet = ss.getSheetByName(sheetDef.templateName);
    if (!sheet) {
      results.push({
        templateName: sheetDef.templateName,
        sheetType: sheetDef.sheetType,
        status: "FAIL",
        missingColumns: expectedHeaders.join(", "),
        unexpectedColumns: "",
        notes: "Template sheet missing"
      });
      return;
    }


    const actualHeaders = getHeaders_(sheet, HEADER_ROW).map(normalizeHeader_).filter(Boolean);
    const actualSet = {};
    actualHeaders.forEach(function(header) {
      actualSet[header] = true;
    });


    const missing = expectedHeaders.filter(function(header) {
      return !actualSet[header];
    });


    const unexpected = actualHeaders.filter(function(header) {
      return !expectedSet[header];
    });


    results.push({
      templateName: sheetDef.templateName,
      sheetType: sheetDef.sheetType,
      status: missing.length ? "FAIL" : "PASS",
      missingColumns: missing.join(", "),
      unexpectedColumns: unexpected.join(", "),
      notes: missing.length ? "Missing required template headers" : "OK"
    });
  });


  return results;
}


function saveDashboardQualityRowsForTemplateHeaderAudit_(results) {
  const rows = [["Template Name", "Sheet Type", "Status", "Missing Columns", "Unexpected Columns", "Quality Notes"]];
  (results || []).forEach(function(item) {
    rows.push([
      item.templateName || "",
      item.sheetType || "",
      item.status || "",
      item.missingColumns || "",
      item.unexpectedColumns || "",
      item.notes || ""
    ]);
  });
  saveDashboardQualitySectionRows_("Template Header Audit", rows);
}




function validateReportTemplates() {
  /*
   * v1.5 Quick QA Section A:
   * Validate operational readiness only: template exists, header count matches,
   * and a template signature is present. Detailed header/format validation remains
   * in Template Header Audit and Workflow QA sections.
   */
  return runFrameworkTimed_("Validate Templates", function(timing) {
    const dashboard = loadDashboardConfig_();
    markFrameworkStep_(timing, "Dashboard loaded");

    const results = sortSheetDefinitionsByProductionOrder_(dashboard.sheetDefinitions).map(function(sheetDef) {
      const ss = SpreadsheetApp.getActiveSpreadsheet();
      const sheet = ss.getSheetByName(sheetDef.templateName);
      const issues = [];

      if (!sheet) {
        issues.push("Template sheet missing");
      } else {
        const expectedHeaders = getHeadersForSheetType_(dashboard, sheetDef.sheetType);
        const actualHeaders = getHeaders_(sheet, dashboard.globals.headerRow || HEADER_ROW)
          .filter(function(header) { return String(header || "").trim() !== ""; });
        if (actualHeaders.length !== expectedHeaders.length) {
          issues.push("Header count mismatch expected " + expectedHeaders.length + " actual " + actualHeaders.length);
        }

        const storedSignature = getStoredTemplateFormatSignatureFromSheet_(sheet) || getStoredTemplateFormatSignature_(sheetDef);
        if (!storedSignature) {
          issues.push("Template signature missing");
        }
      }

      const result = {
        templateName: sheetDef.templateName,
        sheetType: sheetDef.sheetType,
        status: issues.length === 0 ? "PASS" : "FAIL",
        issues: issues.length === 0 ? "OK" : issues.join("; ")
      };

      markFrameworkStep_(timing, "Quick validated: " + sheetDef.templateName);
      return result;
    });

    writeTemplateValidationReport_(results);
    markFrameworkStep_(timing, "Dashboard Quality Section A saved");

    return results;
  });
}

function getDefaultColumnDefinitionRows_() {
  const standards = getColumnStandards_();
  const allHeaders = getAllUniqueHeaders_();
  const rows = [];


  allHeaders.forEach(function(header) {
    const std = standards[header] || {};
    const isDate = isDateLikeHeader_(header);


    rows.push([
      header,
      std.width || 140,
      std.fontSize || 10,
      std.dateColumn !== undefined ? boolText_(std.dateColumn) : boolText_(isDate),
      std.hideColumn !== undefined ? boolText_(std.hideColumn) : "FALSE",
      std.dataWrap || "CLIP",
      std.horizontalAlignment || "left",
      std.verticalAlignment || "middle",
      std.numberFormat || (isDate ? RFF_DEFAULTS.defaultDateFormat : "")
    ]);
  });


  return rows;
}

    headerSets[sheetType].forEach(function(item) {
      const header = normalizeHeader_(item.header);
      if (!header || seen[header]) return;
      seen[header] = true;
      output.push(header);
    });
  });


  return output;
}

function writeDashboardTitle_(sheet, row) {
  sheet.getRange(row, 1).setValue("Report Formatter Framework v" + RFF_VERSION + " - Format Dashboard");
  sheet.getRange(row, 1, 1, 9)
    .merge()
    .setFontFamily("Arial")
    .setFontSize(16)
    .setFontWeight("bold")
    .setBackground("#D9EAF7")
    .setVerticalAlignment("middle");
  sheet.setRowHeight(row, 30);
  return row + 2;
}


function writeDashboardSection_(sheet, startRow, sectionName, headers, rows) {
  sheet.getRange(startRow, 1).setValue(sectionName);
  sheet.getRange(startRow, 1, 1, headers.length)
    .merge()
    .setFontFamily("Arial")
    .setFontSize(12)
    .setFontWeight("bold")
    .setBackground("#B7DEE8")
    .setVerticalAlignment("middle");


  const headerRow = startRow + 1;
  sheet.getRange(headerRow, 1, 1, headers.length)
    .setValues([headers])
    .setFontFamily("Arial")
    .setFontSize(10)
    .setFontWeight("bold")
    .setBackground("#D9EAF7")
    .setWrapStrategy(SpreadsheetApp.WrapStrategy.WRAP)
    .setVerticalAlignment("top");


  if (rows.length > 0) {
    sheet.getRange(headerRow + 1, 1, rows.length, headers.length)
      .setValues(rows)
      .setFontFamily("Arial")
      .setFontSize(10)
      .setVerticalAlignment("middle")
      .setWrapStrategy(SpreadsheetApp.WrapStrategy.CLIP);
  }


  return headerRow + rows.length + 3;
}


function styleDashboard_(sheet) {
  sheet.setFrozenRows(1);


  const lastRow = Math.max(sheet.getLastRow(), 1);
  const lastCol = Math.max(sheet.getLastColumn(), 1);


  sheet.getRange(1, 1, lastRow, lastCol)
    .setFontFamily("Arial")
    .setFontColor("#000000")
    .setHorizontalAlignment("left");


  // Fixed widths are faster and more stable than auto-resize.
  const widths = [250, 160, 240, 180, 130, 120, 160, 150, 140];
  widths.forEach(function(width, index) {
    try {
      sheet.setColumnWidth(index + 1, width);
    } catch (err) {
      Logger.log("Dashboard column width skipped: " + err.message);
    }
  });


  try {
    const filter = sheet.getFilter();
    if (filter) filter.remove();
  } catch (err) {
    Logger.log("Dashboard filter remove skipped: " + err.message);
  }
}

function loadDashboardConfig_(forceRefresh) {
  const cacheKey = getDashboardConfigCacheKey_();


  if (!forceRefresh && RFF_DASHBOARD_CONFIG_CACHE_ && RFF_DASHBOARD_CONFIG_CACHE_KEY_ === cacheKey) {
    return RFF_DASHBOARD_CONFIG_CACHE_;
  }


  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(RFF_DASHBOARD_SHEET);


  if (!sheet) {
    throw new Error("Format Dashboard is missing. Run Setup / Refresh Dashboard first.");
  }


  const globals = loadGlobalSettings_(sheet);
  const sheetDefinitions = loadSheetDefinitions_(sheet);
  const sheetHeaders = loadSheetHeaders_(sheet);
  const columnDefinitions = loadColumnDefinitions_(sheet);
  const behaviors = loadSheetBehaviors_(sheet);


  const dashboard = {
    globals: globals,
    sheetDefinitions: sheetDefinitions,
    sheetHeaders: sheetHeaders,
    columnDefinitions: columnDefinitions,
    behaviors: behaviors
  };


  RFF_DASHBOARD_CONFIG_CACHE_ = dashboard;
  RFF_DASHBOARD_CONFIG_CACHE_KEY_ = cacheKey;


  return dashboard;
}

    })
    .map(function(row) {
      return {

        reportTitle: String(row[1] || row[0] || "").trim(),
        templateName: String(row[2] || ("Template - " + row[0])).trim(),
        outputNamingPattern: String(row[3] || "").trim(),
        baseColor: normalizeHex_(row[4] || "#65A9CC"),
        usePromptDate: parseBoolean_(row[5]),
        endDateSource: String(row[6] || "").trim(),
        templateRowCount: numberOrDefault_(row[7], RFF_DEFAULTS.templateRows),
        templateRowMode: String(row[8] || "FIXED").trim().toUpperCase(),
        minimumRows: numberOrDefault_(row[9], numberOrDefault_(row[7], RFF_DEFAULTS.templateRows)),
        bufferRows: numberOrDefault_(row[10], 100),
        testRows: numberOrDefault_(row[11], 100),
        useTestRows: parseBoolean_(row[12])
      };
    });
}


function loadSheetHeaders_(sheet) {
  const rows = readDashboardSectionRows_(sheet, RFF_SECTION_HEADERS);
  const map = {};


  rows.forEach(function(row) {
    const sheetType = normalizeHeader_(row[0]);
    const order = numberOrDefault_(row[1], 9999);
    const header = normalizeHeader_(row[2]);
    const source = String(row[3] || "").trim();


    if (!sheetType || !header) return;
    if (!map[sheetType]) map[sheetType] = [];


    map[sheetType].push({
      order: order,
      header: header,
      source: source
    });
  });


  Object.keys(map).forEach(function(sheetType) {
    map[sheetType].sort(function(a, b) {
      return a.order - b.order;
    });
  });


  return map;
}


function loadColumnDefinitions_(sheet) {
  const rows = readDashboardSectionRows_(sheet, RFF_SECTION_COLUMNS);
  const map = {};


  rows.forEach(function(row) {
    const header = normalizeHeader_(row[0]);
    if (!header) return;


    map[header] = {
      header: header,
      width: numberOrDefault_(row[1], 140),
      headerFontSize: numberOrDefault_(row[2], 10),
      dateColumn: parseBoolean_(row[3]),
      hideColumn: parseBoolean_(row[4]),
      dataWrap: String(row[5] || "CLIP").trim().toUpperCase(),
      horizontalAlignment: String(row[6] || "left").trim().toLowerCase(),
      verticalAlignment: String(row[7] || "middle").trim().toLowerCase(),
      numberFormat: String(row[8] || "").trim()
    };
  });


  return map;
}


function loadSheetBehaviors_(sheet) {
  const rows = readDashboardSectionRows_(sheet, RFF_SECTION_BEHAVIORS);
  const map = {};


  rows.forEach(function(row) {
    const sheetType = normalizeHeader_(row[0]);
    if (!sheetType) return;


    map[sheetType] = {
      usesTitleRows: parseBoolean_(row[1]),
      usesFilter: parseBoolean_(row[2]),
      usesAlternatingColors: parseBoolean_(row[3]),
      usesSubheaders: parseBoolean_(row[4]),
      hiddenTemplate: parseBoolean_(row[5])
    };
  });


  return map;
}


function readDashboardSectionRows_(sheet, sectionName) {
  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();
  const values = sheet.getRange(1, 1, lastRow, lastCol).getValues();


  let sectionRow = -1;
  for (let i = 0; i < values.length; i++) {
    if (String(values[i][0] || "").trim() === sectionName) {
      sectionRow = i + 1;
      break;
    }
  }


  if (sectionRow === -1) {
    throw new Error("Dashboard section missing: " + sectionName);
  }


  const headerRow = sectionRow + 1;
  const output = [];


  for (let r = headerRow + 1; r <= lastRow; r++) {
    const firstCell = String(sheet.getRange(r, 1).getValue() || "").trim();


    if (firstCell.indexOf("SECTION ") === 0) break;


    const rowValues = sheet.getRange(r, 1, 1, lastCol).getValues()[0];
    const hasAnyValue = rowValues.some(function(value) {
      return String(value === null || value === undefined ? "" : value).trim() !== "";
    });


    if (!hasAnyValue) {
      const nextFirst = r < lastRow ? String(sheet.getRange(r + 1, 1).getValue() || "").trim() : "";
      if (nextFirst.indexOf("SECTION ") === 0 || nextFirst === "") break;
      continue;
    }


    output.push(rowValues);
  }


  return output;
}

function createOrRefreshTemplateFromDashboard_(dashboard, sheetDef, timing) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(sheetDef.templateName);
  const templateExisted = !!sheet;


  if (!sheet) {
    sheet = ss.insertSheet(sheetDef.templateName);
  }
  markFrameworkStep_(timing, "Open/create sheet: " + sheetDef.templateName);


  const headers = getHeadersForSheetType_(dashboard, sheetDef.sheetType);
  const behavior = dashboard.behaviors[sheetDef.sheetType] || getDefaultBehavior_();
  const globals = dashboard.globals;
  const colCount = Math.max(headers.length, 4);
  const rowCount = resolveTemplateRowCount_(sheetDef, globals);
  const expectedSignature = buildTemplateFormatSignature_(dashboard, sheetDef, headers, rowCount, colCount, behavior);
  const propertySignature = getStoredTemplateFormatSignature_(sheetDef);
  const noteSignature = templateExisted ? getStoredTemplateFormatSignatureFromSheet_(sheet) : "";
  const storedSignature = noteSignature || propertySignature;
  const normalizedStoredSignature = normalizeTemplateFormatSignature_(storedSignature);
  const normalizedExpectedSignature = normalizeTemplateFormatSignature_(expectedSignature);


  const signatureMatches = templateExisted &&
    RFF_USE_TEMPLATE_SIGNATURE_CACHE &&
    normalizedStoredSignature !== "" &&
    normalizedStoredSignature === normalizedExpectedSignature;


  const metadataRefreshAllowed =
    RFF_TRUE_SMART_TEMPLATE_REFRESH &&
    RFF_SMART_TEMPLATE_REFRESH &&
    !RFF_FORCE_FULL_TEMPLATE_REBUILD &&
    templateExisted &&
    !!sheet &&
    signatureMatches;


  markFrameworkStep_(timing, signatureMatches
    ? "Template signature matched - metadata refresh eligible: " + sheetDef.templateName
    : "Template signature mismatch - full build required: " + sheetDef.templateName);


  markFrameworkStep_(timing, buildTemplateRefreshDecisionMessage_(
    sheetDef,
    templateExisted,
    signatureMatches,
    metadataRefreshAllowed,
    normalizedStoredSignature,
    normalizedExpectedSignature
  ));


  if (metadataRefreshAllowed) {
    refreshTemplateMetadataOnly_(sheet, dashboard, sheetDef, colCount, expectedSignature, timing);
    return sheet;
  }


  buildTemplateFromDashboard_(sheet, dashboard, sheetDef, headers, rowCount, colCount, behavior, expectedSignature, timing);
  return sheet;
}


function shouldRefreshTemplateMetadataOnly_(sheet, sheetDef, templateExisted, signatureMatches) {
  if (!RFF_TRUE_SMART_TEMPLATE_REFRESH) return false;
  if (!RFF_SMART_TEMPLATE_REFRESH) return false;
  if (RFF_FORCE_FULL_TEMPLATE_REBUILD) return false;
  if (!templateExisted || !sheet) return false;
  if (!signatureMatches) return false;


  // This intentionally avoids expensive row/column/header/filter/freeze checks.
  // The stored dashboard signature is the source of truth for whether a full
  // template rebuild is needed. Validation remains available to catch manual
  // sheet edits or corruption when needed.
  return true;
}


function buildTemplateRefreshDecisionMessage_(sheetDef, templateExisted, signatureMatches, metadataRefreshAllowed, normalizedStoredSignature, normalizedExpectedSignature) {
  const reasons = [];


  if (!templateExisted) reasons.push("template sheet did not exist");
  if (!RFF_TRUE_SMART_TEMPLATE_REFRESH) reasons.push("true smart refresh disabled");
  if (!RFF_SMART_TEMPLATE_REFRESH) reasons.push("smart refresh disabled");
  if (RFF_FORCE_FULL_TEMPLATE_REBUILD) reasons.push("force full rebuild enabled");
  if (!RFF_USE_TEMPLATE_SIGNATURE_CACHE) reasons.push("signature cache disabled");
  if (!normalizedStoredSignature) reasons.push("no stored signature");
  if (normalizedStoredSignature && normalizedExpectedSignature && normalizedStoredSignature !== normalizedExpectedSignature) reasons.push("stored signature differs from expected signature");
  if (signatureMatches && metadataRefreshAllowed) reasons.push("metadata refresh will run");


  const decision = metadataRefreshAllowed ? "METADATA_ONLY" : "FULL_BUILD";
  return "Template refresh decision: " + decision + " - " + sheetDef.templateName + " (" + (reasons.length ? reasons.join("; ") : "no reason captured") + ")";
}




function refreshTemplateMetadataOnly_(sheet, dashboard, sheetDef, colCount, expectedSignature, timing) {
  writeTemplateMetadata_(sheet, dashboard, sheetDef, colCount, expectedSignature);
  storeTemplateFormatSignature_(sheetDef, expectedSignature);
  markFrameworkStep_(timing, "Metadata-only refresh; template unchanged: " + sheetDef.templateName);


  // Do not hide/show, clear, resize, rebuild alternating colors, reapply widths,
  // reapply filters, or re-freeze during metadata-only refresh. Those operations
  // are intentionally reserved for full builds only.
  markFrameworkStep_(timing, "Complete metadata-only template refresh: " + sheetDef.templateName);
}


function buildTemplateFromDashboard_(sheet, dashboard, sheetDef, headers, rowCount, colCount, behavior, expectedSignature, timing) {
  markFrameworkStep_(timing, "Full template build required: " + sheetDef.templateName);


  clearTemplateForFullBuild_(sheet, sheetDef, timing);
  resizeSheet_(sheet, rowCount, colCount);
  markFrameworkStep_(timing, "Resize sheet: " + sheetDef.templateName + " (" + rowCount + " rows x " + colCount + " cols)");


  applyTemplateBaseFormatting_(sheet, dashboard, sheetDef, headers, rowCount, colCount, behavior, timing);
  markFrameworkStep_(timing, "Apply full template formatting: " + sheetDef.templateName);


  writeTemplateMetadata_(sheet, dashboard, sheetDef, colCount, expectedSignature);
  storeTemplateFormatSignature_(sheetDef, expectedSignature);
  markFrameworkStep_(timing, "Write metadata/signature: " + sheetDef.templateName);


  applyTemplateFreezeAndTabColor_(sheet, dashboard, sheetDef, colCount, timing);


  enforceTemplateDateAndNumberFormats_(sheet, dashboard, sheetDef, rowCount);
  markFrameworkStep_(timing, "Final date/number format enforcement: " + sheetDef.templateName);


  if (behavior.hiddenTemplate) {
    hideTemplateIfNeeded_(sheet, sheetDef, timing);
  } else {
    markFrameworkStep_(timing, "Template remains visible: " + sheetDef.templateName);
  }


  markFrameworkStep_(timing, "Complete full template build: " + sheetDef.templateName);
}


function shouldUseCorrectedIndividualTemplateBuild_(sheetDef) {
  /*
   * v1.4.30:
   * Individual template-build branches remain retired. All governed templates use
   * the shared dashboard build sequence. Create and update are now separated by
   * templateExisted inside buildTemplateFromDashboard_: brand-new blank sheets
   * skip clear sheet/formats; existing templates clear only when a full rebuild
   * is required after signature mismatch.
   */
  return false;
}


function clearTemplateForFullBuild_(sheet, sheetDef, timing) {
  /*
   * v1.5:
   * Skip clear on brand-new/blank sheets. Reserve clear/formats for existing
   * template rebuilds where content or formatting must be replaced.
   */
  const lastRow = sheet.getLastRow ? sheet.getLastRow() : 0;
  const lastCol = sheet.getLastColumn ? sheet.getLastColumn() : 0;
  const appearsBlank = lastRow < 1 || lastCol < 1;

  if (appearsBlank) {
    markFrameworkStep_(timing, "Create template fast path - clear skipped for blank sheet: " + sheetDef.templateName);
    return;
  }

  try {
    sheet.clear();
    sheet.setConditionalFormatRules([]);
  } catch (err) {
    sheet.clearContents();
    sheet.clearFormats();
    try {
      sheet.setConditionalFormatRules([]);
    } catch (ruleErr) {
      Logger.log("Conditional format clear skipped for " + sheet.getName() + ": " + ruleErr.message);
    }
  }
  markFrameworkStep_(timing, "Clear sheet/formats for full build: " + sheetDef.templateName);
}



function applyTemplateRowHeights_(sheet, dashboard, sheetDef, rowCount, timing, stepPrefix) {
  /*
   * v1.5:
   * Template row-height processing is intentionally disabled.
   * Empty templates should not pay row-height cost; final output row-height locking
   * belongs only to Raw Data, Demo P, Master List, and Disenrolled output workflows.
   */
  markFrameworkStep_(timing, (stepPrefix || "Apply row heights") + " skipped for template: " + sheetDef.templateName);
}



function lockFinalOutputRowHeights_(sheet, label) {
  if (!sheet) return;


  const dashboard = loadDashboardConfig_();
  const globals = dashboard.globals || {};
  const headerRow = Number(globals.headerRow || HEADER_ROW);
  const dataStartRow = Number(globals.dataStartRow || DATA_START_ROW);
  const lastRow = Math.max(sheet.getLastRow(), dataStartRow);


  try {
    applyRowHeightRuns_(sheet, [
      { start: 1, count: 1, height: Number(globals.row1Height || 25) },
      { start: 2, count: 1, height: Number(globals.row2Height || 20) },
      { start: 3, count: 1, height: Number(globals.row3Height || 10) },
      { start: headerRow, count: 1, height: Number(globals.headerRowHeight || 40) }
    ]);


    if (lastRow >= dataStartRow) {
      safeSetRowHeights_(sheet, dataStartRow, lastRow - dataStartRow + 1, Number(globals.dataRowHeight || 25), "final output data rows");
    }
  } catch (err) {
    Logger.log("Final row height lock skipped for " + (label || sheet.getName()) + ": " + err.message);
  }
}


function safeSetRowHeights_(sheet, startRow, count, height, label) {
  if (!sheet) return 0;


  startRow = Math.max(1, Number(startRow || 1));
  count = Math.max(0, Number(count || 0));
  height = Number(height || 25);


  if (count < 1) return 0;


  const maxRows = Math.max(sheet.getMaxRows(), 1);
  if (startRow > maxRows) return 0;


  const safeCount = Math.min(count, maxRows - startRow + 1);
  if (safeCount < 1) return 0;


  try {
    if (typeof sheet.setRowHeightsForced === "function") {
      sheet.setRowHeightsForced(startRow, safeCount, height);
    } else if (typeof sheet.setRowHeights === "function") {
      sheet.setRowHeights(startRow, safeCount, height);
    } else {
      for (let row = startRow; row < startRow + safeCount; row++) {
        sheet.setRowHeight(row, height);
      }
    }
    return safeCount;
  } catch (err) {
    Logger.log("Row height run skipped for " + (label || sheet.getName()) + " rows " + startRow + "-" + (startRow + safeCount - 1) + ": " + err.message);


    let applied = 0;
    for (let row = startRow; row < startRow + safeCount; row++) {
      try {
        sheet.setRowHeight(row, height);
        applied++;
      } catch (rowErr) {
        Logger.log("Row height skipped for row " + row + ": " + rowErr.message);
      }
    }
    return applied;
  }
}




function applyRowHeightRuns_(sheet, runs) {
  (runs || []).forEach(function(run) {
    if (!run || !run.start || !run.count) return;
    safeSetRowHeights_(sheet, run.start, run.count, run.height, "row-height run");
  });
}


function hideTemplateIfNeeded_(sheet, sheetDef, timing) {
  try {

      markFrameworkStep_(timing, "Template already hidden - skipped hide: " + sheetDef.templateName);
      return;
    }


    hideSheetIfNeeded_(sheet, timing, "Hide template: " + sheetDef.templateName);
  } catch (err) {

    markFrameworkStep_(timing, "Hide template skipped/error: " + sheetDef.templateName);
  }
}




function resolveTemplateRowCount_(sheetDef, globals) {
  const headerRow = globals && globals.headerRow ? globals.headerRow : RFF_DEFAULTS.headerRow;
  const fixedRows = numberOrDefault_(sheetDef.templateRowCount, RFF_DEFAULTS.templateRows);
  const minRows = numberOrDefault_(sheetDef.minimumRows, fixedRows);
  const testRows = numberOrDefault_(sheetDef.testRows, Math.min(fixedRows, 500));
  const bufferRows = numberOrDefault_(sheetDef.bufferRows, 100);
  const mode = String(sheetDef.templateRowMode || "FIXED").trim().toUpperCase();


  if (sheetDef.useTestRows || mode === "TEST") {
    return Math.max(testRows, headerRow);
  }


  if (mode === "DYNAMIC") {
    // Phase 1 templates do not yet receive source data counts. Use the dashboard
    // minimum + buffer until the Phase 2+ report formatters pass actual data row counts.
    return Math.max(minRows + bufferRows, headerRow);
  }


  return Math.max(fixedRows, headerRow);
}

function applyColumnWidths_(sheet, dashboard, headers) {
  const widths = (headers || []).map(function(header) {
    const def = dashboard.columnDefinitions[header] || {};
    return Number(def.width || 140);
  });


  applyColumnWidthsInRuns_(sheet, widths);
}


function applyColumnWidthsInRuns_(sheet, widths) {
  if (!sheet || !widths || widths.length === 0) return;


  let startCol = 1;
  let currentWidth = widths[0];


  for (let i = 1; i <= widths.length; i++) {
    const nextWidth = i < widths.length ? widths[i] : null;


    if (i < widths.length && nextWidth === currentWidth) {
      continue;
    }


    const runLength = i - startCol + 1;


    try {
      if (typeof sheet.setColumnWidths === "function") {
        sheet.setColumnWidths(startCol, runLength, currentWidth);
      } else {
        for (let col = startCol; col < startCol + runLength; col++) {
          sheet.setColumnWidth(col, currentWidth);
        }
      }
    } catch (err) {
      Logger.log("Column width run skipped for columns " + startCol + "-" + (startCol + runLength - 1) + ": " + err.message);
      for (let col = startCol; col < startCol + runLength; col++) {
        try {
          sheet.setColumnWidth(col, currentWidth);
        } catch (colErr) {
          Logger.log("Column width skipped for column " + col + ": " + colErr.message);
        }
      }
    }


    startCol = i + 1;
    currentWidth = nextWidth;
  }
}


function applyDateAndNumberFormats_(sheet, dashboard, headers, rowCount) {

function enforceTemplateDateAndNumberFormats_(sheet, dashboard, sheetDef, rowCount) {
  const headers = getHeadersForSheetType_(dashboard, sheetDef.sheetType);
  enforceDateAndNumberFormatsForHeaders_(sheet, dashboard, headers, rowCount, "Final date/number format enforcement");
}


function enforceDateAndNumberFormatsForHeaders_(sheet, dashboard, headers, rowCount, logPrefix) {
  const globals = dashboard.globals;
  const dataStartRow = globals.dataStartRow;
  const dataRows = Math.max(rowCount - dataStartRow + 1, 1);
  const sampleRows = Math.min(25, dataRows);


  headers.forEach(function(header, index) {
    const def = dashboard.columnDefinitions[header] || {};
    const format = getExpectedNumberFormat_(def, globals);


    if (!format) return;


    const googleSheetsFormat = getGoogleSheetsNumberFormat_(format, def.dateColumn);
    const col = index + 1;


    try {
      // Apply the full template data range first.
      sheet.getRange(dataStartRow, col, dataRows, 1).setNumberFormat(googleSheetsFormat);


      // Then reapply the first sample rows used by validation/testing. This prevents
      // stale or partial number formats in the first data rows from causing false
      // failures such as 0.############### on one date column.
      sheet.getRange(dataStartRow, col, sampleRows, 1).setNumberFormat(googleSheetsFormat);
    } catch (err) {
      Logger.log((logPrefix || "Date/number format") + " skipped for " + header + ": " + err.message);
    }
  });


  if (String(logPrefix || "").indexOf("Final") === 0) {
    SpreadsheetApp.flush();
  }
}


function getExpectedNumberFormat_(columnDef, globals) {
  const def = columnDef || {};
  if (def.numberFormat) return String(def.numberFormat).trim();
  if (def.dateColumn) return String(globals.defaultDateFormat || RFF_DEFAULTS.defaultDateFormat).trim();
  return "";
}


function getGoogleSheetsNumberFormat_(format, isDateColumn) {
  const text = String(format || "").trim();


  if (!text) return "";


  if (isDateColumn || isDateFormatText_(text)) {
    const lower = text.toLowerCase().replace(/\s+/g, "");
    if (lower === "mm/dd/yyyy" || lower === "m/d/yyyy") return "m/d/yyyy";
    if (lower === "mm/dd/yy" || lower === "m/d/yy") return "m/d/yy";
    if (lower === "mm.dd.yy" || lower === "m.d.yy") return "m.d.yy";
    if (lower === "mm.dd.yyyy" || lower === "m.d.yyyy") return "m.d.yyyy";
  }


  return text;
}


function isDateFormatText_(format) {
  const text = String(format || "").toLowerCase();
  return text.indexOf("d") !== -1 && text.indexOf("y") !== -1 && text.indexOf("m") !== -1;
}


function applyHiddenColumnSettings_(sheet, dashboard, headers) {
  const hiddenFlags = (headers || []).map(function(header) {
    const def = dashboard.columnDefinitions[header] || {};
    return !!def.hideColumn;
  });


  applyHiddenColumnSettingsInRuns_(sheet, hiddenFlags);
}


function applyHiddenColumnSettingsInRuns_(sheet, hiddenFlags) {
  if (!sheet || !hiddenFlags || hiddenFlags.length === 0) return;


  let startCol = 1;
  let currentHidden = hiddenFlags[0];


  for (let i = 1; i <= hiddenFlags.length; i++) {
    const nextHidden = i < hiddenFlags.length ? hiddenFlags[i] : null;


    if (i < hiddenFlags.length && nextHidden === currentHidden) {
      continue;
    }


    const runLength = i - startCol + 1;


    try {
      if (currentHidden) {
        sheet.hideColumns(startCol, runLength);
      } else {
        sheet.showColumns(startCol, runLength);
      }
    } catch (err) {
      Logger.log("Hidden column run skipped for columns " + startCol + "-" + (startCol + runLength - 1) + ": " + err.message);
      for (let col = startCol; col < startCol + runLength; col++) {
        try {
          if (currentHidden) {
            sheet.hideColumns(col);
          } else {
            sheet.showColumns(col);
          }
        } catch (colErr) {
          Logger.log("Hidden column setting skipped for column " + col + ": " + colErr.message);
        }
      }
    }


    startCol = i + 1;
    currentHidden = nextHidden;
  }
}

function applyAlternatingColorRules_(sheet, range, theme) {
  try {
    const rules = sheet.getConditionalFormatRules() || [];


    const oddRule = SpreadsheetApp.newConditionalFormatRule()
      .whenFormulaSatisfied("=ISODD(ROW())")
      .setBackground(theme.level3)
      .setRanges([range])
      .build();


    const evenRule = SpreadsheetApp.newConditionalFormatRule()
      .whenFormulaSatisfied("=ISEVEN(ROW())")
      .setBackground(theme.level4)
      .setRanges([range])
      .build();


    rules.push(oddRule);
    rules.push(evenRule);
    sheet.setConditionalFormatRules(rules);
  } catch (err) {
    Logger.log("Alternating color rules skipped for " + sheet.getName() + ": " + err.message);
  }
}

        .setFontSize((globals && globals.standardFontSize) || 10)
        .setFontColor((globals && globals.standardFontColor) || "#000000");

function buildTemplateFormatSignature_(dashboard, sheetDef, headers, rowCount, colCount, behavior) {
  const globals = dashboard.globals || {};
  const parts = [
    "sheetType=" + sheetDef.sheetType,
    "rows=" + rowCount,
    "cols=" + colCount,
    "freeze=" + globals.freezeRows + ":" + globals.freezeColumns,
    "date=" + String(globals.defaultDateFormat || ""),
    "behavior=" + [behavior.usesFilter, behavior.usesAlternatingColors, behavior.usesSubheaders, behavior.hiddenTemplate].join(",")
  ];


  headers.forEach(function(header, index) {
    const def = dashboard.columnDefinitions[header] || {};
    parts.push([
      index + 1,
      normalizeHeader_(header),
      def.width || 140,
      def.headerFontSize || globals.standardFontSize || 10,
      boolText_(!!def.dateColumn),
      boolText_(!!def.hideColumn),
      def.numberFormat || ""
    ].join("~"));
  });


  return parts.join("|");
}


function normalizeTemplateFormatSignature_(signature) {
  // Keep signature comparisons stable across script-version-only updates.
  // A new script version should not force expensive template rebuilds when dashboard
  // structure, headers, widths, hidden columns, and behavior did not change.
  return String(signature || "")
    .split("|")
    .filter(function(part) { return String(part || "").indexOf("v=") !== 0; })
    .join("|");
}


function getTemplateFormatSignatureKey_(sheetDef) {
  return "RFF_TEMPLATE_SIGNATURE_" + String(sheetDef.templateName || sheetDef.sheetType || "")
    .replace(/[^A-Za-z0-9_]/g, "_")
    .toUpperCase();
}

  } catch (err) {

  }

function getStoredTemplateFormatSignatureFromSheet_(sheet) {
  if (!sheet) return "";


  try {
    const lastCol = Math.max(sheet.getLastColumn(), 1);
    const notes = sheet.getRange(1, 1, 1, lastCol).getNotes()[0];
    for (let i = 0; i < notes.length; i++) {
      const note = String(notes[i] || "");
      const match = note.match(/Format Signature:\s*([^\n]+)/);
      if (match && match[1]) return match[1].trim();
    }
  } catch (err) {
    Logger.log("Template signature note read skipped for " + sheet.getName() + ": " + err.message);
  }


  return "";
}

function ensureTemplateFilter_(sheet, headerRow, rowCount, colCount, sheetDef, timing) {
  const expectedRows = Math.max(rowCount - headerRow + 1, 1);
  let existing = null;
  const correctedCreatePath = shouldUseCorrectedIndividualTemplateBuild_(sheetDef);


  try {
    existing = sheet.getFilter();
  } catch (err) {
    Logger.log("Template filter lookup skipped for " + sheet.getName() + ": " + err.message);
  }


  if (existing) {
    try {
      const range = existing.getRange();
      const matches = range.getRow() === headerRow &&
        range.getColumn() === 1 &&
        range.getNumRows() === expectedRows &&
        range.getNumColumns() === colCount;


      if (RFF_FAST_TEMPLATE_REFRESH && matches) {
        markFrameworkStep_(timing, "Filter already correct - skipped recreation: " + sheetDef.templateName);
        return;
      }
    } catch (err) {
      Logger.log("Template filter comparison skipped for " + sheet.getName() + ": " + err.message);
    }


    try {
      existing.remove();
    } catch (err) {
      Logger.log("Template filter removal skipped for " + sheet.getName() + ": " + err.message);
    }
    markFrameworkStep_(timing, "Remove existing filter: " + sheetDef.templateName);
  } else if (!correctedCreatePath) {
    markFrameworkStep_(timing, "No existing filter to remove: " + sheetDef.templateName);
  }


  try {
    sheet.getRange(headerRow, 1, expectedRows, colCount).createFilter();
  } catch (err) {
    Logger.log("Template filter creation skipped for " + sheet.getName() + ": " + err.message);
  }
  markFrameworkStep_(timing, "Create filter: " + sheetDef.templateName);
}


function applyTemplateFreezeAndTabColor_(sheet, dashboard, sheetDef, colCount, timing) {
  const globals = dashboard.globals;
  const expectedFrozenRows = globals.freezeRows;
  const expectedFrozenCols = Math.min(globals.freezeColumns, colCount);
  const expectedTabColor = getThemeColorsFromBase_(sheetDef.baseColor).level1;


  try {

  }
  markFrameworkStep_(timing, "Ensure frozen rows: " + sheetDef.templateName);


  try {
    if (!(RFF_FAST_TEMPLATE_REFRESH && sheet.getFrozenColumns() === expectedFrozenCols)) {
      sheet.setFrozenColumns(expectedFrozenCols);
    }
  } catch (err) {
    Logger.log("Frozen columns skipped for " + sheetDef.templateName + ": " + err.message);
  }
  markFrameworkStep_(timing, "Ensure frozen columns: " + sheetDef.templateName);


  try {
    const currentColor = String(sheet.getTabColor() || "").toUpperCase();

    }
  } catch (err) {

  }
  markFrameworkStep_(timing, "Ensure tab color: " + sheetDef.templateName);
}


function resizeSheet_(sheet, rowCount, colCount) {
  rowCount = Math.max(rowCount, 1);
  colCount = Math.max(colCount, 1);

  const currentRows = sheet.getMaxRows();
  const currentCols = sheet.getMaxColumns();

  if (currentRows < rowCount) {
    sheet.insertRowsAfter(currentRows, rowCount - currentRows);
  } else if (currentRows > rowCount) {
    sheet.deleteRows(rowCount + 1, currentRows - rowCount);
  }

  if (currentCols < colCount) {
    sheet.insertColumnsAfter(currentCols, colCount - currentCols);
  } else if (currentCols > colCount) {
    sheet.deleteColumns(colCount + 1, currentCols - colCount);
  }
}



function resizeSheetRows_(sheet, rowCount) {
  /*
   * v1.4.33:
   * Row resizing is the dominant template-build slow step. New Google Sheets
   * usually start with 1000 rows, so deleting rows down to 500/800 creates
   * avoidable runtime. Only add rows when the sheet has fewer than the dashboard
   * target. Do not delete extra blank rows during template creation.
   */
  rowCount = Math.max(rowCount, 1);
  const currentRows = sheet.getMaxRows();


  if (currentRows < rowCount) {
    sheet.insertRowsAfter(currentRows, rowCount - currentRows);
  }
}




function resizeSheetColumns_(sheet, colCount) {
  colCount = Math.max(colCount, 1);
  const currentCols = sheet.getMaxColumns();


  if (currentCols < colCount) {
    sheet.insertColumnsAfter(currentCols, colCount - currentCols);
  } else if (currentCols > colCount) {
    sheet.deleteColumns(colCount + 1, currentCols - colCount);
  }
}
function getHeadersForSheetType_(dashboard, sheetType) {
  const items = dashboard.sheetHeaders[sheetType] || [];
  return items.map(function(item) {
    return item.header;
  });
}

    usesAlternatingColors: true,
    usesSubheaders: false,
    hiddenTemplate: true
  };
}


function showSheetIfNeeded_(sheet, timing, stepName) {
  if (!sheet) return;
  try {
    if (typeof sheet.isSheetHidden === "function" && sheet.isSheetHidden()) {
      sheet.showSheet();
      if (timing && stepName) markFrameworkStep_(timing, stepName);
    }
  } catch (err) {
    Logger.log("Sheet show skipped for " + sheet.getName() + ": " + err.message);
  }
}


function hideSheetIfNeeded_(sheet, timing, stepName) {
  if (!sheet) return;
  try {
    if (typeof sheet.isSheetHidden === "function" && !sheet.isSheetHidden()) {
      sheet.hideSheet();
      if (timing && stepName) markFrameworkStep_(timing, stepName);
    }
  } catch (err) {
    Logger.log("Sheet hide skipped for " + sheet.getName() + ": " + err.message);
  }
}

function formatBannerReport() {
  return runFrameworkTimed_("Format Banner Report", function(timing) {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const rawSheetOriginal = ss.getActiveSheet();


    if (!rawSheetOriginal) {
      throw new Error("No active sheet found. Open/select the raw Banner import sheet before running Format Banner Report.");
    }


    const reportDate = promptForFormatterReportMonth_();
    const dateLabel = formatReportDateLabel_(reportDate);
    const rawName = "Raw Data - Banners " + Utilities.formatDate(reportDate, Session.getScriptTimeZone(), "MM.yy");
    const outputName = buildBannerReportOutputName_(reportDate);
    const templateName = "Template - Banner Report";


    if (rawSheetOriginal.getName() === templateName) {
      throw new Error("The active sheet is the Banner template. Select the raw Banner import sheet before running the formatter.");
    }
    if (rawSheetOriginal.getName() === RFF_TIMING_SHEET || rawSheetOriginal.getName() === RFF_DASHBOARD_SHEET) {
      throw new Error("The active sheet is not a raw Banner import sheet. Select the raw Banner import sheet before running the formatter.");
    }


    const rawSheet = renameSheetSafely_(ss, rawSheetOriginal, rawName);
    markFrameworkStep_(timing, "Raw sheet renamed: " + rawSheet.getName());


    const templateSheet = ss.getSheetByName(templateName);
    if (!templateSheet) {
      throw new Error(templateName + " is missing. Run Refresh Banner Template once before using Format Banner Report.");
    }
    markFrameworkStep_(timing, "Existing template verified: " + templateName);


    deleteSheetIfExists_(ss, outputName, rawSheet.getName(), templateName);
    const outputSheet = templateSheet.copyTo(ss);
    outputSheet.setName(outputName);
    ss.setActiveSheet(outputSheet);
    markFrameworkStep_(timing, "Template duplicated as output sheet: " + outputName);


    writeBannerReportDates_(outputSheet, reportDate);
    markFrameworkStep_(timing, "Report title dates written");


    copyRawBannerDataToOutput_(rawSheet, outputSheet);
    markFrameworkStep_(timing, "Raw data bulk-copied to duplicated template");


    markFrameworkStep_(timing, "Production output validation skipped");


    if (RFF_ENABLE_AUTO_ARCHIVE_RAW_DATA) {
      archiveRawDataSheet_(rawSheet, rawSheet.getName());
      markFrameworkStep_(timing, "Raw data copied to archive: " + rawSheet.getName());


      if (RFF_DELETE_LOCAL_RAW_AFTER_ARCHIVE) {
        ss.setActiveSheet(outputSheet);
        ss.deleteSheet(rawSheet);
        markFrameworkStep_(timing, "Local raw sheet deleted after archive");
      }
    }


    markFrameworkStep_(timing, "Banner formatter complete");
    return outputSheet;
  });
}




function validateActiveBannerFormatterOutput() {
  return runFrameworkTimed_("Validate Active Banner Output", function(timing) {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getActiveSheet();


    if (!sheet) {
      throw new Error("No active sheet found.");
    }


    if (sheet.getName() === RFF_DASHBOARD_SHEET || sheet.getName() === RFF_TIMING_SHEET) {
      throw new Error("Select a formatted Banner Report sheet before running output validation.");
    }


    validateBannerFormatterOutput_(sheet);
    markFrameworkStep_(timing, "Active Banner output validated: PASS");
    return sheet;
  });
}


function archiveActiveRawDataSheet() {
  return runFrameworkTimed_("Archive Active Raw Data Sheet", function(timing) {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getActiveSheet();


    if (!sheet) throw new Error("No active sheet found.");
    if (sheet.getName() === RFF_TIMING_SHEET || sheet.getName() === RFF_DASHBOARD_SHEET) {
      throw new Error("Select a raw data sheet before running archive.");
    }


    archiveRawDataSheet_(sheet, sheet.getName());
    markFrameworkStep_(timing, "Active sheet archived: " + sheet.getName());
    return sheet;
  });
}


function promptForFormatterReportMonth_() {
  const ui = SpreadsheetApp.getUi();
  const response = ui.prompt("Report Month", RFF_FORMATTER_DATE_PROMPT, ui.ButtonSet.OK_CANCEL);


  if (response.getSelectedButton() !== ui.Button.OK) {
    throw new Error("Formatter cancelled.");
  }


  const month = parseReportMonthInput_(response.getResponseText());
  const year = new Date().getFullYear();
  return new Date(year, month - 1, 1);
}


function parseReportMonthInput_(value) {
  const text = String(value || "").trim();
  if (!text) throw new Error("Report month is required.");


  const numeric = Number(text);
  if (!isNaN(numeric) && numeric >= 1 && numeric <= 12) {
    return Math.floor(numeric);
  }
