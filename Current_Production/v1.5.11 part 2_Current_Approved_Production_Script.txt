  candidates.push(DISENROLLED_EXCLUSION_SHEET);


  for (let c = 0; c < candidates.length; c++) {
    const name = candidates[c];
    const sheet = ss.getSheetByName(name);
    if (!sheet) continue;


    const data = getDataValues_(sheet, HEADER_ROW, DATA_START_ROW);
    const pmrIdx = getPMRIndex_(data.headerMap);
    if (pmrIdx === -1) continue;


    (data.values || []).forEach(function(row) {
      const pmr = normalizePMR_(row[pmrIdx]);
      if (pmr) set.add(pmr);
    });


    if (set.size) return set;
  }


  return set;
}


function mapRowsToHeaders_(sourceRows, sourceHeaders, targetHeaders) {

  return (sourceRows || []).map(row => {
    return (targetHeaders || []).map(header => {
      const sourceIdx = sourceHeaderMap[header];
      return sourceIdx === undefined ? "" : row[sourceIdx];
    });
  });
}

function moveDisenrolledPMRsFromDemoPToExclusion_(demoSheet, monthParts) {
  const result = {
    pmrsRemoved: 0,
    rowsRemoved: 0,
    rowsCopied: 0
  };


  if (!demoSheet) return result;


  const lastRow = demoSheet.getLastRow();
  const lastCol = demoSheet.getLastColumn();


  if (lastRow < DATA_START_ROW || lastCol < 1) return result;


  const headers = getHeaders_(demoSheet, HEADER_ROW);
  const headerMap = buildHeaderIndexMap_(headers);
  const pmrIdx = getPMRIndex_(headerMap);
  const statusIdx = headerMap["Enrollment Status"];


  if (pmrIdx === -1 || statusIdx === undefined) {
    Logger.log("Demo P disenrolled split skipped: missing PMR or Enrollment Status header.");
    return result;
  }


  const dataRange = demoSheet.getRange(DATA_START_ROW, 1, lastRow - DATA_START_ROW + 1, lastCol);
  const values = dataRange.getValues();
  const disenrolledPMRs = new Set();


  values.forEach(row => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr) return;


    const status = String(row[statusIdx] || "").trim().toLowerCase();
    if (status === "disenrolled") disenrolledPMRs.add(pmr);
  });


  if (disenrolledPMRs.size === 0) return result;


  const rowsToRemove = [];
  const rowsByPMR = new Map();


  values.forEach((row, idx) => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr || !disenrolledPMRs.has(pmr)) return;


    rowsToRemove.push(DATA_START_ROW + idx);
    if (!rowsByPMR.has(pmr)) rowsByPMR.set(pmr, []);
    rowsByPMR.get(pmr).push(row);
  });


  result.pmrsRemoved = disenrolledPMRs.size;
  result.rowsRemoved = rowsToRemove.length;
  result.rowsCopied = appendDisenrolledPMRBlocksToExclusion_(demoSheet, headers, rowsByPMR, monthParts);


  deleteRowNumberBatches_(demoSheet, rowsToRemove);
  clearSheetRuntimeCachesForSheet_(demoSheet);


  return result;
}


function appendDisenrolledPMRBlocksToExclusion_(sourceSheet, sourceHeaders, rowsByPMR, monthParts) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const dashboard = loadDashboardConfig_();
  const sheetDef = getSheetDefinitionByType_(dashboard, SHEET_TYPE.DISENROLLED_EXCLUSION);
  const template = ss.getSheetByName(sheetDef.templateName);
  if (!template) {
    throw new Error("Template not found: " + sheetDef.templateName + ". Run Create / Refresh Disenrolled Exclusion List Template first.");
  }

  const exclusionHeaders = getHeadersForSheetType_(dashboard, SHEET_TYPE.DISENROLLED_EXCLUSION);
  const sourceHeaderMap = buildHeaderIndexMap_(sourceHeaders || []);
  const sourcePMRIdx = getPMRIndex_(sourceHeaderMap);
  const exclusionHeaderMap = buildHeaderIndexMap_(exclusionHeaders);
  const exclusionPMRIdx = getPMRIndex_(exclusionHeaderMap);

  const sourceIndexes = exclusionHeaders.map(function(header) {
    return Object.prototype.hasOwnProperty.call(sourceHeaderMap, header) ? sourceHeaderMap[header] : -1;
  });

  const rowsToWrite = [];
  if (rowsByPMR && rowsByPMR.size) {
    rowsByPMR.forEach(function(rows) {
      const sourceRow = rows && rows.length ? rows[0] : null;
      if (!sourceRow) return;

      const output = sourceIndexes.map(function(sourceIdx) {
        return sourceIdx === -1 ? "" : sourceRow[sourceIdx];
      });

      if (exclusionPMRIdx >= 0 && sourcePMRIdx >= 0 && !normalizePMR_(output[exclusionPMRIdx])) {
        output[exclusionPMRIdx] = sourceRow[sourcePMRIdx];
      }
      rowsToWrite.push(output);
    });
  }

  const outputSheetName = buildDatedDisenrolledOutputName_(monthParts);
  const legacy = ss.getSheetByName(DISENROLLED_EXCLUSION_SHEET);
  if (legacy && ss.getSheets().length > 1) ss.deleteSheet(legacy);
  const existing = ss.getSheetByName(outputSheetName);
  if (existing && ss.getSheets().length > 1) ss.deleteSheet(existing);

  const sheet = template.copyTo(ss);
  sheet.showSheet();
  setUniqueSheetName_(sheet, outputSheetName);

  const width = Math.max(exclusionHeaders.length, 1);
  const startDate = monthParts && monthParts.firstDay ? monthParts.firstDay : getTodayLocalDate_();
  const endDate = monthParts && monthParts.lastDay ? monthParts.lastDay : startDate;

  // v1.5: The copied dashboard template already owns headers, row count, column count, and formatting.
  // Do not force row count or clear blank data rows before the first output write.
  sheet.getRange("A1").setValue(DISENROLLED_EXCLUSION_SHEET);
  sheet.getRange("A2:D2").setValues([["Date:", startDate, "to", endDate]]);
  sheet.getRange(HEADER_ROW, 1, 1, width).setValues([exclusionHeaders]);

  if (rowsToWrite.length) {
    sheet.getRange(DATA_START_ROW, 1, rowsToWrite.length, width).setValues(rowsToWrite);
  }

  applyFinalRowHeightLock_(sheet, 25);

  try {
    sheet.getRange("A1").setNote(
      "Rows Processed: " + (rowsByPMR ? rowsByPMR.size : 0) + "\\n" +
      "Rows Written: " + rowsToWrite.length + "\\n" +
      "v1.5 Disenrolled single-write output"
    );
  } catch (err) {
    Logger.log("Disenrolled output note skipped: " + err.message);
  }

  clearSheetRuntimeCachesForSheet_(sheet);
  return rowsToWrite.length;
}

function syncDemoPBannerColumns_(demoSheet, monthParts) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const bannerSheet = getCurrentBannersSheet_(monthParts);


  if (!bannerSheet) {
    notify_("Banners sheet was not found. Demo P Banner sync skipped.");
    return;
  }


  const demoLastRow = demoSheet.getLastRow();
  const demoLastCol = demoSheet.getLastColumn();


  if (demoLastRow < DATA_START_ROW || demoLastCol < 1) return;


  const demoHeaders = getHeaders_(demoSheet, HEADER_ROW);
  const demoHeaderMap = buildHeaderIndexMap_(demoHeaders);


  const demoPMRIdx = getPMRIndex_(demoHeaderMap);
  const demoLastNameIdx = demoHeaderMap["Last Name"];
  const demoFirstNameIdx = demoHeaderMap["First Name"];


  if (demoPMRIdx === -1 || demoLastNameIdx === undefined || demoFirstNameIdx === undefined) {
    notify_("Demo P Banner sync skipped: missing Participant PMR#, Last Name, or First Name.");
    return;
  }


  const bannerMap = buildSourceMapByCompositeKeyForDemoPBanner_(
    bannerSheet,
    HEADER_ROW,
    DATA_START_ROW,
    ["Participant PMR#", "Last Name", "First Name"]
  );


  if (bannerMap.size === 0) {
    notify_("Demo P Banner sync skipped: no matching Banner records were found.");
    return;
  }


  const range = demoSheet.getRange(DATA_START_ROW, 1, demoLastRow - DATA_START_ROW + 1, demoLastCol);
  const values = range.getValues();


  values.forEach(row => {
    const demoDOBIdx = getDOBIndex_(demoHeaderMap);
    if (demoDOBIdx !== -1 && normalizeCompareValue_(row[demoDOBIdx]) === "") return;


    const key = [
      normalizeKeyPart_(row[demoPMRIdx]),
      normalizeKeyPart_(row[demoLastNameIdx]),
      normalizeKeyPart_(row[demoFirstNameIdx])
    ].join("|||");


    if (!bannerMap.has(key)) return;


    const source = bannerMap.get(key);


    DEMO_P_BANNER_SYNC_HEADERS.forEach(header => {
      const targetIdx = demoHeaderMap[header];
      if (targetIdx === undefined) return;
      if (!Object.prototype.hasOwnProperty.call(source, header)) return;
      row[targetIdx] = source[header];
    });
  });


  range.setValues(values);
}


function buildSourceMapByCompositeKeyForDemoPBanner_(sheet, headerRow, dataStartRow, keyHeaders) {
  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();
  const map = new Map();


  if (lastRow < dataStartRow || lastCol < 1) return map;


  const headers = getHeaders_(sheet, headerRow);
  const headerMap = buildHeaderIndexMap_(headers);


  if (keyHeaders.some(header => headerMap[header] === undefined)) return map;


  const data = sheet.getRange(dataStartRow, 1, lastRow - dataStartRow + 1, lastCol).getValues();


  data.forEach(row => {
    const key = keyHeaders
      .map(header => normalizeKeyPart_(row[headerMap[header]]))
      .join("|||");


    if (key.replace(/\|/g, "") === "") return;


    const record = {};
    headers.forEach((header, idx) => {
      if (header) record[header] = row[idx];
    });


    map.set(key, record);
  });


  return map;
}

function formatCarePlanDueOrUnlockedFromDashboard_(sheetType, promptTitle, endDateCellA1) {

    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sourceSheetOriginal = ss.getActiveSheet();


    if (!sourceSheetOriginal) {
      throw new Error("No active sheet found. Select the raw " + sheetType + " import sheet before running " + promptTitle + ".");
    }


    const monthParts = promptForLockedYearReportMonth_(promptTitle);
    if (!monthParts) return null;


    const dashboard = loadDashboardConfig_();
    const sheetDef = getSheetDefinitionByType_(dashboard, sheetType);
    const headers = getHeadersForSheetType_(dashboard, sheetType);
    const outputName = buildDashboardOutputSheetName_(sheetDef, monthParts);
    const rawName = buildRawArchiveNameForSheetType_(sheetType, monthParts);


    const activeName = String(sourceSheetOriginal.getName() || "").trim();
    if (activeName === sheetDef.templateName || activeName === outputName) {
      throw new Error("The active sheet is not a raw " + sheetType + " import sheet. Select the imported source sheet.");
    }


    const originalEndDate =
      coerceToValidDate_(sourceSheetOriginal.getRange(endDateCellA1).getValue()) ||
      coerceToValidDate_(sourceSheetOriginal.getRange(endDateCellA1).getDisplayValue()) ||
      monthParts.lastDay;


    const titleInfoText = collectAndClearMovedTitleInfoCells_(sourceSheetOriginal, sheetType);
    markFrameworkStep_(timing, "Move source title/date cells into dashboard title info");


    /*
     * v1.4.22:
     * Use the same output pattern as Banners. The raw import is normalized only
     * enough to expose the row-4 source headers. Formatting is owned by the
     * verified hidden template, not by the active raw sheet.
     */
    prepareCarePlanSourceSheetForDashboardFormat_(sourceSheetOriginal, sheetType, headers);
    markFrameworkStep_(timing, "Prepare raw source structure");


    const rawSheet = renameSheetSafely_(ss, sourceSheetOriginal, rawName);
    markFrameworkStep_(timing, "Raw sheet renamed: " + rawSheet.getName());


    const rawData = getDataValues_(rawSheet, HEADER_ROW, DATA_START_ROW);
    const outputRows = mapRowsToHeaders_(rawData.values, rawData.headers, headers);
    markFrameworkStep_(timing, "Raw data mapped to dashboard headers");


    deleteSheetIfExists_(ss, outputName, rawSheet.getName(), sheetDef.templateName);
    const outputSheet = createOutputSheetFromDashboardTemplate_(
      sheetType,
      outputName,
      outputRows,
      monthParts.firstDay,
      originalEndDate || monthParts.lastDay
    );
    markFrameworkStep_(timing, "Template duplicated and raw data pasted: " + outputSheet.getName());


    try {
      outputSheet.getRange("A1").setValue(sheetDef.reportTitle || sheetType);
      if (titleInfoText) outputSheet.getRange("A3").setValue(titleInfoText);
    } catch (err) {
      Logger.log("Could not write " + sheetType + " title info: " + err.message);
    }


    lockFinalOutputRowHeights_(outputSheet, sheetType);
    markFrameworkStep_(timing, "Lock final row heights");


    if (RFF_ENABLE_AUTO_ARCHIVE_RAW_DATA) {
      archiveRawDataSheet_(rawSheet, rawSheet.getName());
      markFrameworkStep_(timing, "Raw data copied to archive: " + rawSheet.getName());


      if (RFF_DELETE_LOCAL_RAW_AFTER_ARCHIVE) {
        ss.setActiveSheet(outputSheet);
        ss.deleteSheet(rawSheet);
        markFrameworkStep_(timing, "Local raw sheet deleted after archive");
      }
    }


    clearSheetRuntimeCachesForSheet_(outputSheet);
    markFrameworkStep_(timing, "Clear runtime caches");


    notify_(sheetDef.reportTitle + " formatted using template-copy pattern.\nSheet created: " + outputSheet.getName());
    return outputSheet;
  });
}

function writeChangedColumnsOnly_(sheet, data, beforeValues) {
  if (!sheet || !data || !data.values || !data.values.length) return 0;


  const indexes = Object.keys(beforeValues || {})
    .map(function(key) { return Number(key); })
    .filter(function(idx) {
      return idx >= 0 && data.values.some(function(row, r) {
        return !valuesAreEqual_(row[idx], beforeValues[idx][r]);
      });
    })
    .sort(function(a, b) { return a - b; });


  if (!indexes.length) return 0;


  const groups = [];
  let current = [indexes[0]];


  for (let i = 1; i < indexes.length; i++) {
    if (indexes[i] === current[current.length - 1] + 1) {
      current.push(indexes[i]);
    } else {
      groups.push(current);
      current = [indexes[i]];
    }
  }
  groups.push(current);


  groups.forEach(function(group) {
    const startIdx = group[0];
    const width = group.length;
    const values = data.values.map(function(row) {
      return row.slice(startIdx, startIdx + width);
    });


    sheet.getRange(DATA_START_ROW, startIdx + 1, values.length, width).setValues(values);
  });


  return indexes.length;
}

function applyDashboardTemplateFormattingToActiveReportSheet_(sheet, dashboard, sheetDef, headers, behavior, monthParts, endDate, titleInfoText, timing) {
  /*
   * v1.4.25:
   * Runtime report formatting is restricted. Verified dashboard-built templates
   * are the formatting authority. Normal sheet processes may write data,
   * metadata/date cells, and final row heights only; they may not reapply
   * dashboard formatting, hidden columns, filters, freezes, widths, or
   * date/number formatting outside the template builder.
   */
  if (!sheet || !sheetDef) return sheet;


  try {
    const globals = dashboard && dashboard.globals ? dashboard.globals : RFF_DEFAULTS;
    if (sheetDef.reportTitle) sheet.getRange("A1").setValue(sheetDef.reportTitle);
    if (monthParts && monthParts.firstDay) {
      sheet.getRange("B2").setValue(monthParts.firstDay);
    }
    if (endDate || (monthParts && monthParts.lastDay)) {
      sheet.getRange("D2").setValue(endDate || monthParts.lastDay);
    }
    if (titleInfoText !== undefined && titleInfoText !== null) {
      sheet.getRange("C1").setValue(titleInfoText);
    }
  } catch (err) {
    Logger.log("Dashboard-template runtime metadata update skipped for " + sheet.getName() + ": " + err.message);
  }


  applyFinalRowHeightLockForSheetType_(sheet, sheetDef.sheetType);
  clearSheetRuntimeCachesForSheet_(sheet);
  return sheet;
}


function resizeSheetMinimum_(sheet, rowCount, colCount) {
  const currentRows = sheet.getMaxRows();
  const currentCols = sheet.getMaxColumns();


  if (currentRows < rowCount) {
    sheet.insertRowsAfter(currentRows, rowCount - currentRows);
  }


  if (currentCols < colCount) {
    sheet.insertColumnsAfter(currentCols, colCount - currentCols);
  }
}


function buildDashboardOutputSheetName_(sheetDef, monthParts) {
  const pattern = String(sheetDef.outputNamingPattern || sheetDef.reportTitle || sheetDef.sheetType || "").trim();
  const monthText = Utilities.formatDate(monthParts.firstDay, Session.getScriptTimeZone(), "MM.yy");
  const monthTextNoLeading = Utilities.formatDate(monthParts.firstDay, Session.getScriptTimeZone(), "M.yy");
  const fullStart = formatDateForSheetName_(monthParts.firstDay);
  const fullEnd = formatDateForSheetName_(monthParts.lastDay);


  let name = pattern || String(sheetDef.reportTitle || sheetDef.sheetType || "Report").trim();


  name = name
    .replace(/mm\.yy/g, monthText)
    .replace(/MM\.yy/g, monthText)
    .replace(/m\.yy/g, monthTextNoLeading)
    .replace(/M\.yy/g, monthTextNoLeading)
    .replace(/start/gi, fullStart)
    .replace(/end/gi, fullEnd);


  if (name === pattern && name.toLowerCase().indexOf(monthText.toLowerCase()) === -1 && sheetDef.usePromptDate) {
    name = name + " " + monthText;
  }


  return safeSheetName_(name);
}

function getReportMonthFromB3D3NoPrompt_(sheet, reportName) {
  const startDate = coerceToValidDate_(sheet.getRange("B3").getValue()) ||
    coerceToValidDate_(sheet.getRange("B3").getDisplayValue());


  const endDate = coerceToValidDate_(sheet.getRange("D3").getValue()) ||
    coerceToValidDate_(sheet.getRange("D3").getDisplayValue());

    notify_(`${reportName}: B3 does not contain a valid report start date. The sheet was formatted but not renamed.`);
    return null;

    inputDate: firstDay,

    lastDay: lastDay,

    previousMonthLastDay: new Date(firstDay.getFullYear(), firstDay.getMonth(), 0),

  };
}


function getFirstValidDateFromCells_(sheet, cellList) {
  for (let i = 0; i < cellList.length; i++) {
    try {
      const range = sheet.getRange(cellList[i]);
      const value = range.getValue();
      const display = range.getDisplayValue();


      const parsedValue = coerceToValidDate_(value);
      if (parsedValue) return parsedValue;


      const parsedDisplay = coerceToValidDate_(display);
      if (parsedDisplay) return parsedDisplay;
    } catch (err) {
      Logger.log("Date read skipped for " + cellList[i] + ": " + err.message);
    }
  }


  return null;
}

function removeDisenrolledPMRBlocksFromMasterUsingDemoP_(masterSheet, demoSheet) {
  if (!masterSheet || !demoSheet) return 0;


  const demoHeaders = getHeaders_(demoSheet, HEADER_ROW);
  const demoHeaderMap = buildHeaderIndexMap_(demoHeaders);
  const demoLastRow = demoSheet.getLastRow();
  const demoLastCol = demoSheet.getLastColumn();


  if (demoLastRow < DATA_START_ROW || demoLastCol < 1) return 0;


  const demoValues = demoSheet.getRange(DATA_START_ROW, 1, demoLastRow - DATA_START_ROW + 1, demoLastCol).getValues();
  const disenrolledPMRs = buildDisenrolledPMRSetFromDemoPValues_(demoValues, demoHeaderMap);


  const exclusionPMRs = loadDisenrolledExclusionPMRsForPart3_();
  exclusionPMRs.forEach(pmr => disenrolledPMRs.add(pmr));


  if (disenrolledPMRs.size === 0) return 0;


  const masterHeaders = getHeaders_(masterSheet, HEADER_ROW);
  const masterHeaderMap = buildHeaderIndexMap_(masterHeaders);
  const masterPMRIdx = getPMRIndex_(masterHeaderMap);


  if (masterPMRIdx === -1) return 0;


  const masterLastRow = masterSheet.getLastRow();
  const masterLastCol = masterSheet.getLastColumn();


  if (masterLastRow < DATA_START_ROW || masterLastCol < 1) return 0;


  const range = masterSheet.getRange(DATA_START_ROW, 1, masterLastRow - DATA_START_ROW + 1, masterLastCol);
  const values = range.getValues();


  const keepRows = [];
  let removed = 0;


  values.forEach(row => {
    const pmr = normalizePMR_(row[masterPMRIdx]);
    if (pmr && disenrolledPMRs.has(pmr)) {
      removed++;
      return;
    }
    keepRows.push(row);
  });


  if (removed === 0) return 0;


  range.clearContent();


  if (keepRows.length > 0) {
    masterSheet.getRange(DATA_START_ROW, 1, keepRows.length, masterLastCol).setValues(keepRows);
  }


  return removed;
}

function createMasterList() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const monthParts = promptForLockedYearReportMonth_("Create Master List");
  if (!monthParts) return;


  const timing = startRuntimeTiming_("Create Master List", monthParts);
  let masterSheet = null;
  let copiedRowCount = 0;


  try {
    markRuntimeStep_(timing, "Start");


    const demoSheet = getCurrentDemoPSheet_(monthParts);
    markRuntimeStep_(timing, "Locate current processed Demo P sheet");


    if (!demoSheet) {
      notify_("Demo P sheet for that month was not found. Run Process Demo P first.");
      markRuntimeStep_(timing, "Stopped - missing Demo P");
      writeRuntimeTimingReport_(timing);
      return;
    }


    const masterName = buildMonthlySheetName_(MASTER_LIST_PREFIX, monthParts.firstDay, monthParts.lastDay);


    masterSheet = ss.getSheetByName(masterName);
    if (masterSheet) {
      const ui = SpreadsheetApp.getUi();
      const response = ui.alert(
        "Master List Exists",
        `${masterName} already exists. Replace it?`,
        ui.ButtonSet.YES_NO
      );


      if (response !== ui.Button.YES) {
        notify_("Create Master List cancelled.");
        markRuntimeStep_(timing, "Cancelled - existing Master List not replaced");
        writeRuntimeTimingReport_(timing);
        return;
      }


      ss.deleteSheet(masterSheet);
      markRuntimeStep_(timing, "Delete existing Master List sheet");
    }


    masterSheet = createMasterListSheetFromTemplate_(ss, masterName, monthParts);
    markRuntimeStep_(timing, "Copy Template - Master List as output sheet");


    buildMasterListHeadersBeforeDataCopy_(demoSheet, masterSheet);
    markRuntimeStep_(timing, "Build Master List headers");


    copiedRowCount = copyPrimaryDemoPRowsToMasterListByHeader_(demoSheet, masterSheet, monthParts);
    markRuntimeStep_(timing, "Copy Primary PMR rows from processed Demo P to Master List");


    trimOutputSheetToDataSize_(masterSheet, copiedRowCount || 0, SHEET_TYPE.MASTER_LIST);
    markRuntimeStep_(timing, "Right-size Master List output rows");


    fillMasterListMetadataColumns_(masterSheet, monthParts, demoSheet.getName(), "Created from processed Demo P primary rows");
    markRuntimeStep_(timing, "Fill Master List metadata columns");


    syncMasterListFromUnlockedCarePlan_(masterSheet, monthParts, null);
    markRuntimeStep_(timing, "Sync Unlocked CP to Master List primary rows");


    syncMasterListFromCarePlanDue_(masterSheet, monthParts, null);
    markRuntimeStep_(timing, "Sync Care Plan Due to Master List primary rows");


    setUniqueSheetName_(masterSheet, masterName);
    markRuntimeStep_(timing, "Set final Master List sheet name");


    enforceMasterListFinalDisplay_(masterSheet);
    markRuntimeStep_(timing, "Final Master List display enforcement");


    lockFinalOutputRowHeights_(masterSheet, "Master List");
    markRuntimeStep_(timing, "Lock Master List final row heights");


    writeRuntimeTimingReport_(timing);


    notify_(
      `Master List created. Copied ${copiedRowCount} Primary PMR row(s) from processed Demo P.\n\n` +
      `Runtime: ${formatSeconds_((new Date().getTime() - timing.startMs) / 1000)}`
    );


    return masterSheet;
  } catch (err) {
    markRuntimeStep_(timing, "ERROR - " + err.message);
    try {
      writeRuntimeTimingReport_(timing);
    } catch (reportErr) {
      Logger.log("Could not write timing report: " + reportErr.message);
    }
    throw err;
  }
}




function copyPrimaryDemoPRowsToMasterListByHeader_(demoSheet, masterSheet, monthParts) {
  const demoData = getDataValues_(demoSheet, HEADER_ROW, DATA_START_ROW);

  if (!demoData.values.length) return 0;

  const demoHeaders = demoData.headers;
  const demoHeaderMap = demoData.headerMap;
  const masterHeaders = getHeaders_(masterSheet, HEADER_ROW);
  const masterHeaderMap = getHeaderMap_(masterSheet, HEADER_ROW);

  const pmrIdx = getPMRIndex_(demoHeaderMap);
  if (pmrIdx === -1) {
    throw new Error("Demo P is missing Participant PMR# / PMR # column.");
  }

  const primaryIdx = demoHeaderMap["Primary PMR Row"];
  const dobIdx = getDOBIndex_(demoHeaderMap);

  function getMasterTargetIndex_(header) {
    if (masterHeaderMap[header] !== undefined) return masterHeaderMap[header];

    const aliases = {
      "Participant PMR#": ["PMR #", "PMR#", "Participant PMR"],
      "PMR #": ["Participant PMR#", "PMR#", "Participant PMR"],
      "PMR#": ["Participant PMR#", "PMR #", "Participant PMR"],
      "Participant PMR": ["Participant PMR#", "PMR #", "PMR#"]
    };

    const candidates = aliases[header] || [];
    for (let i = 0; i < candidates.length; i++) {
      if (masterHeaderMap[candidates[i]] !== undefined) return masterHeaderMap[candidates[i]];
    }

    return undefined;
  }

  function buildOutputRows_(mode) {
    const output = [];
    const seenPMRs = new Set();

    demoData.values.forEach(row => {
      const pmr = normalizePMR_(row[pmrIdx]);
      if (!pmr) return;

      let includeRow = false;

      if (mode === "primary") {
        includeRow = primaryIdx !== undefined && isPrimaryPMRRowValue_(row[primaryIdx]);
      } else if (mode === "dob") {
        includeRow = dobIdx !== -1 && normalizeCompareValue_(row[dobIdx]) !== "" && !seenPMRs.has(pmr);
      } else if (mode === "first") {
        includeRow = !seenPMRs.has(pmr);
      }

      if (!includeRow) return;
      seenPMRs.add(pmr);

      const out = new Array(masterHeaders.length).fill("");

      demoHeaders.forEach((header, sourceIdx) => {
        if (!header) return;
        const targetIdx = getMasterTargetIndex_(header);
        if (targetIdx !== undefined) {
          out[targetIdx] = row[sourceIdx];
        }
      });

      output.push(out);
    });

    return output;
  }

  let output = buildOutputRows_("primary");

  /*
   * v1.5.1:
   * Master List creation depends on Primary PMR Row when available. Some v1.5
   * runs produced processed Demo P sheets where the Primary PMR Row column existed
   * but was blank, which caused an empty Master List. Preserve Primary PMR
   * governance first, then fall back to one row per PMR using DOB/first-row
   * selection only when no primary rows are detected.
   */
  if (output.length === 0) output = buildOutputRows_("dob");
  if (output.length === 0) output = buildOutputRows_("first");

  if (output.length > 0) {
    masterSheet.getRange(DATA_START_ROW, 1, output.length, masterHeaders.length).setValues(output);
    clearSheetRuntimeCachesForSheet_(masterSheet);
  }

  return output.length;
}

function fillMasterListMetadataColumns_(sheet, monthParts, sourceSheetName, statusText) {
  const lastRow = sheet.getLastRow();
  if (lastRow < DATA_START_ROW) return;


  const headers = getHeaders_(sheet, HEADER_ROW);
  const headerMap = buildHeaderIndexMap_(headers);


  const required = ["Master List Update Status", "Master List Update Month", "Master List Source Sheet"];
  const missing = required.filter(function(header) { return headerMap[header] === undefined; });
  if (missing.length) {
    Logger.log("Master List metadata write skipped for missing dashboard-defined columns: " + missing.join(", "));
  }


  const rowCount = lastRow - DATA_START_ROW + 1;
  const metadata = {
    "Master List Update Status": statusText || "Created",
    "Master List Update Month": monthParts.firstDay,
    "Master List Source Sheet": sourceSheetName || ""
  };


  const indexes = required
    .map(function(header) { return headerMap[header]; })
    .filter(function(idx) { return idx !== undefined; })
    .sort(function(a, b) { return a - b; });


  if (!indexes.length) return;


  if (indexes[indexes.length - 1] - indexes[0] + 1 === indexes.length) {
    const startIdx = indexes[0];
    const width = indexes.length;
    const rowTemplate = [];
    for (let c = startIdx; c <= indexes[indexes.length - 1]; c++) {
      const header = headers[c];
      rowTemplate.push(Object.prototype.hasOwnProperty.call(metadata, header) ? metadata[header] : "");
    }
    const values = Array.from({ length: rowCount }, function() { return rowTemplate.slice(); });
    sheet.getRange(DATA_START_ROW, startIdx + 1, rowCount, width).setValues(values);
    return;
  }


  indexes.forEach(function(idx) {
    const header = headers[idx];
    const value = Object.prototype.hasOwnProperty.call(metadata, header) ? metadata[header] : "";
    const values = Array.from({ length: rowCount }, function() { return [value]; });
    sheet.getRange(DATA_START_ROW, idx + 1, rowCount, 1).setValues(values);
  });
}

function extractFirstDateFromSheetName_(sheetName) {
  const text = String(sheetName || "").trim();
  const match = text.match(/(\d{1,2})\.(\d{1,2})\.(\d{2}|\d{4})/);
  if (!match) return null;


  const month = Number(match[1]);
  const day = Number(match[2]);
  let year = Number(match[3]);
  if (year < 100) year += 2000;


  const date = createLocalDateOnly_(year, month - 1, day);
  if (
    date.getFullYear() !== year ||
    date.getMonth() !== month - 1 ||
    date.getDate() !== day
  ) {
    return null;
  }


  return date;
}


function parseIndexMonthDate_(value) {
  if (value instanceof Date && !isNaN(value.getTime())) {
    return createLocalDateOnly_(value.getFullYear(), value.getMonth(), value.getDate());
  }


  const text = String(value || "").trim();
  if (!text) return null;


  const dotMatch = text.match(/^(\d{1,2})\.(\d{1,2})\.(\d{2}|\d{4})$/);
  if (dotMatch) {
    const month = Number(dotMatch[1]);
    const day = Number(dotMatch[2]);
    let year = Number(dotMatch[3]);
    if (year < 100) year += 2000;


    const date = createLocalDateOnly_(year, month - 1, day);
    if (
      date.getFullYear() === year &&
      date.getMonth() === month - 1 &&
      date.getDate() === day
    ) {
      return date;
    }
  }


  return coerceToValidDate_(text);
}


function createIndexSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName("Index");


  if (!sheet) {
    sheet = ss.insertSheet("Index", 0);
  } else {
    sheet.clear();
  }


  sheet;


  const rows = [];


  ss.getSheets().forEach(sh => {
    const sheetName = sh.getName();
    if (sheetName === "Index") return;


    const monthDate = extractFirstDateFromSheetName_(sheetName);
    const monthText = monthDate ? formatDateForSheetName_(monthDate) : "";
    const sheetType = getSheetTypeForOrganization_(sheetName);
    const hidden = sh.isSheetHidden() ? "Yes" : "No";


    rows.push([
      monthText,
      sheetName,
      `=HYPERLINK("#gid=${sh.getSheetId()}","Open")`,
      sheetType,
      hidden
    ]);
  });


  rows.sort((a, b) => {
    const aDate = parseIndexMonthDate_(a[0]);
    const bDate = parseIndexMonthDate_(b[0]);


    if (aDate && bDate && aDate.getTime() !== bDate.getTime()) {
      return bDate.getTime() - aDate.getTime();
    }


    const aType = SHEET_TAB_ORDER.indexOf(a[3]);
    const bType = SHEET_TAB_ORDER.indexOf(b[3]);
    const aOrder = aType === -1 ? 999 : aType;
    const bOrder = bType === -1 ? 999 : bType;


    if (aOrder !== bOrder) return aOrder - bOrder;


    return String(a[1]).localeCompare(String(b[1]));
  });


  sheet.getRange(1, 1, 1, INDEX_HEADERS.length)
    .setValues([INDEX_HEADERS]);


  if (rows.length > 0) {
    sheet.getRange(2, 1, rows.length, INDEX_HEADERS.length)
      .setValues(rows);
  }


  applyIndexSheetRowFills_(sheet);


  sheet;
  sheet;
  sheet;
  sheet;
  sheet;
  sheet;


  if (sheet.getFilter()) {
    sheet;
  }


  sheet.getRange(1, 1, Math.max(sheet.getLastRow(), 1), INDEX_HEADERS.length);


  applyCurrentVsOlderTabColors_();
  organizeSheetTabs_();
  applyCurrentVsOlderTabColors_();


  return sheet;


  applyStandardRowHeights_(sheet);
}

function compareRawDemoPForSectionReport_(previousDemo, currentDemo, monthParts) {
  const previousData = getRawDemoPDataForCompare_(previousDemo);
  const currentData = getRawDemoPDataForCompare_(currentDemo);


  const enrollmentPMRs = new Set();
  const disenrollmentPMRs = new Set();
  const demographicPMRs = new Set();
  const caseloadPMRs = new Set();
  const contactPMRs = new Set();


  const demographicChangedColumnsByPMR = new Map();
  const caseloadChangedColumnsByPMR = new Map();
  const contactChangedColumnsByPMR = new Map();


  currentData.rowsByPMR.forEach((currentRows, pmr) => {
    const previousRows = previousData.rowsByPMR.get(pmr) || [];


    const capitationHeader = "Capitation Date";
    const disenrollEffectiveHeader = "Disenrollment Effective Date";


    const mostRecentCapitationDate = getMostRecentDateFromRowsByHeader_(
      currentRows,
      currentData.headerMap,
      capitationHeader
    );


    const isEnrollmentPMR = isSameDate_(mostRecentCapitationDate, monthParts.firstDay);


    const isDisenrollmentPMR =
      currentRows.some(item => {
        const effectiveDate = getFieldValueFromRow_(
          item.values,
          currentData.headerMap,
          disenrollEffectiveHeader
        );


        return isDateInStrictLocalRangeInclusive_(
          effectiveDate,
          monthParts.previousMonthFirstDay,
          monthParts.firstDay
        );
      });


    if (isEnrollmentPMR) enrollmentPMRs.add(pmr);
    if (isDisenrollmentPMR) disenrollmentPMRs.add(pmr);


    if (isEnrollmentPMR || isDisenrollmentPMR) return;


    if (isParticipantEnrollmentStatusDisenrolled_(currentRows, currentData.headerMap)) return;


    const currentDobRows = rowsWithDOBOnlyForSection_(currentRows, currentData.headerMap);
    const previousDobRows = rowsWithDOBOnlyForSection_(previousRows, previousData.headerMap);


    const demographicChangedColumns = getChangedColumnsForSectionRows_(
      currentDobRows,
      previousDobRows,
      currentData.headers,
      previousData.headers,
      RAW_DEMO_P_DEMOGRAPHIC_FIELDS
    );


    if (demographicChangedColumns.size > 0) {
      demographicPMRs.add(pmr);
      demographicChangedColumnsByPMR.set(pmr, demographicChangedColumns);
    }


    const caseloadChangedColumns = getChangedColumnsForSectionRows_(
      currentDobRows,
      previousDobRows,
      currentData.headers,
      previousData.headers,
      RAW_DEMO_P_CASELOAD_FIELDS
    );


    if (caseloadChangedColumns.size > 0) {
      caseloadPMRs.add(pmr);
      caseloadChangedColumnsByPMR.set(pmr, caseloadChangedColumns);
    }


    const contactChangedColumns = getChangedColumnsForSectionRows_(
      currentRows,
      previousRows,
      currentData.headers,
      previousData.headers,
      RAW_DEMO_P_CONTACT_FIELDS
    );


    if (contactChangedColumns.size > 0) {
      contactPMRs.add(pmr);
      contactChangedColumnsByPMR.set(pmr, contactChangedColumns);
    }
  });


  enrollmentPMRs.forEach(pmr => {
    demographicPMRs.delete(pmr);
    caseloadPMRs.delete(pmr);
    contactPMRs.delete(pmr);
  });


  disenrollmentPMRs.forEach(pmr => {
    demographicPMRs.delete(pmr);
    caseloadPMRs.delete(pmr);
    contactPMRs.delete(pmr);
  });


  return {
    previousData: previousData,
    currentData: currentData,
    enrollmentPMRs: enrollmentPMRs,
    disenrollmentPMRs: disenrollmentPMRs,
    demographicPMRs: demographicPMRs,
    caseloadPMRs: caseloadPMRs,
    contactPMRs: contactPMRs,
    demographicChangedColumnsByPMR: demographicChangedColumnsByPMR,
    caseloadChangedColumnsByPMR: caseloadChangedColumnsByPMR,
    contactChangedColumnsByPMR: contactChangedColumnsByPMR
  };
}


function rowsWithDOBOnlyForSection_(items, headerMap) {

function getChangedColumnsForSectionRows_(currentItems, previousItems, currentHeaders, previousHeaders, columnsToCompare) {
  const changed = new Set();


  if (!currentItems || currentItems.length === 0) return changed;


  if (!previousItems || previousItems.length === 0) return changed;


  columnsToCompare.forEach(header => {


    if (currentHeaders.indexOf(header) === -1 || previousHeaders.indexOf(header) === -1) return;


    const currentSig = buildColumnSignatureForSection_(currentItems, currentHeaders, [header]);
    const previousSig = buildColumnSignatureForSection_(previousItems, previousHeaders, [header]);


    if (currentSig !== previousSig) changed.add(header);
  });


  return changed;
}


function buildColumnSignatureForSection_(items, headers, columnsToCompare) {
  const headerMap = buildHeaderIndexMap_(headers);


  return (items || [])
    .map(item => {
      return columnsToCompare.map(header => {
        const idx = headerMap[header];
        if (idx === undefined || idx === -1) return "";
        return normalizeCompareValue_(item.values[idx]);
      }).join("~");
    })
    .sort()
    .join("||");
}

function insertMonthlyDisenrollmentRowsStrict_(reportSheet, currentData, pmrSet, monthParts) {


  if (!monthParts) return;


  const sectionTitle = "Disenrollments";
  const titleRow = findMonthlyChangeSectionTitleRow_(reportSheet, sectionTitle);
  if (titleRow === -1) return;


  const headerRow = titleRow + 2;
  const insertAt = headerRow + 1;
  const reportHeaders = getMonthlyChangeReportHeaders_(currentData.headers);
  const lastCol = reportHeaders.length;


  const effectiveIdx = currentData.headerMap["Disenrollment Effective Date"];
  const pmrIdx = getPMRIndex_(currentData.headerMap);


  if (effectiveIdx === undefined || effectiveIdx === -1 || pmrIdx === -1) return;


  const rowsToInsert = [];
  const usedPMRs = new Set();


  currentData.values.forEach(row => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr || usedPMRs.has(pmr)) return;


    const effectiveDate = row[effectiveIdx];


    if (!isDateDisplayInReportWindow_(
      effectiveDate,
      monthParts.previousMonthFirstDay,
      monthParts.firstDay
    )) {
      return;
    }


    rowsToInsert.push(buildMonthlyChangeReportRow_(
      row,
      currentData.headers,
      reportHeaders,
      new Set()
    ));


    usedPMRs.add(pmr);
  });


  if (rowsToInsert.length === 0) return;


  reportSheet.insertRowsBefore(insertAt, rowsToInsert.length);


  const dataRange = reportSheet.getRange(insertAt, 1, rowsToInsert.length, lastCol);
  dataRange.setValues(rowsToInsert);
  dataRange;


  try {
    reportSheet.getRange(headerRow, 1, 1, lastCol)
      .copyTo(dataRange, SpreadsheetApp.CopyPasteType.PASTE_FORMAT, false);
  } catch (err) {
    Logger.log("Monthly Change disenrollment data format copy skipped: " + err.message);
  }


  dataRange;


  reportSheet;
}

  const disenrollEffectiveIdx = currentData.headerMap["Disenrollment Effective Date"];


  const allowDuplicateRowsForPMR = sectionTitle === "Contact Changes";


  const rowsToInsert = [];
  const pmrForEachInsertedRow = [];


  Array.from(pmrSet).sort().forEach(pmr => {
    const items = currentData.rowsByPMR.get(pmr) || [];
    let insertedForThisPMR = false;


    items.forEach(item => {
      if (!allowDuplicateRowsForPMR && insertedForThisPMR) return;


      if (rowMode === "dobOnly" && (dobIdx === -1 || normalizeCompareValue_(item.values[dobIdx]) === "")) return;


      if (rowMode === "disenrollmentEffectiveRange") {
        if (disenrollEffectiveIdx === undefined || disenrollEffectiveIdx === -1 || !monthParts) return;


        const effectiveDate = item.values[disenrollEffectiveIdx];


        if (!isDateInStrictLocalRangeInclusive_(
          effectiveDate,
          monthParts.previousMonthFirstDay,
          monthParts.firstDay
        )) {
          return;
        }
      }


      const changedColumns = changedColumnsByPMR && changedColumnsByPMR.has(pmr)
        ? changedColumnsByPMR.get(pmr)
        : new Set();


      rowsToInsert.push(buildMonthlyChangeReportRow_(item.values, currentData.headers, reportHeaders, changedColumns));
      pmrForEachInsertedRow.push(pmr);


      if (!allowDuplicateRowsForPMR) insertedForThisPMR = true;
    });
  });


  if (rowsToInsert.length === 0) return;


  reportSheet.insertRowsBefore(insertAt, rowsToInsert.length);


  const dataRange = reportSheet.getRange(insertAt, 1, rowsToInsert.length, lastCol);
  dataRange.setValues(rowsToInsert);
  dataRange;


  const headerFormatRange = reportSheet.getRange(headerRow, 1, 1, lastCol);
  try {
    headerFormatRange.copyTo(dataRange, SpreadsheetApp.CopyPasteType.PASTE_FORMAT, false);
  } catch (err) {
    Logger.log("Monthly Change data format copy skipped: " + err.message);
  }


  dataRange;


  dataRange;


  if (changedColumnsByPMR && changedColumnsByPMR.size > 0) {
    const backgrounds = dataRange.getBackgrounds();


    for (let r = 0; r < backgrounds.length; r++) {
      const pmr = pmrForEachInsertedRow[r];
      const changedColumns = changedColumnsByPMR.get(pmr);


      if (!changedColumns || changedColumns.size === 0) continue;


      reportHeaders.forEach((header, colIndex) => {
        if (changedColumns.has(header)) {
          backgrounds[r][colIndex] = "#f3ffc7";
        }
      });
    }


    dataRange;
  }


  reportSheet;
}


function findMonthlyChangeSectionTitleRow_(sheet, title) {
  const lastRow = sheet.getLastRow();

}

}

function compareRawDemoPForChanges_(previousDemo, currentDemo, monthParts) {
  const previousData = getRawDemoPDataForCompare_(previousDemo);
  const currentData = getRawDemoPDataForCompare_(currentDemo);


  const outputRows = [];
  const changedPMRs = new Map();


  currentData.participantRows.forEach((currentMain, pmr) => {
    const previousMain = previousData.participantRows.get(pmr);


    const capitationIdx = currentData.headerMap["Capitation Date"];
    const currentCapitation = capitationIdx !== undefined ? currentMain.values[capitationIdx] : "";


    const isNewPMR = !previousData.allPMRs.has(pmr);
    const isReportMonthCapitation = isSameMonth_(currentCapitation, monthParts.firstDay);


    if (isNewPMR || isReportMonthCapitation) {
      RAW_DEMO_P_ENROLLMENT_FIELDS.forEach(field => {
        const currentValue = getFieldValueFromRow_(currentMain.values, currentData.headerMap, field);
        if (normalizeCompareValue_(currentValue) !== "") {
          addMCRRow_(
            outputRows,
            changedPMRs,
            CHANGE_SECTION_ENROLLMENTS,
            pmr,
            currentMain,
            null,
            currentData,
            previousData,
            field,
            "",
            currentValue,
            monthParts
          );
        }
      });
    }
  });


  currentData.participantRows.forEach((currentMain, pmr) => {
    const effective = getFieldValueFromRow_(currentMain.values, currentData.headerMap, "Disenrollment Effective Date");
    const disenrollDate = getFieldValueFromRow_(currentMain.values, currentData.headerMap, "Disenrollment Date");
    const deathDate = getFieldValueFromRow_(currentMain.values, currentData.headerMap, "Date of Death");


    const isCurrentEffective = isSameDate_(effective, monthParts.firstDay);
    const isPriorMonthDisenroll = isSameMonth_(disenrollDate, monthParts.previousMonthFirstDay);
    const isPriorMonthDeath = isSameMonth_(deathDate, monthParts.previousMonthFirstDay);


    if (isCurrentEffective || isPriorMonthDisenroll || isPriorMonthDeath) {
      RAW_DEMO_P_DISENROLLMENT_FIELDS.forEach(field => {
        const currentValue = getFieldValueFromRow_(currentMain.values, currentData.headerMap, field);
        if (normalizeCompareValue_(currentValue) !== "") {
          addMCRRow_(
            outputRows,
            changedPMRs,
            CHANGE_SECTION_DISENROLLMENTS,
            pmr,
            currentMain,
            null,
            currentData,
            previousData,
            field,
            "",
            currentValue,
            monthParts
          );
        }
      });
    }
  });


  currentData.participantRows.forEach((currentMain, pmr) => {
    const previousMain = previousData.participantRows.get(pmr);
    if (!previousMain) return;


    RAW_DEMO_P_DEMOGRAPHIC_FIELDS.forEach(field => {
      compareSingleFieldAndAdd_(
        outputRows,
        changedPMRs,
        CHANGE_SECTION_DEMOGRAPHIC,
        pmr,
        previousMain,
        currentMain,
        previousData,
        currentData,
        field,
        monthParts,
        true
      );
    });


    RAW_DEMO_P_CASELOAD_FIELDS.forEach(field => {
      compareSingleFieldAndAdd_(
        outputRows,
        changedPMRs,
        CHANGE_SECTION_CASELOAD,
        pmr,
        previousMain,
        currentMain,
        previousData,
        currentData,
        field,
        monthParts,
        true
      );
    });
  });


  const allCurrentPMRs = Array.from(currentData.allPMRs);
  allCurrentPMRs.forEach(pmr => {
    if (!previousData.allPMRs.has(pmr)) return;


    const previousRows = previousData.rowsByPMR.get(pmr) || [];
    const currentRows = currentData.rowsByPMR.get(pmr) || [];


    const maxLength = Math.max(previousRows.length, currentRows.length);


    for (let i = 0; i < maxLength; i++) {
      const previousItem = previousRows[i] || null;
      const currentItem = currentRows[i] || null;


      RAW_DEMO_P_CONTACT_FIELDS.forEach(field => {
        const previousValue = previousItem ? getFieldValueFromRow_(previousItem.values, previousData.headerMap, field) : "";
        const currentValue = currentItem ? getFieldValueFromRow_(currentItem.values, currentData.headerMap, field) : "";


        if (!valuesAreEqual_(previousValue, currentValue)) {
          addMCRRow_(
            outputRows,
            changedPMRs,
            CHANGE_SECTION_CONTACT,
            pmr,
            currentItem,
            previousItem,
            currentData,
            previousData,
            field,
            previousValue,
            currentValue,
            monthParts
          );
        }
      });
    }
  });


  return {
    rows: outputRows,
    changedPMRs: changedPMRs
  };
}


function getRawDemoPDataForCompare_(sheet) {
  const data = getDataValues_(sheet, HEADER_ROW, DATA_START_ROW);


  const headers = data.headers;
  const headerMap = data.headerMap;
  const pmrIdx = getPMRIndex_(headerMap);
  const dobIdx = getDOBIndex_(headerMap);


  if (pmrIdx === -1) {
    throw new Error(`${sheet.getName()} is missing Participant PMR# / PMR # column.`);
  }


  const values = data.values;


  const rowsByPMR = new Map();
  const participantRows = new Map();
  const allPMRs = new Set();


  values.forEach((row, idx) => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr) return;


    allPMRs.add(pmr);


    const item = {
      rowNumber: DATA_START_ROW + idx,
      values: row
    };


    if (!rowsByPMR.has(pmr)) rowsByPMR.set(pmr, []);
    rowsByPMR.get(pmr).push(item);


    if (!participantRows.has(pmr)) {
      if (dobIdx === -1 || normalizeCompareValue_(row[dobIdx]) !== "") {
        participantRows.set(pmr, item);
      }
    }
  });


  rowsByPMR.forEach((items, pmr) => {
    if (!participantRows.has(pmr) && items.length > 0) {
      participantRows.set(pmr, items[0]);
    }
  });


  return {
    sheet: sheet,
    sheetName: sheet.getName(),
    headers: headers,
    headerMap: headerMap,
    values: values,
    rowsByPMR: rowsByPMR,
    participantRows: participantRows,
    allPMRs: allPMRs
  };
}




function compareSingleFieldAndAdd_(
  outputRows,
  changedPMRs,
  section,
  pmr,
  previousItem,
  currentItem,
  previousData,
  currentData,
  field,
  monthParts,
  skipIfCurrentDOBBlank
) {
  if (!previousItem || !currentItem) return;


  if (skipIfCurrentDOBBlank) {
    const currentDOBIdx = getDOBIndex_(currentData.headerMap);
    if (currentDOBIdx !== -1 && normalizeCompareValue_(currentItem.values[currentDOBIdx]) === "") return;
  }


  if (field !== "Participant PMR#" && currentData.headerMap[field] === undefined && previousData.headerMap[field] === undefined) {
    return;
  }


  const previousValue = getFieldValueFromRow_(previousItem.values, previousData.headerMap, field);
  const currentValue = getFieldValueFromRow_(currentItem.values, currentData.headerMap, field);


  if (!valuesAreEqual_(previousValue, currentValue)) {
    addMCRRow_(
      outputRows,
      changedPMRs,
      section,
      pmr,
      currentItem,
      previousItem,
      currentData,
      previousData,
      field,
      previousValue,
      currentValue,
      monthParts
    );
  }
}


function addMCRRow_(
  outputRows,
  changedPMRs,
  section,
  pmr,
  currentItem,
  previousItem,
  currentData,
  previousData,
  field,
  previousValue,
  currentValue,
  monthParts
) {
  const mainItem = currentItem || previousItem;
  const currentValues = currentItem ? currentItem.values : (mainItem ? mainItem.values : []);
  const currentMap = currentData.headerMap;


  const lastName = getFieldValueFromRow_(currentValues, currentMap, "Last Name");
  const firstName = getFieldValueFromRow_(currentValues, currentMap, "First Name");
  const participantName = buildParticipantName_(currentValues, currentMap);


  outputRows.push([
    section,
    pmr,
    participantName,
    lastName,
    firstName,
    field,
    displayValueForReport_(previousValue),
    displayValueForReport_(currentValue),
    previousItem ? previousItem.rowNumber : "",
    currentItem ? currentItem.rowNumber : "",
    monthParts.firstDay
  ]);


  if (!changedPMRs.has(pmr)) changedPMRs.set(pmr, new Set());
  changedPMRs.get(pmr).add(section);
}


function getFieldValueFromRow_(row, headerMap, field) {
  const idx = headerMap[field];
  if (idx === undefined || idx < 0) return "";
  return row[idx];
}


function buildParticipantName_(row, headerMap) {
  const participantName = getFieldValueFromRow_(row, headerMap, "Participant Name");
  if (normalizeCompareValue_(participantName) !== "") return participantName;


  const last = getFieldValueFromRow_(row, headerMap, "Last Name");
  const first = getFieldValueFromRow_(row, headerMap, "First Name");


  const lastText = String(last || "").trim();
  const firstText = String(first || "").trim();


  if (lastText && firstText) return `${lastText}, ${firstText}`;
  return lastText || firstText || "";
}


function displayValueForReport_(value) {
  if (value instanceof Date && !isNaN(value.getTime())) {
    return formatDateDisplay_(value);
  }
  return value === null || value === undefined ? "" : String(value);
}

function sortMasterListByParticipantNameAndPMR_(sheet) {
  sheet = sheet || SpreadsheetApp.getActiveSheet();


  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();
  if (lastRow < DATA_START_ROW || lastCol < 1) return;


  clearAllRowGroupsIfPossible_(sheet);
  sheet.showRows(DATA_START_ROW, lastRow - DATA_START_ROW + 1);


  const headers = getHeaders_(sheet, HEADER_ROW);
  const headerMap = buildHeaderIndexMap_(headers);
  const pmrIdx = getPMRIndex_(headerMap);
  const lastNameIdx = headerMap["Last Name"];
  const firstNameIdx = headerMap["First Name"];


  if (pmrIdx === -1 || lastNameIdx === undefined || firstNameIdx === undefined) {
    notify_("Sort skipped: missing Last Name, First Name, or Participant PMR#.");
    return;
  }


  const range = sheet.getRange(DATA_START_ROW, 1, lastRow - DATA_START_ROW + 1, lastCol);
  const values = range.getValues();
  const blocksByPMR = new Map();


  values.forEach((row, idx) => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr) return;


    if (!blocksByPMR.has(pmr)) {
      blocksByPMR.set(pmr, {
        pmr: pmr,
        lastName: row[lastNameIdx],
        firstName: row[firstNameIdx],
        rows: []
      });
    }


    const block = blocksByPMR.get(pmr);
    if (!String(block.lastName || "").trim() && String(row[lastNameIdx] || "").trim()) block.lastName = row[lastNameIdx];
    if (!String(block.firstName || "").trim() && String(row[firstNameIdx] || "").trim()) block.firstName = row[firstNameIdx];


    block.rows.push({ row: row, originalIndex: idx });
  });


  const blocks = Array.from(blocksByPMR.values());


  blocks.sort((a, b) => {
    const aLast = normalizeKeyPart_(a.lastName);
    const bLast = normalizeKeyPart_(b.lastName);
    if (aLast !== bLast) return aLast < bLast ? -1 : 1;


    const aFirst = normalizeKeyPart_(a.firstName);
    const bFirst = normalizeKeyPart_(b.firstName);
    if (aFirst !== bFirst) return aFirst < bFirst ? -1 : 1;


    return normalizePMR_(a.pmr) < normalizePMR_(b.pmr) ? -1 : 1;
  });


  const sortedRows = [];


  blocks.forEach(block => {
    block.rows.sort((a, b) => {
      const aScore = getPrimaryRowScore_(a.row, headerMap);
      const bScore = getPrimaryRowScore_(b.row, headerMap);
      if (aScore !== bScore) return bScore - aScore;
      return a.originalIndex - b.originalIndex;
    });


    block.rows.forEach(item => sortedRows.push(item.row));
  });


  if (sortedRows.length > 0) {
    range.setValues(sortedRows);
  }
}


function getPrimaryRowScore_(row, headerMap) {
  let score = 0;
  const checks = [
    ["Date of Birth", 1000],
    ["Enrollment Date", 250],
    ["Capitation Date", 200],
    ["Address Line 1", 100],
    ["Phone Number", 50],
    ["Primary Language", 25],
    ["Participant Name", 20],
    ["PMR #", 10]
  ];


  checks.forEach(item => {
    const idx = headerMap[item[0]];
    if (idx !== undefined && normalizeCompareValue_(row[idx]) !== "") score += item[1];
  });


  return score;
}


function hideNonPrimaryPMRRows_(sheet) {
  sheet = sheet || SpreadsheetApp.getActiveSheet();


  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();
  if (lastRow < DATA_START_ROW || lastCol < 1) return;


  sheet.showRows(DATA_START_ROW, lastRow - DATA_START_ROW + 1);


  const headers = getHeaders_(sheet, HEADER_ROW);
  const headerMap = buildHeaderIndexMap_(headers);
  const pmrIdx = getPMRIndex_(headerMap);
  if (pmrIdx === -1) return;


  const values = sheet.getRange(DATA_START_ROW, 1, lastRow - DATA_START_ROW + 1, lastCol).getValues();
  const seenPMRs = new Set();
  const rowsToHide = [];


  values.forEach((row, offset) => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr) return;


    if (!seenPMRs.has(pmr)) {
      seenPMRs.add(pmr);
      return;
    }


    rowsToHide.push(DATA_START_ROW + offset);
  });


  hideRowNumberBatches_(sheet, rowsToHide);
}


function hideRowNumberBatches_(sheet, rowNumbers) {
  if (!rowNumbers || rowNumbers.length === 0) return;


  const sorted = Array.from(new Set(rowNumbers)).sort((a, b) => a - b);
  let start = sorted[0];
  let previous = sorted[0];


  for (let i = 1; i < sorted.length; i++) {
    const current = sorted[i];
    if (current === previous + 1) {
      previous = current;
      continue;
    }


    sheet.hideRows(start, previous - start + 1);
    start = current;
    previous = current;
  }


  sheet.hideRows(start, previous - start + 1);
}


function clearAllRowGroupsIfPossible_(sheet) {
  const maxRows = sheet.getMaxRows();
  if (maxRows < DATA_START_ROW) return;


  for (let i = 0; i < 8; i++) {
    try {
      sheet.shiftRowGroupDepth(DATA_START_ROW, maxRows - DATA_START_ROW + 1, -1);
    } catch (err) {
      break;
    }
  }
}

}

function syncRowsFromSourceMapData_(data, sourceMap, config, pmrFilter) {
  if (!data || !data.values || !data.values.length) return;


  const headerMap = data.headerMap;
  const fields = normalizeSyncFieldPairs_(config.fields);


  data.values.forEach(row => {
    if (!shouldProcessRowByPMR_(row, headerMap, pmrFilter)) return;


    const key = config.masterKeyHeaders
      .map(header => normalizeKeyPart_(row[headerMap[header]]))
      .join("|||");


    if (key.replace(/\|/g, "") === "") return;
    if (!sourceMap.has(key)) return;


    const source = sourceMap.get(key);


    fields.forEach(pair => {
      const destHeader = pair[0];
      const sourceHeader = pair[1];
      const destIdx = headerMap[destHeader];


      if (destIdx === undefined) return;
      if (!Object.prototype.hasOwnProperty.call(source, sourceHeader)) return;


      row[destIdx] = source[sourceHeader];
    });
  });
}


function updateBannerColumnData_(data, pmrFilter) {
  const headerMap = data.headerMap;
  const bIdx = headerMap["Banner Summary"];


  if (bIdx === undefined) return;


  const bannerHeaders = [
    "Safety - 2 Person",
    "Wanderer",
    "Interpreter Needed",
    "Fall Risk",
    "DPOA or Guardian Active",
    "Palliative Care"
  ];


  data.values.forEach(row => {
    if (!shouldProcessRowByPMR_(row, headerMap, pmrFilter)) return;


    const parts = [];


    bannerHeaders.forEach(header => {
      const idx = headerMap[header];
      if (idx === undefined) return;


      const value = String(row[idx] || "").trim();
      if (value !== "") parts.push(header);
    });


    row[bIdx] = parts.join(" | ");
  });
}


function combineAddressesData_(data, pmrFilter) {
  const headerMap = data.headerMap;
  const a1Idx = headerMap["Address Line 1"];
  const a2Idx = headerMap["Address Line 2"];
  const targetIdx = headerMap["Address 1 - Street"];


  if (a1Idx === undefined || targetIdx === undefined) return;


  data.values.forEach(row => {
    if (!shouldProcessRowByPMR_(row, headerMap, pmrFilter)) return;


    const p1 = String(row[a1Idx] || "").trim();
    const p2 = a2Idx !== undefined ? String(row[a2Idx] || "").trim() : "";


    row[targetIdx] = p1 && p2 ? `${p1} ${p2}` : (p1 || p2);
  });
}


function handleLanguageData_(data, pmrFilter) {
  const headerMap = data.headerMap;
  const labelIdx = headerMap["Custom Field 1 - Label"];
  const valueIdx = headerMap["Custom Field 1 - Value"];
  const langIdx = headerMap["Primary Language"];


  if (labelIdx === undefined || valueIdx === undefined || langIdx === undefined) return;


  data.values.forEach(row => {
    if (!shouldProcessRowByPMR_(row, headerMap, pmrFilter)) return;


    const lang = String(row[langIdx] || "").trim();
    const existingValue = String(row[valueIdx] || "").trim();


    if (lang && lang.toLowerCase() !== "english") {
      row[labelIdx] = "Language";
      row[valueIdx] = lang;
    } else if (existingValue) {
      row[labelIdx] = "Language";
    } else {
      row[labelIdx] = "";
      row[valueIdx] = "";
    }
  });
}


function splitPhoneNumbersData_(data, pmrFilter) {
  const headerMap = data.headerMap;
  const phoneIdx = headerMap["Phone Number"];


  if (phoneIdx === undefined) return;


  const labelHeaders = ["Phone 1 - Label", "Phone 2 - Label", "Phone 3 - Label", "Phone 4 - Label"];
  const valueHeaders = ["Phone 1 - Value", "Phone 2 - Value", "Phone 3 - Value", "Phone 4 - Value"];
  const labels = ["Home", "Mobile", "Other", "Other"];


  data.values.forEach(row => {
    if (!shouldProcessRowByPMR_(row, headerMap, pmrFilter)) return;


    const phoneText = String(row[phoneIdx] || "").trim();


    for (let i = 0; i < 4; i++) {
      const labelIdx = headerMap[labelHeaders[i]];
      const valueIdx = headerMap[valueHeaders[i]];


      if (labelIdx !== undefined) row[labelIdx] = "";
      if (valueIdx !== undefined) row[valueIdx] = "";
    }


    if (!phoneText) return;


    const parts = phoneText.split("_").map(part => part.trim()).filter(Boolean);


    parts.slice(0, 4).forEach((phone, idx) => {
      const labelIdx = headerMap[labelHeaders[idx]];
      const valueIdx = headerMap[valueHeaders[idx]];


      if (labelIdx !== undefined) row[labelIdx] = labels[idx];
      if (valueIdx !== undefined) row[valueIdx] = phone;
    });
  });
}

function combineNotesSummaryData_(data, pmrFilter) {
  if (!data || !data.values || !data.values.length) return;


  const headerMap = data.headerMap;
  const notesIdx = headerMap["Notes"];
  if (notesIdx === undefined) return;


  const sourceHeaders = [
    "Banner Summary",
    "Contact - Summary",
    "Additional Important Information"
  ];


  data.values.forEach(row => {
    if (!shouldProcessRowByPMR_(row, headerMap, pmrFilter)) return;


    const parts = [];


    sourceHeaders.forEach(header => {
      const idx = headerMap[header];
      if (idx === undefined) return;


      const value = String(row[idx] || "").trim();
      if (value) parts.push(value);
    });


    row[notesIdx] = parts.join("\n\n");
  });
}
