function ensureStandardTitleRows_(sheet) {
  sheet.getRange(1, 1, Math.max(sheet.getLastRow(), HEADER_ROW), Math.max(sheet.getLastColumn(), 1)).breakApart();


  if (hasStandardTitleRows_(sheet)) {
    return;
  }


  const headerRow = detectCurrentHeaderRow_(sheet);


  const rowsNeeded = Math.max(0, HEADER_ROW - headerRow);


  if (rowsNeeded > 0) {
    sheet.insertRowsBefore(1, rowsNeeded);
  }


  if (HEADER_ROW > 1) {
    sheet.getRange(1, 1, HEADER_ROW - 1, Math.max(sheet.getLastColumn(), 1))
      .clear({ contentsOnly: true });
  }
}

  return a2 === "date" && c2 === "to";

function moveChangeLogDataOutOfTitleRows_(sheet) {
  const lastCol = Math.max(sheet.getLastColumn(), MASTER_LIST_CHANGE_LOG_HEADERS.length, 1);
  const rowsToMove = [];


  for (let row = 1; row <= 3; row++) {
    const values = sheet.getRange(row, 1, 1, lastCol).getValues()[0];
    const display = values.map(v => String(v || "").trim());
    const hasData = display.some(v => v !== "");


    if (!hasData) continue;


    const firstCell = display[0].toLowerCase();


    const isStandardTitleRow =
      row === 1 && firstCell === "master list change log" ||
      row === 2 && firstCell === "creation date" ||
      row === 3 && display.every(v => v === "");


    if (!isStandardTitleRow) {
      rowsToMove.push(values);
      sheet.getRange(row, 1, 1, lastCol).clearContent();
    }
  }


  if (rowsToMove.length > 0) {
    const appendRow = Math.max(sheet.getLastRow() + 1, DATA_START_ROW);
    sheet.getRange(appendRow, 1, rowsToMove.length, lastCol).setValues(rowsToMove);
  }
}

function enforceDemoPStrictDashboardSchema_(sheet) {
  /*
   * v1.4.24:
   * This function no longer inserts or deletes columns. Dashboard Section C is
   * the column authority; Template Header Audit reports drift.
   */
  if (!sheet) return;


  const dashboard = loadDashboardConfig_();
  const expectedHeaders = getHeadersForSheetType_(dashboard, SHEET_TYPE.DEMO_P)
    .map(normalizeHeader_);
  const expectedSet = {};
  expectedHeaders.forEach(function(header) {
    if (header) expectedSet[header] = true;
  });


  const currentHeaders = getHeaders_(sheet, HEADER_ROW).map(normalizeHeader_);
  const currentSet = {};
  currentHeaders.forEach(function(header) {
    if (header) currentSet[header] = true;
  });


  const missing = expectedHeaders.filter(function(header) {
    return header && !currentSet[header];
  });
  const unexpected = currentHeaders.filter(function(header) {
    return header && !expectedSet[header];
  });


  if (missing.length || unexpected.length) {
    Logger.log(
      "Demo P header drift detected. Missing: " +
      missing.join(", ") +
      " Unexpected: " +
      unexpected.join(", ")
    );
  }


  clearSheetRuntimeCachesForSheet_(sheet);
}

function createOutputSheetFromDashboardTemplate_(sheetType, outputName, outputRows, firstDay, endDate) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const dashboard = loadDashboardConfig_();
  const normalizedSheetType = normalizeDashboardSheetTypeKey_(sheetType);
  const sheetDef = getSheetDefinitionByType_(dashboard, normalizedSheetType);
  const headers = getHeadersForSheetType_(dashboard, normalizedSheetType);
  const template = ss.getSheetByName(sheetDef.templateName);

  if (!template) {
    throw new Error("Template not found: " + sheetDef.templateName + ". Run Create / Refresh All Templates first.");
  }

  deleteSheetIfExists_(ss, outputName, sheetDef.templateName);
  const sheet = template.copyTo(ss);
  sheet.showSheet();
  setUniqueSheetName_(sheet, outputName);
  ss.setActiveSheet(sheet);

  const width = Math.max(headers.length, 1);
  const rows = outputRows || [];
  const requiredRows = rows.length ? DATA_START_ROW + rows.length - 1 : Math.max(sheet.getLastRow(), DATA_START_ROW);

  ensureOutputSheetHasFormattedRows_(sheet, requiredRows, width);

  try {
    sheet.getRange("A1").setValue(sheetDef.reportTitle || normalizedSheetType || outputName);
    if (firstDay) sheet.getRange("B2").setValue(firstDay);
    if (endDate) sheet.getRange("D2").setValue(endDate);
    if (headers.length) sheet.getRange(HEADER_ROW, 1, 1, width).setValues([headers]);
  } catch (err) {
    Logger.log("Dashboard-template output metadata skipped for " + outputName + ": " + err.message);
  }

  if (rows.length) {
    sheet.getRange(DATA_START_ROW, 1, rows.length, width).setValues(rows.map(function(row) {
      const out = (row || []).slice(0, width);
      while (out.length < width) out.push("");
      return out;
    }));
  }

  clearSheetRuntimeCachesForSheet_(sheet);
  return sheet;
}

function createRawDataOutputSheetFromTemplateFast_(sheetType, outputName, outputRows, firstDay, endDate, sourceName) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const dashboard = loadDashboardConfig_();
  const normalizedSheetType = normalizeDashboardSheetTypeKey_(sheetType);
  const sheetDef = getSheetDefinitionByType_(dashboard, normalizedSheetType);
  const headers = getHeadersForSheetType_(dashboard, normalizedSheetType);
  const template = ss.getSheetByName(sheetDef.templateName);

  if (!template) {
    throw new Error("Template not found: " + sheetDef.templateName + ". Run Create / Refresh All Templates first.");
  }

  deleteSheetIfExists_(ss, outputName, sourceName, sheetDef.templateName);
  const sheet = ss.insertSheet(outputName);
  sheet.showSheet();
  ss.setActiveSheet(sheet);

  const width = Math.max(headers.length, 1);
  const rows = outputRows || [];
  const requiredRows = rows.length ? DATA_START_ROW + rows.length - 1 : DATA_START_ROW;

  resizeSheetColumns_(sheet, width);
  if (sheet.getMaxRows() < requiredRows) {
    sheet.insertRowsAfter(sheet.getMaxRows(), requiredRows - sheet.getMaxRows());
  }

  const templateTopRows = Math.max(DATA_START_ROW - 1, HEADER_ROW);
  try {
    template.getRange(1, 1, templateTopRows, width)
      .copyTo(sheet.getRange(1, 1, templateTopRows, width), { contentsOnly: false });
  } catch (err) {
    Logger.log("Raw Data template title/header copy skipped for " + outputName + ": " + err.message);
  }

  const dataRows = Math.max(requiredRows - DATA_START_ROW + 1, 1);
  try {
    template.getRange(DATA_START_ROW, 1, 1, width)
      .copyTo(sheet.getRange(DATA_START_ROW, 1, dataRows, width), { formatOnly: true });
  } catch (err) {
    Logger.log("Raw Data template data-row formatting copy skipped for " + outputName + ": " + err.message);
  }

  try {
    for (let col = 1; col <= width; col++) {
      sheet.setColumnWidth(col, template.getColumnWidth(col));
    }
    sheet.setFrozenRows(template.getFrozenRows());
    sheet.setFrozenColumns(template.getFrozenColumns());
    if (template.getTabColor()) sheet.setTabColor(template.getTabColor());
  } catch (err) {
    Logger.log("Raw Data template sheet settings copy skipped for " + outputName + ": " + err.message);
  }

  try {
    sheet.getRange("A1").setValue(sheetDef.reportTitle || normalizedSheetType || outputName);
    if (firstDay) sheet.getRange("B2").setValue(firstDay);
    if (endDate) sheet.getRange("D2").setValue(endDate);
    if (headers.length) sheet.getRange(HEADER_ROW, 1, 1, width).setValues([headers]);
  } catch (err) {
    Logger.log("Raw Data output metadata skipped for " + outputName + ": " + err.message);
  }

  if (rows.length) {
    sheet.getRange(DATA_START_ROW, 1, rows.length, width).setValues(rows.map(function(row) {
      const out = (row || []).slice(0, width);
      while (out.length < width) out.push("");
      return out;
    }));
  }

  try {
    const existingFilter = sheet.getFilter && sheet.getFilter();
    if (existingFilter) existingFilter.remove();
    sheet.getRange(HEADER_ROW, 1, Math.max(sheet.getMaxRows() - HEADER_ROW + 1, 1), width).createFilter();
  } catch (err) {
    Logger.log("Raw Data output filter creation skipped for " + outputName + ": " + err.message);
  }

  clearSheetRuntimeCachesForSheet_(sheet);
  return sheet;
}

function buildDashboardOutputSheetName_(sheetDef, monthParts) {
  const pattern = String(sheetDef.outputNamingPattern || sheetDef.reportTitle || sheetDef.sheetType || "").trim();
  const monthText = Utilities.formatDate(monthParts.firstDay, Session.getScriptTimeZone(), "MM.yy");
  const monthTextNoLeading = Utilities.formatDate(monthParts.firstDay, Session.getScriptTimeZone(), "M.yy");
  const fullStart = formatDateForSheetName_(monthParts.firstDay);
  const fullEnd = formatDateForSheetName_(monthParts.lastDay);


  let name = pattern || String(sheetDef.reportTitle || sheetDef.sheetType || "Report").trim();


  name = name
    .replace(/mm\.yy/g, monthText)
    .replace(/MM\.yy/g, monthText)
    .replace(/m\.yy/g, monthTextNoLeading)
    .replace(/M\.yy/g, monthTextNoLeading)
    .replace(/start/gi, fullStart)
    .replace(/end/gi, fullEnd);


  if (name === pattern && name.toLowerCase().indexOf(monthText.toLowerCase()) === -1 && sheetDef.usePromptDate) {
    name = name + " " + monthText;
  }


  return safeSheetName_(name);
}

function fillMasterListMetadataColumns_(sheet, monthParts, sourceSheetName, statusText) {
  const lastRow = sheet.getLastRow();
  if (lastRow < DATA_START_ROW) return;


  const headers = getHeaders_(sheet, HEADER_ROW);
  const headerMap = buildHeaderIndexMap_(headers);


  const required = ["Master List Update Status", "Master List Update Month", "Master List Source Sheet"];
  const missing = required.filter(function(header) { return headerMap[header] === undefined; });
  if (missing.length) {
    Logger.log("Master List metadata write skipped for missing dashboard-defined columns: " + missing.join(", "));
  }


  const rowCount = lastRow - DATA_START_ROW + 1;
  const metadata = {
    "Master List Update Status": statusText || "Created",
    "Master List Update Month": monthParts.firstDay,
    "Master List Source Sheet": sourceSheetName || ""
  };


  const indexes = required
    .map(function(header) { return headerMap[header]; })
    .filter(function(idx) { return idx !== undefined; })
    .sort(function(a, b) { return a - b; });


  if (!indexes.length) return;


  if (indexes[indexes.length - 1] - indexes[0] + 1 === indexes.length) {
    const startIdx = indexes[0];
    const width = indexes.length;
    const rowTemplate = [];
    for (let c = startIdx; c <= indexes[indexes.length - 1]; c++) {
      const header = headers[c];
      rowTemplate.push(Object.prototype.hasOwnProperty.call(metadata, header) ? metadata[header] : "");
    }
    const values = Array.from({ length: rowCount }, function() { return rowTemplate.slice(); });
    sheet.getRange(DATA_START_ROW, startIdx + 1, rowCount, width).setValues(values);
    return;
  }


  indexes.forEach(function(idx) {
    const header = headers[idx];
    const value = Object.prototype.hasOwnProperty.call(metadata, header) ? metadata[header] : "";
    const values = Array.from({ length: rowCount }, function() { return [value]; });
    sheet.getRange(DATA_START_ROW, idx + 1, rowCount, 1).setValues(values);
  });
}

    if (!setting) rows.push(["Row " + (index + 1), "FAIL", "Blank Setting", "Format Dashboard Section A contains a blank setting name."]);
    if (setting && seen[setting]) rows.push([setting, "WARNING", "Duplicate Setting", "Format Dashboard Section A contains this setting more than once."]);
    if (setting) seen[setting] = true;
    if (setting && !value) rows.push([setting, "FAIL", "Blank Value", "Format Dashboard Section A setting is blank."]);
  });

  collectFormatDashboardTitleRowsVerificationRows_(dashboardSheet).slice(1).forEach(function(row) {
    rows.push(row);
  });

  return appendDashboardVerificationPassIfNoIssues_(rows, "Global Inputs / Title Rows", "Format Dashboard Section A settings and Section B title rows are present and populated.");
}

function collectFormatDashboardTitleRowsVerificationRows_(dashboardSheet) {
  const rows = [["Title Row", "Status", "Issue", "Quality Notes"]];
  let titleRows = [];
  try {
    titleRows = readDashboardSectionRows_(dashboardSheet, RFF_SECTION_TITLE_ROWS);
  } catch (err) {
    rows.push([RFF_SECTION_TITLE_ROWS, "FAIL", "Missing Section", err.message]);
    return rows;
  }

  if (!titleRows.length) {
    rows.push([RFF_SECTION_TITLE_ROWS, "FAIL", "Blank Section", "Format Dashboard Section B has no title row records."]);
    return rows;
  }

  const requiredRows = { 1: false, 2: false, 3: false, 4: false };
  const validFillLevels = { "LEVEL 1": true, "LEVEL 2": true, "LEVEL 3": true };
  titleRows.forEach(function(row, index) {
    const sheetType = String(row[0] || "").trim();
    const rowNumber = numberOrDefault_(row[1], 0);
    const label = (sheetType || "Row " + (index + 1)) + " title row " + (rowNumber || "?");
    if (!sheetType) rows.push([label, "FAIL", "Blank Sheet Type", "Format Dashboard Section B title row has no sheet type."]);
    if (!rowNumber) rows.push([label, "FAIL", "Invalid Row", "Format Dashboard Section B title row number must be numeric."]);
    if (sheetType.toUpperCase() === "GLOBAL" && requiredRows[rowNumber] !== undefined) requiredRows[rowNumber] = true;
    if (isBlankCell_(row[5])) rows.push([label, "FAIL", "Blank Target Cell", "Format Dashboard Section B title row must define a target cell."]);
    if (isBlankCell_(row[6]) || numberOrDefault_(row[6], 0) <= 0) rows.push([label, "FAIL", "Invalid Height", "Format Dashboard Section B title row height must be greater than zero."]);
    if (isBlankCell_(row[9]) || !validFillLevels[String(row[9] || "").trim().toUpperCase()]) rows.push([label, "FAIL", "Invalid Fill Level", "Fill Level must be Level 1, Level 2, or Level 3."]);
    if (rowNumber === 4 && String(row[11] || "").trim().toUpperCase() !== "WRAP") rows.push([label, "WARNING", "Header Wrap", "Header row should be governed as Wrap."]);
  });

  Object.keys(requiredRows).forEach(function(rowNumber) {
    if (!requiredRows[rowNumber]) rows.push(["GLOBAL row " + rowNumber, "FAIL", "Missing Required Title Row", "Format Dashboard Section B must include GLOBAL rows 1, 2, 3, and 4."]);
  });

  return appendDashboardVerificationPassIfNoIssues_(rows, "Title Rows", "Format Dashboard Section B title row records are present and valid.");
}

function collectFormatDashboardSheetDefinitionVerificationRows_(dashboardSheet) {
  const rows = [["Sheet", "Status", "Issue", "Quality Notes"]];
  let sectionRows = [];
  try {
    sectionRows = readDashboardSectionRows_(dashboardSheet, RFF_SECTION_SHEETS);

}

function collectFormatDashboardSheetHeaderVerificationRows_(dashboardSheet) {
  const rows = [["Header", "Status", "Issue", "Quality Notes"]];
  let sectionRows = [];
  try {
    sectionRows = readDashboardSectionRows_(dashboardSheet, RFF_SECTION_HEADERS);
  } catch (err) {
    rows.push([RFF_SECTION_HEADERS, "FAIL", "Missing Section", err.message]);
    return rows;
  }

  if (!sectionRows.length) {
    rows.push([RFF_SECTION_HEADERS, "FAIL", "Blank Section", "Format Dashboard Section F has no header definition rows."]);
    return rows;
  }

  const seen = {};
  const counts = {};
  sectionRows.forEach(function(row, index) {
    const sheetType = String(row[0] || "").trim();
    const header = String(row[2] || row[1] || "").trim();
    const label = sheetType && header ? sheetType + " - " + header : "Row " + (index + 1);
    if (!sheetType) rows.push([label, "FAIL", "Blank Sheet Type", "Format Dashboard Section F row has no sheet type."]);
    if (!header) rows.push([label, "FAIL", "Blank Header", "Format Dashboard Section F row has no header name."]);
    if (!sheetType || !header) return;
    counts[sheetType] = (counts[sheetType] || 0) + 1;
    const key = sheetType + "||" + header;
    if (seen[key]) rows.push([label, "WARNING", "Duplicate Header", "Header is duplicated within the same Format Dashboard sheet type."]);
    seen[key] = true;
  });

  Object.keys(counts).forEach(function(sheetType) {
    if (counts[sheetType] < 1) rows.push([sheetType, "FAIL", "No Headers", "No headers are defined for this sheet type."]);
  });

  return appendDashboardVerificationPassIfNoIssues_(rows, "Sheet Headers", "No missing, blank, duplicate, or invalid Format Dashboard Section F headers found.");
}

function writeDashboardTitle_(sheet, row) {
  const titleValues = [["Report Formatter Framework", "- v" + RFF_VERSION + " -", "Format Dashboard"]];
  sheet.getRange(row, 1, 1, 9).breakApart();
  sheet.getRange(row, 1, 1, 3).setValues(titleValues);
  sheet.getRange(row, 1, 1, 9)
    .setFontFamily("Arial")
    .setFontSize(16)
    .setFontWeight("bold")
    .setBackground("#D9EAF7")
    .setVerticalAlignment("middle");
  sheet.setRowHeight(row, 30);
  return row + 2;
}

function writeDashboardSection_(sheet, startRow, sectionName, headers, rows) {
  sheet.getRange(startRow, 1).setValue(sectionName);
  sheet.getRange(startRow, 1, 1, headers.length)
    .merge()
    .setFontFamily("Arial")
    .setFontSize(12)
    .setFontWeight("bold")
    .setBackground("#B7DEE8")
    .setVerticalAlignment("middle");


  const headerRow = startRow + 1;
  sheet.getRange(headerRow, 1, 1, headers.length)
    .setValues([headers])
    .setFontFamily("Arial")
    .setFontSize(10)
    .setFontWeight("bold")
    .setBackground("#D9EAF7")
    .setWrapStrategy(SpreadsheetApp.WrapStrategy.WRAP)
    .setVerticalAlignment("top");


  if (rows.length > 0) {
    sheet.getRange(headerRow + 1, 1, rows.length, headers.length)
      .setValues(rows)
      .setFontFamily("Arial")
      .setFontSize(10)
      .setVerticalAlignment("middle")
      .setWrapStrategy(SpreadsheetApp.WrapStrategy.CLIP);
  }


  return headerRow + rows.length + 3;
}

function styleDashboard_(sheet) {
  sheet.setFrozenRows(1);


  const lastRow = Math.max(sheet.getLastRow(), 1);
  const lastCol = Math.max(sheet.getLastColumn(), 1);


  sheet.getRange(1, 1, lastRow, lastCol)
    .setFontFamily("Arial")
    .setFontColor("#000000")
    .setHorizontalAlignment("left");


  // Fixed widths are faster and more stable than auto-resize.
  const widths = [250, 160, 240, 180, 130, 120, 160, 150, 140];
  widths.forEach(function(width, index) {
    try {
      sheet.setColumnWidth(index + 1, width);
    } catch (err) {
      Logger.log("Dashboard column width skipped: " + err.message);
    }
  });


  try {
    const filter = sheet.getFilter();
    if (filter) filter.remove();
  } catch (err) {
    Logger.log("Dashboard filter remove skipped: " + err.message);
  }
}

function appendDashboardSectionRows_(rows, sectionName, headers, dataRows, width) {
  rows.push(normalizeSectionRowForWidth_([sectionName], width));
  rows.push(normalizeSectionRowForWidth_(headers, width));

}

function getResolvedDefaultColumnDefinitionRows_() {
  const defaultWidth = Number(RFF_DEFAULTS.defaultColumnWidth || 105);
  const defaultFontSize = Number(RFF_DEFAULTS.standardFontSize || 10);
  const defaultWrap = String(RFF_DEFAULTS.defaultDataWrap || "CLIP").toUpperCase();
  const defaultHorizontal = String(RFF_DEFAULTS.defaultHorizontalAlignment || "left").toLowerCase();
  const defaultVertical = String(RFF_DEFAULTS.defaultVerticalAlignment || "middle").toLowerCase();
  const defaultNumberFormat = String(RFF_DEFAULTS.defaultNumberFormat || "General");


  return getDefaultColumnDefinitionRows_().map(function(row) {
    const width = Number(row[1]);
    const fontSize = Number(row[2]);
    const dateColumn = row[3] === true;
    const hideColumn = row[4] === true;
    const wrap = String(row[5] || "").toUpperCase();
    const horizontal = String(row[6] || "").toLowerCase();
    const vertical = String(row[7] || "").toLowerCase();
    const numberFormat = String(row[8] || "");


    return [
      row[0],
      width === defaultWidth ? "" : row[1],
      fontSize === defaultFontSize ? "" : row[2],
      dateColumn ? true : "",
      hideColumn ? true : "",
      wrap === defaultWrap ? "" : row[5],
      horizontal === defaultHorizontal ? "" : row[6],
      vertical === defaultVertical ? "" : row[7],
      numberFormat === "" || numberFormat === defaultNumberFormat ? "" : row[8]
    ];
  });
}

function writeDashboardDefaultsFast_(sheet) {
  const width = 14;
  const rows = [];
  rows.push(normalizeSectionRowForWidth_(["Report Formatter Framework", "- v" + RFF_VERSION + " -", "Format Dashboard"], width));
  rows.push(normalizeSectionRowForWidth_([], width));
  appendDashboardSectionRows_(rows, RFF_SECTION_GLOBAL, ["Setting", "Value"], getDefaultGlobalSettingsRows_(), width);
  appendDashboardSectionRows_(rows, RFF_SECTION_TITLE_ROWS, ["Sheet Type", "Row", "Purpose", "Value Source", "Label", "Target Cell", "Height", "Font Size", "Font Weight", "Fill Level", "Alignment", "Wrap", "Notes"], getDefaultTitleRowRows_(), width);
  appendDashboardSectionRows_(rows, RFF_SECTION_SHEETS, ["Sheet Type", "Report Title", "Template Name", "Output Naming Pattern", "Base Color", "Use Prompt Date", "End Date Source", "Template Row Count", "Template Column Count", "Template Row Mode", "Minimum Rows", "Buffer Rows", "Test Rows", "Use Test Rows"], getDefaultSheetDefinitionRowsWithColumnCounts_(), width);
  appendDashboardSectionRows_(rows, RFF_SECTION_COLUMNS, ["Header", "Width", "Header Font Size", "Date Column", "Hide Column", "Data Wrap", "Horizontal Alignment", "Vertical Alignment", "Number Format"], getResolvedDefaultColumnDefinitionRows_(), width);
  appendDashboardSectionRows_(rows, RFF_SECTION_BEHAVIORS, ["Sheet Type", "Uses Title Rows", "Uses Filter", "Uses Alternating Colors", "Uses Subheaders", "Hidden Template"], getDefaultBehaviorRows_(), width);
  appendDashboardSectionRows_(rows, RFF_SECTION_HEADERS, ["Sheet Type", "Column Order", "Header", "Source of Data"], getDefaultSheetHeaderRows_(), width);

  trimSheetToColumnCount_(sheet, width, "Format Dashboard");
  if (sheet.getMaxRows() < rows.length) sheet.insertRowsAfter(sheet.getMaxRows(), rows.length - sheet.getMaxRows());

  const clearRows = Math.max(sheet.getLastRow(), rows.length, 1);
  sheet.getRange(1, 1, clearRows, width).clearContent().breakApart().setBackground("#FFFFFF");
  sheet.getRange(1, 1, rows.length, width).setValues(rows);

  sheet.getRange(1, 1, rows.length, width)
    .setFontFamily("Arial")
    .setFontColor("#000000")
    .setFontSize(10)
    .setHorizontalAlignment("left")
    .setVerticalAlignment("middle")
    .setWrapStrategy(SpreadsheetApp.WrapStrategy.CLIP);

  sheet.getRange(1, 1, 1, width)
    .setFontSize(16)
    .setFontWeight("bold")
    .setBackground("#D9EAF7");

  for (let r = 0; r < rows.length; r++) {
    const first = String(rows[r][0] || "").trim();
    if (first.indexOf("SECTION ") === 0) {
      sheet.getRange(r + 1, 1, 1, width)
        .setFontSize(12)
        .setFontWeight("bold")
        .setBackground("#B7DEE8");
      if (r + 2 <= rows.length) {
        sheet.getRange(r + 2, 1, 1, width)
          .setFontWeight("bold")
          .setBackground("#D9EAF7")
          .setWrapStrategy(SpreadsheetApp.WrapStrategy.WRAP);
      }
    }
  }

  const widths = [250, 160, 240, 180, 130, 120, 160, 150, 140, 120, 120, 120, 120];
  widths.forEach(function(colWidth, index) {
    try {
      sheet.setColumnWidth(index + 1, colWidth);
    } catch (err) {
      Logger.log("Dashboard column width skipped: " + err.message);
    }
  });

  try {
    const filter = sheet.getFilter();
    if (filter) filter.remove();
  } catch (err) {
    Logger.log("Dashboard filter remove skipped: " + err.message);
  }
  sheet.setFrozenRows(1);
}

function getSheetDefinitionByTypeOrNull_(dashboard, sheetType) {
  const normalized = normalizeDashboardSheetTypeKey_(sheetType);
  return (dashboard.sheetDefinitions || []).filter(function(item) {
    return item.sheetType === normalized || normalizeDashboardSheetTypeKey_(item.sheetType) === normalized;
  })[0] || null;
}

function getSheetDefinitionByType_(dashboard, sheetType) {
  const sheetDef = getSheetDefinitionByTypeOrNull_(dashboard, sheetType);
  if (!sheetDef) throw new Error("Sheet definition not found in dashboard: " + sheetType);
  return sheetDef;
}

function sortSheetDefinitionsByProductionOrder_(sheetDefinitions) {
  const orderMap = {};
  getRequiredFrameworkSheetTypes_().forEach(function(sheetType, index) {
    orderMap[normalizeKey_(sheetType)] = index;
  });
  return (sheetDefinitions || []).slice().sort(function(a, b) {
    const aOrder = orderMap[normalizeKey_(a && a.sheetType)] !== undefined ? orderMap[normalizeKey_(a.sheetType)] : 999;
    const bOrder = orderMap[normalizeKey_(b && b.sheetType)] !== undefined ? orderMap[normalizeKey_(b.sheetType)] : 999;
    if (aOrder !== bOrder) return aOrder - bOrder;
    return String(a && a.sheetType || "").localeCompare(String(b && b.sheetType || ""));
  });
}

function createOrRefreshAllReportTemplates() {
  return runFrameworkTimed_("Create / Refresh All Templates", function(timing) {
    const dashboard = loadDashboardConfig_(true);
    markFrameworkStep_(timing, "Dashboard loaded");
    const templateResults = [];
    sortSheetDefinitionsByProductionOrder_(dashboard.sheetDefinitions).forEach(function(sheetDef) {
      try {
        RFF_LAST_TEMPLATE_REFRESH_MODE_ = "";
        createOrRefreshTemplateFromDashboard_(dashboard, sheetDef, timing);
        templateResults.push({
          templateName: sheetDef.templateName,
          mode: RFF_LAST_TEMPLATE_REFRESH_MODE_ || "UNKNOWN",
          status: "PASS",
          issue: ""
        });
      } catch (err) {
        templateResults.push({
          templateName: sheetDef.templateName,
          mode: "FAILED",
          status: "FAIL",
          issue: err.message
        });
        markFrameworkStep_(timing, "ERROR - Template build failed but batch continued: " + sheetDef.templateName + " - " + err.message);
        Logger.log("Template build failed but batch continued for " + sheetDef.templateName + ": " + err.message);
      }
    });
    const failedTemplates = templateResults.filter(function(result) {
      return result.status !== "PASS";
    });
    markFrameworkStep_(timing, "Template timing modes: " + summarizeTemplateRefreshModes_(templateResults));
    if (failedTemplates.length) {
      throw new Error("Template refresh completed with " + failedTemplates.length + " failed template(s): " + failedTemplates.map(function(result) {
        return result.templateName;
      }).join(", "));
    }
    markFrameworkStep_(timing, "All report templates refreshed");
  });
}

function summarizeTemplateRefreshModes_(templateResults) {
  const counts = {};
  (templateResults || []).forEach(function(result) {
    const key = result.mode || "UNKNOWN";
    counts[key] = (counts[key] || 0) + 1;
  });
  return Object.keys(counts).sort().map(function(key) {
    return key + "=" + counts[key];
  }).join("; ");
}

function setReportTemplateVisibility_(dashboard, hidden) {
  /*
   * v1.5.30:
   * The v1.5.28 Apps Script execution report showed standalone Hide/Show
   * Templates above 10 seconds. Build a one-pass sheet map instead of repeatedly
   * resolving every template by name, then only call hide/show when the current
   * hidden state differs from the requested state.
   */
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetMap = {};
  ss.getSheets().forEach(function(sheet) {
    sheetMap[sheet.getName()] = sheet;
  });

  sortSheetDefinitionsByProductionOrder_(dashboard.sheetDefinitions).forEach(function(sheetDef) {
    const sheet = sheetMap[sheetDef.templateName];
    if (!sheet) return;
    if (hidden) {
      hideSheetIfNeeded_(sheet);
    } else {
      showSheetIfNeeded_(sheet);
    }
  });
}

function loadDashboardConfig_(forceRefresh) {
  const cacheKey = getDashboardConfigCacheKey_();


  if (!forceRefresh && RFF_DASHBOARD_CONFIG_CACHE_ && RFF_DASHBOARD_CONFIG_CACHE_KEY_ === cacheKey) {
    return RFF_DASHBOARD_CONFIG_CACHE_;
  }


  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(RFF_DASHBOARD_SHEET);


  if (!sheet) {
    throw new Error("Format Dashboard is missing. Run Setup / Refresh Dashboard first.");
  }


  const dashboardIndex = buildDashboardSectionIndex_(sheet);
  const globals = loadGlobalSettings_(dashboardIndex);
  const titleRows = loadTitleRows_(dashboardIndex, globals);
  const sheetDefinitions = loadSheetDefinitions_(dashboardIndex);
  const sheetHeaders = loadSheetHeaders_(dashboardIndex);
  const columnDefinitions = loadColumnDefinitions_(dashboardIndex);
  const behaviors = loadSheetBehaviors_(dashboardIndex);


  const dashboard = {
    globals: globals,
    titleRows: titleRows,
    sheetDefinitions: sheetDefinitions,
    sheetHeaders: sheetHeaders,
    columnDefinitions: columnDefinitions,
    behaviors: behaviors,
    sectionIndex: dashboardIndex
  };


  RFF_DASHBOARD_CONFIG_CACHE_ = dashboard;
  RFF_DASHBOARD_CONFIG_CACHE_KEY_ = cacheKey;


  return dashboard;
}

function buildDashboardSectionIndex_(sheet) {
  const lastRow = Math.max(sheet.getLastRow(), 1);
  const lastCol = Math.max(1, Math.min(Math.max(sheet.getLastColumn(), 1), RFF_DASHBOARD_CONFIG_MAX_READ_COLS));
  const values = sheet.getRange(1, 1, lastRow, lastCol).getValues();
  const starts = {};


  for (let i = 0; i < values.length; i++) {
    const section = normalizeDashboardSectionTitle_(values[i][0]);
    if (section.indexOf("SECTION ") === 0) {
      starts[section] = i;
    }
  }


  return {
    sheet: sheet,
    values: values,
    starts: starts
  };
}

function loadTitleRows_(sheet, globals) {
  let rows = [];
  try {
    rows = readDashboardSectionRows_(sheet, RFF_SECTION_TITLE_ROWS);
  } catch (err) {
    rows = getDefaultTitleRowRows_();
  }

  const defaultsByRow = {};
  getDefaultTitleRowRows_().forEach(function(row) {
    defaultsByRow[numberOrDefault_(row[1], 0)] = parseTitleRowConfigRow_(row, globals, null);
  });

  const titleRowsBySheetType = {};

  rows.forEach(function(row) {
    const rowNumber = numberOrDefault_(row[1], 0);
    if (!rowNumber) return;
    const sheetTypeRaw = String(row[0] || "GLOBAL").trim() || "GLOBAL";
    const sheetType = sheetTypeRaw.toUpperCase() === "GLOBAL" ? "GLOBAL" : normalizeDashboardSheetTypeKey_(sheetTypeRaw);
    const base = defaultsByRow[rowNumber] || {};
    if (!titleRowsBySheetType[sheetType]) titleRowsBySheetType[sheetType] = {};
    titleRowsBySheetType[sheetType][rowNumber] = parseTitleRowConfigRow_(row, globals, base);
  });

  titleRowsBySheetType.GLOBAL = titleRowsBySheetType.GLOBAL || {};
  Object.keys(defaultsByRow).forEach(function(rowNumber) {
    if (!titleRowsBySheetType.GLOBAL[rowNumber]) {
      titleRowsBySheetType.GLOBAL[rowNumber] = defaultsByRow[rowNumber];
    }
  });

  return titleRowsBySheetType;
}

function parseTitleRowConfigRow_(row, globals, base) {
  row = row || [];
  base = base || {};
  const rowNumber = numberOrDefault_(row[1], base.row || 0);
  const fillLevel = String(row[9] || base.fillLevel || "").trim() || "Level 1";
  const fontWeight = String(row[8] || base.fontWeight || "").trim() || "Normal";
  const alignment = String(row[10] || base.alignment || globals.defaultHorizontalAlignment || RFF_DEFAULTS.defaultHorizontalAlignment).trim().toLowerCase();
  const wrap = String(row[11] || base.wrap || globals.defaultDataWrap || RFF_DEFAULTS.defaultDataWrap).trim().toUpperCase();

  return {
    sheetType: String(row[0] || base.sheetType || "GLOBAL").trim() || "GLOBAL",
    row: rowNumber,
    purpose: String(row[2] || base.purpose || "").trim(),
    valueSource: String(row[3] || base.valueSource || "").trim(),
    label: String(row[4] === "" || row[4] === null || row[4] === undefined ? (base.label || "") : row[4]).trim(),
    targetCell: String(row[5] || base.targetCell || ("Row " + rowNumber)).trim(),
    height: numberOrDefault_(row[6], base.height || RFF_DEFAULTS.dataRowHeight),
    fontSize: numberOrDefault_(row[7], base.fontSize || globals.standardFontSize || RFF_DEFAULTS.standardFontSize),
    fontWeight: fontWeight.toLowerCase() === "bold" ? "bold" : "normal",
    fillLevel: fillLevel,
    alignment: alignment,
    wrap: wrap,
    notes: String(row[12] || base.notes || "").trim()
  };
}

function getTitleRowConfigForSheet_(dashboard, sheetDef, rowNumber) {
  const allTitleRows = (dashboard && dashboard.titleRows) || {};
  const sheetType = sheetDef ? normalizeDashboardSheetTypeKey_(sheetDef.sheetType) : "";
  const sheetRows = allTitleRows[sheetType] || {};
  const globalRows = allTitleRows.GLOBAL || {};
  return sheetRows[rowNumber] || globalRows[rowNumber] || parseTitleRowConfigRow_(getDefaultTitleRowRows_()[Math.max(rowNumber - 1, 0)] || [], dashboard.globals || {}, null);
}

function getThemeFillForTitleRow_(theme, fillLevel) {
  const normalized = String(fillLevel || "").trim().toLowerCase().replace(/\s+/g, "");
  if (normalized === "level3") return theme.level3;
  if (normalized === "level2") return theme.level2;
  if (normalized === "level1") return theme.level1;
  return theme.level1;
}

function loadSheetDefinitions_(sheet) {
  const rows = readDashboardSectionRows_(sheet, RFF_SECTION_SHEETS);
  return rows
    .filter(function(row) {
      return String(row[0] || "").trim();
    })
    .map(function(row) {
      const hasTemplateColumnCount = !isBlankCell_(row[8]) && !isNaN(Number(row[8]));
      const offset = hasTemplateColumnCount ? 1 : 0;
      return {
        sheetType: normalizeDashboardSheetTypeKey_(row[0]),
        reportTitle: String(row[1] || row[0] || "").trim(),
        templateName: String(row[2] || ("Template - " + row[0])).trim(),
        outputNamingPattern: String(row[3] || "").trim(),
        baseColor: normalizeHex_(row[4] || "#65A9CC"),
        usePromptDate: parseBoolean_(row[5]),
        endDateSource: String(row[6] || "").trim(),
        templateRowCount: numberOrDefault_(row[7], RFF_DEFAULTS.templateRows),
        templateColumnCount: hasTemplateColumnCount ? numberOrDefault_(row[8], 0) : 0,
        templateRowMode: String(row[8 + offset] || "FIXED").trim().toUpperCase(),
        minimumRows: numberOrDefault_(row[9 + offset], numberOrDefault_(row[7], RFF_DEFAULTS.templateRows)),
        bufferRows: numberOrDefault_(row[10 + offset], 100),
        testRows: numberOrDefault_(row[11 + offset], 100),
        useTestRows: parseBoolean_(row[12 + offset])
      };
    });
}

function loadSheetHeaders_(sheet) {
  const rows = readDashboardSectionRows_(sheet, RFF_SECTION_HEADERS);
  const map = {};


  rows.forEach(function(row) {
    const sheetType = normalizeDashboardSheetTypeKey_(row[0]);
    const order = numberOrDefault_(row[1], 9999);
    const header = normalizeHeader_(row[2]);
    const source = String(row[3] || "").trim();


    if (!sheetType || !header) return;
    if (!map[sheetType]) map[sheetType] = [];


    map[sheetType].push({
      order: order,
      header: header,
      source: source
    });
  });


  Object.keys(map).forEach(function(sheetType) {
    map[sheetType].sort(function(a, b) {
      return a.order - b.order;
    });
  });


  return map;
}

function loadColumnDefinitions_(sheet) {
  const rows = readDashboardSectionRows_(sheet, RFF_SECTION_COLUMNS);
  const map = {};


  rows.forEach(function(row) {
    const header = normalizeHeader_(row[0]);
    if (!header) return;


    map[header] = {
      header: header,
      width: isBlankCell_(row[1]) ? null : numberOrDefault_(row[1], null),
      headerFontSize: isBlankCell_(row[2]) ? null : numberOrDefault_(row[2], null),
      dateColumn: parseBoolean_(row[3]),
      hideColumn: parseBoolean_(row[4]),
      dataWrap: String(row[5] || "").trim().toUpperCase(),
      horizontalAlignment: String(row[6] || "").trim().toLowerCase(),
      verticalAlignment: String(row[7] || "").trim().toLowerCase(),
      numberFormat: String(row[8] || "").trim()
    };
  });


  return map;
}

function loadSheetBehaviors_(sheet) {
  const rows = readDashboardSectionRows_(sheet, RFF_SECTION_BEHAVIORS);
  const map = {};


  rows.forEach(function(row) {
    const sheetType = normalizeDashboardSheetTypeKey_(row[0]);
    if (!sheetType) return;


    map[sheetType] = {
      usesTitleRows: parseBoolean_(row[1]),
      usesFilter: parseBoolean_(row[2]),
      usesAlternatingColors: parseBoolean_(row[3]),
      usesSubheaders: parseBoolean_(row[4]),
      hiddenTemplate: parseBoolean_(row[5])
    };
  });


  return map;
}

function normalizeDashboardSectionTitle_(value) {
  return String(value === null || value === undefined ? "" : value)
    .replace(/–|—/g, "-")
    .replace(/\s+/g, " ")
    .trim()
    .toUpperCase();
}

function readDashboardSectionRows_(source, sectionName) {
  const index = source && source.values ? source : buildDashboardSectionIndex_(source);
  const values = index.values || [];
  const targetSection = normalizeDashboardSectionTitle_(sectionName);
  let sectionRow = Object.prototype.hasOwnProperty.call(index.starts || {}, targetSection) ? index.starts[targetSection] : -1;


  if (sectionRow === -1) {
    throw new Error("Dashboard section missing: " + sectionName);
  }


  const output = [];


  for (let r = sectionRow + 2; r < values.length; r++) {
    const firstCell = String(values[r][0] || "").trim();
    const firstCellSection = normalizeDashboardSectionTitle_(firstCell);


    if (firstCellSection.indexOf("SECTION ") === 0) break;


    const rowValues = values[r];
    const hasAnyValue = rowValues.some(function(value) {
      return String(value === null || value === undefined ? "" : value).trim() !== "";
    });


    if (!hasAnyValue) {
      const nextFirst = r + 1 < values.length ? String(values[r + 1][0] || "").trim() : "";
      const nextFirstSection = normalizeDashboardSectionTitle_(nextFirst);
      if (nextFirstSection.indexOf("SECTION ") === 0 || nextFirst === "") break;
      continue;
    }


    output.push(rowValues);
  }


  return output;
}


/* ============================================================================
 * TEMPLATE BUILDING
 * ========================================================================== */

function createOrRefreshTemplateFromDashboard_(dashboard, sheetDef, timing) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(sheetDef.templateName);
  const templateExisted = !!sheet;


  if (!sheet) {
    sheet = ss.insertSheet(sheetDef.templateName);
  }
  markFrameworkStep_(timing, "Open/create sheet: " + sheetDef.templateName);


  const headers = getHeadersForSheetType_(dashboard, sheetDef.sheetType);
  const behavior = dashboard.behaviors[sheetDef.sheetType] || getDefaultBehavior_();
  const globals = dashboard.globals;
  const colCount = Math.max(headers.length, 4);
  const rowCount = resolveTemplateRowCount_(sheetDef, globals);
  const expectedSignature = buildTemplateFormatSignature_(dashboard, sheetDef, headers, rowCount, colCount, behavior);
  const propertySignature = getStoredTemplateFormatSignature_(sheetDef);
  const noteSignature = templateExisted ? getStoredTemplateFormatSignatureFromSheet_(sheet) : "";
  const storedSignature = noteSignature || propertySignature;
  const normalizedStoredSignature = normalizeTemplateFormatSignature_(storedSignature);
  const normalizedExpectedSignature = normalizeTemplateFormatSignature_(expectedSignature);


  const signatureMatches = templateExisted &&
    RFF_USE_TEMPLATE_SIGNATURE_CACHE &&
    normalizedStoredSignature !== "" &&
    normalizedStoredSignature === normalizedExpectedSignature;


  const metadataRefreshAllowed =
    RFF_TRUE_SMART_TEMPLATE_REFRESH &&
    RFF_SMART_TEMPLATE_REFRESH &&
    !RFF_FORCE_FULL_TEMPLATE_REBUILD &&
    templateExisted &&
    !!sheet &&
    signatureMatches;


  markFrameworkStep_(timing, signatureMatches
    ? "Template signature matched - metadata refresh eligible: " + sheetDef.templateName
    : "Template signature mismatch - full build required: " + sheetDef.templateName);


  markFrameworkStep_(timing, buildTemplateRefreshDecisionMessage_(
    sheetDef,
    templateExisted,
    signatureMatches,
    metadataRefreshAllowed,
    normalizedStoredSignature,
    normalizedExpectedSignature
  ));
  markFrameworkStep_(timing, "Template timing mode: " + (metadataRefreshAllowed ? "METADATA_ONLY" : (templateExisted ? "FULL_REBUILD" : "FIRST_BUILD")) + " - " + sheetDef.templateName);


  if (metadataRefreshAllowed) {
    refreshTemplateMetadataOnly_(sheet, dashboard, sheetDef, colCount, expectedSignature, timing);
    RFF_LAST_TEMPLATE_REFRESH_MODE_ = "METADATA_ONLY";
    return sheet;
  }


  const builtSheet = shouldUseStagedTemplateBuild_(sheetDef, templateExisted)
    ? buildTemplateFromDashboardSafely_(sheet, dashboard, sheetDef, headers, rowCount, colCount, behavior, expectedSignature, timing, templateExisted)
    : buildTemplateFromDashboard_(sheet, dashboard, sheetDef, headers, rowCount, colCount, behavior, expectedSignature, timing, templateExisted);
  RFF_LAST_TEMPLATE_REFRESH_MODE_ = templateExisted ? "FULL_REBUILD" : "FIRST_BUILD";
  return builtSheet;
}

function shouldUseStagedTemplateBuild_(sheetDef, templateExisted) {
  if (!RFF_TEMPLATE_STAGED_BUILD_ENABLED) return false;
  if (templateExisted) return true;
  if (!RFF_TEMPLATE_STAGED_FIRST_BUILD_FOR_HIGH_RISK_ONLY) return true;
  return RFF_TEMPLATE_ALWAYS_STAGED_TEMPLATE_NAMES.indexOf(String(sheetDef && sheetDef.templateName || "").trim()) !== -1;
}

function shouldRefreshTemplateMetadataOnly_(sheet, sheetDef, templateExisted, signatureMatches) {
  if (!RFF_TRUE_SMART_TEMPLATE_REFRESH) return false;
  if (!RFF_SMART_TEMPLATE_REFRESH) return false;
  if (RFF_FORCE_FULL_TEMPLATE_REBUILD) return false;
  if (!templateExisted || !sheet) return false;
  if (!signatureMatches) return false;


  // This intentionally avoids expensive row/column/header/filter/freeze checks.
  // The stored dashboard signature is the source of truth for whether a full
  // template rebuild is needed. Validation remains available to catch manual
  // sheet edits or corruption when needed.
  return true;
}

function buildTemplateRefreshDecisionMessage_(sheetDef, templateExisted, signatureMatches, metadataRefreshAllowed, normalizedStoredSignature, normalizedExpectedSignature) {
  const reasons = [];


  if (!templateExisted) reasons.push("template sheet did not exist");
  if (!RFF_TRUE_SMART_TEMPLATE_REFRESH) reasons.push("true smart refresh disabled");
  if (!RFF_SMART_TEMPLATE_REFRESH) reasons.push("smart refresh disabled");
  if (RFF_FORCE_FULL_TEMPLATE_REBUILD) reasons.push("force full rebuild enabled");
  if (!RFF_USE_TEMPLATE_SIGNATURE_CACHE) reasons.push("signature cache disabled");
  if (!normalizedStoredSignature) reasons.push("no stored signature");
  if (normalizedStoredSignature && normalizedExpectedSignature && normalizedStoredSignature !== normalizedExpectedSignature) reasons.push("stored signature differs from expected signature");
  if (signatureMatches && metadataRefreshAllowed) reasons.push("metadata refresh will run");


  const decision = metadataRefreshAllowed ? "METADATA_ONLY" : "FULL_BUILD";
  return "Template refresh decision: " + decision + " - " + sheetDef.templateName + " (" + (reasons.length ? reasons.join("; ") : "no reason captured") + ")";
}

function refreshTemplateMetadataOnly_(sheet, dashboard, sheetDef, colCount, expectedSignature, timing) {
  writeTemplateMetadata_(sheet, dashboard, sheetDef, colCount, expectedSignature);
  storeTemplateFormatSignature_(sheetDef, expectedSignature);
  markFrameworkStep_(timing, "Metadata-only refresh; template unchanged: " + sheetDef.templateName);


  // Do not hide/show, clear, resize, rebuild alternating colors, reapply widths,
  // reapply filters, or re-freeze during metadata-only refresh. Those operations
  // are intentionally reserved for full builds only.
  markFrameworkStep_(timing, "Complete metadata-only template refresh: " + sheetDef.templateName);
}

function buildTemplateFromDashboardSafely_(existingSheet, dashboard, sheetDef, headers, rowCount, colCount, behavior, expectedSignature, timing, templateExisted) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const tempName = getTemplateBuildSheetName_(sheetDef.templateName);
  const staleTemp = ss.getSheetByName(tempName);
  if (staleTemp) deleteSheetIfExists_(ss, tempName, sheetDef.templateName, "");


  const buildSheet = ss.insertSheet(tempName);
  markFrameworkStep_(timing, "Staged template build sheet created: " + sheetDef.templateName);


  try {
    buildTemplateFromDashboard_(buildSheet, dashboard, sheetDef, headers, rowCount, colCount, behavior, expectedSignature, timing, false);
    validateBuiltTemplateMinimumStructure_(buildSheet, dashboard, sheetDef, headers, colCount);
    promoteStagedTemplateBuild_(ss, buildSheet, existingSheet, sheetDef, timing);
    markFrameworkStep_(timing, "Staged template build promoted: " + sheetDef.templateName);
    return ss.getSheetByName(sheetDef.templateName) || buildSheet;
  } catch (err) {
    storeTemplateFormatSignature_(sheetDef, "");
    try {
      buildSheet.setName(tempName + " FAILED");
      buildSheet.showSheet();
    } catch (renameErr) {
      Logger.log("Failed staged template rename skipped for " + sheetDef.templateName + ": " + renameErr.message);
    }
    throw err;
  }
}

function getTemplateBuildSheetName_(templateName) {
  return String(templateName || "Template") + " __BUILDING";
}

function promoteStagedTemplateBuild_(ss, buildSheet, existingSheet, sheetDef, timing) {
  buildSheet.showSheet();
  if (existingSheet && existingSheet.getSheetId() !== buildSheet.getSheetId()) {
    ss.setActiveSheet(buildSheet);
    ss.deleteSheet(existingSheet);
    markFrameworkStep_(timing, "Previous template replaced after staged validation: " + sheetDef.templateName);
  }
  buildSheet.setName(sheetDef.templateName);
  if (typeof buildSheet.hideSheet === "function") {
    hideSheetIfNeeded_(buildSheet, timing, "Hide promoted template: " + sheetDef.templateName);
  }
}

function validateBuiltTemplateMinimumStructure_(sheet, dashboard, sheetDef, headers, colCount) {
  const globals = dashboard.globals || RFF_DEFAULTS;
  const headerRow = Number(globals.headerRow || RFF_DEFAULTS.headerRow);
  const expectedFrozenRows = Number(globals.freezeRows || RFF_DEFAULTS.freezeRows);
  if (sheet.getMaxColumns() < colCount) {
    throw new Error("Staged template has " + sheet.getMaxColumns() + " column(s), expected at least " + colCount + ": " + sheetDef.templateName);
  }
  if (sheet.getFrozenRows() !== expectedFrozenRows) {
    throw new Error("Staged template frozen rows expected " + expectedFrozenRows + " but found " + sheet.getFrozenRows() + ": " + sheetDef.templateName);
  }
  const actualHeaders = sheet.getRange(headerRow, 1, 1, Math.max(headers.length, 1)).getDisplayValues()[0];
  for (let i = 0; i < headers.length; i++) {
    if (String(actualHeaders[i] || "").trim() !== String(headers[i] || "").trim()) {
      throw new Error("Staged template header mismatch col " + (i + 1) + ": expected " + headers[i] + ", found " + actualHeaders[i] + ": " + sheetDef.templateName);
    }
  }
}

function buildTemplateFromDashboard_(sheet, dashboard, sheetDef, headers, rowCount, colCount, behavior, expectedSignature, timing, templateExisted) {
  markFrameworkStep_(timing, "Full template build required: " + sheetDef.templateName);


  const useCorrectedIndividualBuild = shouldUseCorrectedIndividualTemplateBuild_(sheetDef);
  const isTemplateCreate = !templateExisted;


  if (rowCount > 1000) {
    resizeSheetRows_(sheet, rowCount);
    markFrameworkStep_(timing, "Set template row count from dashboard: " + sheetDef.templateName + " (" + rowCount + " rows)");
  } else {
    markFrameworkStep_(timing, "Template row count already covered by default sheet size: " + sheetDef.templateName + " (" + rowCount + " rows)");
  }


  resizeSheetColumns_(sheet, colCount);
  markFrameworkStep_(timing, "Set template column count from dashboard headers: " + sheetDef.templateName + " (" + colCount + " cols)");


  if (isTemplateCreate) {
    markFrameworkStep_(timing, "Create template fast path - clear skipped for blank sheet: " + sheetDef.templateName);
  } else if (!useCorrectedIndividualBuild) {
    clearTemplateForFullBuild_(sheet, sheetDef, timing, templateExisted);
  }


  applyTemplateBaseFormatting_(sheet, dashboard, sheetDef, headers, rowCount, colCount, behavior, timing);
  if (!useCorrectedIndividualBuild) {
    markFrameworkStep_(timing, "Apply full template formatting: " + sheetDef.templateName);
  }


  writeTemplateMetadata_(sheet, dashboard, sheetDef, colCount, expectedSignature);
  storeTemplateFormatSignature_(sheetDef, expectedSignature);
  markFrameworkStep_(timing, "Write metadata/signature: " + sheetDef.templateName);


  applyTemplateFreezeAndTabColor_(sheet, dashboard, sheetDef, colCount, timing);


  if (RFF_SKIP_DUPLICATE_TEMPLATE_FORMAT_ENFORCEMENT) {
    markFrameworkStep_(timing, "Final date/number format enforcement skipped - already applied during template base formatting: " + sheetDef.templateName);
  } else {
    enforceTemplateDateAndNumberFormats_(sheet, dashboard, sheetDef, rowCount);
    markFrameworkStep_(timing, "Final date/number format enforcement: " + sheetDef.templateName);
  }


  if (behavior.hiddenTemplate) {
    hideTemplateIfNeeded_(sheet, sheetDef, timing);
  } else {
    markFrameworkStep_(timing, "Template remains visible: " + sheetDef.templateName);
  }


  markFrameworkStep_(timing, "Complete full template build: " + sheetDef.templateName);
  return sheet;
}

function shouldSkipTemplateResize_(sheetDef) {
  const templateName = String(sheetDef && sheetDef.templateName || "").trim();
  return RFF_TEMPLATE_RESIZE_EXEMPT_TEMPLATE_NAMES.indexOf(templateName) !== -1;
}

function shouldUseCorrectedIndividualTemplateBuild_(sheetDef) {
  /*
   * v1.4.30:
   * Individual template-build branches remain retired. All governed templates use
   * the shared dashboard build sequence. Create and update are now separated by
   * templateExisted inside buildTemplateFromDashboard_: brand-new blank sheets
   * skip clear sheet/formats; existing templates clear only when a full rebuild
   * is required after signature mismatch.
   */
  return false;
}

function clearTemplateForFullBuild_(sheet, sheetDef, timing, templateExisted) {
  /*
   * v1.5.30:
   * New templates are known blank immediately after insertSheet(). The v1.5.28
   * timing report flagged the blank-sheet clear-skip branch as the only slow
   * template step, so the create path avoids getLastRow()/getLastColumn() probes.
   * Existing templates still clear before full rebuilds.
   */
  if (!templateExisted) {
    markFrameworkStep_(timing, "Create template fast path - clear skipped for new sheet: " + sheetDef.templateName);
    return;
  }

  try {
    sheet.clear();
    sheet.setConditionalFormatRules([]);
  } catch (err) {
    sheet.clearContents();
    sheet.clearFormats();
    try {
      sheet.setConditionalFormatRules([]);
    } catch (ruleErr) {
      Logger.log("Conditional format clear skipped for " + sheet.getName() + ": " + ruleErr.message);
    }
  }
  markFrameworkStep_(timing, "Clear sheet/formats for full build: " + sheetDef.templateName);
}

function applyTemplateRowHeights_(sheet, dashboard, sheetDef, rowCount, timing, stepPrefix) {
  /*
   * v1.5:
   * Template row-height processing is intentionally disabled.
   * Empty templates should not pay row-height cost; final output row-height locking
   * belongs only to Raw Data, Demo P, Master List, and Disenrolled output workflows.
   */
  markFrameworkStep_(timing, (stepPrefix || "Apply row heights") + " skipped for template: " + sheetDef.templateName);
}

function applyFinalRowHeightLockForSheetType_(sheet, sheetType) {
  lockFinalOutputRowHeights_(sheet, sheetType);
}

function lockFinalOutputRowHeights_(sheet, label) {
  if (!sheet) return;


  const dashboard = loadDashboardConfig_();
  const globals = dashboard.globals || {};
  const headerRow = Number(globals.headerRow || HEADER_ROW);
  const dataStartRow = Number(globals.dataStartRow || DATA_START_ROW);
  const lastRow = Math.max(sheet.getLastRow(), dataStartRow);


  try {
    applyRowHeightRuns_(sheet, [
      { start: 1, count: 1, height: Number(globals.row1Height || 25) },
      { start: 2, count: 1, height: Number(globals.row2Height || 20) },
      { start: 3, count: 1, height: Number(globals.row3Height || 10) },
      { start: headerRow, count: 1, height: Number(globals.headerRowHeight || 40) }
    ]);


    if (lastRow >= dataStartRow) {
      safeSetRowHeights_(sheet, dataStartRow, lastRow - dataStartRow + 1, Number(globals.dataRowHeight || 25), "final output data rows");
    }
  } catch (err) {
    Logger.log("Final row height lock skipped for " + (label || sheet.getName()) + ": " + err.message);
  }
}

function safeSetRowHeights_(sheet, startRow, count, height, label) {
  if (!sheet) return 0;


  startRow = Math.max(1, Number(startRow || 1));
  count = Math.max(0, Number(count || 0));
  height = Number(height || 25);


  if (count < 1) return 0;


  const maxRows = Math.max(sheet.getMaxRows(), 1);
  if (startRow > maxRows) return 0;


  const safeCount = Math.min(count, maxRows - startRow + 1);
  if (safeCount < 1) return 0;


  try {
    if (typeof sheet.setRowHeightsForced === "function") {
      sheet.setRowHeightsForced(startRow, safeCount, height);
    } else if (typeof sheet.setRowHeights === "function") {
      sheet.setRowHeights(startRow, safeCount, height);
    } else {
      for (let row = startRow; row < startRow + safeCount; row++) {
        sheet.setRowHeight(row, height);
      }
    }
    return safeCount;
  } catch (err) {
    Logger.log("Row height run skipped for " + (label || sheet.getName()) + " rows " + startRow + "-" + (startRow + safeCount - 1) + ": " + err.message);


    let applied = 0;
    for (let row = startRow; row < startRow + safeCount; row++) {
      try {
        sheet.setRowHeight(row, height);
        applied++;
      } catch (rowErr) {
        Logger.log("Row height skipped for row " + row + ": " + rowErr.message);
      }
    }
    return applied;
  }
}

function applyRowHeightRuns_(sheet, runs) {
  (runs || []).forEach(function(run) {
    if (!run || !run.start || !run.count) return;
    safeSetRowHeights_(sheet, run.start, run.count, run.height, "row-height run");
  });
}

function hideTemplateIfNeeded_(sheet, sheetDef, timing) {
  try {
    if (typeof sheet.isSheetHidden === "function" && sheet.isSheetHidden()) {
      markFrameworkStep_(timing, "Template already hidden - skipped hide: " + sheetDef.templateName);
      return;
    }


    hideSheetIfNeeded_(sheet, timing, "Hide template: " + sheetDef.templateName);
  } catch (err) {
    Logger.log("Template hide skipped for " + sheetDef.templateName + ": " + err.message);
    markFrameworkStep_(timing, "Hide template skipped/error: " + sheetDef.templateName);
  }
}

function resolveTemplateRowCount_(sheetDef, globals) {
  const headerRow = globals && globals.headerRow ? globals.headerRow : RFF_DEFAULTS.headerRow;
  const fixedRows = numberOrDefault_(sheetDef.templateRowCount, RFF_DEFAULTS.templateRows);
  const minRows = numberOrDefault_(sheetDef.minimumRows, fixedRows);
  const testRows = numberOrDefault_(sheetDef.testRows, Math.min(fixedRows, 500));
  const bufferRows = numberOrDefault_(sheetDef.bufferRows, 100);
  const mode = String(sheetDef.templateRowMode || "FIXED").trim().toUpperCase();


  if (sheetDef.useTestRows || mode === "TEST") {
    return Math.max(testRows, headerRow);
  }


  if (mode === "DYNAMIC") {
    // Phase 1 templates do not yet receive source data counts. Use the dashboard
    // minimum + buffer until the Phase 2+ report formatters pass actual data row counts.
    return Math.max(minRows + bufferRows, headerRow);
  }


  return Math.max(fixedRows, headerRow);
}

function applyTemplateBaseFormatting_(sheet, dashboard, sheetDef, headers, rowCount, colCount, behavior, timing) {
  const globals = dashboard.globals;
  const theme = getThemeColorsFromBase_(sheetDef.baseColor);
  const headerRow = globals.headerRow;
  const dataStartRow = globals.dataStartRow;


  sheet.getRange(1, 1, rowCount, colCount)
    .setFontFamily(globals.standardFont)
    .setFontColor(globals.standardFontColor)
    .setFontSize(globals.standardFontSize)
    .setHorizontalAlignment(globals.defaultHorizontalAlignment || "left")
    .setVerticalAlignment(globals.defaultVerticalAlignment || "middle")
    .setWrapStrategy(toWrapStrategy_(globals.defaultDataWrap || "CLIP"));


  applyTitleRows_(sheet, dashboard, sheetDef, theme, colCount);
  markFrameworkStep_(timing, "Apply title rows: " + sheetDef.templateName);


  applyHeaderRow_(sheet, dashboard, sheetDef, headers, theme, colCount);
  markFrameworkStep_(timing, "Apply header row: " + sheetDef.templateName);


  applyDataRows_(sheet, dashboard, sheetDef, headers, rowCount, colCount, behavior, timing);
  markFrameworkStep_(timing, "Apply data row base style: " + sheetDef.templateName);


  applyColumnWidths_(sheet, dashboard, headers);
  markFrameworkStep_(timing, "Apply column widths: " + sheetDef.templateName);


  applyDateAndNumberFormats_(sheet, dashboard, headers, rowCount);
  markFrameworkStep_(timing, "Apply date/number formats: " + sheetDef.templateName);


  applyHiddenColumnSettings_(sheet, dashboard, headers);
  markFrameworkStep_(timing, "Apply hidden/show column settings: " + sheetDef.templateName);


  if (behavior.usesFilter) {
    ensureTemplateFilter_(sheet, headerRow, rowCount, colCount, sheetDef, timing);
  }


  if (behavior.usesSubheaders) {
    applyMonthlyChangeSubheaderGuide_(sheet, dashboard, sheetDef, colCount);
    markFrameworkStep_(timing, "Apply Monthly Change subheader guide: " + sheetDef.templateName);
  }


  applyTemplateRowHeights_(sheet, dashboard, sheetDef, rowCount, timing, "Apply row heights");
}

function applyTitleRows_(sheet, dashboard, sheetDef, theme, colCount) {
  const globals = dashboard.globals;
  const row1 = getTitleRowConfigForSheet_(dashboard, sheetDef, 1);
  const row2 = getTitleRowConfigForSheet_(dashboard, sheetDef, 2);
  const row3 = getTitleRowConfigForSheet_(dashboard, sheetDef, 3);
  const row4 = getTitleRowConfigForSheet_(dashboard, sheetDef, 4);


  [row1, row2, row3, row4].forEach(function(rowConfig) {
    if (!rowConfig || !rowConfig.row) return;
    try {
      sheet.setRowHeight(rowConfig.row, rowConfig.height);
    } catch (err) {
      Logger.log("Title row height skipped for row " + rowConfig.row + ": " + err.message);
    }
    sheet.getRange(rowConfig.row, 1, 1, colCount)
      .setBackground(getThemeFillForTitleRow_(theme, rowConfig.fillLevel))
      .setFontSize(rowConfig.fontSize)
      .setFontWeight(rowConfig.fontWeight)
      .setHorizontalAlignment(rowConfig.alignment)
      .setVerticalAlignment("middle")
      .setWrapStrategy(toWrapStrategy_(rowConfig.wrap));
  });


  sheet.getRange(row1.targetCell || "A1")
    .setValue(sheetDef.reportTitle)
    .setFontSize(row1.fontSize || globals.titleFontSize)
    .setFontWeight(row1.fontWeight || "bold")
    .setWrapStrategy(toWrapStrategy_(row1.wrap || "CLIP"))
    .setVerticalAlignment("middle");


  try {
    sheet.getRange("C1:D1").breakApart();
    sheet.getRange("C1:D1").merge();
    sheet.getRange("C1:D1")
      .setValue("")
      .setFontSize(globals.titleInfoFontSize)
      .setWrapStrategy(SpreadsheetApp.WrapStrategy.WRAP)
      .setVerticalAlignment("middle");
  } catch (err) {
    Logger.log("Title info merge skipped for " + sheet.getName() + ": " + err.message);
  }


  sheet.getRange("A2").setValue("Date");
  if (row2.label) sheet.getRange("A2").setValue(row2.label);
  sheet.getRange("B2").setValue("").setNumberFormat(globals.defaultDateFormat);
  sheet.getRange("C2").setValue("to").setHorizontalAlignment("center");
  sheet.getRange("D2").setValue("").setNumberFormat(globals.defaultDateFormat);
}

function applyColumnWidths_(sheet, dashboard, headers) {
  const globals = dashboard.globals || {};
  const widths = (headers || []).map(function(header) {
    const def = dashboard.columnDefinitions[header] || {};
    return Number(def.width || globals.defaultColumnWidth || RFF_DEFAULTS.defaultColumnWidth);
  });


  applyColumnWidthsInRuns_(sheet, widths);
}

function applyColumnWidthsInRuns_(sheet, widths) {
  if (!sheet || !widths || widths.length === 0) return;


  let startCol = 1;
  let currentWidth = widths[0];


  for (let i = 1; i <= widths.length; i++) {
    const nextWidth = i < widths.length ? widths[i] : null;


    if (i < widths.length && nextWidth === currentWidth) {
      continue;
    }


    const runLength = i - startCol + 1;


    try {
      if (typeof sheet.setColumnWidths === "function") {
        sheet.setColumnWidths(startCol, runLength, currentWidth);
      } else {
        for (let col = startCol; col < startCol + runLength; col++) {
          sheet.setColumnWidth(col, currentWidth);
        }
      }
    } catch (err) {
      Logger.log("Column width run skipped for columns " + startCol + "-" + (startCol + runLength - 1) + ": " + err.message);
      for (let col = startCol; col < startCol + runLength; col++) {
        try {
          sheet.setColumnWidth(col, currentWidth);
        } catch (colErr) {
          Logger.log("Column width skipped for column " + col + ": " + colErr.message);
        }
      }
    }


    startCol = i + 1;
    currentWidth = nextWidth;
  }
}

function enforceTemplateDateAndNumberFormats_(sheet, dashboard, sheetDef, rowCount) {
  const headers = getHeadersForSheetType_(dashboard, sheetDef.sheetType);
  enforceDateAndNumberFormatsForHeaders_(sheet, dashboard, headers, rowCount, "Final date/number format enforcement");
}

function applyHiddenColumnSettings_(sheet, dashboard, headers) {
  const hiddenFlags = (headers || []).map(function(header) {
    const def = dashboard.columnDefinitions[header] || {};
    return !!def.hideColumn;
  });


  applyHiddenColumnSettingsInRuns_(sheet, hiddenFlags);
}

function applyHiddenColumnSettingsInRuns_(sheet, hiddenFlags) {
  if (!sheet || !hiddenFlags || hiddenFlags.length === 0) return;


  let startCol = 1;
  let currentHidden = hiddenFlags[0];


  for (let i = 1; i <= hiddenFlags.length; i++) {
    const nextHidden = i < hiddenFlags.length ? hiddenFlags[i] : null;


    if (i < hiddenFlags.length && nextHidden === currentHidden) {
      continue;
    }


    const runLength = i - startCol + 1;


    try {
      if (currentHidden) {
        sheet.hideColumns(startCol, runLength);
      } else {
        sheet.showColumns(startCol, runLength);
      }
    } catch (err) {
      Logger.log("Hidden column run skipped for columns " + startCol + "-" + (startCol + runLength - 1) + ": " + err.message);
      for (let col = startCol; col < startCol + runLength; col++) {
        try {
          if (currentHidden) {
            sheet.hideColumns(col);
          } else {
            sheet.showColumns(col);
          }
        } catch (colErr) {
          Logger.log("Hidden column setting skipped for column " + col + ": " + colErr.message);
        }
      }
    }


    startCol = i + 1;
    currentHidden = nextHidden;
  }
}

function applyAlternatingColorRules_(sheet, range, theme) {
  try {
    const rules = sheet.getConditionalFormatRules() || [];


    const oddRule = SpreadsheetApp.newConditionalFormatRule()
      .whenFormulaSatisfied("=ISODD(ROW())")
      .setBackground(theme.level3)
      .setRanges([range])
      .build();


    const evenRule = SpreadsheetApp.newConditionalFormatRule()
      .whenFormulaSatisfied("=ISEVEN(ROW())")
      .setBackground(theme.level4)
      .setRanges([range])
      .build();


    rules.push(oddRule);
    rules.push(evenRule);
    sheet.setConditionalFormatRules(rules);
  } catch (err) {
    Logger.log("Alternating color rules skipped for " + sheet.getName() + ": " + err.message);
  }
}

function buildTemplateFormatSignature_(dashboard, sheetDef, headers, rowCount, colCount, behavior) {
  const globals = dashboard.globals || {};
  const parts = [
    "sheetType=" + sheetDef.sheetType,
    "rows=" + rowCount,
    "cols=" + colCount,
    "freeze=" + globals.freezeRows + ":" + globals.freezeColumns,
    "date=" + String(globals.defaultDateFormat || ""),
    "defaultWidth=" + String(globals.defaultColumnWidth || RFF_DEFAULTS.defaultColumnWidth),
    "wrap=" + String(globals.defaultDataWrap || RFF_DEFAULTS.defaultDataWrap),
    "align=" + String(globals.defaultHorizontalAlignment || RFF_DEFAULTS.defaultHorizontalAlignment) + ":" + String(globals.defaultVerticalAlignment || RFF_DEFAULTS.defaultVerticalAlignment),
    "titleRows=" + JSON.stringify(dashboard.titleRows || {}),
    "behavior=" + [behavior.usesFilter, behavior.usesAlternatingColors, behavior.usesSubheaders, behavior.hiddenTemplate].join(",")
  ];


  headers.forEach(function(header, index) {
    const def = dashboard.columnDefinitions[header] || {};
    parts.push([
      index + 1,
      normalizeHeader_(header),
      def.width || globals.defaultColumnWidth || RFF_DEFAULTS.defaultColumnWidth,
      def.headerFontSize || globals.standardFontSize || 10,
      boolText_(!!def.dateColumn),
      boolText_(!!def.hideColumn),
      def.numberFormat || ""
    ].join("~"));
  });


  return parts.join("|");
}

function normalizeTemplateFormatSignature_(signature) {
  // Keep signature comparisons stable across script-version-only updates.
  // A new script version should not force expensive template rebuilds when dashboard
  // structure, headers, widths, hidden columns, and behavior did not change.
  return String(signature || "")
    .split("|")
    .filter(function(part) { return String(part || "").indexOf("v=") !== 0; })
    .join("|");
}

function getTemplateFormatSignatureKey_(sheetDef) {
  return "RFF_TEMPLATE_SIGNATURE_" + String(sheetDef.templateName || sheetDef.sheetType || "")
    .replace(/[^A-Za-z0-9_]/g, "_")
    .toUpperCase();
}

function getStoredTemplateFormatSignature_(sheetDef) {
  try {
    return PropertiesService.getDocumentProperties().getProperty(getTemplateFormatSignatureKey_(sheetDef)) || "";
  } catch (err) {
    Logger.log("Template signature read skipped for " + sheetDef.templateName + ": " + err.message);
    return "";
  }
}

function getStoredTemplateFormatSignatureFromSheet_(sheet) {
  if (!sheet) return "";


  try {
    const lastCol = Math.max(sheet.getLastColumn(), 1);
    const notes = sheet.getRange(1, 1, 1, lastCol).getNotes()[0];
    for (let i = 0; i < notes.length; i++) {
      const note = String(notes[i] || "");
      const match = note.match(/Format Signature:\s*([^\n]+)/);
      if (match && match[1]) return match[1].trim();
    }
  } catch (err) {
    Logger.log("Template signature note read skipped for " + sheet.getName() + ": " + err.message);
  }


  return "";
}

function storeTemplateFormatSignature_(sheetDef, signature) {
  try {
    PropertiesService.getDocumentProperties().setProperty(getTemplateFormatSignatureKey_(sheetDef), signature || "");
  } catch (err) {
    Logger.log("Template signature store skipped for " + sheetDef.templateName + ": " + err.message);
  }
}

function ensureTemplateFilter_(sheet, headerRow, rowCount, colCount, sheetDef, timing) {
  const expectedRows = Math.max(rowCount - headerRow + 1, 1);
  let existing = null;
  const correctedCreatePath = shouldUseCorrectedIndividualTemplateBuild_(sheetDef);


  try {
    existing = sheet.getFilter();
  } catch (err) {
    Logger.log("Template filter lookup skipped for " + sheet.getName() + ": " + err.message);
  }


  if (existing) {
    try {
      const range = existing.getRange();
      const matches = range.getRow() === headerRow &&
        range.getColumn() === 1 &&
        range.getNumRows() === expectedRows &&
        range.getNumColumns() === colCount;


      if (RFF_FAST_TEMPLATE_REFRESH && matches) {
        markFrameworkStep_(timing, "Filter already correct - skipped recreation: " + sheetDef.templateName);
        return;
      }
    } catch (err) {
      Logger.log("Template filter comparison skipped for " + sheet.getName() + ": " + err.message);
    }


    try {
      existing.remove();
    } catch (err) {
      Logger.log("Template filter removal skipped for " + sheet.getName() + ": " + err.message);
    }
    markFrameworkStep_(timing, "Remove existing filter: " + sheetDef.templateName);
  } else if (!correctedCreatePath) {
    markFrameworkStep_(timing, "No existing filter to remove: " + sheetDef.templateName);
  }


  try {
    sheet.getRange(headerRow, 1, expectedRows, colCount).createFilter();
  } catch (err) {
    Logger.log("Template filter creation skipped for " + sheet.getName() + ": " + err.message);
  }
  markFrameworkStep_(timing, "Create filter: " + sheetDef.templateName);
}

function applyTemplateFreezeAndTabColor_(sheet, dashboard, sheetDef, colCount, timing) {
  const globals = dashboard.globals;
  const expectedFrozenRows = globals.freezeRows;
  const expectedFrozenCols = Math.min(globals.freezeColumns, colCount);
  const expectedTabColor = getThemeColorsFromBase_(sheetDef.baseColor).level1;


  try {
    if (!(RFF_FAST_TEMPLATE_REFRESH && sheet.getFrozenRows() === expectedFrozenRows)) {
      sheet.setFrozenRows(expectedFrozenRows);
    }
  } catch (err) {
    Logger.log("Frozen rows skipped for " + sheetDef.templateName + ": " + err.message);
  }
  markFrameworkStep_(timing, "Ensure frozen rows: " + sheetDef.templateName);


  try {
    if (!(RFF_FAST_TEMPLATE_REFRESH && sheet.getFrozenColumns() === expectedFrozenCols)) {
      sheet.setFrozenColumns(expectedFrozenCols);
    }
  } catch (err) {
    Logger.log("Frozen columns skipped for " + sheetDef.templateName + ": " + err.message);
  }
  markFrameworkStep_(timing, "Ensure frozen columns: " + sheetDef.templateName);


  try {
    const currentColor = String(sheet.getTabColor() || "").toUpperCase();
    if (!(RFF_FAST_TEMPLATE_REFRESH && currentColor === expectedTabColor)) {
      sheet.setTabColor(expectedTabColor);
    }
  } catch (err) {
    Logger.log("Tab color skipped for " + sheetDef.templateName + ": " + err.message);
  }
  markFrameworkStep_(timing, "Ensure tab color: " + sheetDef.templateName);
}

function resizeSheet_(sheet, rowCount, colCount, options) {
  rowCount = Math.max(rowCount, 1);
  colCount = Math.max(colCount, 1);
  options = options || {};

  const currentRows = sheet.getMaxRows();
  const currentCols = sheet.getMaxColumns();
  const skipShrink = !!options.skipShrink;
  const hideExtraColumns = !!options.hideExtraColumns;

  if (currentRows < rowCount) {
    sheet.insertRowsAfter(currentRows, rowCount - currentRows);
  } else if (!skipShrink && currentRows > rowCount) {
    sheet.deleteRows(rowCount + 1, currentRows - rowCount);
  }

  if (currentCols < colCount) {
    sheet.insertColumnsAfter(currentCols, colCount - currentCols);
  } else if (!skipShrink && currentCols > colCount) {
    sheet.deleteColumns(colCount + 1, currentCols - colCount);
  } else if (skipShrink && hideExtraColumns && currentCols > colCount) {
    sheet.hideColumns(colCount + 1, currentCols - colCount);
  }
}

function resizeSheetRows_(sheet, rowCount) {
  if (Number(rowCount) <= 1000) return;

  /*
   * v1.4.33:
   * Row resizing is the dominant template-build slow step. New Google Sheets
   * usually start with 1000 rows, so deleting rows down to 500/800 creates
   * avoidable runtime. Only add rows when the sheet has fewer than the dashboard
   * target. Do not delete extra blank rows during template creation.
   */
  rowCount = Math.max(rowCount, 1);
  const currentRows = sheet.getMaxRows();


  if (currentRows < rowCount) {
    sheet.insertRowsAfter(currentRows, rowCount - currentRows);
  }
}

function resizeSheetColumns_(sheet, colCount) {
  colCount = Math.max(colCount, 1);
  const currentCols = sheet.getMaxColumns();


  if (currentCols < colCount) {
    sheet.insertColumnsAfter(currentCols, colCount - currentCols);
  } else if (currentCols > colCount) {
    sheet.deleteColumns(colCount + 1, currentCols - colCount);
  }
}

  };

function testDateFormattingForTemplate_(dashboard, sheetDef) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(sheetDef.templateName);
  const globals = dashboard.globals;
  const headers = getHeadersForSheetType_(dashboard, sheetDef.sheetType);
  const results = [];


  if (!sheet) {
    results.push({
      templateName: sheetDef.templateName,
      sheetType: sheetDef.sheetType,
      header: "",
      column: "",
      expectedFormat: "",
      actualFormats: "Template missing",
      status: "FAIL"
    });
    return results;
  }


  const dateChecks = [];
  headers.forEach(function(header, index) {
    const def = dashboard.columnDefinitions[header] || {};
    if (!def.dateColumn) return;
    const expectedFormat = getExpectedNumberFormat_(def, globals);
    const expectedSheetsFormat = getGoogleSheetsNumberFormat_(expectedFormat, true);
    dateChecks.push({
      header: header,
      column: index + 1,
      expectedFormat: expectedFormat,
      expectedSheetsFormat: expectedSheetsFormat
    });
  });


  if (dateChecks.length === 0) return results;


  if (sheet.getMaxColumns() < headers.length) {
    dateChecks.forEach(function(check) {
      results.push({
        templateName: sheetDef.templateName,
        sheetType: sheetDef.sheetType,
        header: check.header,
        column: check.column,
        expectedFormat: check.expectedFormat,
        actualFormats: "Template has " + sheet.getMaxColumns() + " column(s), expected at least " + headers.length,
        status: "FAIL"
      });
    });
    return results;
  }


  const sampleRows = Math.min(10, Math.max(sheet.getMaxRows() - globals.dataStartRow + 1, 1));
  const lastCol = Math.max(headers.length, 1);
  let allFormats = [];


  try {
    allFormats = sheet.getRange(globals.dataStartRow, 1, sampleRows, lastCol).getNumberFormats();
  } catch (err) {
    dateChecks.forEach(function(check) {
      results.push({
        templateName: sheetDef.templateName,
        sheetType: sheetDef.sheetType,
        header: check.header,
        column: check.column,
        expectedFormat: check.expectedFormat,
        actualFormats: "Read error: " + err.message,
        status: "FAIL"
      });
    });
    return results;
  }


  dateChecks.forEach(function(check) {
    let uniqueActualFormats = collectUniqueDateFormatsForCheck_(allFormats, check.column);
    let failedFormats = uniqueActualFormats.filter(function(actualFormat) {
      return !numberFormatsMatch_(actualFormat, check.expectedSheetsFormat);
    });
    let repaired = false;

    if (failedFormats.length && RFF_REPAIR_DATE_FORMATS_DURING_SECTION_G) {
      repaired = repairDateFormatForSectionG_(sheet, globals, check);
      if (repaired) {
        try {
          const repairedFormats = sheet.getRange(globals.dataStartRow, check.column, sampleRows, 1).getNumberFormats();
          uniqueActualFormats = collectUniqueDateFormatsForCheck_(repairedFormats, 1);
          failedFormats = uniqueActualFormats.filter(function(actualFormat) {
            return !numberFormatsMatch_(actualFormat, check.expectedSheetsFormat);
          });
        } catch (readErr) {
          Logger.log("Section G repaired format read skipped for " + sheetDef.templateName + " / " + check.header + ": " + readErr.message);
        }
      }
    }

    results.push({
      templateName: sheetDef.templateName,
      sheetType: sheetDef.sheetType,
      header: check.header,
      column: check.column,
      expectedFormat: check.expectedFormat,
      actualFormats: uniqueActualFormats.join(", ") + (repaired ? " (repaired)" : ""),
      status: failedFormats.length === 0 ? "PASS" : "FAIL"
    });
  });


  return results;
}

function validateTemplateFromDashboard_(dashboard, sheetDef, options) {
  options = options || {};
  const strict = !!options.strict;


  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(sheetDef.templateName);
  const issues = [];
  const globals = dashboard.globals;
  const expectedHeaders = getHeadersForSheetType_(dashboard, sheetDef.sheetType);
  const behavior = dashboard.behaviors[sheetDef.sheetType] || getDefaultBehavior_();
  const colCount = Math.max(expectedHeaders.length, 4);
  const rowCount = resolveTemplateRowCount_(sheetDef, globals);
  const expectedSignature = normalizeTemplateFormatSignature_(buildTemplateFormatSignature_(dashboard, sheetDef, expectedHeaders, rowCount, colCount, behavior));


  if (!sheet) {
    return {
      templateName: sheetDef.templateName,
      sheetType: sheetDef.sheetType,
      status: "FAIL",
      issues: "Template missing"
    };
  }


  if (expectedHeaders.length === 0) {
    issues.push("No headers defined in dashboard");
  }


  if (!strict) {
    try {
      const storedSignature = normalizeTemplateFormatSignature_(getStoredTemplateFormatSignature_(sheetDef) || getStoredTemplateFormatSignatureFromSheet_(sheet));
      if (storedSignature && storedSignature === expectedSignature && expectedHeaders.length > 0) {
        return {
          templateName: sheetDef.templateName,
          sheetType: sheetDef.sheetType,
          status: "PASS",
          issues: "OK"
        };
      }
      // Signature drift should trigger template refresh, not fail Section F structure validation.
      // Structural checks below remain authoritative for Dashboard Quality Section F.
    } catch (err) {
      issues.push("Could not compare template signature: " + err.message);
    }
  }


  try {
    const frozenRows = sheet.getFrozenRows();
    const frozenColumns = sheet.getFrozenColumns();


    if (frozenRows !== globals.freezeRows) {
      issues.push("Frozen rows expected " + globals.freezeRows + " but found " + frozenRows);
    }


    if (expectedHeaders.length > 0 && frozenColumns > globals.freezeColumns) {
      issues.push("Frozen columns expected " + globals.freezeColumns + " or less but found " + frozenColumns);
    }
  } catch (err) {
    issues.push("Could not check frozen settings: " + err.message);
  }



  try {
    const actualHeaders = sheet.getRange(globals.headerRow, 1, 1, Math.max(expectedHeaders.length, 1))
      .getValues()[0]
      .map(normalizeHeader_);


    expectedHeaders.forEach(function(expected, index) {
      if (actualHeaders[index] !== expected) {
        issues.push("Header mismatch col " + (index + 1) + ": expected " + expected + ", found " + actualHeaders[index]);
      }
    });
  } catch (err) {
    issues.push("Could not check headers: " + err.message);
  }


  try {
    expectedHeaders.forEach(function(header, index) {
      const def = dashboard.columnDefinitions[header] || {};
      if (def.width && sheet.getColumnWidth(index + 1) !== def.width) {
        issues.push("Width mismatch for " + header);
      }
    });
  } catch (err) {
    issues.push("Could not check column widths: " + err.message);
  }


  try {
    expectedHeaders.forEach(function(header, index) {
      const def = dashboard.columnDefinitions[header] || {};
      const expectedFormat = getExpectedNumberFormat_(def, globals);
      if (!expectedFormat) return;


      const expectedSheetsFormat = getGoogleSheetsNumberFormat_(expectedFormat, def.dateColumn);
      const sampleRows = Math.min(10, Math.max(sheet.getMaxRows() - globals.dataStartRow + 1, 1));
      const actualFormats = sheet.getRange(globals.dataStartRow, index + 1, sampleRows, 1).getNumberFormats();
      const mismatches = [];


      actualFormats.forEach(function(row) {
        const actualFormat = row[0];
        if (!numberFormatsMatch_(actualFormat, expectedSheetsFormat)) {
          if (mismatches.indexOf(actualFormat) === -1) mismatches.push(actualFormat);
        }
      });


      if (mismatches.length > 0) {
        issues.push("Date/number format mismatch for " + header + " expected " + expectedFormat + " but found " + mismatches.join(", "));
      }
    });
  } catch (err) {
    issues.push("Could not check date/number formats: " + err.message);
  }


  try {
    if (behavior.usesFilter && !sheet.getFilter()) {
      issues.push("Filter missing");
    }
  } catch (err) {
    issues.push("Could not check filter: " + err.message);
  }


  return {
    templateName: sheetDef.templateName,
    sheetType: sheetDef.sheetType,
    status: issues.length === 0 ? "PASS" : "FAIL",
    issues: issues.length === 0 ? "OK" : issues.join("; ")
  };
}

function auditDashboardConfiguration_(dashboard) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(RFF_DASHBOARD_SHEET);
  if (!sheet) {
    return [{ area: "Dashboard", item: RFF_DASHBOARD_SHEET, status: "FAIL", notes: "Format Dashboard sheet is missing" }];
  }


  const results = [];


  function add(area, item, status, notes) {
    results.push({ area: area, item: item, status: status, notes: notes || "OK" });
  }


  const sheetRows = readDashboardSectionRows_(sheet, RFF_SECTION_SHEETS);
  const headerRows = readDashboardSectionRows_(sheet, RFF_SECTION_HEADERS);
  const columnRows = readDashboardSectionRows_(sheet, RFF_SECTION_COLUMNS);
  const behaviorRows = readDashboardSectionRows_(sheet, RFF_SECTION_BEHAVIORS);


  const sheetTypeSeen = {};
  sheetRows.forEach(function(row) {
    const sheetType = normalizeHeader_(row[0]);
    const templateName = String(row[2] || "").trim();
    if (!sheetType) return;
    if (sheetTypeSeen[sheetType]) add("Duplicate Sheet Definition", sheetType, "FAIL", "Sheet type appears more than once");
    sheetTypeSeen[sheetType] = true;
    if (!templateName) add("Sheet Definition", sheetType, "FAIL", "Template Name is blank");
  });


  const behaviorSeen = {};
  behaviorRows.forEach(function(row) {
    const sheetType = normalizeHeader_(row[0]);
    if (!sheetType) return;
    if (behaviorSeen[sheetType]) add("Duplicate Behavior", sheetType, "FAIL", "Behavior row appears more than once");
    behaviorSeen[sheetType] = true;
  });


  Object.keys(sheetTypeSeen).forEach(function(sheetType) {
    if (!dashboard.sheetHeaders[sheetType] || dashboard.sheetHeaders[sheetType].length === 0) {
      add("Missing Headers", sheetType, "FAIL", "No headers defined for sheet type");
    }
    if (!behaviorSeen[sheetType]) {
      add("Missing Behavior", sheetType, "FAIL", "No Sheet Behaviors row found for sheet type");
    }
  });


  const columnDefSeen = {};
  columnRows.forEach(function(row) {
    const header = normalizeHeader_(row[0]);
    if (!header) return;
    if (columnDefSeen[header]) add("Duplicate Column Definition", header, "FAIL", "Column definition appears more than once");
    columnDefSeen[header] = true;


    const width = isBlankCell_(row[1]) ? null : Number(row[1]);
    const fontSize = isBlankCell_(row[2]) ? null : Number(row[2]);
    const dateColumn = parseBoolean_(row[3]);
    const numberFormat = String(row[8] || "").trim();


    if (width !== null && (isNaN(width) || width < 1)) add("Invalid Width", header, "FAIL", "Width override must be a positive number when populated; blank inherits Default Column Width");
    if (fontSize !== null && (isNaN(fontSize) || fontSize < 1)) add("Invalid Font Size", header, "FAIL", "Header Font Size override must be a positive number when populated");
    if (dateColumn && numberFormat && !isDateFormatText_(numberFormat)) add("Invalid Date Format", header, "FAIL", "Date Column is TRUE but format does not look like a date format");
  });


  headerRows.forEach(function(row) {
    const sheetType = normalizeHeader_(row[0]);
    const header = normalizeHeader_(row[2]);
    if (!sheetType || !header) return;
    if (!columnDefSeen[header]) add("Missing Column Definition", sheetType + " / " + header, "FAIL", "Header is used by a sheet but has no Column Definitions row");
  });


  if (!results.some(function(item) { return item.status !== "PASS"; })) {
    add("Dashboard Audit", "Overall", "PASS", "No dashboard configuration issues found");
  }


  return results;
}

function writeCombinedDashboardReportSection_(sheet, row, sectionTitle, sourceSheetName) {
  sheet.getRange(row, 1, 1, 8).merge().setValue(sectionTitle).setFontWeight("bold").setBackground("#B7DEE8");
  row++;


  const compactRows = getCompactReportRowsForDashboard_(sourceSheetName);
  if (!compactRows.length) {
    sheet.getRange(row, 1, 1, 2).setValues([[sourceSheetName, "Report not found or empty"]]);
    return row + 2;
  }


  const width = Math.min(Math.max.apply(null, compactRows.map(function(r) { return r.length; })), 8);
  const normalized = compactRows.map(function(r) {
    const out = r.slice(0, width);
    while (out.length < width) out.push("");
    return out;
  });


  sheet.getRange(row, 1, normalized.length, width).setValues(normalized);
  if (normalized.length > 1) sheet.getRange(row, 1, 1, width).setFontWeight("bold").setBackground("#D9EAF7");
  return row + normalized.length + 2;
}

function getCompactReportRowsForDashboard_(sourceSheetName) {
  const storedRows = getDashboardQualitySectionRows_(sourceSheetName);
  if (storedRows && storedRows.length) return storedRows;


  const source = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sourceSheetName);
  if (!source) return [];


  if (sourceSheetName === RFF_VALIDATION_SHEET) {
    return getCompactValidationRows_(source);
  }
  if (sourceSheetName === RFF_TEST6_DATE_FORMAT_SHEET) {
    return getCompactStatusRows_(source, 7, ["Template", "Sheet Type", "Header", "Expected", "Actual", "Status", "Issues"], 12);
  }
  if (sourceSheetName === RFF_TEST7_VALIDATION_FAILURE_SHEET) {
    return getCompactStatusRows_(source, 7, ["Template", "Sheet Type", "Header", "Expected", "Forced", "Status", "Notes"], 8);
  }
  if (sourceSheetName === RFF_TEST9_SUBHEADER_SHEET) {
    return getCompactTest9Rows_(source);
  }
  if (sourceSheetName === RFF_TEST10_DASHBOARD_AUDIT_SHEET) {
    return getCompactStatusRows_(source, 3, ["Area", "Item", "Status", "Quality Notes"], 10);
  }


  return getCompactGenericRows_(source, 12);
}

function writeCombinedDashboardSignOff_(sheet, row) {
  const summary = buildFrameworkSummaryRows_();
  const overall = summary[0];
  sheet.getRange(row, 1, 1, 4).merge().setValue("SECTION H - FRAMEWORK SIGN-OFF").setFontWeight("bold").setBackground("#B7DEE8");
  row++;
  sheet.getRange(row, 1, 4, 2).setValues([
    ["Framework Status", overall[1]],
    ["Ready for Phase 2", overall[1] === "READY FOR PHASE 2" ? "YES" : "NO"],
    ["Remaining Action", overall[2]],
    ["Testing Required Before Production", "YES"]
  ]);
  return row + 6;
}

function styleCombinedFrameworkTestDashboard_(sheet) {
  applyDashboardQualityReportColumnSettings_(sheet);

  try {
    const lastRow = Math.max(sheet.getLastRow(), 1);
    const lastCol = Math.min(Math.max(sheet.getLastColumn(), 1), RFF_DASHBOARD_QUALITY_COL_WIDTHS.length);
    sheet.getRange(1, 1, lastRow, lastCol)
      .setFontFamily("Arial")
      .setFontSize(10)
      .setFontColor("#000000")
      .setVerticalAlignment("middle");
    if (sheet.getFrozenRows() !== 1) sheet.setFrozenRows(1);
  } catch (err) {
    Logger.log("Dashboard Quality styling skipped: " + err.message);
  }
}
