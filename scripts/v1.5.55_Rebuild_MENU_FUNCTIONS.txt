// -----------------------------------------------------------------------------
// Apps Script open trigger
// -----------------------------------------------------------------------------
function onOpen() {
  const ui = SpreadsheetApp.getUi();

  const startUpMenu = ui.createMenu("Start-Up")
    .addItem("Start Up / Verify Framework Configuration", "verifyFrameworkConfiguration")
    .addItem("Rebuild Format Dashboard Defaults", "setupReportFormattingDashboardFromScriptDefaults_")
    .addItem("Dashboard Quality Start up", "runDashboardQualityStartUp")
    .addItem("Create / Refresh All Templates", "createOrRefreshAllReportTemplates")
    .addItem("Dashboard Quality Validate Templates", "runDashboardQualityValidateTemplates");
  const setupMenu = ui.createMenu("Setup")
    .addItem("Start Up / Verify Framework Configuration", "verifyFrameworkConfiguration")
    .addItem("Rebuild Format Dashboard Defaults", "setupReportFormattingDashboardFromScriptDefaults_")
    .addItem("Create / Refresh All Templates", "createOrRefreshAllReportTemplates");

  const qualityMenu = ui.createMenu("Quality")
    .addItem("Dashboard Quality Start up", "runDashboardQualityStartUp")
    .addItem("Dashboard Quality Validate Templates", "runDashboardQualityValidateTemplates")
    .addItem("Dashboard Quality Workflow", "runDashboardQualityFull");

  const workflowFormatMenu = ui.createMenu("Format")
    .addItem("Format Banner Report", "formatBannerReport")
    .addItem("Format Care Plan Due", "formatCarePlanDueReport")
    .addItem("Format Unlocked Care Plan", "formatUnlockedCarePlanReport")
    .addItem("Format Raw Data", "formatRawData");

  const workflowCreateMenu = ui.createMenu("Create")
    .addItem("Create Disenrolled List", "createDisenrolledList")
    .addItem("Process Demo P", "processDemoP")
    .addItem("Create Master List", "createMasterList");

  const workflowUpdateMenu = ui.createMenu("Update")
    .addItem("Monthly Change", "buildMonthlyChangeReportFromDemoP")
    .addItem("Master List Change Log", "runUpdateMasterListMergeAudit");

  const workflowMenu = ui.createMenu("Workflow")
    .addSubMenu(workflowFormatMenu)
    .addSubMenu(workflowCreateMenu)
    .addSubMenu(workflowUpdateMenu)
    .addItem("Dashboard Quality Workflow", "runDashboardQualityFull");

  const templateManagementMenu = ui.createMenu("Templates")
    .addItem("Hide Templates", "hideReportTemplates")
    .addItem("Show Templates", "showReportTemplates");

  const systemSheetMenu = ui.createMenu("System Sheets")
    .addItem("Show System Sheets", "showSystemSheetsNow")
    .addItem("Hide System Sheets", "hideSystemSheetsNow");

  const sheetManagementMenu = ui.createMenu("Sheet Management")
    .addSubMenu(templateManagementMenu)
    .addSubMenu(systemSheetMenu);

  ui.createMenu("Master List Tools")
    .addSubMenu(startUpMenu)
    .addSubMenu(setupMenu)
    .addSubMenu(qualityMenu)
    .addSubMenu(workflowMenu)
    .addSubMenu(sheetManagementMenu)
    .addToUi();
}
