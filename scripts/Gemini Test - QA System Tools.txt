function combineNotesSummaryData_(data, pmrFilter) {
  if (!data || !data.values || !data.values.length) return;


  const headerMap = data.headerMap;
  const notesIdx = headerMap["Notes"];
  if (notesIdx === undefined) return;


  const sourceHeaders = [
    "Banner Summary",
    "Contact - Summary",
    "Additional Important Information"
  ];


  data.values.forEach(row => {
    if (!shouldProcessRowByPMR_(row, headerMap, pmrFilter)) return;


    const parts = [];


    sourceHeaders.forEach(header => {
      const idx = headerMap[header];
      if (idx === undefined) return;


      const value = String(row[idx] || "").trim();
      if (value) parts.push(value);
    });


    row[notesIdx] = parts.join("\n\n");
  });
}

function isSystemOrTestingSheet_(sheet) {
  if (!sheet) return false;


  const sheetName = String(sheet.getName() || "").trim();
  const lower = sheetName.toLowerCase();
  const explicitNames = getSystemAndTestingSheetNames_();


  // Do not treat Template - sheets as system sheets. Templates have their own
  // separate show/hide menu actions and triggers.
  return explicitNames.has(sheetName) ||
    lower.indexOf("framework test") !== -1 ||
    lower.indexOf("template validation") !== -1 ||
    lower.indexOf("framework health check") !== -1 ||
    lower.indexOf("dashboard audit") !== -1 ||
    lower.endsWith(" audit") ||
    lower.endsWith(" validation") ||
    lower.endsWith(" diagnostics") ||
    lower.endsWith(" health check") ||
    lower.indexOf("timing") !== -1;
}

function hideSystemSheets_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let hiddenCount = 0;


  ss.getSheets().forEach(sheet => {
    if (!isSystemOrTestingSheet_(sheet)) return;


    try {
      if (!sheet.isSheetHidden()) {
        sheet.hideSheet();
        hiddenCount++;
      }
    } catch (err) {
      Logger.log("Could not hide system/testing sheet " + sheet.getName() + ": " + err.message);
    }
  });


  return hiddenCount;
}

function runUpdateMasterListMergeAudit() {
  const monthParts = promptForLockedYearReportMonth_("Update Master List Merge Audit");
  if (!monthParts) return;


  const secondarySheet = buildUpdateMasterListMergeAuditForMonth_(monthParts);
  const primarySheet = buildUpdateMasterListPrimaryRowAuditForMonth_(monthParts);


  if (secondarySheet) secondarySheet.showSheet();
  if (primarySheet) {
    primarySheet.showSheet();
    primarySheet.activate();
  } else if (secondarySheet) {
    secondarySheet.activate();
  }


  if (secondarySheet || primarySheet) {
    notify_("Update Master List Merge Audit complete. Review the secondary-row audit and primary-row audit sheets.");
  }
}

function buildUpdateMasterListMergeAuditForMonth_(monthParts) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const currentDemo = getCurrentDemoPSheet_(monthParts);
  const sourceMaster = getPreviousMasterListSheet_(monthParts);


  if (!currentDemo) {
    notify_("Current month Demo P was not found. Format Demo P first.");
    return null;
  }


  if (!sourceMaster) {
    notify_("Previous month Master List was not found for merge audit.");
    return null;
  }


  const reportSheet = getMonthlySheetByPrefixAndDate_(
    ss,
    MONTHLY_CHANGE_REPORT_PREFIX,
    monthParts.firstDay,
    monthParts.lastDay
  );


  if (!reportSheet) {
    notify_("Monthly Change Report was not found. Create Monthly Change Report first.");
    return null;
  }


  const changedInfo = getChangedPMRsFromMonthlyChangeReport_(reportSheet);
  const contactChangePMRs = Array.from(changedInfo.keys())
    .filter(pmr => changedInfo.get(pmr) && changedInfo.get(pmr).has("Contact Changes"))
    .sort();


  const demoData = getDataValues_(currentDemo, HEADER_ROW, DATA_START_ROW);
  const masterData = getDataValues_(sourceMaster, HEADER_ROW, DATA_START_ROW);


  const demoBlocks = buildMergeRowsByPMRFromData_(demoData);
  const masterBlocks = buildMergeRowsByPMRFromData_(masterData);


  const summaryRows = [];
  const detailRows = [];


  contactChangePMRs.forEach(pmr => {
    const demoRows = demoBlocks.get(pmr) || [];
    const masterRows = masterBlocks.get(pmr) || [];
    const participantName = getMergeAuditParticipantName_(demoRows, demoData.headerMap, masterRows, masterData.headerMap);


    const demoKeys = buildSecondaryMergeKeyMapForRows_(demoRows, demoData.headerMap, pmr);
    const masterKeys = buildSecondaryMergeKeyMapForRows_(masterRows, masterData.headerMap, pmr);


    const contactKeys = new Set(
      [].concat(Array.from(demoKeys.keys()), Array.from(masterKeys.keys()))
    );


    let matchedCount = 0;
    let updateCount = 0;
    let addCount = 0;
    let removeCount = 0;
    let noActionCount = 0;


    contactKeys.forEach(key => {
      const demoItem = demoKeys.get(key);
      const masterItem = masterKeys.get(key);
      const masterSummary = masterItem ? buildMergeAuditContactSummary_(masterItem.values, masterData.headerMap) : "";
      const demoSummary = demoItem ? buildMergeAuditContactSummary_(demoItem.values, demoData.headerMap) : "";


      let auditAction = "";
      let differenceFound = "";
      let recommendedAction = "";


      if (demoItem && masterItem) {
        matchedCount++;
        const changedFields = getMergeAuditChangedFields_(masterItem.values, masterData.headerMap, demoItem.values, demoData.headerMap);


        if (changedFields.length > 0) {
          auditAction = "Update";
          differenceFound = "Contact exists in both; changed fields: " + changedFields.join(", ");
          recommendedAction = "Update existing Master List contact row";
          updateCount++;
        } else {
          auditAction = "No action";
          differenceFound = "Contact exists in both; no compared field differences";
          recommendedAction = "No Master List contact row change needed";
          noActionCount++;
        }
      } else if (demoItem && !masterItem) {
        auditAction = "Add";
        differenceFound = "Contact exists in Demo P only";
        recommendedAction = "Add new contact row to Master List";
        addCount++;
      } else if (!demoItem && masterItem) {
        auditAction = "Remove";
        differenceFound = "Contact exists in previous Master List only";
        recommendedAction = "Remove contact row from Master List";
        removeCount++;
      }


      detailRows.push([
        pmr,
        participantName.lastName,
        participantName.firstName,
        auditAction,
        differenceFound,
        recommendedAction,
        masterItem ? masterItem.rowNumber : "",
        demoItem ? demoItem.rowNumber : "",
        "Secondary/Contact",
        "Yes",
        masterSummary,
        demoSummary,
        key
      ]);
    });


    let recommendation = "No contact row change needed";
    if (addCount || removeCount || updateCount) {
      const pieces = [];
      if (updateCount) pieces.push(updateCount + " update");
      if (addCount) pieces.push(addCount + " add");
      if (removeCount) pieces.push(removeCount + " remove");
      recommendation = pieces.join(", ");
    }


    summaryRows.push([
      pmr,
      participantName.lastName,
      participantName.firstName,
      masterRows.length,
      demoRows.length,
      matchedCount,
      updateCount,
      addCount,
      removeCount,
      noActionCount,
      recommendation
    ]);
  });


  const auditName = "Update Master List Merge Audit";
  let audit = ss.getSheetByName(auditName);
  if (audit) ss.deleteSheet(audit);
  audit = ss.insertSheet(auditName);


  audit.getRange("A1").setValue(auditName);
  audit.getRange("A2").setValue("Date");
  audit.getRange("B2").setValue(monthParts.firstDay);

  audit.getRange("D2").setValue(monthParts.lastDay);

  audit.getRange("E3").setValue("Source Demo P");
  audit.getRange("F3").setValue(currentDemo.getName());

    "PMR #",
    "Last Name",
    "First Name",
    "Master Row Count",
    "Demo P Row Count",
    "Contacts Matched",
    "Contacts To Update",
    "Contacts To Add",
    "Contacts To Remove",
    "Contacts With No Action",
    "Merge Recommendation"
  ];


  audit.getRange(5, 1, 1, summaryHeaders.length).setValues([summaryHeaders]);

    "Last Name",
    "First Name",

    "Demo P Contact Summary",
    "Contact Match Key"

  if (detailRows.length) {

  }

    title: auditName,
    startDate: monthParts.firstDay,
    endDate: monthParts.lastDay,
    renameSheet: false

// v1.1.0 removed orphaned legacy formatting statement.
  return audit;

function formatMergeAuditValueForDisplay_(value) {
  if (value instanceof Date && !isNaN(value.getTime())) {
    return formatDateDisplay_(value);
  }


  if (value === null || value === undefined) return "";
  return String(value);
}

function getMergeAuditParticipantName_(demoRows, demoHeaderMap, masterRows, masterHeaderMap) {
  const demoName = getMergeAuditParticipantNameFromRows_(demoRows, demoHeaderMap);
  if (demoName.lastName || demoName.firstName) return demoName;
  return getMergeAuditParticipantNameFromRows_(masterRows, masterHeaderMap);
}

function getMergeAuditParticipantNameFromRows_(rows, headerMap) {
  const lastIdx = headerMap ? headerMap["Last Name"] : undefined;
  const firstIdx = headerMap ? headerMap["First Name"] : undefined;
  const primaryIdx = headerMap ? headerMap["Primary PMR Row"] : undefined;


  const primaryItem = (rows || []).find(item => {
    const row = item.values || [];
    return primaryIdx !== undefined && String(row[primaryIdx] || "").trim().toLowerCase() === "yes";
  }) || (rows || [])[0];


  const row = primaryItem && primaryItem.values ? primaryItem.values : [];


  return {
    lastName: lastIdx === undefined ? "" : String(row[lastIdx] || "").trim(),
    firstName: firstIdx === undefined ? "" : String(row[firstIdx] || "").trim()
  };
}

function buildMergeAuditContactSummary_(row, headerMap) {
  if (!row || !headerMap) return "";


  const parts = [];
  const contactLast = getMergeRowValue_(row, headerMap, ["Contact - Last Name"]);
  const contactFirst = getMergeRowValue_(row, headerMap, ["Contact - First Name"]);
  const relationship = getMergeRowValue_(row, headerMap, ["Relationship"]);
  const type = getMergeRowValue_(row, headerMap, ["Type of Contact"]);
  const phone = getMergeRowValue_(row, headerMap, ["AD1 - Phone", "Phone Number", "Phone 1 - Value"]);
  const notes = getMergeRowValue_(row, headerMap, ["Contact - Notes"]);


  const name = [contactFirst, contactLast].map(v => String(v || "").trim()).filter(Boolean).join(" ");
  if (name) parts.push(name);
  if (relationship) parts.push("Relationship: " + relationship);
  if (type) parts.push("Type: " + type);
  if (phone) parts.push("Phone: " + phone);
  if (notes) parts.push("Notes: " + notes);


  return parts.join(" | ");
}

function getMergeAuditChangedFields_(masterRow, masterHeaderMap, demoRow, demoHeaderMap) {
  const headersToCompare = [
    "Contact - Last Name",
    "Contact - First Name",
    "Type of Contact",
    "Contact - Primary Language",
    "Relationship",
    "AD1 - Phone",
    "AD1 - Phone Valid Dates From",
    "AD1 - Phone Valid Dates To",
    "AD2 - Phone",
    "AD2 - Phone Valid Dates From",
    "AD2 - Phone Valid Dates To",
    "AD3 - Phone",
    "AD3 - Phone Valid Dates From",
    "AD3 - Phone Valid Dates To",
    "Company",
    "Contact - Notes"
  ];


  return headersToCompare.filter(header => {
    const masterIdx = masterHeaderMap[header];
    const demoIdx = demoHeaderMap[header];
    if (masterIdx === undefined && demoIdx === undefined) return false;


    const masterValue = masterIdx === undefined ? "" : masterRow[masterIdx];
    const demoValue = demoIdx === undefined ? "" : demoRow[demoIdx];
    return !valuesAreEqual_(masterValue, demoValue);
  });
}

function writeContactSummaryToRows_(data, headerMap, pmrFilter) {
  const summaryIdx = headerMap["Contact - Summary"];
  const bannerIdx = headerMap["Banner Summary"];
  const contactNotesIdx = headerMap["Contact - Notes"];


  if (summaryIdx === undefined) return;


  const contactColIdxs = [1, 2, 3, 4, 5, 6, 7, 8]
    .map(num => headerMap[`Contact - ${num}`])
    .filter(idx => idx !== undefined);


  data.forEach(row => {
    if (!shouldProcessRowByPMR_(row, headerMap, pmrFilter)) return;


    const parts = [];


    if (bannerIdx !== undefined && String(row[bannerIdx] || "").trim()) {
      parts.push(String(row[bannerIdx]).trim());
    }


    contactColIdxs.forEach(idx => {
      const value = String(row[idx] || "").trim();
      if (value) parts.push(value);
    });


    if (contactNotesIdx !== undefined && String(row[contactNotesIdx] || "").trim()) {
      parts.push(String(row[contactNotesIdx]).trim());
    }


    row[summaryIdx] = parts.join("\n");
  });
}

function combineNotesSummary(sheet, pmrFilter) {
  sheet = sheet || getActiveMasterListOrSheet_();


  const lastRow = sheet.getLastRow();
  if (lastRow < DATA_START_ROW) return;


  ensureHeaders_(sheet, ["Notes"], HEADER_ROW);


  const headers = getHeaders_(sheet, HEADER_ROW);
  const headerMap = buildHeaderIndexMap_(headers);


  const notesIdx = headerMap["Notes"];
  if (notesIdx === undefined) return;


  const sourceHeaders = [
    "Banner Summary",
    "Contact - Summary",
    "Additional Important Information"
  ];


  const dataRange = sheet.getRange(DATA_START_ROW, 1, lastRow - DATA_START_ROW + 1, sheet.getLastColumn());
  const data = dataRange.getValues();


  data.forEach(row => {
    if (!shouldProcessRowByPMR_(row, headerMap, pmrFilter)) return;


    const parts = [];


    sourceHeaders.forEach(header => {
      const idx = headerMap[header];
      if (idx === undefined) return;


      const value = String(row[idx] || "").trim();
      if (value) parts.push(value);
    });


    row[notesIdx] = parts.join("\n\n");
  });


  dataRange.setValues(data);
}

function collectMasterListHealthCheckRows_(masterSheet) {
  const outputRows = [["Check", "Status", "Count", "Quality Notes"]];

  if (!masterSheet) {
    outputRows.push(["Master List Sheet", "FAIL", 0, "Master List sheet was not found."]);
    return outputRows;
  }

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const data = getDataValues_(masterSheet, HEADER_ROW, DATA_START_ROW);
  const rows = data.values || [];
  const headerMap = data.headerMap || {};
  const pmrIdx = getPMRIndex_(headerMap);
  const bannerIdx = headerMap["Banner Summary"];

  const carePlanHeaders = [
    "Participant Name",
    "IDT Meeting Date",
    "Care Plan Start Date",
    "Enrollment Date",
    "Last Care Plan",
    "Next Care Plan Due",
    "CP Type"
  ];

  const detailRows = [];
  const byPMR = new Map();
  const duplicatePMRs = [];
  let rowsMissingPMR = 0;
  let rowsMissingBannerSummary = 0;
  let rowsMissingCarePlanFields = 0;

  rows.forEach(function(row, offset) {
    const rowNumber = DATA_START_ROW + offset;
    const pmr = pmrIdx === -1 ? "" : normalizePMR_(row[pmrIdx]);

    if (!pmr) {
      rowsMissingPMR++;
      detailRows.push(["Rows Missing PMR Detail", "FAIL", rowNumber, "Participant PMR# / PMR # is blank."]);
      return;
    }

    if (!byPMR.has(pmr)) byPMR.set(pmr, []);
    byPMR.get(pmr).push({ rowNumber: rowNumber, row: row });

    if (bannerIdx !== undefined && String(row[bannerIdx] || "").trim() === "") {
      rowsMissingBannerSummary++;
      detailRows.push(["Missing Banner Summary", "WARNING", pmr, "Master List row " + rowNumber + " has no Banner Summary."]);
    }

    const missingCareFields = carePlanHeaders.filter(function(header) {
      const idx = headerMap[header];
      return idx !== undefined && String(row[idx] || "").trim() === "";
    });

    if (missingCareFields.length > 0) {
      rowsMissingCarePlanFields++;
      detailRows.push(["Missing Care Plan Field(s)", "WARNING", pmr, missingCareFields.join(", ")]);
    }
  });

  byPMR.forEach(function(items, pmr) {
    if (items.length > 1) {
      duplicatePMRs.push(pmr);
      detailRows.push(["Duplicate PMR", "WARNING", pmr, "Rows: " + items.map(function(item) { return item.rowNumber; }).join(", ")]);
    }
  });

  const masterSuffix = getWorkflowMonthSuffixFromSheet_(masterSheet);
  const demoSheet = masterSuffix
    ? findMonthlyOutputSheetByPrefixAndSuffix_(ss, DEMO_P_PREFIX, masterSuffix)
    : getNewestFormattedMonthlySheetByPrefix_(ss, DEMO_P_PREFIX);

  let demoPrimaryPMRs = new Set();
  if (demoSheet) {
    const demoData = getDataValues_(demoSheet, HEADER_ROW, DATA_START_ROW);
    const demoPMRIdx = getPMRIndex_(demoData.headerMap || {});
    const demoPrimaryIdx = (demoData.headerMap || {})["Primary PMR Row"];
    (demoData.values || []).forEach(function(row) {
      if (demoPrimaryIdx !== undefined && String(row[demoPrimaryIdx] || "").trim().toLowerCase() !== "yes") return;
      const pmr = demoPMRIdx === -1 ? "" : normalizePMR_(row[demoPMRIdx]);
      if (pmr) demoPrimaryPMRs.add(pmr);
    });
  }

  const masterPMRs = new Set(Array.from(byPMR.keys()));
  const demoMasterMatches = countSetMatches_(masterPMRs, demoPrimaryPMRs);

  outputRows.push(["Master List Sheet", "PASS", 1, masterSheet.getName()]);
  outputRows.push(["Master List Month", masterSuffix ? "PASS" : "WARNING", masterSuffix || "Unknown", "Primary PMR validation is not performed here; Primary PMR is owned by Demo P Section M."]);
  outputRows.push(["Total Master List Rows", "INFO", rows.length, masterSheet.getName()]);
  outputRows.push(["Unique PMRs", "INFO", byPMR.size, ""]);
  outputRows.push(["Rows Missing PMR", rowsMissingPMR === 0 ? "PASS" : "FAIL", rowsMissingPMR, ""]);
  outputRows.push(["Duplicate PMRs", duplicatePMRs.length === 0 ? "PASS" : "WARNING", duplicatePMRs.length, ""]);
  outputRows.push(["Rows Missing Banner Summary", rowsMissingBannerSummary === 0 ? "PASS" : "WARNING", rowsMissingBannerSummary, ""]);
  outputRows.push(["Rows Missing Care Plan Fields", rowsMissingCarePlanFields === 0 ? "PASS" : "WARNING", rowsMissingCarePlanFields, ""]);
  outputRows.push(["Master List / Demo P Reconciliation", demoSheet && demoMasterMatches === masterPMRs.size ? "PASS" : "WARNING", demoMasterMatches, "Master PMRs: " + masterPMRs.size + "; Demo P primary PMRs: " + demoPrimaryPMRs.size + (demoSheet ? "; Demo P: " + demoSheet.getName() : "; Demo P not found")]);

  if (detailRows.length) {
    outputRows.push(["Detail", "INFO", detailRows.length, "Detailed issues follow."]);
    detailRows.slice(0, 100).forEach(function(row) {
      outputRows.push(row);
    });
  }

  return outputRows;
}

  const unlockedMatches = countSetMatches_(masterPMRKeys, unlockedInfo.keys);
  const unlockedUnmatched = Math.max(0, masterPMRKeys.size - unlockedMatches);
  summaryRows.push(["Unlocked Rows Read", "INFO", unlockedInfo.rowsRead, "Header key: PMR #"]);
  summaryRows.push(["Unlocked Unique PMR Keys", "INFO", unlockedInfo.keys.size, ""]);
  summaryRows.push(["Unlocked Matches To Master List", unlockedMatches > 0 ? "PASS" : "WARNING", unlockedMatches, "Master PMR keys: " + masterPMRKeys.size]);
  summaryRows.push(["Unlocked Unmatched Master PMRs", unlockedUnmatched === 0 ? "PASS" : "WARNING", unlockedUnmatched, "Master PMRs not found in Unlocked CP."]);

  const dueInfo = getDiagnosticSourceKeyInfo_(dueSheet, HEADER_ROW, DATA_START_ROW, "Participant Name", false);
  const dueMatches = countSetMatches_(masterNameKeys, dueInfo.keys);
  const dueUnmatched = Math.max(0, masterNameKeys.size - dueMatches);
  summaryRows.push(["Care Plan Due Rows Read", "INFO", dueInfo.rowsRead, "Header key: Participant Name"]);
  summaryRows.push(["Care Plan Due Unique Name Keys", "INFO", dueInfo.keys.size, ""]);
  summaryRows.push(["Care Plan Due Matches To Master List", dueMatches > 0 ? "PASS" : "WARNING", dueMatches, "Master Participant Name keys: " + masterNameKeys.size]);
  summaryRows.push(["Care Plan Due Unmatched Master Names", dueUnmatched === 0 ? "PASS" : "WARNING", dueUnmatched, "Master names not found in CP Due."]);

  unlockedInfo.missingHeaders.forEach(function(header) {
    summaryRows.push(["Unlocked Missing Header", "FAIL", header, unlockedSheet ? unlockedSheet.getName() : "Not found"]);
  });
  dueInfo.missingHeaders.forEach(function(header) {
    summaryRows.push(["Care Plan Due Missing Header", "FAIL", header, dueSheet ? dueSheet.getName() : "Not found"]);
  });

  deleteLegacyQualityReportSheet_("Care Plan Sync Diagnostics");
  return summaryRows;

function runMonthlyChangeValidation() {
  const monthParts = promptForLockedYearReportMonth_("Monthly Change Validation");
  if (!monthParts) return;


  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const reportSheet = getMonthlySheetByPrefixAndDate_(ss, MONTHLY_CHANGE_REPORT_PREFIX, monthParts.firstDay, monthParts.lastDay);


  if (!reportSheet) {
    notify_("Monthly Change Report was not found for the selected month.");
    return;
  }


  const sections = ["Enrollments", "Disenrollments", "Demographic Changes", "Caseload Changes", "Contact Changes"];
  const summaryRows = [];
  const detailRows = [];


  sections.forEach(section => {
    const titleRow = findMonthlyChangeSectionTitleRow_(reportSheet, section);
    if (titleRow === -1) {
      summaryRows.push([section, "FAIL", 0, "Section title not found."]);
      detailRows.push(["FAIL", section, "Missing section title."]);
      return;
    }


    const headerRow = titleRow + 2;
    const dataStartRow = headerRow + 1;
    const nextTitleRow = findNextMonthlyChangeSectionTitleRow_(reportSheet, titleRow);
    const dataEndRow = nextTitleRow === -1 ? reportSheet.getLastRow() : nextTitleRow - 1;


    if (dataEndRow < dataStartRow) {
      summaryRows.push([section, "PASS", 0, "No rows in section."]);
      return;
    }


    const lastCol = Math.max(reportSheet.getLastColumn(), 1);
    const headers = reportSheet.getRange(headerRow, 1, 1, lastCol).getValues()[0].map(h => String(h || "").trim());
    const headerMap = buildHeaderIndexMap_(headers);
    const pmrIdx = getPMRIndex_(headerMap);


    const values = reportSheet.getRange(dataStartRow, 1, dataEndRow - dataStartRow + 1, lastCol).getValues();
    const nonBlankRows = values.filter(row => row.some(value => String(value || "").trim() !== ""));
    const uniquePMRs = new Set();
    let missingPMR = 0;


    nonBlankRows.forEach((row, idx) => {
      const rowNumber = dataStartRow + idx;
      const pmr = pmrIdx === -1 ? "" : normalizePMR_(row[pmrIdx]);
      if (!pmr) {
        missingPMR++;
        detailRows.push(["WARNING", section, "Missing PMR on row " + rowNumber]);
      } else {
        uniquePMRs.add(pmr);
      }
    });


    summaryRows.push([section, missingPMR === 0 ? "PASS" : "WARNING", nonBlankRows.length, "Unique PMRs: " + uniquePMRs.size + "; Missing PMRs: " + missingPMR]);
  });


  writeDiagnosticReport_(
    "Monthly Change Validation",
    "Monthly Change Validation",
    monthParts,
    ["Section", "Status", "Rows", "Details"],
    summaryRows,
    ["Status", "Section", "Details"],
    detailRows
  );


  notify_("Monthly Change Validation complete.");
}

function getDiagnosticSourceKeyInfo_(sheet, headerRow, dataStartRow, keyHeader, usePMR) {
  const result = {
    rowsRead: 0,
    keys: new Set(),
    missingHeaders: []
  };


  if (!sheet) {
    result.missingHeaders.push("Source sheet not found");
    return result;
  }


  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();


  if (lastRow < dataStartRow || lastCol < 1) return result;


  const headers = getHeaders_(sheet, headerRow);
  const headerMap = buildHeaderIndexMap_(headers);
  const keyIdx = headerMap[keyHeader];


  if (keyIdx === undefined) {
    result.missingHeaders.push(keyHeader);
    return result;
  }


  const values = sheet.getRange(dataStartRow, 1, lastRow - dataStartRow + 1, lastCol).getValues();
  result.rowsRead = values.length;


  values.forEach(row => {
    const key = usePMR ? normalizePMR_(row[keyIdx]) : normalizeKeyPart_(row[keyIdx]);
    if (key) result.keys.add(key);
  });


  return result;
}

function writeDiagnosticReport_(sheetName, title, monthParts, summaryHeaders, summaryRows, detailHeaders, detailRows) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(sheetName);


  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
  } else {
    sheet.clear();
  }


  sheet;


  sheet.getRange("A1").setValue(title);
  sheet.getRange("A2").setValue("Date");
  sheet.getRange("B2").setValue(monthParts.firstDay);
  sheet.getRange("C2").setValue("to");
  sheet.getRange("D2").setValue(monthParts.lastDay);


  let row = 4;
  sheet.getRange(row, 1, 1, summaryHeaders.length).setValues([summaryHeaders]);
  sheet.getRange(row, 1, 1, summaryHeaders.length);
  row++;


  if (summaryRows && summaryRows.length > 0) {
    sheet.getRange(row, 1, summaryRows.length, summaryHeaders.length).setValues(summaryRows);
    row += summaryRows.length;
  }


  row += 2;
  sheet.getRange(row, 1).setValue("Details");
  row++;
  sheet.getRange(row, 1, 1, detailHeaders.length).setValues([detailHeaders]);
  sheet.getRange(row, 1, 1, detailHeaders.length);
  row++;


  if (detailRows && detailRows.length > 0) {
    sheet.getRange(row, 1, detailRows.length, detailHeaders.length).setValues(detailRows);
  } else {
    sheet.getRange(row, 1).setValue("No issues found.");
  }


  const lastRow = Math.max(sheet.getLastRow(), 1);
  const lastCol = Math.max(sheet.getLastColumn(), 1);


  sheet.getRange(1, 1, lastRow, lastCol);


  sheet;
  applyStandardRowHeights_(sheet);


  for (let col = 1; col <= lastCol; col++) {
    sheet;
  }


  if (sheet.getFilter()) sheet;
  sheet.getRange(4, 1, Math.max(lastRow - 3, 1), lastCol);
}

function findFrameworkTimingSectionRow_(sheet, sectionTitle) {
  if (!sheet) return 0;
  const target = String(sectionTitle || "").trim();
  const lastRow = sheet.getLastRow();
  if (lastRow < 1) return 0;

  const values = sheet.getRange(1, 1, lastRow, 1).getDisplayValues();
  for (let i = 0; i < values.length; i++) {
    if (String(values[i][0] || "").trim() === target) return i + 1;
  }
  return 0;
}

function findNextFrameworkTimingSectionRow_(sheet, startRow) {
  if (!sheet) return 0;
  const lastRow = sheet.getLastRow();
  if (startRow > lastRow) return 0;

  const values = sheet.getRange(startRow, 1, lastRow - startRow + 1, 1).getDisplayValues();
  for (let i = 0; i < values.length; i++) {
    const value = String(values[i][0] || "").trim();
    if (/^SECTION [A-Z] - /.test(value)) return startRow + i;
  }
  return 0;
}

function collectExistingFrameworkTimingSectionBlocks_(sheet) {
  const blocks = {};
  if (!sheet || sheet.getLastRow() < 1) return blocks;
  const width = 8;

  getFrameworkTimingSectionRegistry_().forEach(function(section) {
    const titleRow = findFrameworkTimingSectionRow_(sheet, section.title);
    if (!titleRow) return;

    const nextRow = findNextFrameworkTimingSectionRow_(sheet, titleRow + 1);
    const endRow = nextRow ? Math.max(titleRow, nextRow - 2) : sheet.getLastRow();
    const rowCount = Math.max(1, endRow - Math.max(1, titleRow - 1) + 1);
    const values = sheet.getRange(Math.max(1, titleRow - 1), 1, rowCount, width).getValues();
    blocks[section.id] = trimTrailingBlankRows_(values);
  });

  return blocks;
}

function buildDefaultFrameworkTimingSectionBlock_(section) {
  return [
    [""],
    [section.title],
    ["Last Updated", "Date and time"],
    section.headers,
    ["Enter Data here"],
    [""],
    [""]
  ];
}

function normalizeFrameworkTimingSectionBlock_(section, block) {
  let rows = trimTrailingBlankRows_(block || []);
  let titleIndex = -1;
  for (let i = 0; i < rows.length; i++) {
    if (String((rows[i] || [])[0] || "").trim() === section.title) {
      titleIndex = i;
      break;
    }
  }
  if (titleIndex < 0) rows = buildDefaultFrameworkTimingSectionBlock_(section);
  else rows = rows.slice(titleIndex);

  const dataRows = [];
  for (let i = 1; i < rows.length; i++) {
    const joined = (rows[i] || []).join(" ").toLowerCase();
    if (joined.indexOf("runtime") !== -1 || joined.indexOf("threshold") !== -1 || joined.indexOf("step seconds") !== -1 || joined.indexOf("recommendation") !== -1) {
      for (let r = i + 1; r < rows.length; r++) {
        if (rowHasAnyValue_(rows[r])) dataRows.push(rows[r]);
      }
      break;
    }
  }
  if (!dataRows.length) dataRows.push(["Enter Data here"]);

  const output = [[""], [section.title], ["Last Updated", "Date and time"], section.headers].concat(dataRows).concat([[""], [""]]);
  return output.map(function(row) {
    return normalizeSectionRowForWidth_(row, 8);
  });
}

function rebuildFrameworkTimingReportShellCompact_(sheet) {
  if (!sheet) return;

  const requiredCols = 8;
  if (sheet.getMaxColumns() < requiredCols) {
    sheet.insertColumnsAfter(sheet.getMaxColumns(), requiredCols - sheet.getMaxColumns());
  }

  const existingBlocks = collectExistingFrameworkTimingSectionBlocks_(sheet);
  const rows = [];
  rows.push(normalizeSectionRowForWidth_(["Framework Timing Report", "- v" + MASTER_LIST_MERGE_ML_VERSION + " -", "Report Timing Framework"], requiredCols));
  rows.push(normalizeSectionRowForWidth_(["Report Actions", "Refresh Framework Timing Report", "Write Performance Recommendations"], requiredCols));

  getFrameworkTimingSectionRegistry_().forEach(function(section) {
    const block = normalizeFrameworkTimingSectionBlock_(section, existingBlocks[section.id]);
    block.forEach(function(row) {
      rows.push(normalizeSectionRowForWidth_(row, requiredCols));
    });
  });

  if (sheet.getMaxRows() < rows.length) {
    sheet.insertRowsAfter(sheet.getMaxRows(), rows.length - sheet.getMaxRows());
  }

  const clearRows = Math.max(sheet.getLastRow(), rows.length, 1);
  sheet.getRange(1, 1, clearRows, requiredCols).clearContent().breakApart();
  sheet.getRange(1, 1, rows.length, requiredCols).setValues(rows);

  try {
    sheet.getRange(1, 1, 1, requiredCols)
      .setFontFamily("Arial")
      .setFontSize(12)
      .setFontWeight("bold")
      .setBackground(RFF_SYSTEM_SHEET_TITLE_COLOR);
    sheet.setFrozenRows(1);
  } catch (err) {
    Logger.log("Framework Timing compact shell formatting skipped: " + err.message);
  }

  styleFrameworkTimingReport_(sheet);
}

function hasFrameworkTimingReportShell_(sheet) {
  if (!sheet) return false;

  const titleValues = sheet.getRange(1, 1, 1, 3).getDisplayValues()[0];
  if (String(titleValues[0] || "").trim() !== "Framework Timing Report") return false;
  if (String(titleValues[1] || "").trim() !== "- v" + MASTER_LIST_MERGE_ML_VERSION + " -") return false;
  if (String(titleValues[2] || "").trim() !== "Report Timing Framework") return false;

  const sections = getFrameworkTimingSectionRegistry_();
  for (let i = 0; i < sections.length; i++) {
    const titleRow = findFrameworkTimingSectionRow_(sheet, sections[i].title);
    if (!titleRow) return false;

    const headerRow = titleRow + 2;
    if (headerRow > sheet.getLastRow()) return false;

    const actualHeaders = sheet.getRange(headerRow, 1, 1, sections[i].headers.length).getDisplayValues()[0]
      .map(function(value) { return String(value || "").trim(); });
    for (let h = 0; h < sections[i].headers.length; h++) {
      if (actualHeaders[h] !== sections[i].headers[h]) return false;
    }
  }

  return true;
}

function styleFrameworkTimingReport_(sheet, sectionIds) {
  if (!sheet) return;

  const widths = [180, 180, 260, 140, 140, 260, 120, 120];
  trimSheetToColumnCount_(sheet, widths.length, "Framework Timing Report");
  for (let i = 0; i < widths.length; i++) {
    try {
      if (sheet.getColumnWidth(i + 1) !== widths[i]) sheet.setColumnWidth(i + 1, widths[i]);
    } catch (err) {
      Logger.log("Framework Timing width skipped for column " + (i + 1) + ": " + err.message);
    }
  }

  const lastRow = Math.max(sheet.getLastRow(), 1);
  const width = widths.length;
  const registry = getFrameworkTimingSectionRegistry_();
  const sectionTitles = {};
  registry.forEach(function(section) {
    sectionTitles[section.title] = true;
  });

  const requested = sectionIds && sectionIds.length ? sectionIds : registry.map(function(section) { return section.id; });
  const ranges = [];
  if (!sectionIds || !sectionIds.length) {
    ranges.push({ startRow: 1, rowCount: lastRow });
  } else {
    requested.forEach(function(sectionId) {
      const section = getFrameworkTimingSectionForId_(sectionId);
      if (!section) return;
      const titleRow = findFrameworkTimingSectionRow_(sheet, section.title);
      if (!titleRow) return;
      const replaceStartRow = Math.max(2, titleRow - 1);
      const nextRow = findNextFrameworkTimingSectionRow_(sheet, titleRow + 1);
      const endRow = nextRow ? Math.max(replaceStartRow, nextRow - 2) : lastRow;
      ranges.push({ startRow: replaceStartRow, rowCount: Math.max(1, endRow - replaceStartRow + 1) });
    });
  }
  if (!ranges.length) return;

  try {
    ranges.forEach(function(item) {
      const rowCount = Math.max(1, Math.min(item.rowCount, lastRow - item.startRow + 1));
      if (rowCount <= 0) return;
      const range = sheet.getRange(item.startRow, 1, rowCount, width);
      const values = range.getDisplayValues();
      const backgrounds = [];
      const fontWeights = [];
      const fontStyles = [];
      const fontSizes = [];

      for (let r = 0; r < rowCount; r++) {
        const absoluteRow = item.startRow + r;
        const first = String((values[r] || [])[0] || "").trim();
        const previousFirst = r > 0 ? String((values[r - 1] || [])[0] || "").trim() : "";
        const twoBackFirst = r > 1 ? String((values[r - 2] || [])[0] || "").trim() : "";
        let background = "#FFFFFF";
        let weight = "normal";
        let style = "normal";
        let size = 10;

        if (absoluteRow === 1) {
          background = RFF_SYSTEM_SHEET_TITLE_COLOR;
          weight = "bold";
          size = 12;
        } else if (sectionTitles[first]) {
          background = RFF_SYSTEM_SHEET_SECTION_COLOR;
          weight = "bold";
        } else if (sectionTitles[previousFirst]) {
          background = RFF_SYSTEM_SHEET_SUBHEADER_COLOR;
          style = "italic";
        } else if (sectionTitles[twoBackFirst]) {
          background = RFF_SYSTEM_SHEET_SUBHEADER_COLOR;
          weight = "bold";
        }

        backgrounds.push(new Array(width).fill(background));
        fontWeights.push(new Array(width).fill(weight));
        fontStyles.push(new Array(width).fill(style));
        fontSizes.push(new Array(width).fill(size));
      }

      range
        .setFontFamily("Arial")
        .setVerticalAlignment("top")
        .setHorizontalAlignment("left")
        .setBorder(true, true, true, true, true, true, RFF_SYSTEM_SHEET_BORDER_COLOR, SpreadsheetApp.BorderStyle.DOTTED)
        .setBackgrounds(backgrounds)
        .setFontWeights(fontWeights)
        .setFontStyles(fontStyles)
        .setFontSizes(fontSizes);

      sheet.getRange(item.startRow, 1, rowCount, 1).setWrapStrategy(SpreadsheetApp.WrapStrategy.OVERFLOW);
      [2, 3, 4, 5, 6, 7, 8].forEach(function(col) {
        sheet.getRange(item.startRow, col, rowCount, 1).setWrapStrategy(SpreadsheetApp.WrapStrategy.WRAP);
      });
    });

    sheet.setFrozenRows(1);
  } catch (err) {
    Logger.log("Framework Timing styling skipped: " + err.message);
  }
}

function getFrameworkTimingSectionForId_(sectionId) {
  const id = String(sectionId || "").trim().toUpperCase();
  const matches = getFrameworkTimingSectionRegistry_().filter(function(section) {
    return section.id === id || section.title === sectionId;
  });
  return matches.length ? matches[0] : null;
}

function replaceFrameworkTimingSectionRows_(sheet, sectionId, rows, options) {
  if (!sheet) return;

  const section = getFrameworkTimingSectionForId_(sectionId);
  if (!section) throw new Error("Unknown Framework Timing Report section: " + sectionId);

  if (!hasFrameworkTimingReportShell_(sheet)) {
    rebuildFrameworkTimingReportShellCompact_(sheet);
  }

  const width = 8;
  let titleRow = findFrameworkTimingSectionRow_(sheet, section.title);
  if (!titleRow) throw new Error("Framework Timing section not found: " + section.title);

  const normalized = (rows || []).map(function(row) {
    return normalizeSectionRowForWidth_(row || [], width);
  });

  const replaceStartRow = Math.max(2, titleRow - 1);
  const nextRow = findNextFrameworkTimingSectionRow_(sheet, titleRow + 1);
  const replaceEndRow = nextRow ? Math.max(replaceStartRow, nextRow - 2) : sheet.getLastRow();
  let oldCount = Math.max(1, replaceEndRow - replaceStartRow + 1);
  const output = [normalizeSectionRowForWidth_([], width)]
    .concat([
      normalizeSectionRowForWidth_([section.title], width),
      normalizeSectionRowForWidth_(["Last Updated", new Date()], width),
      normalizeSectionRowForWidth_(section.headers, width)
    ])
    .concat(normalized.length ? normalized : [normalizeSectionRowForWidth_(["Enter Data here"], width)])
    .concat([normalizeSectionRowForWidth_([], width), normalizeSectionRowForWidth_([], width)]);

  if (output.length > oldCount) {
    sheet.insertRowsAfter(replaceStartRow + oldCount - 1, output.length - oldCount);
    oldCount = output.length;
  } else if (output.length < oldCount) {
    sheet.deleteRows(replaceStartRow + output.length, oldCount - output.length);
    oldCount = output.length;
  }

  sheet.getRange(replaceStartRow, 1, oldCount, width).clearContent();
  sheet.getRange(replaceStartRow, 1, oldCount, width).setValues(output);
  sheet.getRange(replaceStartRow + 2, 2).setNumberFormat("mm/dd/yyyy hh:mm:ss");
  if (!options || !options.skipStyle) styleFrameworkTimingReport_(sheet, [section.id]);
}

function getFrameworkTimingBenchmarkForProcess_(processName, mode) {

  if (text.indexOf("dashboard quality quick") !== -1) return 20;
  if (text.indexOf("dashboard quality workflow") !== -1) return 30;
  if (text.indexOf("dashboard quality validate") !== -1) return 30;
  if (text.indexOf("verify framework") !== -1) return 15;
  if (text.indexOf("template") !== -1) {
    if (refreshMode === "FIRST_BUILD") return 240;
    if (refreshMode === "FULL_REBUILD") return 180;
    if (refreshMode === "METADATA_ONLY" || refreshMode === "NO_CHANGE") return 45;
    return 45;
  }
  if (text.indexOf("master list") !== -1) return 60;
  if (text.indexOf("demo p") !== -1) return 60;
  return "";
}

function getFrameworkTimingDetailStartRow_(sheet) {
  if (!sheet) return 16;
  const section = getFrameworkTimingSectionForId_("D");
  const titleRow = section ? findFrameworkTimingSectionRow_(sheet, section.title) : 0;
  return titleRow ? titleRow + 3 : 16;
}

function getFrameworkTimingDetailRows_(sheet) {
  if (!sheet || sheet.getLastRow() < 1) return [];
  const section = getFrameworkTimingSectionForId_("D");
  if (!section) return [];

  const titleRow = findFrameworkTimingSectionRow_(sheet, section.title);
  if (!titleRow) return [];

  const dataStart = titleRow + 3;
  const nextRow = findNextFrameworkTimingSectionRow_(sheet, titleRow + 1);
  const endRow = nextRow ? nextRow - 1 : sheet.getLastRow();
  const dataRows = Math.max(0, endRow - dataStart + 1);
  if (dataRows <= 0) return [];

  const lookbackRows = Math.min(dataRows, Math.max(Number(RFF_TIMING_SUMMARY_LOOKBACK_ROWS) || 750, 50));
  const readStart = endRow - lookbackRows + 1;
  const values = sheet.getRange(readStart, 1, lookbackRows, section.headers.length).getValues();
  return values.filter(function(row) {
    return row.some(function(value) {
      return String(value === null || value === undefined ? "" : value).trim() !== "" && String(value).trim() !== "Enter Data here";
    });
  });
}

function getLatestFrameworkTimingRowsByProcess_(detailRows) {
  const rows = detailRows || [];
  const processStarts = {};
  rows.forEach(function(row, index) {
    const process = String(row[1] || "").trim();
    const step = String(row[2] || "").trim().toLowerCase();
    if (!process) return;
    if (step === "start") processStarts[process] = index;
    else if (processStarts[process] === undefined) processStarts[process] = index;
  });

  const latestRows = [];
  rows.forEach(function(row, index) {
    const process = String(row[1] || "").trim();
    if (!process) return;
    const startIndex = processStarts[process];
    if (startIndex === undefined || index >= startIndex) latestRows.push(row);
  });
  return latestRows.length ? latestRows : rows;
}

function getFrameworkTimingBenchmarkSeverity_(runtime, benchmark) {
  const value = Number(runtime || 0);
  const target = Number(benchmark || 0);
  if (!target || value <= target) return "OK";
  const variance = value - target;
  if (variance >= 60 || value >= 180) return "CRITICAL";
  if (variance >= 30 || value >= 90) return "BOTTLENECK";
  return "SLOW";
}

function getFrameworkTimingModeForStep_(stepText) {
  const text = String(stepText || "").toLowerCase();
  if (text.indexOf("metadata_only") !== -1 || text.indexOf("metadata-only") !== -1) return "METADATA_ONLY";
  if (text.indexOf("template sheet did not exist") !== -1 || text.indexOf("no stored signature") !== -1) return "FIRST_BUILD";
  if (text.indexOf("full_build") !== -1 || text.indexOf("full build required") !== -1) return "FULL_REBUILD";
  if (text.indexOf("no_change") !== -1 || text.indexOf("template unchanged") !== -1) return "NO_CHANGE";
  return "";
}

function mergeFrameworkTimingModes_(current, next) {
  const rank = { "": 0, "NO_CHANGE": 1, "METADATA_ONLY": 2, "FULL_REBUILD": 3, "FIRST_BUILD": 4 };
  return (rank[next] || 0) > (rank[current] || 0) ? next : current;
}

function buildFrameworkTimingProcessSummaryRows_(detailRows) {
  const processMap = {};
  const currentRunByProcess = {};
  let globalRunSequence = 0;


  (detailRows || []).forEach(function(row) {
    const process = String(row[1] || "").trim();
    if (!process) return;


    const step = String(row[2] || "");
    const normalizedStep = step.trim().toLowerCase();
    const stepSeconds = Number(row[3]) || 0;
    const severity = String((row.length >= 7 ? row[5] : row[4]) || "OK").trim() || "OK";
    const timestamp = row[0] || "";
    const mode = getFrameworkTimingModeForStep_(step);


    if (normalizedStep === "start" || !currentRunByProcess[process]) {
      globalRunSequence += 1;
      currentRunByProcess[process] = {
        key: process + "||" + globalRunSequence,
        started: timestamp || new Date()
      };
    }


    const runInfo = currentRunByProcess[process];
    if (!processMap[runInfo.key]) {
      processMap[runInfo.key] = {
        process: process,
        steps: 0,
        runtime: 0,
        slowSteps: 0,
        worstSeverity: "OK",
        firstTimestamp: runInfo.started || timestamp,
        lastTimestamp: timestamp,
        mode: ""
      };
    }


    const item = processMap[runInfo.key];
    item.steps += 1;
    item.runtime += stepSeconds;
    item.lastTimestamp = timestamp || item.lastTimestamp;
    item.mode = mergeFrameworkTimingModes_(item.mode, mode);
    if (severity !== "OK") item.slowSteps += 1;
    item.worstSeverity = worseTimingSeverity_(item.worstSeverity, severity);
  });


  return Object.keys(processMap).sort(function(a, b) {
    const left = processMap[a].firstTimestamp;
    const right = processMap[b].firstTimestamp;
    const leftTime = left instanceof Date ? left.getTime() : new Date(left).getTime();
    const rightTime = right instanceof Date ? right.getTime() : new Date(right).getTime();
    return (isNaN(leftTime) ? 0 : leftTime) - (isNaN(rightTime) ? 0 : rightTime);
  }).map(function(key) {
    const item = processMap[key];
    const benchmark = getFrameworkTimingBenchmarkForProcess_(item.process, item.mode);
    const variance = benchmark ? item.runtime - benchmark : "";
    const benchmarkSeverity = getFrameworkTimingBenchmarkSeverity_(item.runtime, benchmark);
    const status = worseTimingSeverity_(item.worstSeverity, benchmarkSeverity);
    const notes = [];
    if (item.firstTimestamp) notes.push("Run Started: " + formatTimingTimestampForSummary_(item.firstTimestamp));
    if (item.mode) notes.push("Mode: " + item.mode);
    if (variance !== "" && variance > 0) notes.push("Exceeded benchmark by " + Number(variance.toFixed(3)) + " sec");
    if (item.slowSteps) notes.push(item.slowSteps + " slow/bottleneck/critical step(s)");
    if (!notes.length) notes.push("Within target");
    return [
      item.process,
      Number(item.runtime.toFixed(3)),
      status === "OK" ? "PASS" : status,
      benchmark || "",
      variance === "" ? "" : Number(variance.toFixed(3)),
      notes.join("; ")
    ];
  });
}

function formatTimingTimestampForSummary_(value) {
  try {
    const date = value instanceof Date ? value : new Date(value);
    if (isNaN(date.getTime())) return String(value || "");
    return Utilities.formatDate(date, Session.getScriptTimeZone(), "MM/dd/yyyy HH:mm:ss");
  } catch (err) {
    return String(value || "");
  }
}

function buildFrameworkTimingIssueRows_(detailRows, processRows) {
  const rows = (detailRows || []).filter(function(row) {
    const severity = String((row.length >= 7 ? row[5] : row[4]) || "").trim().toUpperCase();
    return severity === "SLOW" || severity === "BOTTLENECK" || severity === "CRITICAL";
  }).map(function(row) {
    const severity = String((row.length >= 7 ? row[5] : row[4]) || "");
    const recommendation = getPerformanceRecommendationForTimingStep_(String(row[1] || ""), String(row[2] || ""), Number(row[3]) || 0, severity);
    return [severity, row[1], row[3], getFrameworkTimingThresholdForSeverity_(severity), row[2], recommendation];
  });

  (processRows || []).forEach(function(row) {
    const status = String(row[2] || "").trim().toUpperCase();
    const process = String(row[0] || "").trim();
    const runtime = Number(row[1]) || 0;
    const benchmark = Number(row[3]) || 0;
    const variance = Number(row[4]) || 0;
    if (!process || status === "PASS" || status === "OK" || !benchmark || variance <= 0) return;
    rows.push([
      status,
      process,
      runtime,
      "Benchmark " + benchmark + " sec",
      "Process exceeded benchmark by " + Number(variance.toFixed(3)) + " sec",
      getPerformanceRecommendationForTimingStep_(process, "process exceeded benchmark", runtime, status)
    ]);
  });

  return rows;
}

function buildFrameworkTimingRecommendationRows_(issueRows) {
  const seen = {};
  const rows = [];

  (issueRows || []).forEach(function(row) {
    const process = String(row[1] || "").trim();
    const priority = String(row[0] || "").trim();
    const recommendation = String(row[5] || "").trim();
    const key = process + "|" + priority + "|" + recommendation;
    if (!process || seen[key]) return;
    seen[key] = true;
    rows.push([process, row[4] || "Performance issue", priority, recommendation, priority, "Open"]);
  });

  return rows;
}

function getPerformanceRecommendationForTimingStep_(process, step, seconds, severity) {
  const text = String(step || "").toLowerCase();


  if (text.indexOf("process exceeded benchmark") !== -1) {
    return "Separate first-build/full-rebuild timing from metadata-only refresh, apply the governed benchmark for that mode, and optimize the largest full-build steps before treating this as a steady-state failure.";
  }
  if (text.indexOf("dashboard loaded for sections f-g") !== -1) {
    return "Dashboard config reads are capped to governed dashboard columns and Sections F-G are batch-written; rerun Validate Templates and compare this step against the v1.5.38 baseline.";
  }
  if (text.indexOf("dashboard quality sections a-e updated") !== -1) {
    return "A-E section writes now skip redundant unmerge work in batch mode and merge adjacent style ranges to reduce Apps Script range calls.";
  }
  if (text.indexOf("error -") !== -1) {
    return "Resolve the runtime exception before performance tuning this workflow; timed errors are forced to CRITICAL even when the failure is fast.";
  }
  if (text.indexOf("resize rows") !== -1) {
    return "Review dashboard Template Row Count and row mode. Row resizing is the likely source of this resize cost.";
  }
  if (text.indexOf("resize columns") !== -1) {
    return "Review Dashboard Section C header count and remove obsolete columns only through Sheet Headers.";
  }
  if (text.indexOf("resize sheet") !== -1) {
    return "Split resize timing into rows and columns; v1.4.30 logs those separately for future runs.";
  }
  if (text.indexOf("dashboard quality") !== -1 && text.indexOf("saved") !== -1) {
    return "Use the contiguous A-O Dashboard Quality shell and repair stale shells before section writes. Keep section writes scoped between adjacent section titles.";
  }
  if (text.indexOf("apply row heights") !== -1) {
    return "Keep row-height work out of CP Due/Unlocked CP/Banner/Change Log outputs. Use final forced 25px row lock only after Raw Data, Demo P, Master List, and Disenrolled processing because processing/data writes can expand rows.";
  }
  if (text.indexOf("create filter") !== -1) {
    return "Skip filter recreation when the existing filter already matches the governed template range.";
  }
  if (text.indexOf("clear sheet") !== -1 || text.indexOf("clear formats") !== -1) {
    return "Use create-template fast path for blank sheets and reserve clear/rebuild only for existing template updates with signature mismatch.";
  }


  return "Review this slow step and confirm it is necessary for the governed dashboard/template path.";
}

function worseTimingSeverity_(current, next) {
  const rank = { "OK": 0, "PASS": 0, "INFO": 0, "WARNING": 1, "SLOW": 2, "BOTTLENECK": 3, "CRITICAL": 4 };
  const c = rank[String(current || "OK").toUpperCase()] || 0;
  const n = rank[String(next || "OK").toUpperCase()] || 0;
  return n > c ? String(next || "OK") : String(current || "OK");
}

function runDashboardQualityDashboardVerificationSections_(timing, qualitySheet) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const quality = qualitySheet || ensureDashboardQualityReportSheet_();
  const dashboard = ss.getSheetByName(RFF_DASHBOARD_SHEET);

  if (!dashboard) {
    const missingRows = [
      ["Verification Area", "Verification Item", "Expected Value", "Actual Value", "Status", "Severity", "Quality Notes"],
      ["Format Dashboard", "Sheet Exists", RFF_DASHBOARD_SHEET, "Missing", "FAIL", "Critical", "Create or rebuild Format Dashboard before template creation."]
    ];
    saveDashboardQualitySectionRows_(RFF_DASHBOARD_VERIFY_GLOBAL_KEY, missingRows, { deferSheetWrite: true });
    replaceDashboardQualitySectionRows_(
      quality,
      getDashboardQualitySectionTitleForKey_(RFF_DASHBOARD_VERIFY_GLOBAL_KEY),
      buildTimestampedDashboardQualitySectionRows_(RFF_DASHBOARD_VERIFY_GLOBAL_KEY, missingRows)
    );
    markFrameworkStep_(timing, "Format Dashboard missing");
    return missingRows;
  }

  const sectionPayloads = [
    { key: RFF_DASHBOARD_VERIFY_GLOBAL_KEY, rows: collectFormatDashboardGlobalInputVerificationRows_(dashboard), label: "Section A Global Inputs and Section B Title Rows verified" },
    { key: RFF_DASHBOARD_VERIFY_SHEETS_KEY, rows: collectFormatDashboardSheetDefinitionVerificationRows_(dashboard), label: "Section C Sheet Definitions verified" },
    { key: RFF_DASHBOARD_VERIFY_HEADERS_KEY, rows: collectFormatDashboardSheetHeaderVerificationRows_(dashboard), label: "Section F Sheet Headers verified" },
    { key: RFF_DASHBOARD_VERIFY_COLUMNS_KEY, rows: collectFormatDashboardColumnDefinitionVerificationRows_(dashboard), label: "Section D Column Definitions verified" },
    { key: RFF_DASHBOARD_VERIFY_BEHAVIORS_KEY, rows: collectFormatDashboardSheetBehaviorVerificationRows_(dashboard), label: "Section E Sheet Behaviors verified" }
  ];

  const sectionWrites = [];
  sectionPayloads.forEach(function(item) {
    const previousRows = getDashboardQualitySectionRows_(item.key);
    const changed = !dashboardQualityRowsEqualValues_(previousRows, item.rows);
    saveDashboardQualitySectionRows_(item.key, item.rows, { deferSheetWrite: true });
    markFrameworkStep_(timing, item.label + (changed ? "" : " skipped - unchanged"));
    if (changed) {
      sectionWrites.push({
        title: getDashboardQualitySectionTitleForKey_(item.key),
        rows: buildTimestampedDashboardQualitySectionRows_(item.key, item.rows)
      });
    }
  });

  if (sectionWrites.length) {
    replaceDashboardQualitySectionsRows_(quality, sectionWrites);
    markFrameworkStep_(timing, "Dashboard Quality Sections A-F batch written");
  } else {
    markFrameworkStep_(timing, "Dashboard Quality Sections A-F sheet write skipped - unchanged");
  }

  return true;
}

function validateBuiltTemplateMinimumStructure_(sheet, dashboard, sheetDef, headers, colCount) {
  const globals = dashboard.globals || RFF_DEFAULTS;
  const headerRow = Number(globals.headerRow || RFF_DEFAULTS.headerRow);
  const expectedFrozenRows = Number(globals.freezeRows || RFF_DEFAULTS.freezeRows);
  if (sheet.getMaxColumns() < colCount) {
    throw new Error("Staged template has " + sheet.getMaxColumns() + " column(s), expected at least " + colCount + ": " + sheetDef.templateName);
  }
  if (sheet.getFrozenRows() !== expectedFrozenRows) {
    throw new Error("Staged template frozen rows expected " + expectedFrozenRows + " but found " + sheet.getFrozenRows() + ": " + sheetDef.templateName);
  }
  const actualHeaders = sheet.getRange(headerRow, 1, 1, Math.max(headers.length, 1)).getDisplayValues()[0];
  for (let i = 0; i < headers.length; i++) {
    if (String(actualHeaders[i] || "").trim() !== String(headers[i] || "").trim()) {

function testDateFormattingForTemplate_(dashboard, sheetDef) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(sheetDef.templateName);
  const globals = dashboard.globals;
  const headers = getHeadersForSheetType_(dashboard, sheetDef.sheetType);
  const results = [];


  if (!sheet) {
    results.push({
      templateName: sheetDef.templateName,
      sheetType: sheetDef.sheetType,
      header: "",
      column: "",
      expectedFormat: "",
      actualFormats: "Template missing",
      status: "FAIL"
    });
    return results;
  }


  const dateChecks = [];
  headers.forEach(function(header, index) {
    const def = dashboard.columnDefinitions[header] || {};
    if (!def.dateColumn) return;
    const expectedFormat = getExpectedNumberFormat_(def, globals);
    const expectedSheetsFormat = getGoogleSheetsNumberFormat_(expectedFormat, true);
    dateChecks.push({
      header: header,
      column: index + 1,
      expectedFormat: expectedFormat,
      expectedSheetsFormat: expectedSheetsFormat
    });
  });


  if (dateChecks.length === 0) return results;


  if (sheet.getMaxColumns() < headers.length) {
    dateChecks.forEach(function(check) {
      results.push({
        templateName: sheetDef.templateName,
        sheetType: sheetDef.sheetType,
        header: check.header,
        column: check.column,
        expectedFormat: check.expectedFormat,
        actualFormats: "Template has " + sheet.getMaxColumns() + " column(s), expected at least " + headers.length,
        status: "FAIL"
      });
    });
    return results;
  }


  const sampleRows = Math.min(10, Math.max(sheet.getMaxRows() - globals.dataStartRow + 1, 1));
  const lastCol = Math.max(headers.length, 1);
  let allFormats = [];


  try {
    allFormats = sheet.getRange(globals.dataStartRow, 1, sampleRows, lastCol).getNumberFormats();
  } catch (err) {
    dateChecks.forEach(function(check) {
      results.push({
        templateName: sheetDef.templateName,
        sheetType: sheetDef.sheetType,
        header: check.header,
        column: check.column,
        expectedFormat: check.expectedFormat,
        actualFormats: "Read error: " + err.message,
        status: "FAIL"
      });
    });
    return results;
  }


  dateChecks.forEach(function(check) {
    let uniqueActualFormats = collectUniqueDateFormatsForCheck_(allFormats, check.column);
    let failedFormats = uniqueActualFormats.filter(function(actualFormat) {
      return !numberFormatsMatch_(actualFormat, check.expectedSheetsFormat);
    });
    let repaired = false;

    if (failedFormats.length && RFF_REPAIR_DATE_FORMATS_DURING_SECTION_G) {
      repaired = repairDateFormatForSectionG_(sheet, globals, check);
      if (repaired) {
        try {
          const repairedFormats = sheet.getRange(globals.dataStartRow, check.column, sampleRows, 1).getNumberFormats();
          uniqueActualFormats = collectUniqueDateFormatsForCheck_(repairedFormats, 1);
          failedFormats = uniqueActualFormats.filter(function(actualFormat) {
            return !numberFormatsMatch_(actualFormat, check.expectedSheetsFormat);
          });
        } catch (readErr) {
          Logger.log("Section G repaired format read skipped for " + sheetDef.templateName + " / " + check.header + ": " + readErr.message);
        }
      }
    }

    results.push({
      templateName: sheetDef.templateName,
      sheetType: sheetDef.sheetType,
      header: check.header,
      column: check.column,
      expectedFormat: check.expectedFormat,
      actualFormats: uniqueActualFormats.join(", ") + (repaired ? " (repaired)" : ""),
      status: failedFormats.length === 0 ? "PASS" : "FAIL"
    });
  });


  return results;
}

function runFrameworkTest7IntentionalValidationFailure() {
  return runFrameworkTimed_("Framework Test 7 - Intentional Validation Failure", function(timing) {
    const dashboard = loadDashboardConfig_();
    markFrameworkStep_(timing, "Dashboard loaded");


    const sheetDef = getSheetDefinitionByType_(dashboard, RFF_SHEET_TYPES.BANNER) || dashboard.sheetDefinitions[0];
    if (!sheetDef) throw new Error("No sheet definitions found for Framework Test 7.");


    let sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetDef.templateName);
    if (!sheet) {
      createOrRefreshTemplateFromDashboard_(dashboard, sheetDef, timing);
      sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetDef.templateName);
      markFrameworkStep_(timing, "Missing Framework Test 7 template created: " + sheetDef.templateName);
    }
    if (!sheet) throw new Error("Template missing for Framework Test 7: " + sheetDef.templateName);


    const headers = getHeadersForSheetType_(dashboard, sheetDef.sheetType);
    if (headers.length === 0) throw new Error("No headers available for Framework Test 7: " + sheetDef.sheetType);


    const testColumn = 1;
    const testHeader = headers[testColumn - 1];
    const expectedWidth = (dashboard.columnDefinitions[testHeader] || {}).width || sheet.getColumnWidth(testColumn);
    const originalWidth = sheet.getColumnWidth(testColumn);
    const forcedBadWidth = expectedWidth + 17;


    let brokenResult;
    let restoredResult;
    let status = "FAIL";
    let notes = "";


    try {
      showSheetIfNeeded_(sheet);
      sheet.setColumnWidth(testColumn, forcedBadWidth);
      SpreadsheetApp.flush();
      markFrameworkStep_(timing, "Intentional width mismatch created: " + sheetDef.templateName + " / " + testHeader);


      brokenResult = validateTemplateFromDashboard_(dashboard, sheetDef, { strict: true });
      markFrameworkStep_(timing, "Validation run against intentionally broken template");


      sheet.setColumnWidth(testColumn, originalWidth);
      SpreadsheetApp.flush();
      markFrameworkStep_(timing, "Template width restored");


      restoredResult = validateTemplateFromDashboard_(dashboard, sheetDef, { strict: true });
      markFrameworkStep_(timing, "Validation run after restore");


      const caughtFailure = brokenResult.status === "FAIL" && String(brokenResult.issues || "").indexOf("Width mismatch") !== -1;
      const restoredIssuesText = String(restoredResult.issues || "");
      const restoredWidthMismatch = restoredIssuesText.indexOf("Width mismatch") !== -1;
      const restoredPass = restoredResult.status === "PASS" || (caughtFailure && !restoredWidthMismatch);
      /*
       * v1.5.1:
       * Test 7 intentionally creates a temporary width mismatch. Template-level row
       * height governance was retired, so strict row-height findings after the
       * width restore must not keep Section D failed. Section D passes when the
       * forced width mismatch is detected and no width mismatch remains after
       * restore.
       */
      status = restoredPass ? "PASS" : "FAIL";
      notes =
        "Intentional width mismatch forced from " + expectedWidth + " to " + forcedBadWidth +
        " | Broken validation: " + brokenResult.status + " - " + brokenResult.issues +
        " | Strict mismatch detected: " + (caughtFailure ? "YES" : "NO") +
        " | Restored validation: " + restoredResult.status + " - " + restoredResult.issues;
    } catch (err) {
      try {
        sheet.setColumnWidth(testColumn, originalWidth);
      } catch (restoreErr) {
        Logger.log("Framework Test 7 restore failed: " + restoreErr.message);
      }
      notes = err.message;
      status = "FAIL";
    }


    const rows = [[
      sheetDef.templateName,
      sheetDef.sheetType,
      testHeader,
      expectedWidth,
      forcedBadWidth,
      status,
      notes
    ]];


    writeFrameworkTest7ValidationFailureReport_(rows);
    markFrameworkStep_(timing, "Dashboard Quality Section D saved");


    return rows;
  });
}

function validateTemplateFromDashboard_(dashboard, sheetDef, options) {
  options = options || {};
  const strict = !!options.strict;


  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(sheetDef.templateName);
  const issues = [];
  const globals = dashboard.globals;
  const expectedHeaders = getHeadersForSheetType_(dashboard, sheetDef.sheetType);
  const behavior = dashboard.behaviors[sheetDef.sheetType] || getDefaultBehavior_();
  const colCount = Math.max(expectedHeaders.length, 4);
  const rowCount = resolveTemplateRowCount_(sheetDef, globals);
  const expectedSignature = normalizeTemplateFormatSignature_(buildTemplateFormatSignature_(dashboard, sheetDef, expectedHeaders, rowCount, colCount, behavior));


  if (!sheet) {
    return {
      templateName: sheetDef.templateName,
      sheetType: sheetDef.sheetType,
      status: "FAIL",
      issues: "Template missing"
    };
  }


  if (expectedHeaders.length === 0) {
    issues.push("No headers defined in dashboard");
  }


  if (!strict) {
    try {
      const storedSignature = normalizeTemplateFormatSignature_(getStoredTemplateFormatSignature_(sheetDef) || getStoredTemplateFormatSignatureFromSheet_(sheet));
      if (storedSignature && storedSignature === expectedSignature && expectedHeaders.length > 0) {
        return {
          templateName: sheetDef.templateName,
          sheetType: sheetDef.sheetType,
          status: "PASS",
          issues: "OK"
        };
      }
      // Signature drift should trigger template refresh, not fail Section F structure validation.
      // Structural checks below remain authoritative for Dashboard Quality Section F.
    } catch (err) {
      issues.push("Could not compare template signature: " + err.message);
    }
  }


  try {
    const frozenRows = sheet.getFrozenRows();
    const frozenColumns = sheet.getFrozenColumns();


    if (frozenRows !== globals.freezeRows) {
      issues.push("Frozen rows expected " + globals.freezeRows + " but found " + frozenRows);
    }


    if (expectedHeaders.length > 0 && frozenColumns > globals.freezeColumns) {
      issues.push("Frozen columns expected " + globals.freezeColumns + " or less but found " + frozenColumns);
    }
  } catch (err) {
    issues.push("Could not check frozen settings: " + err.message);
  }



  try {
    const actualHeaders = sheet.getRange(globals.headerRow, 1, 1, Math.max(expectedHeaders.length, 1))
      .getValues()[0]
      .map(normalizeHeader_);


    expectedHeaders.forEach(function(expected, index) {
      if (actualHeaders[index] !== expected) {
        issues.push("Header mismatch col " + (index + 1) + ": expected " + expected + ", found " + actualHeaders[index]);
      }
    });
  } catch (err) {
    issues.push("Could not check headers: " + err.message);
  }


  try {
    expectedHeaders.forEach(function(header, index) {
      const def = dashboard.columnDefinitions[header] || {};
      if (def.width && sheet.getColumnWidth(index + 1) !== def.width) {
        issues.push("Width mismatch for " + header);
      }
    });
  } catch (err) {
    issues.push("Could not check column widths: " + err.message);
  }


  try {
    expectedHeaders.forEach(function(header, index) {
      const def = dashboard.columnDefinitions[header] || {};
      const expectedFormat = getExpectedNumberFormat_(def, globals);
      if (!expectedFormat) return;


      const expectedSheetsFormat = getGoogleSheetsNumberFormat_(expectedFormat, def.dateColumn);
      const sampleRows = Math.min(10, Math.max(sheet.getMaxRows() - globals.dataStartRow + 1, 1));
      const actualFormats = sheet.getRange(globals.dataStartRow, index + 1, sampleRows, 1).getNumberFormats();
      const mismatches = [];


      actualFormats.forEach(function(row) {
        const actualFormat = row[0];
        if (!numberFormatsMatch_(actualFormat, expectedSheetsFormat)) {
          if (mismatches.indexOf(actualFormat) === -1) mismatches.push(actualFormat);
        }
      });


      if (mismatches.length > 0) {
        issues.push("Date/number format mismatch for " + header + " expected " + expectedFormat + " but found " + mismatches.join(", "));
      }
    });
  } catch (err) {
    issues.push("Could not check date/number formats: " + err.message);
  }


  try {
    if (behavior.usesFilter && !sheet.getFilter()) {
      issues.push("Filter missing");
    }
  } catch (err) {
    issues.push("Could not check filter: " + err.message);
  }


  return {
    templateName: sheetDef.templateName,
    sheetType: sheetDef.sheetType,
    status: issues.length === 0 ? "PASS" : "FAIL",
    issues: issues.length === 0 ? "OK" : issues.join("; ")
  };
}

function collectCarePlanSyncValidationMetrics_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const metrics = {
    unlockedMatches: 0,
    unlockedUnmatched: 0,
    dueMatches: 0,
    dueUnmatched: 0,
    missingIDT: 0,
    missingCarePlanStart: 0,
    missingNextDue: 0,
    missingCPType: 0,
    rowsWithCPData: 0,
    rowsMissingCPData: 0
  };


  const masterSheet = getNewestFormattedMonthlySheetByPrefix_(ss, MASTER_LIST_PREFIX);
  if (!masterSheet) return metrics;


  const master = getDataValues_(masterSheet, HEADER_ROW, DATA_START_ROW);
  if (!master.values.length) return metrics;


  const masterPMRIdx = findHeaderIndex_(master.headerMap, ["Participant PMR#", "PMR #"]);
  const masterNameIdx = master.headerMap["Participant Name"];
  const capDateIdx = master.headerMap["Capitation Date"];


  const idtIdx = master.headerMap["IDT Meeting Date"];
  const cpsIdx = master.headerMap["Care Plan Start Date"];
  const nextDueIdx = master.headerMap["Next Care Plan Due"];
  const cpTypeIdx = master.headerMap["CP Type"];


  const unlockedSheet = getNewestFormattedMonthlySheetByPrefix_(ss, "Unlocked CP");
  const dueSheet = getNewestFormattedMonthlySheetByPrefix_(ss, "CP Due");


  const unlockedSet = new Set();
  if (unlockedSheet) {
    const unlocked = getDataValues_(unlockedSheet, HEADER_ROW, DATA_START_ROW);
    const sourcePMRIdx = findHeaderIndex_(unlocked.headerMap, ["PMR #", "Participant PMR#"]);
    if (sourcePMRIdx !== -1) {
      unlocked.values.forEach(function(row) {
        const pmr = normalizePMR_(row[sourcePMRIdx]);
        if (pmr) unlockedSet.add(pmr);
      });
    }
  }


  const dueSet = new Set();
  if (dueSheet) {
    const due = getDataValues_(dueSheet, HEADER_ROW, DATA_START_ROW);
    const nameIdx = due.headerMap["Participant Name"];
    const enrollmentIdx = due.headerMap["Enrollment Date"];
    if (nameIdx !== undefined && enrollmentIdx !== undefined) {
      due.values.forEach(function(row) {
        const key = normalizeKeyPart_(row[nameIdx]) + "|||" + normalizeKeyPart_(row[enrollmentIdx]);
        if (key.replace(/\|/g, "")) dueSet.add(key);
      });
    }
  }


  master.values.forEach(function(row) {
    const pmr = masterPMRIdx === -1 ? "" : normalizePMR_(row[masterPMRIdx]);
    if (pmr && unlockedSet.size) {
      if (unlockedSet.has(pmr)) metrics.unlockedMatches++;
      else metrics.unlockedUnmatched++;
    }


    const dueKey = (masterNameIdx === undefined || capDateIdx === undefined)
      ? ""
      : normalizeKeyPart_(row[masterNameIdx]) + "|||" + normalizeKeyPart_(row[capDateIdx]);
    if (dueKey && dueKey.replace(/\|/g, "") && dueSet.size) {
      if (dueSet.has(dueKey)) metrics.dueMatches++;
      else metrics.dueUnmatched++;
    }


    const hasIDT = idtIdx !== undefined && normalizeCompareValue_(row[idtIdx]) !== "";
    const hasCPS = cpsIdx !== undefined && normalizeCompareValue_(row[cpsIdx]) !== "";
    const hasNextDue = nextDueIdx !== undefined && normalizeCompareValue_(row[nextDueIdx]) !== "";
    const hasCPType = cpTypeIdx !== undefined && normalizeCompareValue_(row[cpTypeIdx]) !== "";


    if (!hasIDT) metrics.missingIDT++;
    if (!hasCPS) metrics.missingCarePlanStart++;
    if (!hasNextDue) metrics.missingNextDue++;
    if (!hasCPType) metrics.missingCPType++;


    if (hasIDT || hasCPS || hasNextDue || hasCPType) metrics.rowsWithCPData++;
    else metrics.rowsMissingCPData++;
  });


  return metrics;
}

function auditDashboardConfiguration_(dashboard) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(RFF_DASHBOARD_SHEET);
  if (!sheet) {
    return [{ area: "Dashboard", item: RFF_DASHBOARD_SHEET, status: "FAIL", notes: "Format Dashboard sheet is missing" }];
  }


  const results = [];


  function add(area, item, status, notes) {
    results.push({ area: area, item: item, status: status, notes: notes || "OK" });
  }


  const sheetRows = readDashboardSectionRows_(sheet, RFF_SECTION_SHEETS);
  const headerRows = readDashboardSectionRows_(sheet, RFF_SECTION_HEADERS);
  const columnRows = readDashboardSectionRows_(sheet, RFF_SECTION_COLUMNS);
  const behaviorRows = readDashboardSectionRows_(sheet, RFF_SECTION_BEHAVIORS);


  const sheetTypeSeen = {};
  sheetRows.forEach(function(row) {
    const sheetType = normalizeHeader_(row[0]);
    const templateName = String(row[2] || "").trim();
    if (!sheetType) return;
    if (sheetTypeSeen[sheetType]) add("Duplicate Sheet Definition", sheetType, "FAIL", "Sheet type appears more than once");
    sheetTypeSeen[sheetType] = true;
    if (!templateName) add("Sheet Definition", sheetType, "FAIL", "Template Name is blank");
  });


  const behaviorSeen = {};
  behaviorRows.forEach(function(row) {
    const sheetType = normalizeHeader_(row[0]);
    if (!sheetType) return;
    if (behaviorSeen[sheetType]) add("Duplicate Behavior", sheetType, "FAIL", "Behavior row appears more than once");
    behaviorSeen[sheetType] = true;
  });


  Object.keys(sheetTypeSeen).forEach(function(sheetType) {
    if (!dashboard.sheetHeaders[sheetType] || dashboard.sheetHeaders[sheetType].length === 0) {
      add("Missing Headers", sheetType, "FAIL", "No headers defined for sheet type");
    }
    if (!behaviorSeen[sheetType]) {
      add("Missing Behavior", sheetType, "FAIL", "No Sheet Behaviors row found for sheet type");
    }
  });


  const columnDefSeen = {};
  columnRows.forEach(function(row) {
    const header = normalizeHeader_(row[0]);
    if (!header) return;
    if (columnDefSeen[header]) add("Duplicate Column Definition", header, "FAIL", "Column definition appears more than once");
    columnDefSeen[header] = true;


    const width = isBlankCell_(row[1]) ? null : Number(row[1]);
    const fontSize = isBlankCell_(row[2]) ? null : Number(row[2]);
    const dateColumn = parseBoolean_(row[3]);
    const numberFormat = String(row[8] || "").trim();


    if (width !== null && (isNaN(width) || width < 1)) add("Invalid Width", header, "FAIL", "Width override must be a positive number when populated; blank inherits Default Column Width");
    if (fontSize !== null && (isNaN(fontSize) || fontSize < 1)) add("Invalid Font Size", header, "FAIL", "Header Font Size override must be a positive number when populated");
    if (dateColumn && numberFormat && !isDateFormatText_(numberFormat)) add("Invalid Date Format", header, "FAIL", "Date Column is TRUE but format does not look like a date format");
  });


  headerRows.forEach(function(row) {
    const sheetType = normalizeHeader_(row[0]);
    const header = normalizeHeader_(row[2]);
    if (!sheetType || !header) return;
    if (!columnDefSeen[header]) add("Missing Column Definition", sheetType + " / " + header, "FAIL", "Header is used by a sheet but has no Column Definitions row");
  });


  if (!results.some(function(item) { return item.status !== "PASS"; })) {
    add("Dashboard Audit", "Overall", "PASS", "No dashboard configuration issues found");
  }


  return results;
}

function collectDashboardQualityPerformanceSummaryRows_() {
  const sheet = ensureFrameworkTimingReport_();
  const detailRows = getFrameworkTimingDetailRows_(sheet);
  const latestRows = getLatestFrameworkTimingRowsByProcess_(detailRows.slice(-500));
  const processRows = buildFrameworkTimingProcessSummaryRows_(latestRows);
  const rows = [["Process", "Steps Logged", "Total Runtime", "Slow/Bottleneck/Critical Steps", "Worst Severity", "Last Timestamp", "Quality Notes"]];

  if (!processRows.length) {
    rows.push(["Framework Timing", 0, 0, 0, "OK", "", "No timing rows logged yet."]);
    return rows;
  }

  const processMap = {};
  (detailRows || []).forEach(function(row) {
    const process = String(row[1] || "").trim();
    if (!process) return;
    if (!processMap[process]) {
      processMap[process] = {
        steps: 0,
        runtime: 0,
        issueSteps: 0,
        worst: "OK",
        lastTimestamp: row[0] || ""
      };
    }
    const item = processMap[process];
    const severity = String((row.length >= 7 ? row[5] : row[4]) || "OK").trim() || "OK";
    item.steps += 1;
    item.runtime += Number(row[3]) || 0;
    item.issueSteps += severity === "OK" ? 0 : 1;
    item.worst = worseTimingSeverity_(item.worst, severity);
    item.lastTimestamp = row[0] || item.lastTimestamp;
  });

  Object.keys(processMap).sort().forEach(function(process) {
    const item = processMap[process];
    rows.push([
      process,
      item.steps,
      Number(item.runtime.toFixed(3)),
      item.issueSteps,
      item.worst,
      item.lastTimestamp,
      item.issueSteps ? "Review timing issue details in Framework Timing Report." : "Within target"
    ]);
  });

  return rows;
}

function collectExistingDashboardQualitySectionBlocks_(sheet) {
  const blocks = {};
  if (!sheet || sheet.getLastRow() < 1) return blocks;

  RFF_DASHBOARD_QUALITY_SECTIONS.forEach(function(section) {
    const titleRow = findDashboardQualitySectionRow_(sheet, section.title);
    if (!titleRow) return;

    const nextRow = findNextDashboardQualitySectionRow_(sheet, titleRow + 1);
    const endRow = nextRow ? nextRow - 1 : sheet.getLastRow();
    const rowCount = Math.max(1, endRow - titleRow + 1);
    const values = sheet.getRange(titleRow, 1, rowCount, RFF_DASHBOARD_QUALITY_COL_WIDTHS.length).getValues();
    blocks[section.title] = trimTrailingBlankRows_(values);
  });

  return blocks;
}

function normalizeDashboardQualitySectionBlock_(section, block) {
  /*
   * v1.5.24:
   * Dashboard Quality sections are dynamic subheader groups, not fixed blocks.
   * Each group has one clear Arial 10 row above, a three-row subheader
   * (section / last updated / detail header), writes data on group row 5,
   * and keeps two clear rows below the section data.
   */
  let rows = trimTrailingBlankRows_(block || []);
  let titleIndex = -1;
  for (let i = 0; i < rows.length; i++) {
    if (String((rows[i] || [])[0] || "").trim() === section.title) {
      titleIndex = i;
      break;
    }
  }

  if (titleIndex < 0) rows = buildDefaultDashboardQualitySectionBlock_(section);
  else rows = rows.slice(titleIndex);

  let detailHeaderIndex = -1;
  for (let i = 1; i < rows.length; i++) {
    const joined = (rows[i] || []).join(" ").toLowerCase();
    if (joined.indexOf("quality notes") !== -1 || joined.indexOf("status") !== -1) {
      detailHeaderIndex = i;
      break;
    }
  }

  const dataRows = [];
  if (detailHeaderIndex >= 0) {
    for (let r = detailHeaderIndex + 1; r < rows.length; r++) {
      if (rowHasAnyValue_(rows[r])) dataRows.push(rows[r]);
    }
  }
  if (!dataRows.length) dataRows.push(["Enter Data here"]);

  const output = [[""], [section.title], ["Last Updated", "Date and time"], getDefaultDashboardQualityDetailHeader_(section)]
    .concat(dataRows)
    .concat([[""], [""]]);
  return output.map(function(row) {
    return normalizeSectionRowForWidth_(row, RFF_DASHBOARD_QUALITY_COL_WIDTHS.length);
  });
}

function rebuildDashboardQualityReportShellCompact_(sheet) {
  if (!sheet) return;

  const requiredCols = RFF_DASHBOARD_QUALITY_COL_WIDTHS.length;
  if (sheet.getMaxColumns() < requiredCols) {
    sheet.insertColumnsAfter(sheet.getMaxColumns(), requiredCols - sheet.getMaxColumns());
  }

  const existingBlocks = collectExistingDashboardQualitySectionBlocks_(sheet);
  const rows = [normalizeSectionRowForWidth_(["Dashboard Quality Report", "- v" + MASTER_LIST_MERGE_ML_VERSION + " -", "Report Formatter Framework"], requiredCols)];

  RFF_DASHBOARD_QUALITY_SECTIONS.forEach(function(section) {
    const block = normalizeDashboardQualitySectionBlock_(section, existingBlocks[section.title]);
    block.forEach(function(row) {
      rows.push(normalizeSectionRowForWidth_(row, requiredCols));
    });
  });

  if (sheet.getMaxRows() < rows.length) {
    sheet.insertRowsAfter(sheet.getMaxRows(), rows.length - sheet.getMaxRows());
  }

  const clearRows = Math.max(sheet.getLastRow(), rows.length, 1);
  sheet.getRange(1, 1, clearRows, requiredCols).clearContent().breakApart();
  sheet.getRange(1, 1, rows.length, requiredCols).setValues(rows);

  try {
    sheet.getRange(1, 1, 1, requiredCols)
      .setFontFamily("Arial")
      .setFontSize(12)
      .setFontWeight("bold")
      .setBackground(RFF_SYSTEM_SHEET_TITLE_COLOR);
    sheet.setFrozenRows(1);
  } catch (err) {
    Logger.log("Dashboard Quality shell title formatting skipped: " + err.message);
  }

  applyDashboardQualityReportColumnSettings_(sheet);
  styleDashboardQualityUpdatedSections_(sheet, RFF_DASHBOARD_QUALITY_SECTIONS.map(function(section) {
    return section.title;
  }));
}

function applyDashboardQualityReportColumnSettings_(sheet, options) {
  if (!sheet) return;

  trimSheetToColumnCount_(sheet, RFF_DASHBOARD_QUALITY_COL_WIDTHS.length, "Dashboard Quality Report");

  for (let i = 0; i < RFF_DASHBOARD_QUALITY_COL_WIDTHS.length; i++) {
    const width = RFF_DASHBOARD_QUALITY_COL_WIDTHS[i];
    if (!width) continue;
    try {
      if (sheet.getColumnWidth(i + 1) !== width) {
        sheet.setColumnWidth(i + 1, width);
      }
    } catch (err) {
      Logger.log("Dashboard Quality width skipped for column " + (i + 1) + ": " + err.message);
    }
  }

  if (options && options.skipRangeFormat) {
    try {
      sheet.setFrozenRows(1);
    } catch (err) {
      Logger.log("Dashboard Quality freeze skipped: " + err.message);
    }
    return;
  }

  const lastRow = Math.max(sheet.getLastRow(), 1);
  const lastCol = RFF_DASHBOARD_QUALITY_COL_WIDTHS.length;

  try {
    sheet.getRange(1, 1, lastRow, lastCol)
      .setFontFamily("Arial")
      .setFontSize(10)
      .setVerticalAlignment("top")
      .setHorizontalAlignment("left")
      .setBorder(true, true, true, true, true, true, RFF_SYSTEM_SHEET_BORDER_COLOR, SpreadsheetApp.BorderStyle.DOTTED);

    sheet.getRange(1, 1, lastRow, 1).setWrapStrategy(SpreadsheetApp.WrapStrategy.OVERFLOW);
    RFF_DASHBOARD_QUALITY_WRAP_COLUMNS.forEach(function(col) {
      sheet.getRange(1, col, lastRow, 1).setWrapStrategy(SpreadsheetApp.WrapStrategy.WRAP);
    });

    sheet.setFrozenRows(1);
  } catch (err) {
    Logger.log("Dashboard Quality base style skipped: " + err.message);
  }
}

function isDashboardQualityNotesLabel_(value) {

function normalizeDashboardQualityOutputRow_(row, governedHeader) {
  const width = RFF_DASHBOARD_QUALITY_COL_WIDTHS.length;
  const source = row || [];
  const out = source.slice(0, width);
  while (out.length < width) out.push("");

  let notesIndex = -1;
  for (let i = 0; i < source.length; i++) {
    if (isDashboardQualityNotesLabel_(source[i])) {
      notesIndex = i;
      break;
    }
  }

  if (notesIndex >= 0 && notesIndex !== 3) {
    out[3] = source[notesIndex] || "Quality Notes";
    if (notesIndex < width) out[notesIndex] = "";
  } else if (notesIndex === 3) {
    out[3] = source[notesIndex] || "Quality Notes";
  }

  if (governedHeader && governedHeader.length) {
    for (let i = 0; i < governedHeader.length && i < width; i++) {
      out[i] = governedHeader[i] || "";
    }
  }

  for (let i = 4; i < width; i++) out[i] = "";
  return out;
}

function normalizeDashboardQualityRowsForSection_(section, rows) {
  const input = rows || [];
  const governedHeader = getDefaultDashboardQualityDetailHeader_(section);
  if (!input.length) return [governedHeader, normalizeDashboardQualityOutputRow_(["Enter Data here"], null)];

  const letter = getDashboardQualitySectionLetter_(section);
  const output = [governedHeader];

  for (let r = 1; r < input.length; r++) {
    const row = input[r] || [];
    let mapped;
    if (letter === "A") mapped = [row[1] || row[0] || "", row[4] || row[1] || "", normalizeDashboardQualityIssueValue_(row[3], row[2]), row[6] || row[3] || ""];
    else if (letter === "B") mapped = [row[1] || row[0] || "", row[4] || row[1] || "", normalizeDashboardQualityIssueValue_(row[3], row[2]), row[6] || row[3] || ""];
    else if (letter === "C") mapped = [row[1] ? String(row[0] || "") + " - " + String(row[1]) : row[0] || "", row[4] || row[1] || "", normalizeDashboardQualityIssueValue_(row[3], row[2]), row[6] || row[3] || ""];
    else if (letter === "D") mapped = [row[1] ? String(row[0] || "") + " - " + String(row[1]) : row[0] || "", row[4] || row[1] || "", normalizeDashboardQualityIssueValue_(row[3], row[2]), row[6] || row[3] || ""];
    else if (letter === "E") mapped = [row[1] ? String(row[0] || "") + " - " + String(row[1]) : row[0] || "", row[4] || row[1] || "", normalizeDashboardQualityIssueValue_(row[3], row[2]), row[6] || row[3] || ""];
    else if (letter === "F") mapped = [row[0] || "", row[2] || row[1] || "", row[3] || row[4] || "", row[6] || row[3] || ""];
    else if (letter === "G") mapped = [row[2] || row[0] || "", row[3] || row[1] || "", row[5] || row[2] || "", row[6] || row[4] || ""];
    else if (letter === "H") mapped = [row[0] || "", row[1] || "", row[2] || "", row[3] || ""];
    else if (letter === "I") mapped = [row[1] || row[0] || "", row[2] || row[1] || "", row[3] || "", row[4] || row[3] || ""];
    else if (letter === "J") mapped = [row[0] || "", row[1] || "", row[2] || "", row[3] || ""];
    else if (letter === "K") mapped = [row[0] || "", row[3] || row[2] || "", row[1] || "", row[2] || row[3] || ""];
    else if (letter === "L") mapped = [row[0] || "", row[2] || row[1] || "", row[4] || row[2] || "", row[6] || row[3] || ""];
    else if (letter === "M" || letter === "N" || letter === "O") mapped = [row[0] || "", row[1] || "", row[2] || "", row[3] || ""];
    else mapped = [row[0] || "", row[1] || "", row[2] || "", row[3] || ""];

    output.push(normalizeDashboardQualityOutputRow_(mapped, null));
  }

  if (output.length === 1) output.push(normalizeDashboardQualityOutputRow_(["Enter Data here"], null));
  return output;
}

function getStatusFromDashboardQualityRows_(rows) {
  if (!rows || !rows.length) return "MISSING";
  for (let i = 1; i < rows.length; i++) {
    const row = rows[i] || [];
    for (let c = 0; c < row.length; c++) {
      const value = String(row[c] || "").trim().toUpperCase();
      if (value === "FAIL") return "FAIL";
    }
  }
  for (let i = 1; i < rows.length; i++) {
    const row = rows[i] || [];
    for (let c = 0; c < row.length; c++) {
      const value = String(row[c] || "").trim().toUpperCase();
      if (value === "WARNING" || value === "MISSING") return value;
    }
  }
  return "PASS";
}

function getDashboardQualitySectionBoundsMap_(sheet) {
  const bounds = {};
  if (!sheet) return bounds;

  const lastRow = Math.max(sheet.getLastRow(), 1);
  const values = sheet.getRange(1, 1, lastRow, 1).getDisplayValues();
  const sectionRows = [];

  for (let i = 0; i < values.length; i++) {
    const title = String(values[i][0] || "").trim();
    if (/^SECTION [A-Z] - /.test(title)) {
      sectionRows.push({ title: title, row: i + 1 });
    }
  }

  sectionRows.forEach(function(section, index) {
    const replaceStartRow = Math.max(2, section.row - 1);
    const nextTitleRow = index + 1 < sectionRows.length ? sectionRows[index + 1].row : 0;
    const replaceEndRow = nextTitleRow ? Math.max(replaceStartRow, nextTitleRow - 2) : lastRow;
    bounds[section.title] = {
      titleRow: section.row,
      replaceStartRow: replaceStartRow,
      oldCount: Math.max(1, replaceEndRow - replaceStartRow + 1)
    };
  });

  return bounds;
}

function replaceDashboardQualitySectionsRows_(sheet, sectionWrites) {
  if (!sheet || !sectionWrites || !sectionWrites.length) return;

  if (!hasDashboardQualityTemplateShell_(sheet)) {
    rebuildDashboardQualityReportShellCompact_(sheet);
  }

  const bounds = getDashboardQualitySectionBoundsMap_(sheet);
  sectionWrites.slice().sort(function(a, b) {
    const aBound = bounds[a.title] || { replaceStartRow: 0 };
    const bBound = bounds[b.title] || { replaceStartRow: 0 };
    return bBound.replaceStartRow - aBound.replaceStartRow;
  }).forEach(function(sectionWrite) {
    replaceDashboardQualitySectionRows_(sheet, sectionWrite.title, sectionWrite.rows, { sectionBounds: bounds, deferStyle: true, skipBreakApart: true });
  });
  styleDashboardQualityUpdatedSections_(sheet, sectionWrites.map(function(sectionWrite) { return sectionWrite.title; }));
}

function replaceDashboardQualitySectionRows_(sheet, sectionTitle, rows, options) {
  if (!sheet || !rows || !rows.length) return;

  if (!hasDashboardQualityTemplateShell_(sheet)) {
    rebuildDashboardQualityReportShellCompact_(sheet);
  }

  const width = RFF_DASHBOARD_QUALITY_COL_WIDTHS.length;
  const normalized = rows.map(function(row) {
    return normalizeSectionRowForWidth_(row || [], width);
  });

  const bounds = options && options.sectionBounds ? options.sectionBounds : getDashboardQualitySectionBoundsMap_(sheet);
  let sectionBound = bounds[sectionTitle];

  if (!sectionBound) {
    rebuildDashboardQualityReportShellCompact_(sheet);
    sectionBound = getDashboardQualitySectionBoundsMap_(sheet)[sectionTitle];
  }
  if (!sectionBound) throw new Error("Dashboard Quality section not found after shell repair: " + sectionTitle);

  const replaceStartRow = sectionBound.replaceStartRow;
  let oldCount = sectionBound.oldCount;

  const output = [normalizeSectionRowForWidth_([], width)].concat(normalized).concat([
    normalizeSectionRowForWidth_([], width),
    normalizeSectionRowForWidth_([], width)
  ]);

  if (output.length > oldCount) {
    sheet.insertRowsAfter(replaceStartRow + oldCount - 1, output.length - oldCount);
    oldCount = output.length;
  } else if (output.length < oldCount) {
    sheet.deleteRows(replaceStartRow + output.length, oldCount - output.length);
    oldCount = output.length;
  }

  const targetRange = sheet.getRange(replaceStartRow, 1, oldCount, width);
  if (dashboardQualitySectionContentMatches_(targetRange, output)) {
    try {
      sheet.getRange(replaceStartRow + 2, 2).setValue(output[2][1]).setNumberFormat("mm/dd/yyyy hh:mm:ss");
    } catch (err) {
      Logger.log("Dashboard Quality Last Updated date-time format skipped for unchanged " + sectionTitle + ": " + err.message);
    }
    return;
  }

  targetRange.clearContent();
  if (!options || !options.skipBreakApart) targetRange.breakApart();
  targetRange.setValues(output);

  try {
    sheet.getRange(replaceStartRow + 2, 2).setNumberFormat("mm/dd/yyyy hh:mm:ss");
  } catch (err) {
    Logger.log("Dashboard Quality Last Updated date-time format skipped for " + sectionTitle + ": " + err.message);
  }

  if (!options || !options.deferStyle) {
    styleDashboardQualityUpdatedSections_(sheet, [sectionTitle]);
  }
}

function findDashboardQualitySectionRow_(sheet, sectionTitle) {
  if (!sheet) return 0;
  const target = String(sectionTitle || "").trim();
  if (!target) return 0;

  const lastRow = sheet.getLastRow();
  if (lastRow < 1) return 0;

  const values = sheet.getRange(1, 1, lastRow, 1).getDisplayValues();
  for (let i = 0; i < values.length; i++) {
    if (String(values[i][0] || "").trim() === target) return i + 1;
  }
  return 0;
}

function findNextDashboardQualitySectionRow_(sheet, startRow) {
  const lastRow = sheet.getLastRow();
  if (startRow > lastRow) return 0;
  const values = sheet.getRange(startRow, 1, lastRow - startRow + 1, 1).getDisplayValues();
  for (let i = 0; i < values.length; i++) {
    const value = String(values[i][0] || "").trim();
    if (/^SECTION [A-Z] - /.test(value)) return startRow + i;
  }
  return 0;
}

function dashboardQualitySectionContentMatches_(range, output) {
  if (!range || !output || !output.length) return false;
  try {
    const existing = range.getValues();
    if (existing.length !== output.length) return false;
    for (let r = 0; r < output.length; r++) {
      for (let c = 0; c < output[r].length; c++) {
        if (r === 2 && c === 1) continue; // Last Updated timestamp changes every run.
        const oldValue = existing[r] && existing[r][c] instanceof Date ? existing[r][c].getTime() : String(existing[r] && existing[r][c] !== undefined ? existing[r][c] : "");
        const newValue = output[r] && output[r][c] instanceof Date ? output[r][c].getTime() : String(output[r] && output[r][c] !== undefined ? output[r][c] : "");
        if (oldValue !== newValue) return false;
      }
    }
    return true;
  } catch (err) {
    Logger.log("Dashboard Quality unchanged-section comparison skipped: " + err.message);
    return false;
  }
}

function mergeDashboardQualityStyleRanges_(ranges) {
  const sorted = (ranges || []).slice().sort(function(a, b) { return a.startRow - b.startRow; });
  const merged = [];
  sorted.forEach(function(item) {
    const start = Math.max(1, Number(item.startRow) || 1);
    const count = Math.max(1, Number(item.rowCount) || 1);
    const end = start + count - 1;
    const previous = merged.length ? merged[merged.length - 1] : null;
    if (previous && start <= previous.endRow + 2) {
      previous.endRow = Math.max(previous.endRow, end);
      previous.rowCount = previous.endRow - previous.startRow + 1;
    } else {
      merged.push({ startRow: start, endRow: end, rowCount: count });
    }
  });
  return merged;
}

function styleDashboardQualityUpdatedSections_(sheet, sectionTitles, options) {
  if (!sheet) return;
  const width = RFF_DASHBOARD_QUALITY_COL_WIDTHS.length;
  const lastRow = Math.max(sheet.getLastRow(), 1);

  const requested = sectionTitles && sectionTitles.length ? sectionTitles : RFF_DASHBOARD_QUALITY_SECTIONS.map(function(section) {
    return section.title;
  });
  const bounds = getDashboardQualitySectionBoundsMap_(sheet);
  const sectionMap = {};
  RFF_DASHBOARD_QUALITY_SECTIONS.forEach(function(section) {
    sectionMap[section.title] = true;
  });

  let ranges = [];
  requested.forEach(function(title) {
    const bound = bounds[title];
    if (bound) {
      ranges.push({ startRow: bound.replaceStartRow, rowCount: bound.oldCount });
    }
  });
  if (!ranges.length) ranges.push({ startRow: 1, rowCount: lastRow });
  ranges = mergeDashboardQualityStyleRanges_(ranges);

  try {
    ranges.forEach(function(item) {
      const rowCount = Math.max(1, Math.min(item.rowCount, lastRow - item.startRow + 1));
      if (rowCount <= 0) return;
      const range = sheet.getRange(item.startRow, 1, rowCount, width);
      const values = range.getDisplayValues();
      const backgrounds = [];
      const fontWeights = [];
      const fontStyles = [];
      const fontSizes = [];

      for (let r = 0; r < rowCount; r++) {
        const absoluteRow = item.startRow + r;
        const first = String((values[r] || [])[0] || "").trim();
        const previousFirst = r > 0 ? String((values[r - 1] || [])[0] || "").trim() : "";
        const twoBackFirst = r > 1 ? String((values[r - 2] || [])[0] || "").trim() : "";
        let background = "#FFFFFF";
        let weight = "normal";
        let style = "normal";
        let size = 10;

        if (absoluteRow === 1) {
          background = RFF_SYSTEM_SHEET_TITLE_COLOR;
          weight = "bold";
          size = 12;
        } else if (sectionMap[first]) {
          background = RFF_SYSTEM_SHEET_SECTION_COLOR;
          weight = "bold";
        } else if (sectionMap[previousFirst]) {
          background = RFF_SYSTEM_SHEET_SUBHEADER_COLOR;
          style = "italic";
        } else if (sectionMap[twoBackFirst]) {
          background = RFF_SYSTEM_SHEET_SUBHEADER_COLOR;
          weight = "bold";
        }

        if (options && options.highlightProblemRows && values[r].some(function(cell) {
          return /^(FAIL|WARNING|MISSING|ACTION REQUIRED)$/i.test(String(cell || "").trim());
        })) {
          background = "#efd0d0";
        }

        backgrounds.push(new Array(width).fill(background));
        fontWeights.push(new Array(width).fill(weight));
        fontStyles.push(new Array(width).fill(style));
        fontSizes.push(new Array(width).fill(size));
      }

      range
        .setFontFamily("Arial")
        .setVerticalAlignment("top")
        .setHorizontalAlignment("left")
        .setBackgrounds(backgrounds)
        .setFontWeights(fontWeights)
        .setFontStyles(fontStyles)
        .setFontSizes(fontSizes);
    });
  } catch (err) {
    Logger.log("Dashboard Quality section styling skipped: " + err.message);
  }
}

    const results = collectFrameworkHealthCheckRows_();

    if (issues.length) {
      markFrameworkStep_(
        timing,
        "Framework Health Check issues detected: " + issues.length,
        formatFrameworkHealthCheckIssuesForTiming_(issues)
      );
    }


    const failed = issues.filter(function(row) {
      return String(row[2] || "").trim().toUpperCase() === "FAIL";
    });


    if (failed.length) {
      const err = new Error("Framework Health Check failed: " + failed.length + " failure(s)");
      err.timingDetails = formatFrameworkHealthCheckIssuesForTiming_(failed);
      throw err;
    }


    return results;
  });
}

function formatFrameworkHealthCheckIssuesForTiming_(issues) {
  return (issues || []).map(function(row) {
    const area = row && row[0] !== undefined ? row[0] : "";
    const item = row && row[1] !== undefined ? row[1] : "";
    const status = row && row[2] !== undefined ? row[2] : "";
    const notes = row && row[3] !== undefined ? row[3] : "";
    return [area, item, status, notes].filter(function(value) {
      return String(value || "").trim() !== "";
    }).join(" - ");
  }).join(" | ");
}

  cpVerification.rows.forEach(function(row) {
    push(row[0], row[1], row[2], row[3]);
  });

  const disenrolledVerification = verifyDisenrolledWorkflowOutputForDashboardQuality_(disenrolledSheet);
  disenrolledVerification.rows.forEach(function(row) {
    push(row[0], row[1], row[2], row[3]);
  });

  const monthlyVerification = verifyMonthlyChangeWorkflowOutputForDashboardQuality_(monthlyChangeSheet, monthSuffix);
  monthlyVerification.rows.forEach(function(row) {
    push(row[0], row[1], row[2], row[3]);
  });

  const overall = allStatuses.indexOf("FAIL") !== -1
    ? "FAIL"
    : allStatuses.indexOf("WARNING") !== -1
      ? "WARNING"
      : "PASS";

  outputRows.splice(1, 0, ["OVERALL WORKFLOW VERIFICATION", overall, allStatuses.length, "Section M verifies Demo P processing, Care Plan sync, Primary PMR assignment, Master List, Disenrolled, and Monthly Change outputs."]);

  return outputRows;
}

function verifyDemoPNotesSummary_(values, headerMap) {
  const notesIdx = headerMap["Notes"];
  if (notesIdx === undefined) {
    return [["Notes Summary", "FAIL", "Missing Notes column", "Notes output column was not found."]];
  }

  let rowsWithNotes = 0;
  (values || []).forEach(function(row) {
    if (String(row[notesIdx] || "").trim() !== "") rowsWithNotes++;
  });

  return [["Notes Summary", rowsWithNotes > 0 ? "PASS" : "WARNING", rowsWithNotes, "Rows with Notes populated."]];
}

function verifyDemoPBannerSummary_(values, headerMap) {
  const bannerIdx = headerMap["Banner Summary"];
  const primaryIdx = headerMap["Primary PMR Row"];

  if (bannerIdx === undefined) {
    return [["Banner Summary", "FAIL", "Missing Banner Summary column", "Banner Summary output column was not found."]];
  }

  let checkedRows = 0;
  let rowsWithBanner = 0;
  let missingBanner = 0;

  (values || []).forEach(function(row) {
    const isPrimary = primaryIdx === undefined || String(row[primaryIdx] || "").trim().toLowerCase() === "yes";
    if (!isPrimary) return;
    checkedRows++;
    if (String(row[bannerIdx] || "").trim() !== "") rowsWithBanner++;
    else missingBanner++;
  });

  return [
    ["Banner Summary - Primary Rows Checked", checkedRows > 0 ? "PASS" : "WARNING", checkedRows, "Primary Demo P rows checked."],
    ["Banner Summary - Missing", missingBanner === 0 ? "PASS" : "WARNING", missingBanner, "Rows with Banner Summary populated: " + rowsWithBanner]
  ];
}

function verifyMasterListWorkflowOutputForDashboardQuality_(masterSheet, demoSheet, monthSuffix) {
  const rows = [];

  if (!masterSheet) {
    return { rows: [["Master List Output", "FAIL", "Not found", monthSuffix ? "Expected Master List " + monthSuffix : "Master List output not found."]] };
  }

  const masterData = getDataValues_(masterSheet, HEADER_ROW, DATA_START_ROW);
  const masterRows = countNonBlankRows_(masterData.values || []);
  const masterPMRIdx = getPMRIndex_(masterData.headerMap || {});
  const masterPMRs = collectKeysFromValuesByIndex_(masterData.values || [], masterPMRIdx, true);

  let demoPrimaryPMRs = new Set();
  if (demoSheet) {
    const demoData = getDataValues_(demoSheet, HEADER_ROW, DATA_START_ROW);
    const demoPMRIdx = getPMRIndex_(demoData.headerMap || {});
    const demoPrimaryIdx = (demoData.headerMap || {})["Primary PMR Row"];
    (demoData.values || []).forEach(function(row) {
      if (demoPrimaryIdx !== undefined && String(row[demoPrimaryIdx] || "").trim().toLowerCase() !== "yes") return;
      const pmr = demoPMRIdx === -1 ? "" : normalizePMR_(row[demoPMRIdx]);
      if (pmr) demoPrimaryPMRs.add(pmr);
    });
  }

  const matches = countSetMatches_(masterPMRs, demoPrimaryPMRs);

  rows.push(["Master List Output Sheet", "PASS", masterSheet.getName(), monthSuffix ? "Expected month " + monthSuffix : "Newest Master List used."]);
  rows.push(["Master List Rows", masterRows > 0 ? "PASS" : "FAIL", masterRows, "Rows with data below header."]);
  rows.push(["Master List Unique PMRs", masterPMRs.size > 0 ? "PASS" : "FAIL", masterPMRs.size, "Unique PMRs in Master List."]);
  rows.push(["Master List / Demo P Primary PMR Match", matches === masterPMRs.size && masterPMRs.size > 0 ? "PASS" : "WARNING", matches, "Master PMRs: " + masterPMRs.size + "; Demo P primary PMRs: " + demoPrimaryPMRs.size]);

  return { rows: rows };
}

function verifyCarePlanWorkflowSyncForDashboardQuality_(masterSheet, cpDueSheet, unlockedSheet, monthSuffix) {
  const rows = [];

  if (!masterSheet) {
    return { rows: [["Care Plan Sync", "FAIL", "Master List not found", "Care Plan sync verification requires Master List."]] };
  }

  const masterData = getDataValues_(masterSheet, HEADER_ROW, DATA_START_ROW);
  const masterHeaderMap = masterData.headerMap || {};
  const masterValues = masterData.values || [];
  const masterPMRIdx = getPMRIndex_(masterHeaderMap);
  const masterNameIdx = masterHeaderMap["Participant Name"];
  const masterPrimaryIdx = masterHeaderMap["Primary PMR Row"];
  const idtIdx = masterHeaderMap["IDT Meeting Date"];
  const lastCpIdx = masterHeaderMap["Last Care Plan"];
  const nextCpIdx = masterHeaderMap["Next Care Plan Due"];
  const cpTypeIdx = masterHeaderMap["CP Type"];

  // v1.5.5: Master List is primary-row output; Primary PMR ownership is verified in Demo P Section M.
  const primaryRows = masterValues;

  const masterPMRs = new Set();
  const masterNames = new Set();
  let missingIdt = 0;
  let missingLast = 0;
  let missingNext = 0;
  let missingType = 0;

  primaryRows.forEach(function(row) {
    const pmr = masterPMRIdx === -1 ? "" : normalizePMR_(row[masterPMRIdx]);
    if (pmr) masterPMRs.add(pmr);

    const name = masterNameIdx === undefined ? "" : normalizeKeyPart_(row[masterNameIdx]);
    if (name) masterNames.add(name);

    if (idtIdx !== undefined && String(row[idtIdx] || "").trim() === "") missingIdt++;
    if (lastCpIdx !== undefined && String(row[lastCpIdx] || "").trim() === "") missingLast++;
    if (nextCpIdx !== undefined && String(row[nextCpIdx] || "").trim() === "") missingNext++;
    if (cpTypeIdx !== undefined && String(row[cpTypeIdx] || "").trim() === "") missingType++;
  });

  const unlockedInfo = getDiagnosticSourceKeyInfo_(unlockedSheet, HEADER_ROW, DATA_START_ROW, "PMR #", true);
  const dueInfo = getDiagnosticSourceKeyInfo_(cpDueSheet, HEADER_ROW, DATA_START_ROW, "Participant Name", false);
  const unlockedMatches = countSetMatches_(masterPMRs, unlockedInfo.keys);
  const dueMatches = countSetMatches_(masterNames, dueInfo.keys);

  rows.push(["Care Plan Sync - Expected Month", monthSuffix ? "PASS" : "WARNING", monthSuffix || "Unknown", "CP Due / Unlocked CP sheets are matched to the Master List month."]);
  rows.push(["Unlocked CP Sheet", unlockedSheet ? "PASS" : "FAIL", unlockedSheet ? unlockedSheet.getName() : "Not found", monthSuffix ? "Expected Unlocked CP " + monthSuffix : ""]);
  rows.push(["Unlocked CP Matches", unlockedMatches > 0 ? "PASS" : "WARNING", unlockedMatches, "Master primary PMRs: " + masterPMRs.size + "; Unlocked rows read: " + unlockedInfo.rowsRead]);
  rows.push(["CP Due Sheet", cpDueSheet ? "PASS" : "FAIL", cpDueSheet ? cpDueSheet.getName() : "Not found", monthSuffix ? "Expected CP Due " + monthSuffix : ""]);
  rows.push(["CP Due Matches", dueMatches > 0 ? "PASS" : "WARNING", dueMatches, "Master primary names: " + masterNames.size + "; CP Due rows read: " + dueInfo.rowsRead]);
  rows.push(["Care Plan Fields - Missing IDT Meeting Date", missingIdt === 0 ? "PASS" : "WARNING", missingIdt, "Primary Master List rows."]);
  rows.push(["Care Plan Fields - Missing Last Care Plan", missingLast === 0 ? "PASS" : "WARNING", missingLast, "Primary Master List rows."]);
  rows.push(["Care Plan Fields - Missing Next Care Plan Due", missingNext === 0 ? "PASS" : "WARNING", missingNext, "Primary Master List rows."]);
  rows.push(["Care Plan Fields - Missing CP Type", missingType === 0 ? "PASS" : "WARNING", missingType, "Primary Master List rows."]);

  unlockedInfo.missingHeaders.forEach(function(header) {
    rows.push(["Unlocked CP Missing Header", "FAIL", header, unlockedSheet ? unlockedSheet.getName() : "Not found"]);
  });
  dueInfo.missingHeaders.forEach(function(header) {
    rows.push(["CP Due Missing Header", "FAIL", header, cpDueSheet ? cpDueSheet.getName() : "Not found"]);
  });

  return { rows: rows };
}
