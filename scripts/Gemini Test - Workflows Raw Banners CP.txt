function hideBannerReportForPreviousMonth_(ss, monthParts, hiddenNames, notHiddenNames) {
  const previousStartText = formatDateForSheetName_(monthParts.previousMonthFirstDay);
  const previousEndText = formatDateForSheetName_(monthParts.previousMonthLastDay);
  const previousRangeText = previousStartText + " - " + previousEndText;


  ss.getSheets().forEach(sheet => {
    const sheetName = String(sheet.getName() || "").trim();
    const lower = sheetName.toLowerCase();


    if (isSystemSheetName_(sheetName)) return;
    if (lower.indexOf("banner") === -1) return;
    if (sheetName.indexOf(previousRangeText) === -1) return;
    if (hiddenNames.indexOf(sheetName) !== -1) return;


    hideSheetSafely_(sheet, hiddenNames, notHiddenNames);
  });
}

function isParticipantEnrollmentStatusDisenrolled_(rows, headerMap) {
  const idx = headerMap["Enrollment Status"];
  if (idx === undefined || idx < 0) return false;


  return (rows || []).some(item => {
    const value = item && item.values ? item.values[idx] : "";
    return String(value || "").trim().toLowerCase() === "disenrolled";
  });
}

function convertMonthlyChangeReportDateValues_(rowValues, reportHeaders) {
  const output = rowValues.slice();


  GLOBAL_DATE_FORMAT_HEADERS.forEach(headerName => {
    const idx = reportHeaders.indexOf(headerName);
    if (idx === -1 || idx >= output.length) return;


    const value = output[idx];


    if (value instanceof Date && !isNaN(value.getTime())) return;


    if (typeof value === "number" && !isNaN(value) && value > 20000) {
      const convertedDate = spreadsheetSerialDateToLocalDate_(value);
      if (convertedDate) output[idx] = convertedDate;
      return;
    }


    const parsed = coerceToValidDate_(value);
    if (parsed) output[idx] = parsed;
  });


  return output;
}

function buildMonthlyChangeReportRow_(sourceRow, sourceHeaders, reportHeaders, changedColumns) {
  const output = sourceRow.slice(0, sourceHeaders.length);


  while (output.length < reportHeaders.length) {
    output.push("");
  }


  const changeIdx = reportHeaders.indexOf("Columns With Change");
  if (changeIdx !== -1) {
    output[changeIdx] = changedColumns && changedColumns.size > 0
      ? Array.from(changedColumns).sort().join(", ")
      : "";
  }


  return convertMonthlyChangeReportDateValues_(output, reportHeaders);
}

  const values = sheet.getRange(1, 1, maxRowsToCheck, maxCols).getDisplayValues();


  for (let r = 0; r < values.length; r++) {
    const row = values[r].map(v => String(v || "").trim());


    if (
      row.indexOf("Last Name") !== -1 &&
      row.indexOf("First Name") !== -1 &&
      (
        row.indexOf("Participant PMR#") !== -1 ||
        row.indexOf("PMR #") !== -1
      )
    ) {
      return r + 1;
    }
  }


  return HEADER_ROW;
}

function formatRawData() {
  const sourceSheet = SpreadsheetApp.getActiveSheet();
  const monthParts = promptForLockedYearReportMonth_("Format Raw Data");
  if (!monthParts) return;


  const timing = startRuntimeTiming_("Format Raw Data", monthParts);


  try {
    if (!sourceSheet) {
      throw new Error("No active sheet found. Select the imported Raw Data sheet before running Format Raw Data.");
    }


    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const dashboard = loadDashboardConfig_();
    const sheetDef = getSheetDefinitionByType_(dashboard, SHEET_TYPE.RAW_DATA);
    const outputName = buildRawDataSheetName_(monthParts);
    const sourceName = sourceSheet.getName();


    if (sourceName === sheetDef.templateName || sourceName === outputName) {
      throw new Error("The active sheet is not a Raw Data import sheet. Select the imported Raw Data source sheet before formatting.");
    }


    ensureRawDataHeaderRows_(sourceSheet);
    markRuntimeStep_(timing, "Ensure Raw Data title/header row structure");


    let headers = getRawDataCurrentHeadersOrDefaults_(sourceSheet);
    headers = prepareRawDataSourceSheetForDashboardFormat_(sourceSheet, headers);
    ensurePrimaryPMRRowColumn_(sourceSheet);
    headers = getHeaders_(sourceSheet, HEADER_ROW);
    markRuntimeStep_(timing, "Normalize Raw Data headers and ensure Primary PMR Row");


    const templateHeaders = getHeadersForSheetType_(dashboard, sheetDef.sheetType);
    const rawData = getDataValues_(sourceSheet, HEADER_ROW, DATA_START_ROW);
    const outputRows = mapRowsToHeaders_(rawData.values, rawData.headers, templateHeaders);


    deleteSheetIfExists_(ss, outputName, sourceName, sheetDef.templateName);
    const outputSheet = createRawDataOutputSheetFromTemplateFast_(
      sheetDef.sheetType,
      outputName,
      outputRows,
      monthParts.firstDay,
      monthParts.lastDay,
      sourceName
    );
    markRuntimeStep_(timing, "Create Raw Data output from template skeleton and paste values");


    processRawDataApprovedSyncColumns_(outputSheet, monthParts);
    markRuntimeStep_(timing, "Populate Raw Data approved sync column: Primary PMR Row");


    if (sourceSheet.getSheetId() !== outputSheet.getSheetId() && ss.getSheets().length > 1) {
      ss.setActiveSheet(outputSheet);
      ss.deleteSheet(sourceSheet);
      markRuntimeStep_(timing, "Delete source Raw Data import sheet");
    }


    lockFinalOutputRowHeights_(outputSheet, "Raw Data");
    markRuntimeStep_(timing, "Lock Raw Data final row heights");


    clearSheetRuntimeCachesForSheet_(outputSheet);
    writeRuntimeTimingReport_(timing);


    notify_(
      "Raw Data formatted as " + outputName + ".\n" +
      "Template-driven output created from Template - Raw Data.\n" +
      "Primary PMR Row was added/populated. Run Process Demo P next when ready."
    );


    return outputSheet;
  } catch (err) {
    markRuntimeStep_(timing, "ERROR - " + err.message);
    writeRuntimeTimingReport_(timing);
    notify_("Format Raw Data stopped: " + err.message + "\nTiming log updated: " + FORMAT_DEMO_P_TIMING_SHEET);
    throw err;
  }
}

function ensureRawDataHeaderRows_(sheet) {
  if (!sheet) return;


  const row4Headers = sheet.getRange(HEADER_ROW, 1, 1, Math.max(sheet.getLastColumn(), 1)).getValues()[0];
  if (rowLooksLikeParticipantHeader_(row4Headers)) return;


  const row1Headers = sheet.getRange(1, 1, 1, Math.max(sheet.getLastColumn(), 1)).getValues()[0];
  if (rowLooksLikeParticipantHeader_(row1Headers)) {
    sheet.insertRowsBefore(1, HEADER_ROW - 1);
    clearSheetRuntimeCachesForSheet_(sheet);
    return;
  }


  // If neither row clearly looks like the source header, do not destructively
  // move rows. The existing formatter will surface missing-header errors later.
}

function getRawDataCurrentHeadersOrDefaults_(sheet) {
  const current = getHeaders_(sheet, HEADER_ROW)
    .map(function(header) { return String(header || "").trim(); })
    .filter(function(header) { return header !== ""; });
  return current.length ? current : getRawDataDemoPSourceHeaders_();
}

function enforceDemoPStrictDashboardSchema_(sheet) {
  /*
   * v1.4.24:
   * This function no longer inserts or deletes columns. Dashboard Section C is
   * the column authority; Template Header Audit reports drift.
   */
  if (!sheet) return;


  const dashboard = loadDashboardConfig_();
  const expectedHeaders = getHeadersForSheetType_(dashboard, SHEET_TYPE.DEMO_P)
    .map(normalizeHeader_);
  const expectedSet = {};
  expectedHeaders.forEach(function(header) {
    if (header) expectedSet[header] = true;
  });


  const currentHeaders = getHeaders_(sheet, HEADER_ROW).map(normalizeHeader_);
  const currentSet = {};
  currentHeaders.forEach(function(header) {
    if (header) currentSet[header] = true;
  });


  const missing = expectedHeaders.filter(function(header) {
    return header && !currentSet[header];
  });
  const unexpected = currentHeaders.filter(function(header) {
    return header && !expectedSet[header];
  });


  if (missing.length || unexpected.length) {
    Logger.log(
      "Demo P header drift detected. Missing: " +
      missing.join(", ") +
      " Unexpected: " +
      unexpected.join(", ")
    );
  }


  clearSheetRuntimeCachesForSheet_(sheet);
}

function processDemoP() {
  const monthParts = promptForLockedYearReportMonth_("Process Demo P");
  if (!monthParts) return;


  const timing = startRuntimeTiming_("Process Demo P", monthParts);
  let movedResult = {
    pmrsRemoved: 0,
    rowsRemoved: 0,
    rowsCopied: 0
  };


  try {
    const buildResult = getOrCreateDemoPProcessingSheet_(monthParts, timing);
    const sheet = buildResult && buildResult.sheet ? buildResult.sheet : buildResult;


    if (buildResult && buildResult.movedResult) {
      movedResult = buildResult.movedResult;
    }


    ensureStandardTitleRows_(sheet);
    markRuntimeStep_(timing, "Prepare Demo P title/header rows");


    enforceDemoPStrictDashboardSchema_(sheet);
    clearSheetRuntimeCachesForSheet_(sheet);
    markRuntimeStep_(timing, "Enforce Demo P strict dashboard schema and clear caches");


    if (!buildResult || !buildResult.sourceWasRawData) {
      movedResult = moveDisenrolledPMRsFromDemoPToExclusion_(sheet, monthParts);
      markRuntimeStep_(
        timing,
        "Move disenrolled PMRs to exclusion list - PMRs removed: " + movedResult.pmrsRemoved +
          ", rows copied: " + movedResult.rowsCopied +
          ", rows removed: " + movedResult.rowsRemoved
      );


      processDemoPAsWorkingSource_(sheet, monthParts, timing);
    } else {
      markRuntimeStep_(
        timing,
        "Disenrolled exclusion built from Raw Data - PMRs excluded from Demo P: " + movedResult.pmrsRemoved +
          ", rows copied: " + movedResult.rowsCopied
      );


      if (buildResult.preProcessedWorkingSource) {
        markRuntimeStep_(timing, "Demo P Working Source processing skipped - already completed in optimized Raw Data single-pass build");
      } else {
        processDemoPAsWorkingSource_(sheet, monthParts, timing);
      }
    }


    if (buildResult && buildResult.createdFromDashboardTemplate) {
      markRuntimeStep_(timing, "Apply Demo P dashboard template formatting skipped - template copy already formatted");
    } else {
      applyDemoPTemplateToSheet_(sheet, monthParts);
      markRuntimeStep_(timing, "Apply Demo P dashboard template formatting");
    }


    setUniqueSheetName_(
      sheet,
      buildMonthlySheetNameNoDashAfterPrefix_(
        "Demo P",
        monthParts.firstDay,
        monthParts.lastDay
      )
    );
    markRuntimeStep_(timing, "Rename processed Demo P sheet");


    lockFinalOutputRowHeights_(sheet, "Demo P");
    markRuntimeStep_(timing, "Lock Demo P final row heights");


    writeRuntimeTimingReport_(timing);


    notify_(
      "Demo P processed as the working source.\n" +
      "Raw Data preserved when source sheet was Raw Data.\n" +
      "Demo P contains active participants only when built from Raw Data.\n" +
      "Disenrolled PMRs excluded from Demo P: " + movedResult.pmrsRemoved + "\n" +
      "Disenrolled primary rows copied to exclusion list: " + movedResult.rowsCopied + "\n" +
      "Contact, phone, address, language, notes, banner, and Primary PMR fields were processed on Demo P.\n" +
      "Care Plan Due and Unlocked Care Plan remain Master List sync functions.\n" +
      "Master List can now be created from Primary PMR rows only."
    );


    return sheet;
  } catch (err) {
    markRuntimeStep_(timing, "ERROR - " + err.message);
    writeRuntimeTimingReport_(timing);
    notify_("Process Demo P stopped: " + err.message + "\nTiming log updated: " + FORMAT_DEMO_P_TIMING_SHEET);
    throw err;
  }
}

function getOrCreateDemoPProcessingSheet_(monthParts, timing) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const active = ss.getActiveSheet();
  const rawName = buildRawDataSheetName_(monthParts);
  const targetName = buildMonthlySheetNameNoDashAfterPrefix_("Demo P", monthParts.firstDay, monthParts.lastDay);


  if (!active) {
    throw new Error("No active sheet found. Select the Raw Data sheet or existing Demo P sheet before running Process Demo P.");
  }


  const activeName = String(active.getName() || "").trim();
  const activeLower = activeName.toLowerCase();


  if (activeName === targetName || activeLower.indexOf("demo p") === 0) {
    markRuntimeStep_(timing, "Using active Demo P sheet: " + activeName);
    return {
      sheet: active,
      sourceWasRawData: false,
      movedResult: {
        pmrsRemoved: 0,
        rowsRemoved: 0,
        rowsCopied: 0
      },
      preProcessedWorkingSource: false
    };
  }


  let rawSheet = null;


  if (activeName === rawName || activeLower.indexOf("raw data") === 0) {
    rawSheet = active;
  } else {
    rawSheet = ss.getSheetByName(rawName);
  }


  if (!rawSheet) {
    throw new Error("Raw Data sheet was not found for " + Utilities.formatDate(monthParts.firstDay, Session.getScriptTimeZone(), "MM.yy") + ". Run Format Raw Data first.");
  }


  deleteSheetIfExistsForDemoPProcess_(ss, targetName, rawSheet.getName());


  const buildResult = createActiveDemoPFromRawData_(rawSheet, targetName, monthParts, timing);
  const demoSheet = buildResult && buildResult.sheet ? buildResult.sheet : buildResult;


  ss.setActiveSheet(demoSheet);
  clearSheetRuntimeCachesForSheet_(demoSheet);
  markRuntimeStep_(timing, "Active-only processed Demo P built from protected Raw Data: " + targetName);


  return {
    sheet: demoSheet,
    sourceWasRawData: true,
    createdFromDashboardTemplate: true,
    preProcessedWorkingSource: !!(buildResult && buildResult.preProcessedWorkingSource),
    movedResult: (buildResult && buildResult.movedResult) ? buildResult.movedResult : getLastRawDataDisenrolledBuildResult_()
  };
}

function createActiveDemoPFromRawData_(rawSheet, targetName, monthParts, timing) {
  const rawData = getDataValues_(rawSheet, HEADER_ROW, DATA_START_ROW);

  if (!rawData.values.length) {
    throw new Error("Raw Data has no participant rows to process.");
  }

  const pmrIdx = getPMRIndex_(rawData.headerMap);
  if (pmrIdx === -1) {
    throw new Error("Raw Data is missing Participant PMR# / PMR # column.");
  }

  const disenrolledPMRSet = loadDisenrolledPMRSetForMonth_(monthParts);
  const split = splitRawDataRowsIntoActiveAndDisenrolled_(rawData, disenrolledPMRSet);
  const movedResult = {
    pmrsRemoved: split.disenrolledPMRs.size,
    rowsRemoved: 0,
    rowsCopied: split.disenrolledRows.length,
    primaryRowsOnly: true
  };

  setLastRawDataDisenrolledBuildResult_(movedResult);

  movedResult.outputRowsWritten = 0;
  markRuntimeStep_(timing, "Disenrolled PMRs identified and excluded from Demo P - " +
    (disenrolledPMRSet && disenrolledPMRSet.size ? "used Disenrolled List PMR set" : "used Raw Data disenrollment fields") +
    " | Rows Read: " + rawData.values.length +
    "; Rows Processed: " + split.activeRows.length +
    "; Rows Excluded: " + split.disenrolledRows.length);

  const dashboard = loadDashboardConfig_();
  const sheetDef = getSheetDefinitionByType_(dashboard, SHEET_TYPE.DEMO_P);
  const headers = getHeadersForSheetType_(dashboard, SHEET_TYPE.DEMO_P);
  const outputRows = mapRowsToHeaders_(split.activeRows, rawData.headers, headers);

  const workingData = {
    headers: headers,
    headerMap: buildHeaderIndexMap_(headers),
    values: outputRows,
    range: null
  };

  populateDemoPUpdateColumns_(workingData, monthParts, rawSheet.getName(), "Created");
  markRuntimeStep_(timing, "Demo P Working Source - populate Demo P update tracking columns | Rows Processed: " + workingData.values.length + "; Write Mode: in-memory only");

  const demoSheet = createOutputSheetFromDashboardTemplate_(
    SHEET_TYPE.DEMO_P,
    targetName,
    workingData.values,
    monthParts.firstDay,
    monthParts.lastDay
  );
  markRuntimeStep_(timing, "Demo P Working Source - copy active Raw Data rows to Demo P template | Rows Read: " + rawData.values.length + "; Rows Written: " + workingData.values.length + "; Write Mode: single output write");

  try {
    demoSheet.getRange("A1").setValue(sheetDef.reportTitle || "Demo P");
  } catch (err) {
    Logger.log("Could not set Demo P title: " + err.message);
  }

  clearSheetRuntimeCachesForSheet_(demoSheet);

  processDemoPAsWorkingSource_(demoSheet, monthParts, timing);

  return {
    sheet: demoSheet,
    movedResult: movedResult,
    createdFromDashboardTemplate: true,

function splitRawDataRowsIntoActiveAndDisenrolled_(rawData, disenrolledPMRSet) {
  const headerMap = rawData.headerMap;
  const pmrIdx = getPMRIndex_(headerMap);
  const statusIdx = headerMap["Enrollment Status"];
  const disenrollmentDateIdx = findHeaderIndex_(headerMap, ["Disenrollment Effective Date", "Disenrollment Date", "Date of Death"]);
  const useExternalDisenrolledSet = !!(disenrolledPMRSet && disenrolledPMRSet.size);


  const activeRows = [];
  const disenrolledRows = [];
  const disenrolledRowsByPMR = new Map();
  const disenrolledPMRs = new Set();


  const values = rawData.values || [];
  let i = 0;


  while (i < values.length) {
    const firstRow = values[i];
    const pmr = normalizePMR_(firstRow[pmrIdx]);


    if (!pmr) {
      i++;
      continue;
    }


    const block = [];
    let blockIsDisenrolled = useExternalDisenrolledSet && disenrolledPMRSet.has(pmr);


    while (i < values.length && normalizePMR_(values[i][pmrIdx]) === pmr) {
      const row = values[i];
      block.push(row);


      if (!useExternalDisenrolledSet) {
        const status = statusIdx === undefined ? "" : String(row[statusIdx] || "").trim().toLowerCase();
        const disenrollmentValue = disenrollmentDateIdx === -1 ? "" : row[disenrollmentDateIdx];


        if (status === "disenrolled" || normalizeCompareValue_(disenrollmentValue) !== "") {
          blockIsDisenrolled = true;
        }
      }


      i++;
    }


    if (blockIsDisenrolled) {
      disenrolledPMRs.add(pmr);
      const primaryRow = block[0];
      disenrolledRows.push(primaryRow);
      disenrolledRowsByPMR.set(pmr, [primaryRow]);
    } else {
      for (let b = 0; b < block.length; b++) {
        activeRows.push(block[b]);
      }
    }
  }


  return {
    activeRows: activeRows,
    disenrolledRows: disenrolledRows,
    disenrolledRowsByPMR: disenrolledRowsByPMR,
    disenrolledPMRs: disenrolledPMRs
  };
}

function loadDisenrolledPMRSetForMonth_(monthParts) {

  const candidates = [];

    if (set.size) return set;
  }


  return set;
}

function createRawDataOutputSheetFromTemplateFast_(sheetType, outputName, outputRows, firstDay, endDate, sourceName) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const dashboard = loadDashboardConfig_();
  const normalizedSheetType = normalizeDashboardSheetTypeKey_(sheetType);
  const sheetDef = getSheetDefinitionByType_(dashboard, normalizedSheetType);
  const headers = getHeadersForSheetType_(dashboard, normalizedSheetType);
  const template = ss.getSheetByName(sheetDef.templateName);

  if (!template) {
    throw new Error("Template not found: " + sheetDef.templateName + ". Run Create / Refresh All Templates first.");
  }

  deleteSheetIfExists_(ss, outputName, sourceName, sheetDef.templateName);
  const sheet = ss.insertSheet(outputName);
  sheet.showSheet();
  ss.setActiveSheet(sheet);

  const width = Math.max(headers.length, 1);
  const rows = outputRows || [];
  const requiredRows = rows.length ? DATA_START_ROW + rows.length - 1 : DATA_START_ROW;

  resizeSheetColumns_(sheet, width);
  if (sheet.getMaxRows() < requiredRows) {
    sheet.insertRowsAfter(sheet.getMaxRows(), requiredRows - sheet.getMaxRows());
  }

  const templateTopRows = Math.max(DATA_START_ROW - 1, HEADER_ROW);
  try {
    template.getRange(1, 1, templateTopRows, width)
      .copyTo(sheet.getRange(1, 1, templateTopRows, width), { contentsOnly: false });
  } catch (err) {
    Logger.log("Raw Data template title/header copy skipped for " + outputName + ": " + err.message);
  }

  const dataRows = Math.max(requiredRows - DATA_START_ROW + 1, 1);
  try {
    template.getRange(DATA_START_ROW, 1, 1, width)
      .copyTo(sheet.getRange(DATA_START_ROW, 1, dataRows, width), { formatOnly: true });
  } catch (err) {
    Logger.log("Raw Data template data-row formatting copy skipped for " + outputName + ": " + err.message);
  }

  try {
    for (let col = 1; col <= width; col++) {
      sheet.setColumnWidth(col, template.getColumnWidth(col));
    }
    sheet.setFrozenRows(template.getFrozenRows());
    sheet.setFrozenColumns(template.getFrozenColumns());
    if (template.getTabColor()) sheet.setTabColor(template.getTabColor());
  } catch (err) {
    Logger.log("Raw Data template sheet settings copy skipped for " + outputName + ": " + err.message);
  }

  try {
    sheet.getRange("A1").setValue(sheetDef.reportTitle || normalizedSheetType || outputName);
    if (firstDay) sheet.getRange("B2").setValue(firstDay);
    if (endDate) sheet.getRange("D2").setValue(endDate);
    if (headers.length) sheet.getRange(HEADER_ROW, 1, 1, width).setValues([headers]);
  } catch (err) {
    Logger.log("Raw Data output metadata skipped for " + outputName + ": " + err.message);
  }

  if (rows.length) {
    sheet.getRange(DATA_START_ROW, 1, rows.length, width).setValues(rows.map(function(row) {
      const out = (row || []).slice(0, width);
      while (out.length < width) out.push("");
      return out;
    }));
  }

  try {
    const existingFilter = sheet.getFilter && sheet.getFilter();
    if (existingFilter) existingFilter.remove();
    sheet.getRange(HEADER_ROW, 1, Math.max(sheet.getMaxRows() - HEADER_ROW + 1, 1), width).createFilter();
  } catch (err) {
    Logger.log("Raw Data output filter creation skipped for " + outputName + ": " + err.message);
  }

  clearSheetRuntimeCachesForSheet_(sheet);
  return sheet;
}

function moveDisenrolledPMRsFromDemoPToExclusion_(demoSheet, monthParts) {
  const result = {
    pmrsRemoved: 0,
    rowsRemoved: 0,
    rowsCopied: 0
  };


  if (!demoSheet) return result;


  const lastRow = demoSheet.getLastRow();
  const lastCol = demoSheet.getLastColumn();


  if (lastRow < DATA_START_ROW || lastCol < 1) return result;


  const headers = getHeaders_(demoSheet, HEADER_ROW);
  const headerMap = buildHeaderIndexMap_(headers);
  const pmrIdx = getPMRIndex_(headerMap);
  const statusIdx = headerMap["Enrollment Status"];


  if (pmrIdx === -1 || statusIdx === undefined) {
    Logger.log("Demo P disenrolled split skipped: missing PMR or Enrollment Status header.");
    return result;
  }


  const dataRange = demoSheet.getRange(DATA_START_ROW, 1, lastRow - DATA_START_ROW + 1, lastCol);
  const values = dataRange.getValues();
  const disenrolledPMRs = new Set();


  values.forEach(row => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr) return;


    const status = String(row[statusIdx] || "").trim().toLowerCase();
    if (status === "disenrolled") disenrolledPMRs.add(pmr);
  });


  if (disenrolledPMRs.size === 0) return result;


  const rowsToRemove = [];
  const rowsByPMR = new Map();


  values.forEach((row, idx) => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr || !disenrolledPMRs.has(pmr)) return;


    rowsToRemove.push(DATA_START_ROW + idx);
    if (!rowsByPMR.has(pmr)) rowsByPMR.set(pmr, []);
    rowsByPMR.get(pmr).push(row);
  });


  result.pmrsRemoved = disenrolledPMRs.size;
  result.rowsRemoved = rowsToRemove.length;
  result.rowsCopied = appendDisenrolledPMRBlocksToExclusion_(demoSheet, headers, rowsByPMR, monthParts);


  deleteRowNumberBatches_(demoSheet, rowsToRemove);
  clearSheetRuntimeCachesForSheet_(demoSheet);


  return result;
}

function appendDisenrolledPMRBlocksToExclusion_(sourceSheet, sourceHeaders, rowsByPMR, monthParts) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const dashboard = loadDashboardConfig_();
  const sheetDef = getSheetDefinitionByType_(dashboard, SHEET_TYPE.DISENROLLED_EXCLUSION);
  const template = ss.getSheetByName(sheetDef.templateName);
  if (!template) {
    throw new Error("Template not found: " + sheetDef.templateName + ". Run Create / Refresh Disenrolled Exclusion List Template first.");
  }

  const exclusionHeaders = getHeadersForSheetType_(dashboard, SHEET_TYPE.DISENROLLED_EXCLUSION);
  const sourceHeaderMap = buildHeaderIndexMap_(sourceHeaders || []);
  const sourcePMRIdx = getPMRIndex_(sourceHeaderMap);
  const exclusionHeaderMap = buildHeaderIndexMap_(exclusionHeaders);
  const exclusionPMRIdx = getPMRIndex_(exclusionHeaderMap);

  const sourceIndexes = exclusionHeaders.map(function(header) {
    return Object.prototype.hasOwnProperty.call(sourceHeaderMap, header) ? sourceHeaderMap[header] : -1;
  });

  const rowsToWrite = [];
  if (rowsByPMR && rowsByPMR.size) {
    rowsByPMR.forEach(function(rows) {
      const sourceRow = rows && rows.length ? rows[0] : null;
      if (!sourceRow) return;

      const output = sourceIndexes.map(function(sourceIdx) {
        return sourceIdx === -1 ? "" : sourceRow[sourceIdx];
      });

      if (exclusionPMRIdx >= 0 && sourcePMRIdx >= 0 && !normalizePMR_(output[exclusionPMRIdx])) {
        output[exclusionPMRIdx] = sourceRow[sourcePMRIdx];
      }
      rowsToWrite.push(output);
    });
  }

  const outputSheetName = buildDatedDisenrolledOutputName_(monthParts);
  const legacy = ss.getSheetByName(DISENROLLED_EXCLUSION_SHEET);
  if (legacy && ss.getSheets().length > 1) ss.deleteSheet(legacy);
  const existing = ss.getSheetByName(outputSheetName);
  if (existing && ss.getSheets().length > 1) ss.deleteSheet(existing);

  const sheet = template.copyTo(ss);
  sheet.showSheet();
  setUniqueSheetName_(sheet, outputSheetName);

  const width = Math.max(exclusionHeaders.length, 1);
  const startDate = monthParts && monthParts.firstDay ? monthParts.firstDay : getTodayLocalDate_();
  const endDate = monthParts && monthParts.lastDay ? monthParts.lastDay : startDate;

  // v1.5: The copied dashboard template already owns headers, row count, column count, and formatting.
  // Do not force row count or clear blank data rows before the first output write.
  sheet.getRange("A1").setValue(DISENROLLED_EXCLUSION_SHEET);
  sheet.getRange("A2:D2").setValues([["Date:", startDate, "to", endDate]]);
  sheet.getRange(HEADER_ROW, 1, 1, width).setValues([exclusionHeaders]);

  if (rowsToWrite.length) {
    sheet.getRange(DATA_START_ROW, 1, rowsToWrite.length, width).setValues(rowsToWrite);
  }

  lockFinalOutputRowHeights_(sheet, SHEET_TYPE.DISENROLLED_EXCLUSION);

  try {
    sheet.getRange("A1").setNote(
      "Rows Processed: " + (rowsByPMR ? rowsByPMR.size : 0) + "\\n" +
      "Rows Written: " + rowsToWrite.length + "\\n" +
      "v1.5 Disenrolled single-write output"
    );
  } catch (err) {
    Logger.log("Disenrolled output note skipped: " + err.message);
  }

  clearSheetRuntimeCachesForSheet_(sheet);
  return rowsToWrite.length;
}

function syncDemoPBannerColumns_(demoSheet, monthParts) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const bannerSheet = getCurrentBannersSheet_(monthParts);


  if (!bannerSheet) {
    notify_("Banners sheet was not found. Demo P Banner sync skipped.");
    return;
  }


  const demoLastRow = demoSheet.getLastRow();
  const demoLastCol = demoSheet.getLastColumn();


  if (demoLastRow < DATA_START_ROW || demoLastCol < 1) return;


  const demoHeaders = getHeaders_(demoSheet, HEADER_ROW);
  const demoHeaderMap = buildHeaderIndexMap_(demoHeaders);


  const demoPMRIdx = getPMRIndex_(demoHeaderMap);
  const demoLastNameIdx = demoHeaderMap["Last Name"];
  const demoFirstNameIdx = demoHeaderMap["First Name"];


  if (demoPMRIdx === -1 || demoLastNameIdx === undefined || demoFirstNameIdx === undefined) {
    notify_("Demo P Banner sync skipped: missing Participant PMR#, Last Name, or First Name.");
    return;
  }


  const bannerMap = buildSourceMapByCompositeKeyForDemoPBanner_(
    bannerSheet,
    HEADER_ROW,
    DATA_START_ROW,
    ["Participant PMR#", "Last Name", "First Name"]
  );


  if (bannerMap.size === 0) {
    notify_("Demo P Banner sync skipped: no matching Banner records were found.");
    return;
  }


  const range = demoSheet.getRange(DATA_START_ROW, 1, demoLastRow - DATA_START_ROW + 1, demoLastCol);
  const values = range.getValues();


  values.forEach(row => {
    const demoDOBIdx = getDOBIndex_(demoHeaderMap);
    if (demoDOBIdx !== -1 && normalizeCompareValue_(row[demoDOBIdx]) === "") return;


    const key = [
      normalizeKeyPart_(row[demoPMRIdx]),
      normalizeKeyPart_(row[demoLastNameIdx]),
      normalizeKeyPart_(row[demoFirstNameIdx])
    ].join("|||");


    if (!bannerMap.has(key)) return;


    const source = bannerMap.get(key);


    DEMO_P_BANNER_SYNC_HEADERS.forEach(header => {
      const targetIdx = demoHeaderMap[header];
      if (targetIdx === undefined) return;
      if (!Object.prototype.hasOwnProperty.call(source, header)) return;
      row[targetIdx] = source[header];
    });
  });


  range.setValues(values);
}

function buildSourceMapByCompositeKeyForDemoPBanner_(sheet, headerRow, dataStartRow, keyHeaders) {
  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();
  const map = new Map();


  if (lastRow < dataStartRow || lastCol < 1) return map;


  const headers = getHeaders_(sheet, headerRow);
  const headerMap = buildHeaderIndexMap_(headers);


  if (keyHeaders.some(header => headerMap[header] === undefined)) return map;


  const data = sheet.getRange(dataStartRow, 1, lastRow - dataStartRow + 1, lastCol).getValues();


  data.forEach(row => {
    const key = keyHeaders
      .map(header => normalizeKeyPart_(row[headerMap[header]]))
      .join("|||");


    if (key.replace(/\|/g, "") === "") return;


    const record = {};
    headers.forEach((header, idx) => {
      if (header) record[header] = row[idx];
    });


    map.set(key, record);
  });


  return map;
}

function buildDisenrolledPMRSetFromDemoPValues_(demoValues, demoHeaderMap) {

function removeDisenrolledPMRBlocksFromMasterUsingDemoP_(masterSheet, demoSheet) {
  if (!masterSheet || !demoSheet) return 0;


  const demoHeaders = getHeaders_(demoSheet, HEADER_ROW);
  const demoHeaderMap = buildHeaderIndexMap_(demoHeaders);
  const demoLastRow = demoSheet.getLastRow();
  const demoLastCol = demoSheet.getLastColumn();


  if (demoLastRow < DATA_START_ROW || demoLastCol < 1) return 0;


  const demoValues = demoSheet.getRange(DATA_START_ROW, 1, demoLastRow - DATA_START_ROW + 1, demoLastCol).getValues();
  const disenrolledPMRs = buildDisenrolledPMRSetFromDemoPValues_(demoValues, demoHeaderMap);


  const exclusionPMRs = loadDisenrolledExclusionPMRsForPart3_();
  exclusionPMRs.forEach(pmr => disenrolledPMRs.add(pmr));


  if (disenrolledPMRs.size === 0) return 0;


  const masterHeaders = getHeaders_(masterSheet, HEADER_ROW);
  const masterHeaderMap = buildHeaderIndexMap_(masterHeaders);
  const masterPMRIdx = getPMRIndex_(masterHeaderMap);


  if (masterPMRIdx === -1) return 0;


  const masterLastRow = masterSheet.getLastRow();
  const masterLastCol = masterSheet.getLastColumn();


  if (masterLastRow < DATA_START_ROW || masterLastCol < 1) return 0;


  const range = masterSheet.getRange(DATA_START_ROW, 1, masterLastRow - DATA_START_ROW + 1, masterLastCol);
  const values = range.getValues();


  const keepRows = [];
  let removed = 0;


  values.forEach(row => {
    const pmr = normalizePMR_(row[masterPMRIdx]);
    if (pmr && disenrolledPMRs.has(pmr)) {
      removed++;
      return;
    }
    keepRows.push(row);
  });


  if (removed === 0) return 0;


  range.clearContent();


  if (keepRows.length > 0) {
    masterSheet.getRange(DATA_START_ROW, 1, keepRows.length, masterLastCol).setValues(keepRows);
  }


  return removed;
}

function createMasterList() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const monthParts = promptForLockedYearReportMonth_("Create Master List");
  if (!monthParts) return;


  const timing = startRuntimeTiming_("Create Master List", monthParts);
  let masterSheet = null;
  let copiedRowCount = 0;


  try {
    markRuntimeStep_(timing, "Start");


    const demoSheet = getCurrentDemoPSheet_(monthParts);
    markRuntimeStep_(timing, "Locate current processed Demo P sheet");


    if (!demoSheet) {
      notify_("Demo P sheet for that month was not found. Run Process Demo P first.");
      markRuntimeStep_(timing, "Stopped - missing Demo P");
      writeRuntimeTimingReport_(timing);
      return;
    }


    const masterName = buildMonthlySheetName_(MASTER_LIST_PREFIX, monthParts.firstDay, monthParts.lastDay);


    masterSheet = ss.getSheetByName(masterName);
    if (masterSheet) {
      const ui = SpreadsheetApp.getUi();
      const response = ui.alert(
        "Master List Exists",
        `${masterName} already exists. Replace it?`,
        ui.ButtonSet.YES_NO
      );


      if (response !== ui.Button.YES) {
        notify_("Create Master List cancelled.");
        markRuntimeStep_(timing, "Cancelled - existing Master List not replaced");
        writeRuntimeTimingReport_(timing);
        return;
      }


      ss.deleteSheet(masterSheet);
      markRuntimeStep_(timing, "Delete existing Master List sheet");
    }


    masterSheet = createMasterListSheetFromTemplate_(ss, masterName, monthParts);
    markRuntimeStep_(timing, "Copy Template - Master List as output sheet");


    buildMasterListHeadersBeforeDataCopy_(demoSheet, masterSheet);
    markRuntimeStep_(timing, "Build Master List headers");


    copiedRowCount = copyPrimaryDemoPRowsToMasterListByHeader_(demoSheet, masterSheet, monthParts);
    markRuntimeStep_(timing, "Copy Primary PMR rows from processed Demo P to Master List");


    markRuntimeStep_(timing, "Master List output row trimming skipped - template controls sheet size");


    fillMasterListMetadataColumns_(masterSheet, monthParts, demoSheet.getName(), "Created from processed Demo P primary rows");
    markRuntimeStep_(timing, "Fill Master List metadata columns");


    syncMasterListFromUnlockedCarePlan_(masterSheet, monthParts, null);
    markRuntimeStep_(timing, "Sync Unlocked CP to Master List primary rows");


    syncMasterListFromCarePlanDue_(masterSheet, monthParts, null);
    markRuntimeStep_(timing, "Sync Care Plan Due to Master List primary rows");


    setUniqueSheetName_(masterSheet, masterName);
    markRuntimeStep_(timing, "Set final Master List sheet name");


    enforceMasterListFinalDisplay_(masterSheet);
    markRuntimeStep_(timing, "Final Master List display enforcement");


    lockFinalOutputRowHeights_(masterSheet, "Master List");
    markRuntimeStep_(timing, "Lock Master List final row heights");


    writeRuntimeTimingReport_(timing);


    notify_(
      `Master List created. Copied ${copiedRowCount} Primary PMR row(s) from processed Demo P.\n\n` +
      `Runtime: ${formatSeconds_((new Date().getTime() - timing.startMs) / 1000)}`
    );


    return masterSheet;
  } catch (err) {
    markRuntimeStep_(timing, "ERROR - " + err.message);
    try {
      writeRuntimeTimingReport_(timing);
    } catch (reportErr) {
      Logger.log("Could not write timing report: " + reportErr.message);
    }
    throw err;
  }
}

function copyPrimaryDemoPRowsToMasterListByHeader_(demoSheet, masterSheet, monthParts) {
  const demoData = getDataValues_(demoSheet, HEADER_ROW, DATA_START_ROW);

  if (!demoData.values.length) return 0;

  const demoHeaders = demoData.headers;
  const demoHeaderMap = demoData.headerMap;
  const masterHeaders = getHeaders_(masterSheet, HEADER_ROW);
  const masterHeaderMap = getHeaderMap_(masterSheet, HEADER_ROW);

  const pmrIdx = getPMRIndex_(demoHeaderMap);
  if (pmrIdx === -1) {
    throw new Error("Demo P is missing Participant PMR# / PMR # column.");
  }

  const primaryIdx = demoHeaderMap["Primary PMR Row"];
  const dobIdx = getDOBIndex_(demoHeaderMap);

  function getMasterTargetIndex_(header) {
    if (masterHeaderMap[header] !== undefined) return masterHeaderMap[header];

    const aliases = {
      "Participant PMR#": ["PMR #", "PMR#", "Participant PMR"],
      "PMR #": ["Participant PMR#", "PMR#", "Participant PMR"],
      "PMR#": ["Participant PMR#", "PMR #", "Participant PMR"],
      "Participant PMR": ["Participant PMR#", "PMR #", "PMR#"]
    };

    const candidates = aliases[header] || [];
    for (let i = 0; i < candidates.length; i++) {
      if (masterHeaderMap[candidates[i]] !== undefined) return masterHeaderMap[candidates[i]];
    }

    return undefined;
  }

  function buildOutputRows_(mode) {
    const output = [];
    const seenPMRs = new Set();

    demoData.values.forEach(row => {
      const pmr = normalizePMR_(row[pmrIdx]);
      if (!pmr) return;

      let includeRow = false;

      if (mode === "primary") {
        includeRow = primaryIdx !== undefined && isPrimaryPMRRowValue_(row[primaryIdx]);
      } else if (mode === "dob") {
        includeRow = dobIdx !== -1 && normalizeCompareValue_(row[dobIdx]) !== "" && !seenPMRs.has(pmr);
      } else if (mode === "first") {
        includeRow = !seenPMRs.has(pmr);
      }

      if (!includeRow) return;
      seenPMRs.add(pmr);

      const out = new Array(masterHeaders.length).fill("");

      demoHeaders.forEach((header, sourceIdx) => {
        if (!header) return;
        const targetIdx = getMasterTargetIndex_(header);
        if (targetIdx !== undefined) {
          out[targetIdx] = row[sourceIdx];
        }
      });

      output.push(out);
    });

    return output;
  }

  let output = buildOutputRows_("primary");

  /*
   * v1.5.1:
   * Master List creation depends on Primary PMR Row when available. Some v1.5
   * runs produced processed Demo P sheets where the Primary PMR Row column existed
   * but was blank, which caused an empty Master List. Preserve Primary PMR
   * governance first, then fall back to one row per PMR using DOB/first-row
   * selection only when no primary rows are detected.
   */
  if (output.length === 0) output = buildOutputRows_("dob");
  if (output.length === 0) output = buildOutputRows_("first");

  if (output.length > 0) {
    masterSheet.getRange(DATA_START_ROW, 1, output.length, masterHeaders.length).setValues(output);
    clearSheetRuntimeCachesForSheet_(masterSheet);
  }

  return output.length;
}

function fillMasterListMetadataColumns_(sheet, monthParts, sourceSheetName, statusText) {
  const lastRow = sheet.getLastRow();
  if (lastRow < DATA_START_ROW) return;


  const headers = getHeaders_(sheet, HEADER_ROW);
  const headerMap = buildHeaderIndexMap_(headers);


  const required = ["Master List Update Status", "Master List Update Month", "Master List Source Sheet"];
  const missing = required.filter(function(header) { return headerMap[header] === undefined; });
  if (missing.length) {
    Logger.log("Master List metadata write skipped for missing dashboard-defined columns: " + missing.join(", "));
  }


  const rowCount = lastRow - DATA_START_ROW + 1;
  const metadata = {
    "Master List Update Status": statusText || "Created",
    "Master List Update Month": monthParts.firstDay,
    "Master List Source Sheet": sourceSheetName || ""
  };


  const indexes = required
    .map(function(header) { return headerMap[header]; })
    .filter(function(idx) { return idx !== undefined; })
    .sort(function(a, b) { return a - b; });


  if (!indexes.length) return;


  if (indexes[indexes.length - 1] - indexes[0] + 1 === indexes.length) {
    const startIdx = indexes[0];
    const width = indexes.length;
    const rowTemplate = [];
    for (let c = startIdx; c <= indexes[indexes.length - 1]; c++) {
      const header = headers[c];
      rowTemplate.push(Object.prototype.hasOwnProperty.call(metadata, header) ? metadata[header] : "");
    }
    const values = Array.from({ length: rowCount }, function() { return rowTemplate.slice(); });
    sheet.getRange(DATA_START_ROW, startIdx + 1, rowCount, width).setValues(values);
    return;
  }


  indexes.forEach(function(idx) {
    const header = headers[idx];
    const value = Object.prototype.hasOwnProperty.call(metadata, header) ? metadata[header] : "";
    const values = Array.from({ length: rowCount }, function() { return [value]; });
    sheet.getRange(DATA_START_ROW, idx + 1, rowCount, 1).setValues(values);
  });
}

  values.forEach(row => {

  });


  return pmrs;
}

  if (!monthParts) return null;

  const previousDemo = getPreviousDemoPSheet_(monthParts);

    );
    return null;
  }


  if (!previousDemo) {
    notify_(
      "Previous month Demo P was not found.\n\n" +

      `Previous PMRs found: ${sectionData.previousData.rowsByPMR.size}\n\n` +
      "For enrollments, Capitation Date must equal the report date. " +
      "For disenrollments, Disenrollment Effective Date must be between the previous month first day and the report date."
    );

  }


  const reportName = buildMonthlySheetName_(MONTHLY_CHANGE_REPORT_PREFIX, monthParts.firstDay, monthParts.lastDay);

function compareRawDemoPForSectionReport_(previousDemo, currentDemo, monthParts) {
  const previousData = getRawDemoPDataForCompare_(previousDemo);
  const currentData = getRawDemoPDataForCompare_(currentDemo);


  const enrollmentPMRs = new Set();
  const disenrollmentPMRs = new Set();
  const demographicPMRs = new Set();
  const caseloadPMRs = new Set();
  const contactPMRs = new Set();


  const demographicChangedColumnsByPMR = new Map();
  const caseloadChangedColumnsByPMR = new Map();
  const contactChangedColumnsByPMR = new Map();


  currentData.rowsByPMR.forEach((currentRows, pmr) => {
    const previousRows = previousData.rowsByPMR.get(pmr) || [];


    const capitationHeader = "Capitation Date";
    const disenrollEffectiveHeader = "Disenrollment Effective Date";


    const mostRecentCapitationDate = getMostRecentDateFromRowsByHeader_(
      currentRows,
      currentData.headerMap,
      capitationHeader
    );


    const isEnrollmentPMR = isSameDate_(mostRecentCapitationDate, monthParts.firstDay);


    const isDisenrollmentPMR =
      currentRows.some(item => {
        const effectiveDate = getFieldValueFromRow_(
          item.values,
          currentData.headerMap,
          disenrollEffectiveHeader
        );


        return isDateInStrictLocalRangeInclusive_(
          effectiveDate,
          monthParts.previousMonthFirstDay,
          monthParts.firstDay
        );
      });


    if (isEnrollmentPMR) enrollmentPMRs.add(pmr);
    if (isDisenrollmentPMR) disenrollmentPMRs.add(pmr);


    if (isEnrollmentPMR || isDisenrollmentPMR) return;


    if (isParticipantEnrollmentStatusDisenrolled_(currentRows, currentData.headerMap)) return;


    const currentDobRows = rowsWithDOBOnlyForSection_(currentRows, currentData.headerMap);
    const previousDobRows = rowsWithDOBOnlyForSection_(previousRows, previousData.headerMap);


    const demographicChangedColumns = getChangedColumnsForSectionRows_(
      currentDobRows,
      previousDobRows,
      currentData.headers,
      previousData.headers,
      RAW_DEMO_P_DEMOGRAPHIC_FIELDS
    );


    if (demographicChangedColumns.size > 0) {
      demographicPMRs.add(pmr);
      demographicChangedColumnsByPMR.set(pmr, demographicChangedColumns);
    }


    const caseloadChangedColumns = getChangedColumnsForSectionRows_(
      currentDobRows,
      previousDobRows,
      currentData.headers,
      previousData.headers,
      RAW_DEMO_P_CASELOAD_FIELDS
    );


    if (caseloadChangedColumns.size > 0) {
      caseloadPMRs.add(pmr);
      caseloadChangedColumnsByPMR.set(pmr, caseloadChangedColumns);
    }


    const contactChangedColumns = getChangedColumnsForSectionRows_(
      currentRows,
      previousRows,
      currentData.headers,
      previousData.headers,
      RAW_DEMO_P_CONTACT_FIELDS
    );


    if (contactChangedColumns.size > 0) {
      contactPMRs.add(pmr);
      contactChangedColumnsByPMR.set(pmr, contactChangedColumns);
    }
  });


  enrollmentPMRs.forEach(pmr => {
    demographicPMRs.delete(pmr);
    caseloadPMRs.delete(pmr);
    contactPMRs.delete(pmr);
  });


  disenrollmentPMRs.forEach(pmr => {
    demographicPMRs.delete(pmr);
    caseloadPMRs.delete(pmr);
    contactPMRs.delete(pmr);
  });


  return {
    previousData: previousData,
    currentData: currentData,
    enrollmentPMRs: enrollmentPMRs,
    disenrollmentPMRs: disenrollmentPMRs,
    demographicPMRs: demographicPMRs,
    caseloadPMRs: caseloadPMRs,
    contactPMRs: contactPMRs,
    demographicChangedColumnsByPMR: demographicChangedColumnsByPMR,
    caseloadChangedColumnsByPMR: caseloadChangedColumnsByPMR,
    contactChangedColumnsByPMR: contactChangedColumnsByPMR
  };
}

function insertMonthlyDisenrollmentRowsStrict_(reportSheet, currentData, pmrSet, monthParts) {


  if (!monthParts) return;


  const sectionTitle = "Disenrollments";
  const titleRow = findMonthlyChangeSectionTitleRow_(reportSheet, sectionTitle);
  if (titleRow === -1) return;


  const headerRow = titleRow + 2;
  const insertAt = headerRow + 1;
  const reportHeaders = getMonthlyChangeReportHeaders_(currentData.headers);
  const lastCol = reportHeaders.length;


  const effectiveIdx = currentData.headerMap["Disenrollment Effective Date"];
  const pmrIdx = getPMRIndex_(currentData.headerMap);


  if (effectiveIdx === undefined || effectiveIdx === -1 || pmrIdx === -1) return;


  const rowsToInsert = [];
  const usedPMRs = new Set();


  currentData.values.forEach(row => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr || usedPMRs.has(pmr)) return;


    const effectiveDate = row[effectiveIdx];


    if (!isDateDisplayInReportWindow_(
      effectiveDate,
      monthParts.previousMonthFirstDay,
      monthParts.firstDay
    )) {
      return;
    }


    rowsToInsert.push(buildMonthlyChangeReportRow_(
      row,
      currentData.headers,
      reportHeaders,
      new Set()
    ));


    usedPMRs.add(pmr);
  });


  if (rowsToInsert.length === 0) return;


  reportSheet.insertRowsBefore(insertAt, rowsToInsert.length);


  const dataRange = reportSheet.getRange(insertAt, 1, rowsToInsert.length, lastCol);
  dataRange.setValues(rowsToInsert);
  dataRange;


  try {
    reportSheet.getRange(headerRow, 1, 1, lastCol)
      .copyTo(dataRange, SpreadsheetApp.CopyPasteType.PASTE_FORMAT, false);
  } catch (err) {
    Logger.log("Monthly Change disenrollment data format copy skipped: " + err.message);
  }


  dataRange;


  reportSheet;
}

function insertMonthlyChangeSectionRows_(reportSheet, currentData, pmrSet, sectionTitle, rowMode, changedColumnsByPMR, monthParts) {
  if (!pmrSet || pmrSet.size === 0) return;


  const titleRow = findMonthlyChangeSectionTitleRow_(reportSheet, sectionTitle);
  if (titleRow === -1) return;


  const headerRow = titleRow + 2;
  const insertAt = headerRow + 1;
  const reportHeaders = getMonthlyChangeReportHeaders_(currentData.headers);
  const lastCol = reportHeaders.length;
  const dobIdx = getDOBIndex_(currentData.headerMap);
  const disenrollEffectiveIdx = currentData.headerMap["Disenrollment Effective Date"];


  const allowDuplicateRowsForPMR = sectionTitle === "Contact Changes";


  const rowsToInsert = [];
  const pmrForEachInsertedRow = [];


  Array.from(pmrSet).sort().forEach(pmr => {
    const items = currentData.rowsByPMR.get(pmr) || [];
    let insertedForThisPMR = false;


    items.forEach(item => {
      if (!allowDuplicateRowsForPMR && insertedForThisPMR) return;


      if (rowMode === "dobOnly" && (dobIdx === -1 || normalizeCompareValue_(item.values[dobIdx]) === "")) return;


      if (rowMode === "disenrollmentEffectiveRange") {
        if (disenrollEffectiveIdx === undefined || disenrollEffectiveIdx === -1 || !monthParts) return;


        const effectiveDate = item.values[disenrollEffectiveIdx];


        if (!isDateInStrictLocalRangeInclusive_(
          effectiveDate,
          monthParts.previousMonthFirstDay,
          monthParts.firstDay
        )) {
          return;
        }
      }


      const changedColumns = changedColumnsByPMR && changedColumnsByPMR.has(pmr)
        ? changedColumnsByPMR.get(pmr)
        : new Set();


      rowsToInsert.push(buildMonthlyChangeReportRow_(item.values, currentData.headers, reportHeaders, changedColumns));
      pmrForEachInsertedRow.push(pmr);


      if (!allowDuplicateRowsForPMR) insertedForThisPMR = true;
    });
  });


  if (rowsToInsert.length === 0) return;


  reportSheet.insertRowsBefore(insertAt, rowsToInsert.length);


  const dataRange = reportSheet.getRange(insertAt, 1, rowsToInsert.length, lastCol);
  dataRange.setValues(rowsToInsert);
  dataRange;


  const headerFormatRange = reportSheet.getRange(headerRow, 1, 1, lastCol);
  try {
    headerFormatRange.copyTo(dataRange, SpreadsheetApp.CopyPasteType.PASTE_FORMAT, false);
  } catch (err) {
    Logger.log("Monthly Change data format copy skipped: " + err.message);
  }


  dataRange;


  dataRange;


  if (changedColumnsByPMR && changedColumnsByPMR.size > 0) {
    const backgrounds = dataRange.getBackgrounds();


    for (let r = 0; r < backgrounds.length; r++) {
      const pmr = pmrForEachInsertedRow[r];
      const changedColumns = changedColumnsByPMR.get(pmr);


      if (!changedColumns || changedColumns.size === 0) continue;


      reportHeaders.forEach((header, colIndex) => {
        if (changedColumns.has(header)) {
          backgrounds[r][colIndex] = "#f3ffc7";
        }
      });
    }


    dataRange;
  }


  reportSheet;
}

function findMonthlyChangeSectionTitleRow_(sheet, title) {
  const lastRow = sheet.getLastRow();
  if (lastRow < 1) return -1;


  const values = sheet.getRange(1, 1, lastRow, 1).getValues();


  for (let i = 0; i < values.length; i++) {
    if (String(values[i][0] || "").trim() === title) return i + 1;
  }


  return -1;
}

    const value = String(values[i][0] || "").trim();
    if (titles.has(value)) return afterRow + 1 + i;
  }


  return -1;
}

function compareRawDemoPForChanges_(previousDemo, currentDemo, monthParts) {
  const previousData = getRawDemoPDataForCompare_(previousDemo);
  const currentData = getRawDemoPDataForCompare_(currentDemo);


  const outputRows = [];
  const changedPMRs = new Map();


  currentData.participantRows.forEach((currentMain, pmr) => {
    const previousMain = previousData.participantRows.get(pmr);


    const capitationIdx = currentData.headerMap["Capitation Date"];
    const currentCapitation = capitationIdx !== undefined ? currentMain.values[capitationIdx] : "";


    const isNewPMR = !previousData.allPMRs.has(pmr);
    const isReportMonthCapitation = isSameMonth_(currentCapitation, monthParts.firstDay);


    if (isNewPMR || isReportMonthCapitation) {
      RAW_DEMO_P_ENROLLMENT_FIELDS.forEach(field => {
        const currentValue = getFieldValueFromRow_(currentMain.values, currentData.headerMap, field);
        if (normalizeCompareValue_(currentValue) !== "") {
          addMCRRow_(
            outputRows,
            changedPMRs,
            CHANGE_SECTION_ENROLLMENTS,
            pmr,
            currentMain,
            null,
            currentData,
            previousData,
            field,
            "",
            currentValue,
            monthParts
          );
        }
      });
    }
  });


  currentData.participantRows.forEach((currentMain, pmr) => {
    const effective = getFieldValueFromRow_(currentMain.values, currentData.headerMap, "Disenrollment Effective Date");
    const disenrollDate = getFieldValueFromRow_(currentMain.values, currentData.headerMap, "Disenrollment Date");
    const deathDate = getFieldValueFromRow_(currentMain.values, currentData.headerMap, "Date of Death");


    const isCurrentEffective = isSameDate_(effective, monthParts.firstDay);
    const isPriorMonthDisenroll = isSameMonth_(disenrollDate, monthParts.previousMonthFirstDay);
    const isPriorMonthDeath = isSameMonth_(deathDate, monthParts.previousMonthFirstDay);


    if (isCurrentEffective || isPriorMonthDisenroll || isPriorMonthDeath) {
      RAW_DEMO_P_DISENROLLMENT_FIELDS.forEach(field => {
        const currentValue = getFieldValueFromRow_(currentMain.values, currentData.headerMap, field);
        if (normalizeCompareValue_(currentValue) !== "") {
          addMCRRow_(
            outputRows,
            changedPMRs,
            CHANGE_SECTION_DISENROLLMENTS,
            pmr,
            currentMain,
            null,
            currentData,
            previousData,
            field,
            "",
            currentValue,
            monthParts
          );
        }
      });
    }
  });


  currentData.participantRows.forEach((currentMain, pmr) => {
    const previousMain = previousData.participantRows.get(pmr);
    if (!previousMain) return;


    RAW_DEMO_P_DEMOGRAPHIC_FIELDS.forEach(field => {
      compareSingleFieldAndAdd_(
        outputRows,
        changedPMRs,
        CHANGE_SECTION_DEMOGRAPHIC,
        pmr,
        previousMain,
        currentMain,
        previousData,
        currentData,
        field,
        monthParts,
        true
      );
    });


    RAW_DEMO_P_CASELOAD_FIELDS.forEach(field => {
      compareSingleFieldAndAdd_(
        outputRows,
        changedPMRs,
        CHANGE_SECTION_CASELOAD,
        pmr,
        previousMain,
        currentMain,
        previousData,
        currentData,
        field,
        monthParts,
        true
      );
    });
  });


  const allCurrentPMRs = Array.from(currentData.allPMRs);
  allCurrentPMRs.forEach(pmr => {
    if (!previousData.allPMRs.has(pmr)) return;


    const previousRows = previousData.rowsByPMR.get(pmr) || [];
    const currentRows = currentData.rowsByPMR.get(pmr) || [];


    const maxLength = Math.max(previousRows.length, currentRows.length);


    for (let i = 0; i < maxLength; i++) {
      const previousItem = previousRows[i] || null;
      const currentItem = currentRows[i] || null;


      RAW_DEMO_P_CONTACT_FIELDS.forEach(field => {
        const previousValue = previousItem ? getFieldValueFromRow_(previousItem.values, previousData.headerMap, field) : "";
        const currentValue = currentItem ? getFieldValueFromRow_(currentItem.values, currentData.headerMap, field) : "";


        if (!valuesAreEqual_(previousValue, currentValue)) {
          addMCRRow_(
            outputRows,
            changedPMRs,
            CHANGE_SECTION_CONTACT,
            pmr,
            currentItem,
            previousItem,
            currentData,
            previousData,
            field,
            previousValue,
            currentValue,
            monthParts
          );
        }
      });
    }
  });


  return {
    rows: outputRows,
    changedPMRs: changedPMRs
  };
}

function getRawDemoPDataForCompare_(sheet) {
  const data = getDataValues_(sheet, HEADER_ROW, DATA_START_ROW);


  const headers = data.headers;
  const headerMap = data.headerMap;
  const pmrIdx = getPMRIndex_(headerMap);
  const dobIdx = getDOBIndex_(headerMap);


  if (pmrIdx === -1) {
    throw new Error(`${sheet.getName()} is missing Participant PMR# / PMR # column.`);
  }


  const values = data.values;


  const rowsByPMR = new Map();
  const participantRows = new Map();
  const allPMRs = new Set();


  values.forEach((row, idx) => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr) return;


    allPMRs.add(pmr);


    const item = {
      rowNumber: DATA_START_ROW + idx,
      values: row
    };


    if (!rowsByPMR.has(pmr)) rowsByPMR.set(pmr, []);
    rowsByPMR.get(pmr).push(item);


    if (!participantRows.has(pmr)) {
      if (dobIdx === -1 || normalizeCompareValue_(row[dobIdx]) !== "") {
        participantRows.set(pmr, item);
      }
    }
  });


  rowsByPMR.forEach((items, pmr) => {
    if (!participantRows.has(pmr) && items.length > 0) {
      participantRows.set(pmr, items[0]);
    }
  });


  return {
    sheet: sheet,
    sheetName: sheet.getName(),
    headers: headers,
    headerMap: headerMap,
    values: values,
    rowsByPMR: rowsByPMR,
    participantRows: participantRows,
    allPMRs: allPMRs
  };
}

  return reportSheet;
}

function copyPreviousMasterListToCurrentMonth_(previousMaster, monthParts) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const targetName = buildMonthlySheetName_(MASTER_LIST_PREFIX, monthParts.firstDay, monthParts.lastDay);
  const existing = ss.getSheetByName(targetName);


  if (existing) {
    ss.deleteSheet(existing);
  }


  const currentMaster = createMasterListSheetFromTemplate_(ss, targetName, monthParts);


  const lastRow = Math.max(previousMaster.getLastRow(), HEADER_ROW);
  const lastCol = Math.max(previousMaster.getLastColumn(), 1);


  if (lastCol > currentMaster.getMaxColumns()) {
    currentMaster.insertColumnsAfter(currentMaster.getMaxColumns(), lastCol - currentMaster.getMaxColumns());
  }


  if (lastRow > currentMaster.getMaxRows()) {
    currentMaster.insertRowsAfter(currentMaster.getMaxRows(), lastRow - currentMaster.getMaxRows());
  }


  const values = previousMaster.getRange(1, 1, lastRow, lastCol).getValues();
  currentMaster.getRange(1, 1, lastRow, lastCol).setValues(values);


  updateCopiedMasterListHeader_(currentMaster, monthParts);
  applyMasterListTemplateHeaderFormatting_(currentMaster, monthParts);


  ensureHeaders_(currentMaster, MASTER_LIST_ADDED_HEADERS, HEADER_ROW);
  ensureBannerSummaryOutputHeaders_(currentMaster);
  ensureContactOutputHeaders_(currentMaster);
  clearSheetRuntimeCachesForSheet_(currentMaster);


  return currentMaster;
}

function sortMasterListByParticipantNameAndPMR_(sheet) {
  sheet = sheet || SpreadsheetApp.getActiveSheet();


  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();
  if (lastRow < DATA_START_ROW || lastCol < 1) return;


  clearAllRowGroupsIfPossible_(sheet);
  sheet.showRows(DATA_START_ROW, lastRow - DATA_START_ROW + 1);


  const headers = getHeaders_(sheet, HEADER_ROW);
  const headerMap = buildHeaderIndexMap_(headers);
  const pmrIdx = getPMRIndex_(headerMap);
  const lastNameIdx = headerMap["Last Name"];
  const firstNameIdx = headerMap["First Name"];


  if (pmrIdx === -1 || lastNameIdx === undefined || firstNameIdx === undefined) {
    notify_("Sort skipped: missing Last Name, First Name, or Participant PMR#.");
    return;
  }


  const range = sheet.getRange(DATA_START_ROW, 1, lastRow - DATA_START_ROW + 1, lastCol);
  const values = range.getValues();
  const blocksByPMR = new Map();


  values.forEach((row, idx) => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr) return;


    if (!blocksByPMR.has(pmr)) {
      blocksByPMR.set(pmr, {
        pmr: pmr,
        lastName: row[lastNameIdx],
        firstName: row[firstNameIdx],
        rows: []
      });
    }


    const block = blocksByPMR.get(pmr);
    if (!String(block.lastName || "").trim() && String(row[lastNameIdx] || "").trim()) block.lastName = row[lastNameIdx];
    if (!String(block.firstName || "").trim() && String(row[firstNameIdx] || "").trim()) block.firstName = row[firstNameIdx];


    block.rows.push({ row: row, originalIndex: idx });
  });


  const blocks = Array.from(blocksByPMR.values());


  blocks.sort((a, b) => {
    const aLast = normalizeKeyPart_(a.lastName);
    const bLast = normalizeKeyPart_(b.lastName);
    if (aLast !== bLast) return aLast < bLast ? -1 : 1;


    const aFirst = normalizeKeyPart_(a.firstName);
    const bFirst = normalizeKeyPart_(b.firstName);
    if (aFirst !== bFirst) return aFirst < bFirst ? -1 : 1;


    return normalizePMR_(a.pmr) < normalizePMR_(b.pmr) ? -1 : 1;
  });


  const sortedRows = [];


  blocks.forEach(block => {
    block.rows.sort((a, b) => {
      const aScore = getPrimaryRowScore_(a.row, headerMap);
      const bScore = getPrimaryRowScore_(b.row, headerMap);
      if (aScore !== bScore) return bScore - aScore;
      return a.originalIndex - b.originalIndex;
    });


    block.rows.forEach(item => sortedRows.push(item.row));
  });


  if (sortedRows.length > 0) {
    range.setValues(sortedRows);
  }
}

function updateBannerColumnData_(data, pmrFilter) {
  const headerMap = data.headerMap;
  const bIdx = headerMap["Banner Summary"];


  if (bIdx === undefined) return;


  const bannerHeaders = [
    "Safety - 2 Person",
    "Wanderer",
    "Interpreter Needed",
    "Fall Risk",
    "DPOA or Guardian Active",
    "Palliative Care"
  ];


  data.values.forEach(row => {
    if (!shouldProcessRowByPMR_(row, headerMap, pmrFilter)) return;


    const parts = [];


    bannerHeaders.forEach(header => {
      const idx = headerMap[header];
      if (idx === undefined) return;


      const value = String(row[idx] || "").trim();
      if (value !== "") parts.push(header);
    });


    row[bIdx] = parts.join(" | ");
  });
}

function updateMasterListFromMonthlyChangeActions_(currentDemo, currentMaster, reportSheet, monthParts, changedInfo, timing) {
  const enrollmentPMRs = getPMRsForMonthlyChangeSections_(changedInfo, ["Enrollments"]);
  const disenrollmentPMRs = getPMRsForMonthlyChangeSections_(changedInfo, ["Disenrollments"]);
  const primaryUpdatePMRs = getPMRsForMonthlyChangeSections_(changedInfo, ["Demographic Changes", "Caseload Changes", "Banner Changes"]);
  const contactChangePMRs = getPMRsForMonthlyChangeSections_(changedInfo, ["Contact Changes"]);
  const affectedPMRs = new Set();


  disenrollmentPMRs.forEach(pmr => affectedPMRs.add(pmr));
  enrollmentPMRs.forEach(pmr => affectedPMRs.add(pmr));
  primaryUpdatePMRs.forEach(pmr => affectedPMRs.add(pmr));
  contactChangePMRs.forEach(pmr => affectedPMRs.add(pmr));


  deletePMRBlocksFromMasterListBySet_(currentMaster, disenrollmentPMRs, reportSheet, monthParts, "Disenrollments - remove PMR block");
  markRuntimeStep_(timing, "Update - remove Disenrollment PMR blocks (" + disenrollmentPMRs.size + " PMRs)");


  deletePMRBlocksFromMasterListBySet_(currentMaster, enrollmentPMRs, reportSheet, monthParts, "Enrollments - replace/add PMR block");
  const addedRows = copyChangedPMRsFromDemoPToMasterList_(currentDemo, currentMaster, enrollmentPMRs, monthParts);
  markRuntimeStep_(timing, "Update - add Enrollment PMR blocks (" + enrollmentPMRs.size + " PMRs, " + addedRows + " rows)");


  const primaryResult = updatePrimaryRowsFromDemoPForPMRs_(currentDemo, currentMaster, primaryUpdatePMRs, monthParts);
  markRuntimeStep_(timing, "Update - primary row updates (" + primaryResult.rowsUpdated + " rows, " + primaryResult.columnsUpdated + " columns)");


  const contactResult = mergeSecondaryRowsFromDemoPForPMRs_(currentDemo, currentMaster, contactChangePMRs, monthParts);
  markRuntimeStep_(timing, "Update - contact row merge (" + contactResult.rowsUpdated + " updates, " + contactResult.rowsAdded + " adds, " + contactResult.rowsRemoved + " removes)");


  return {
    affectedPMRs: affectedPMRs,
    enrollmentPMRs: enrollmentPMRs,
    disenrollmentPMRs: disenrollmentPMRs,
    primaryUpdatePMRs: primaryUpdatePMRs,
    contactChangePMRs: contactChangePMRs,
    primaryResult: primaryResult,
    contactResult: contactResult
  };
}

function getPMRsForMonthlyChangeSections_(changedInfo, sectionsToMatch) {
  const targets = new Set(sectionsToMatch || []);
  const result = new Set();


  (changedInfo || new Map()).forEach((sections, pmr) => {
    if (!sections) return;
    const hasMatch = Array.from(sections).some(section => targets.has(section));
    if (hasMatch) result.add(pmr);
  });


  return result;
}

function deletePMRBlocksFromMasterListBySet_(masterSheet, pmrSet, reportSheet, monthParts, reasonText) {
  if (!masterSheet || !pmrSet || pmrSet.size === 0) return 0;


  const rowsByPMR = buildRowsByPMR_(masterSheet, HEADER_ROW, DATA_START_ROW);
  const headers = getHeaders_(masterSheet, HEADER_ROW);
  const logRows = [];
  const rowsToDelete = [];
  const timestamp = new Date();


  pmrSet.forEach(pmr => {
    const items = rowsByPMR.get(pmr) || [];
    items.forEach(item => {
      logRows.push([
        timestamp,
        monthParts.firstDay,
        pmr,
        reasonText || "Monthly Change action",
        masterSheet.getName(),
        item.rowNumber,
        reportSheet ? reportSheet.getName() : "",
        JSON.stringify(rowObjectFromHeaders_(headers, item.values))
      ]);
      rowsToDelete.push(item.rowNumber);
    });
  });


  appendMasterListChangeLog_(logRows);
  deleteRowNumberBatches_(masterSheet, rowsToDelete);
  clearSheetRuntimeCachesForSheet_(masterSheet);
  return rowsToDelete.length;
}

function updatePrimaryRowsFromDemoPForPMRs_(demoSheet, masterSheet, pmrSet, monthParts) {
  const result = { rowsUpdated: 0, columnsUpdated: 0, missingInMaster: 0, missingInDemo: 0 };
  if (!demoSheet || !masterSheet || !pmrSet || pmrSet.size === 0) return result;


  const demoData = getDataValues_(demoSheet, HEADER_ROW, DATA_START_ROW);
  const masterData = getDataValues_(masterSheet, HEADER_ROW, DATA_START_ROW);
  const demoBlocks = buildMergeRowsByPMRFromData_(demoData);
  const masterBlocks = buildMergeRowsByPMRFromData_(masterData);
  const masterHeaders = masterData.headers;
  const masterHeaderMap = masterData.headerMap;
  const sourceSheetName = demoSheet.getName();


  pmrSet.forEach(pmr => {
    const demoPrimary = getPrimaryMergeRowItem_(demoBlocks.get(pmr) || [], demoData.headerMap);
    const masterPrimary = getPrimaryMergeRowItem_(masterBlocks.get(pmr) || [], masterHeaderMap);


    if (!demoPrimary) {
      result.missingInDemo++;
      return;
    }
    if (!masterPrimary) {
      result.missingInMaster++;
      return;
    }


    const changedColumns = getPrimaryRowChangedColumnDetails_(
      masterPrimary.values,
      masterHeaderMap,
      demoPrimary.values,
      demoData.headerMap
    ).map(item => item.column);


    if (changedColumns.length === 0) return;


    setMasterRowColumnsFromDemoRow_(
      masterSheet,
      masterPrimary.rowNumber,
      masterHeaders,
      masterHeaderMap,
      demoPrimary.values,
      demoData.headerMap,
      changedColumns,
      monthParts,
      sourceSheetName,
      "Primary row updated from Demo P"
    );


    result.rowsUpdated++;
    result.columnsUpdated += changedColumns.length;
  });


  clearSheetRuntimeCachesForSheet_(masterSheet);
  return result;
}

function mergeSecondaryRowsFromDemoPForPMRs_(demoSheet, masterSheet, pmrSet, monthParts) {
  const result = { rowsUpdated: 0, rowsAdded: 0, rowsRemoved: 0, missingInDemo: 0 };
  if (!demoSheet || !masterSheet || !pmrSet || pmrSet.size === 0) return result;


  const demoData = getDataValues_(demoSheet, HEADER_ROW, DATA_START_ROW);
  let masterData = getDataValues_(masterSheet, HEADER_ROW, DATA_START_ROW);
  const demoBlocks = buildMergeRowsByPMRFromData_(demoData);
  let masterBlocks = buildMergeRowsByPMRFromData_(masterData);
  const masterHeaders = masterData.headers;
  const masterHeaderMap = masterData.headerMap;
  const sourceSheetName = demoSheet.getName();
  const rowsToDelete = [];
  const rowsToAppend = [];


  pmrSet.forEach(pmr => {
    const demoRows = demoBlocks.get(pmr) || [];
    const masterRows = masterBlocks.get(pmr) || [];
    if (!demoRows.length) {
      result.missingInDemo++;
      return;
    }


    const demoKeys = buildSecondaryMergeKeyMapForRows_(demoRows, demoData.headerMap, pmr);
    const masterKeys = buildSecondaryMergeKeyMapForRows_(masterRows, masterHeaderMap, pmr);
    const allKeys = new Set([].concat(Array.from(demoKeys.keys()), Array.from(masterKeys.keys())));


    allKeys.forEach(key => {
      const demoItem = demoKeys.get(key);
      const masterItem = masterKeys.get(key);


      if (demoItem && masterItem) {
        const changedFields = getMergeAuditChangedFields_(masterItem.values, masterHeaderMap, demoItem.values, demoData.headerMap);
        if (changedFields.length > 0) {
          setMasterRowColumnsFromDemoRow_(
            masterSheet,
            masterItem.rowNumber,
            masterHeaders,
            masterHeaderMap,
            demoItem.values,
            demoData.headerMap,
            changedFields,
            monthParts,
            sourceSheetName,
            "Contact row updated from Demo P"
          );
          result.rowsUpdated++;
        }
      } else if (demoItem && !masterItem) {
        rowsToAppend.push(buildMappedMasterRowFromDemoRow_(
          demoItem.values,
          demoData.headers,
          demoData.headerMap,
          masterHeaders,
          masterHeaderMap,
          monthParts,
          sourceSheetName,
          "Contact row added from Demo P"
        ));
        result.rowsAdded++;
      } else if (!demoItem && masterItem) {
        rowsToDelete.push(masterItem.rowNumber);
        result.rowsRemoved++;
      }
    });
  });


  if (rowsToAppend.length) {
    const appendRow = Math.max(masterSheet.getLastRow() + 1, DATA_START_ROW);
    masterSheet.getRange(appendRow, 1, rowsToAppend.length, masterHeaders.length).setValues(rowsToAppend);
  }


  if (rowsToDelete.length) {
    deleteRowNumberBatches_(masterSheet, rowsToDelete);
  }


  clearSheetRuntimeCachesForSheet_(masterSheet);
  return result;
}

function buildMappedMasterRowFromDemoRow_(demoRow, demoHeaders, demoHeaderMap, masterHeaders, masterHeaderMap, monthParts, sourceSheetName, statusText) {
  const output = new Array(masterHeaders.length).fill("");


  demoHeaders.forEach((header, sourceIdx) => {
    if (!header) return;
    const targetIdx = masterHeaderMap[header];
    if (targetIdx !== undefined) output[targetIdx] = demoRow[sourceIdx];
  });


  applyMasterListUpdateMetadataToRow_(output, masterHeaderMap, monthParts, sourceSheetName, statusText);
  return output;
}

function setMasterRowColumnsFromDemoRow_(masterSheet, rowNumber, masterHeaders, masterHeaderMap, demoRow, demoHeaderMap, headersToUpdate, monthParts, sourceSheetName, statusText) {
  if (!masterSheet || !rowNumber || rowNumber < DATA_START_ROW) return;
  const rowValues = masterSheet.getRange(rowNumber, 1, 1, masterHeaders.length).getValues()[0];


  (headersToUpdate || []).forEach(header => {
    const masterIdx = masterHeaderMap[header];
    const demoIdx = demoHeaderMap[header];
    if (masterIdx === undefined || demoIdx === undefined) return;
    rowValues[masterIdx] = demoRow[demoIdx];
  });


  applyMasterListUpdateMetadataToRow_(rowValues, masterHeaderMap, monthParts, sourceSheetName, statusText);
  masterSheet.getRange(rowNumber, 1, 1, masterHeaders.length).setValues([rowValues]);
}

function applyMasterListUpdateMetadataToRow_(rowValues, headerMap, monthParts, sourceSheetName, statusText) {
  const statusIdx = headerMap["Master List Update Status"];
  const monthIdx = headerMap["Master List Update Month"];
  const sourceIdx = headerMap["Master List Source Sheet"];


  if (statusIdx !== undefined) rowValues[statusIdx] = statusText || "Updated from Demo P";
  if (monthIdx !== undefined) rowValues[monthIdx] = monthParts ? monthParts.firstDay : new Date();
  if (sourceIdx !== undefined) rowValues[sourceIdx] = sourceSheetName || "";
}

function buildUpdateMasterListMergeAuditForMonth_(monthParts) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const currentDemo = getCurrentDemoPSheet_(monthParts);
  const sourceMaster = getPreviousMasterListSheet_(monthParts);


  if (!currentDemo) {
    notify_("Current month Demo P was not found. Format Demo P first.");
    return null;
  }


  if (!sourceMaster) {
    notify_("Previous month Master List was not found for merge audit.");
    return null;
  }


  const reportSheet = getMonthlySheetByPrefixAndDate_(
    ss,
    MONTHLY_CHANGE_REPORT_PREFIX,
    monthParts.firstDay,
    monthParts.lastDay
  );


  if (!reportSheet) {
    notify_("Monthly Change Report was not found. Create Monthly Change Report first.");
    return null;
  }


  const changedInfo = getChangedPMRsFromMonthlyChangeReport_(reportSheet);
  const contactChangePMRs = Array.from(changedInfo.keys())
    .filter(pmr => changedInfo.get(pmr) && changedInfo.get(pmr).has("Contact Changes"))
    .sort();


  const demoData = getDataValues_(currentDemo, HEADER_ROW, DATA_START_ROW);
  const masterData = getDataValues_(sourceMaster, HEADER_ROW, DATA_START_ROW);


  const demoBlocks = buildMergeRowsByPMRFromData_(demoData);
  const masterBlocks = buildMergeRowsByPMRFromData_(masterData);


  const summaryRows = [];
  const detailRows = [];


  contactChangePMRs.forEach(pmr => {
    const demoRows = demoBlocks.get(pmr) || [];
    const masterRows = masterBlocks.get(pmr) || [];
    const participantName = getMergeAuditParticipantName_(demoRows, demoData.headerMap, masterRows, masterData.headerMap);


    const demoKeys = buildSecondaryMergeKeyMapForRows_(demoRows, demoData.headerMap, pmr);
    const masterKeys = buildSecondaryMergeKeyMapForRows_(masterRows, masterData.headerMap, pmr);


    const contactKeys = new Set(
      [].concat(Array.from(demoKeys.keys()), Array.from(masterKeys.keys()))
    );


    let matchedCount = 0;
    let updateCount = 0;
    let addCount = 0;
    let removeCount = 0;
    let noActionCount = 0;


    contactKeys.forEach(key => {
      const demoItem = demoKeys.get(key);
      const masterItem = masterKeys.get(key);
      const masterSummary = masterItem ? buildMergeAuditContactSummary_(masterItem.values, masterData.headerMap) : "";
      const demoSummary = demoItem ? buildMergeAuditContactSummary_(demoItem.values, demoData.headerMap) : "";


      let auditAction = "";
      let differenceFound = "";
      let recommendedAction = "";


      if (demoItem && masterItem) {
        matchedCount++;
        const changedFields = getMergeAuditChangedFields_(masterItem.values, masterData.headerMap, demoItem.values, demoData.headerMap);


        if (changedFields.length > 0) {
          auditAction = "Update";
          differenceFound = "Contact exists in both; changed fields: " + changedFields.join(", ");
          recommendedAction = "Update existing Master List contact row";
          updateCount++;
        } else {
          auditAction = "No action";
          differenceFound = "Contact exists in both; no compared field differences";
          recommendedAction = "No Master List contact row change needed";
          noActionCount++;
        }
      } else if (demoItem && !masterItem) {
        auditAction = "Add";
        differenceFound = "Contact exists in Demo P only";
        recommendedAction = "Add new contact row to Master List";
        addCount++;
      } else if (!demoItem && masterItem) {
        auditAction = "Remove";
        differenceFound = "Contact exists in previous Master List only";
        recommendedAction = "Remove contact row from Master List";

function updateMasterListForMonth_(monthParts) {
  if (!monthParts) return null;


  const timing = startRuntimeTiming_("Update Master List", monthParts);
  const hiddenSheetNamesAtStart = captureHiddenSheetNames_();
  let currentMaster = null;
  let rebuiltCount = 0;


  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();


    markRuntimeStep_(timing, "Start");
    const currentDemo = getCurrentDemoPSheet_(monthParts);
    const previousDemo = getPreviousDemoPSheet_(monthParts);
    const previousMaster = getPreviousMasterListSheet_(monthParts);
    markRuntimeStep_(timing, "Locate monthly source sheets");


    if (!currentDemo) {
      notify_("Current month Demo P was not found. Format Demo P first.");
      markRuntimeStep_(timing, "Stopped - missing current Demo P");
      return null;
    }


    if (!previousMaster) {
      notify_("Previous month Master List was not found. Create the previous month Master List first.");
      markRuntimeStep_(timing, "Stopped - missing previous Master List");
      return null;
    }


    currentMaster = copyPreviousMasterListToCurrentMonth_(previousMaster, monthParts);
    markRuntimeStep_(timing, "Copy previous Master List to current month");


    let reportSheet;
    try {
      reportSheet = getOrBuildMonthlyChangeReport_(ss, monthParts, currentDemo, previousDemo);
      markRuntimeStep_(timing, "Get or build Monthly Change Report");
    } catch (err) {
      markRuntimeStep_(timing, "Stopped - Monthly Change Report error");
      notify_(err.message);
      return null;
    }


    if (reportSheet) {
      rebuiltCount = rebuildChangedPMRsOnExistingMaster_(
        currentDemo,
        currentMaster,
        reportSheet,
        monthParts,
        true,
        timing
      );
      markRuntimeStep_(timing, "Rebuild changed PMRs on copied Master List");
    } else {
      processMasterListFull_(currentMaster, monthParts, null);
      markRuntimeStep_(timing, "No change report rows - full current-month processing");
    }


    finalizeWorkflowAfterCreateOrUpdate_();
    markRuntimeStep_(timing, "Finalize workflow after create/update");


    hidePreviousMonthSheets_(monthParts);
    markRuntimeStep_(timing, "Hide previous month sheets");


    currentMaster.activate();


    removeDisenrolledPMRBlocksFromMasterUsingDemoP_(currentMaster, currentDemo);
    markRuntimeStep_(timing, "Remove disenrolled PMR blocks");


    prepareMasterListSortOrderBeforeFormatting_(currentMaster);
    markRuntimeStep_(timing, "Prepare Sort Order before formatting");


    applyMasterListTemplateModeUpdateDisplay_(currentMaster);
    markRuntimeStep_(timing, "Apply template-mode update display settings");


    applyFinalMasterListColorAndDisplay_(currentMaster);
    markRuntimeStep_(timing, "Apply final Sort Order coloring and display lock");


    autoHidePreviousMonthSheetsAfterWorkflow_(monthParts);
    markRuntimeStep_(timing, "Auto-hide previous month sheets");


    enforceMasterListFinalDisplay_(currentMaster);
    markRuntimeStep_(timing, "Final Master List display enforcement");


    restorePreviouslyHiddenSheets_(hiddenSheetNamesAtStart, currentMaster);
    markRuntimeStep_(timing, "Restore previously hidden sheets");


    writeRuntimeTimingReport_(timing);
    hideSystemSheets_();


    notify_(
      "Update Master List complete.\\n\\n" +
      `New Master List: ${currentMaster.getName()}\\n` +
      `Changed PMRs rebuilt: ${rebuiltCount}\\n` +
      `Runtime: ${formatSeconds_((new Date().getTime() - timing.startMs) / 1000)}\\n` +
      "Previous month sheets hidden."
    );


    return currentMaster;
  } catch (err) {
    markRuntimeStep_(timing, "ERROR - " + err.message);
    try {
      writeRuntimeTimingReport_(timing);
    } catch (reportErr) {
      Logger.log("Could not write timing report: " + reportErr.message);
    }
    throw err;
  }
}

function getChangedPMRsFromMonthlyChangeReport_(reportSheet) {
  const result = new Map();
  const lastRow = reportSheet.getLastRow();
  const lastCol = reportSheet.getLastColumn();


  if (lastRow < 1 || lastCol < 1) return result;


  const sectionTitles = ["Enrollments", "Disenrollments", "Demographic Changes", "Caseload Changes", "Banner Changes", "Contact Changes"];


  let foundSectionLayout = false;


  sectionTitles.forEach(section => {
    const titleRow = findMonthlyChangeSectionTitleRow_(reportSheet, section);
    if (titleRow === -1) return;


    foundSectionLayout = true;


    const headerRow = titleRow + 2;
    const dataStartRow = headerRow + 1;
    const nextTitleRow = findNextMonthlyChangeSectionTitleRow_(reportSheet, titleRow);
    const dataEndRow = nextTitleRow === -1 ? lastRow : nextTitleRow - 1;


    if (dataEndRow < dataStartRow) return;


    const headers = reportSheet.getRange(headerRow, 1, 1, lastCol)
      .getValues()[0]
      .map(h => String(h || "").trim());
    const headerMap = buildHeaderIndexMap_(headers);
    const pmrIdx = getPMRIndex_(headerMap);


    if (pmrIdx === -1) return;


    const values = reportSheet.getRange(dataStartRow, 1, dataEndRow - dataStartRow + 1, lastCol).getValues();


    values.forEach(row => {
      const pmr = normalizePMR_(row[pmrIdx]);
      if (!pmr) return;


      if (!result.has(pmr)) result.set(pmr, new Set());
      result.get(pmr).add(section);
    });
  });


  if (foundSectionLayout) return result;


  if (lastRow < 5) return result;


  const headers = reportSheet.getRange(4, 1, 1, lastCol)
    .getValues()[0]
    .map(h => String(h || "").trim());
  const headerMap = buildHeaderIndexMap_(headers);


  const sectionIdx = headerMap["Section"];
  const pmrIdx = headerMap["Participant PMR#"];


  if (sectionIdx === undefined || pmrIdx === undefined) return result;


  const values = reportSheet.getRange(5, 1, lastRow - 4, lastCol).getValues();


  values.forEach(row => {
    const pmr = normalizePMR_(row[pmrIdx]);
    const section = String(row[sectionIdx] || "").trim();


    if (!pmr) return;


    if (!result.has(pmr)) result.set(pmr, new Set());
    if (section) result.get(pmr).add(section);
  });


  return result;
}

function copyChangedPMRsFromDemoPToMasterList_(demoSheet, masterSheet, pmrsToCopy, monthParts) {
  if (!demoSheet || !masterSheet || !pmrsToCopy || pmrsToCopy.size === 0) return 0;


  const demoLastRow = demoSheet.getLastRow();
  const demoLastCol = demoSheet.getLastColumn();


  if (demoLastRow < DATA_START_ROW || demoLastCol < 1) return 0;


  const demoHeaders = getHeaders_(demoSheet, HEADER_ROW);
  const demoHeaderMap = buildHeaderIndexMap_(demoHeaders);
  const masterHeaders = getHeaders_(masterSheet, HEADER_ROW);
  const masterHeaderMap = buildHeaderIndexMap_(masterHeaders);
  const pmrIdx = getPMRIndex_(demoHeaderMap);


  if (pmrIdx === -1) throw new Error("Demo P is missing Participant PMR# / PMR # column.");


  const demoValues = demoSheet.getRange(DATA_START_ROW, 1, demoLastRow - DATA_START_ROW + 1, demoLastCol).getValues();
  const disenrolledPMRs = buildDisenrolledPMRSetFromDemoPValues_(demoValues, demoHeaderMap);
  const exclusionPMRs = loadDisenrolledExclusionPMRsForPart3_();
  exclusionPMRs.forEach(pmr => disenrolledPMRs.add(pmr));


  const output = [];


  demoValues.forEach(row => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr) return;
    if (!pmrsToCopy.has(pmr)) return;
    if (disenrolledPMRs.has(pmr)) return;


    const out = new Array(masterHeaders.length).fill("");
    demoHeaders.forEach((header, sourceIdx) => {
      if (!header) return;
      const targetIdx = masterHeaderMap[header];
      if (targetIdx !== undefined) out[targetIdx] = row[sourceIdx];
    });


    const statusIdx = masterHeaderMap["Master List Update Status"];
    const monthIdx = masterHeaderMap["Master List Update Month"];
    const sourceIdx = masterHeaderMap["Master List Source Sheet"];


    if (statusIdx !== undefined) out[statusIdx] = "Updated from Demo P";
    if (monthIdx !== undefined) out[monthIdx] = monthParts.firstDay;
    if (sourceIdx !== undefined) out[sourceIdx] = demoSheet.getName();


    output.push(out);
  });


  if (output.length === 0) return 0;


  const appendRow = Math.max(masterSheet.getLastRow() + 1, DATA_START_ROW);
  masterSheet.getRange(appendRow, 1, output.length, masterHeaders.length).setValues(output);


  return output.length;
}

      const value = String(row[idx] || "").trim();
      if (value !== "") parts.push(header);
    });


    row[bIdx] = parts.join(" | ");
  });


  dataRange.setValues(data);
}

function formatContactEntryForMasterList_(entry) {
  if (!entry) return "";


  const name = `${entry.first || ""} ${entry.last || ""}`.trim();
  const relationships = Array.from(entry.relationships).join(", ");
  const phones = Array.from(entry.phones).slice(0, 4).join(", ");
  const types = Array.from(entry.types).join(", ");


  let output = `${name} - ${relationships} - ${phones}---${types}`.trim();


  if (entry.language) {
    output += ` Language--${entry.language}`;
  }


  return output;
}

function collectMasterListHealthCheckRows_(masterSheet) {
  const outputRows = [["Check", "Status", "Count", "Quality Notes"]];

  if (!masterSheet) {
    outputRows.push(["Master List Sheet", "FAIL", 0, "Master List sheet was not found."]);
    return outputRows;
  }

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const data = getDataValues_(masterSheet, HEADER_ROW, DATA_START_ROW);
  const rows = data.values || [];
  const headerMap = data.headerMap || {};
  const pmrIdx = getPMRIndex_(headerMap);
  const bannerIdx = headerMap["Banner Summary"];

  const carePlanHeaders = [
    "Participant Name",
    "IDT Meeting Date",
    "Care Plan Start Date",
    "Enrollment Date",
    "Last Care Plan",
    "Next Care Plan Due",
    "CP Type"
  ];

  const detailRows = [];
  const byPMR = new Map();
  const duplicatePMRs = [];
  let rowsMissingPMR = 0;
  let rowsMissingBannerSummary = 0;
  let rowsMissingCarePlanFields = 0;

  rows.forEach(function(row, offset) {
    const rowNumber = DATA_START_ROW + offset;
    const pmr = pmrIdx === -1 ? "" : normalizePMR_(row[pmrIdx]);

    if (!pmr) {
      rowsMissingPMR++;
      detailRows.push(["Rows Missing PMR Detail", "FAIL", rowNumber, "Participant PMR# / PMR # is blank."]);
      return;
    }

    if (!byPMR.has(pmr)) byPMR.set(pmr, []);
    byPMR.get(pmr).push({ rowNumber: rowNumber, row: row });

    if (bannerIdx !== undefined && String(row[bannerIdx] || "").trim() === "") {
      rowsMissingBannerSummary++;
      detailRows.push(["Missing Banner Summary", "WARNING", pmr, "Master List row " + rowNumber + " has no Banner Summary."]);
    }

    const missingCareFields = carePlanHeaders.filter(function(header) {
      const idx = headerMap[header];
      return idx !== undefined && String(row[idx] || "").trim() === "";
    });

    if (missingCareFields.length > 0) {
      rowsMissingCarePlanFields++;
      detailRows.push(["Missing Care Plan Field(s)", "WARNING", pmr, missingCareFields.join(", ")]);
    }
  });

  byPMR.forEach(function(items, pmr) {
    if (items.length > 1) {
      duplicatePMRs.push(pmr);
      detailRows.push(["Duplicate PMR", "WARNING", pmr, "Rows: " + items.map(function(item) { return item.rowNumber; }).join(", ")]);
    }
  });

  const masterSuffix = getWorkflowMonthSuffixFromSheet_(masterSheet);
  const demoSheet = masterSuffix
    ? findMonthlyOutputSheetByPrefixAndSuffix_(ss, DEMO_P_PREFIX, masterSuffix)
    : getNewestFormattedMonthlySheetByPrefix_(ss, DEMO_P_PREFIX);

  let demoPrimaryPMRs = new Set();
  if (demoSheet) {
    const demoData = getDataValues_(demoSheet, HEADER_ROW, DATA_START_ROW);
    const demoPMRIdx = getPMRIndex_(demoData.headerMap || {});
    const demoPrimaryIdx = (demoData.headerMap || {})["Primary PMR Row"];
    (demoData.values || []).forEach(function(row) {
      if (demoPrimaryIdx !== undefined && String(row[demoPrimaryIdx] || "").trim().toLowerCase() !== "yes") return;
      const pmr = demoPMRIdx === -1 ? "" : normalizePMR_(row[demoPMRIdx]);
      if (pmr) demoPrimaryPMRs.add(pmr);
    });
  }

  const masterPMRs = new Set(Array.from(byPMR.keys()));
  const demoMasterMatches = countSetMatches_(masterPMRs, demoPrimaryPMRs);

  outputRows.push(["Master List Sheet", "PASS", 1, masterSheet.getName()]);
  outputRows.push(["Master List Month", masterSuffix ? "PASS" : "WARNING", masterSuffix || "Unknown", "Primary PMR validation is not performed here; Primary PMR is owned by Demo P Section M."]);
  outputRows.push(["Total Master List Rows", "INFO", rows.length, masterSheet.getName()]);
  outputRows.push(["Unique PMRs", "INFO", byPMR.size, ""]);
  outputRows.push(["Rows Missing PMR", rowsMissingPMR === 0 ? "PASS" : "FAIL", rowsMissingPMR, ""]);
  outputRows.push(["Duplicate PMRs", duplicatePMRs.length === 0 ? "PASS" : "WARNING", duplicatePMRs.length, ""]);
  outputRows.push(["Rows Missing Banner Summary", rowsMissingBannerSummary === 0 ? "PASS" : "WARNING", rowsMissingBannerSummary, ""]);
  outputRows.push(["Rows Missing Care Plan Fields", rowsMissingCarePlanFields === 0 ? "PASS" : "WARNING", rowsMissingCarePlanFields, ""]);
  outputRows.push(["Master List / Demo P Reconciliation", demoSheet && demoMasterMatches === masterPMRs.size ? "PASS" : "WARNING", demoMasterMatches, "Master PMRs: " + masterPMRs.size + "; Demo P primary PMRs: " + demoPrimaryPMRs.size + (demoSheet ? "; Demo P: " + demoSheet.getName() : "; Demo P not found")]);

  if (detailRows.length) {
    outputRows.push(["Detail", "INFO", detailRows.length, "Detailed issues follow."]);
    detailRows.slice(0, 100).forEach(function(row) {
      outputRows.push(row);
    });
  }

  return outputRows;
}

function runMonthlyChangeValidation() {
  const monthParts = promptForLockedYearReportMonth_("Monthly Change Validation");
  if (!monthParts) return;


  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const reportSheet = getMonthlySheetByPrefixAndDate_(ss, MONTHLY_CHANGE_REPORT_PREFIX, monthParts.firstDay, monthParts.lastDay);


  if (!reportSheet) {
    notify_("Monthly Change Report was not found for the selected month.");
    return;
  }


  const sections = ["Enrollments", "Disenrollments", "Demographic Changes", "Caseload Changes", "Contact Changes"];
  const summaryRows = [];
  const detailRows = [];


  sections.forEach(section => {
    const titleRow = findMonthlyChangeSectionTitleRow_(reportSheet, section);
    if (titleRow === -1) {
      summaryRows.push([section, "FAIL", 0, "Section title not found."]);
      detailRows.push(["FAIL", section, "Missing section title."]);
      return;
    }


    const headerRow = titleRow + 2;
    const dataStartRow = headerRow + 1;
    const nextTitleRow = findNextMonthlyChangeSectionTitleRow_(reportSheet, titleRow);
    const dataEndRow = nextTitleRow === -1 ? reportSheet.getLastRow() : nextTitleRow - 1;


    if (dataEndRow < dataStartRow) {
      summaryRows.push([section, "PASS", 0, "No rows in section."]);
      return;
    }


    const lastCol = Math.max(reportSheet.getLastColumn(), 1);
    const headers = reportSheet.getRange(headerRow, 1, 1, lastCol).getValues()[0].map(h => String(h || "").trim());
    const headerMap = buildHeaderIndexMap_(headers);
    const pmrIdx = getPMRIndex_(headerMap);


    const values = reportSheet.getRange(dataStartRow, 1, dataEndRow - dataStartRow + 1, lastCol).getValues();
    const nonBlankRows = values.filter(row => row.some(value => String(value || "").trim() !== ""));
    const uniquePMRs = new Set();
    let missingPMR = 0;


    nonBlankRows.forEach((row, idx) => {
      const rowNumber = dataStartRow + idx;
      const pmr = pmrIdx === -1 ? "" : normalizePMR_(row[pmrIdx]);
      if (!pmr) {
        missingPMR++;
        detailRows.push(["WARNING", section, "Missing PMR on row " + rowNumber]);
      } else {
        uniquePMRs.add(pmr);
      }
    });


    summaryRows.push([section, missingPMR === 0 ? "PASS" : "WARNING", nonBlankRows.length, "Unique PMRs: " + uniquePMRs.size + "; Missing PMRs: " + missingPMR]);
  });


  writeDiagnosticReport_(
    "Monthly Change Validation",
    "Monthly Change Validation",
    monthParts,
    ["Section", "Status", "Rows", "Details"],
    summaryRows,
    ["Status", "Section", "Details"],
    detailRows
  );


  notify_("Monthly Change Validation complete.");
}

function formatMonthlyChangeSubsectionBlock_(sheet, titleRow, label, colCount, dashboard, sheetDef, writeLabel) {
  const theme = getThemeColorsFromBase_(sheetDef.baseColor);
  const globals = dashboard.globals;
  const spacerRow = titleRow + 1;
  const headerRow = titleRow + 2;


  sheet.getRange(titleRow, 1, 1, colCount)
    .setBackground(theme.level3)
    .setFontFamily(globals.standardFont)
    .setFontSize(14)
    .setFontWeight("bold")
    .setFontColor(globals.standardFontColor)
    .setWrapStrategy(SpreadsheetApp.WrapStrategy.WRAP)
    .setHorizontalAlignment("left")
    .setVerticalAlignment("middle");


  applyMonthlyChangeSpacerRow3Format_(sheet, spacerRow, colCount, globals);


  sheet.getRange(headerRow, 1, 1, colCount)
    .setBackground(theme.level2)
    .setFontFamily(globals.standardFont)
    .setFontSize(8)
    .setFontWeight("bold")
    .setFontColor(globals.standardFontColor)
    .setWrapStrategy(SpreadsheetApp.WrapStrategy.WRAP)
    .setHorizontalAlignment("left")
    .setVerticalAlignment("top");


  if (writeLabel) {
    sheet.getRange(titleRow, 1).setValue(label || "");


    const monthlyHeaders = getHeadersForSheetType_(dashboard, SHEET_TYPE.MONTHLY_CHANGE).slice(0, colCount);
    while (monthlyHeaders.length < colCount) monthlyHeaders.push("");


    if (monthlyHeaders.length > 0) {
      sheet.getRange(headerRow, 1, 1, colCount).setValues([monthlyHeaders]);
    }
  }

function copyRawBannerDataToOutput_(rawSheet, outputSheet) {
  const rawLastRow = rawSheet.getLastRow();
  const rawLastCol = rawSheet.getLastColumn();


  if (rawLastRow < 2 || rawLastCol < 1) {
    throw new Error("Raw Banner sheet does not contain enough data rows to format.");
  }


  // Bounded read: only read the actual used rows/columns instead of getDataRange().
  const rawValues = rawSheet.getRange(1, 1, rawLastRow, rawLastCol).getValues();
  const outputHeaderRow = RFF_DEFAULTS.headerRow;
  const outputDataStartRow = RFF_DEFAULTS.dataStartRow;
  const outputLastCol = outputSheet.getLastColumn();
  const outputHeaders = outputSheet
    .getRange(outputHeaderRow, 1, 1, outputLastCol)
    .getValues()[0]
    .map(normalizeHeader_);
  const rawHeaders = rawValues[0].map(normalizeHeader_);
  const rawIndexByHeader = {};


  rawHeaders.forEach(function(header, index) {
    if (header && rawIndexByHeader[header] === undefined) {
      rawIndexByHeader[header] = index;
    }
  });


  const sourceData = rawValues.slice(1).filter(function(row) {
    return row.some(function(value) {
      return String(value === null || value === undefined ? "" : value).trim() !== "";
    });
  });


  const mapped = sourceData.map(function(row) {
    return outputHeaders.map(function(header) {
      const rawIndex = rawIndexByHeader[header];
      return rawIndex === undefined ? "" : row[rawIndex];
    });
  });


  if (mapped.length > 0) {
    ensureSheetHasAtLeastRows_(outputSheet, outputDataStartRow + mapped.length - 1);
    // Do not clear the duplicated template data area first. setValues overwrites the target rows.
    outputSheet.getRange(outputDataStartRow, 1, mapped.length, outputLastCol).setValues(mapped);
  }
}

  results.push(["Care Plan Sync Health", "Care Plan Due Unmatched", metrics.dueUnmatched === 0 ? "PASS" : "WARNING", String(metrics.dueUnmatched)]);
  results.push(["Care Plan Sync Health", "Rows With CP Data", metrics.rowsWithCPData > 0 ? "PASS" : "WARNING", String(metrics.rowsWithCPData)]);
  results.push(["Care Plan Sync Health", "Rows Missing CP Data", metrics.rowsMissingCPData === 0 ? "PASS" : "WARNING", String(metrics.rowsMissingCPData)]);
}

function collectCarePlanSyncValidationMetrics_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const metrics = {
    unlockedMatches: 0,
    unlockedUnmatched: 0,
    dueMatches: 0,
    dueUnmatched: 0,
    missingIDT: 0,
    missingCarePlanStart: 0,
    missingNextDue: 0,
    missingCPType: 0,
    rowsWithCPData: 0,
    rowsMissingCPData: 0
  };


  const masterSheet = getNewestFormattedMonthlySheetByPrefix_(ss, MASTER_LIST_PREFIX);
  if (!masterSheet) return metrics;


  const master = getDataValues_(masterSheet, HEADER_ROW, DATA_START_ROW);
  if (!master.values.length) return metrics;


  const masterPMRIdx = findHeaderIndex_(master.headerMap, ["Participant PMR#", "PMR #"]);
  const masterNameIdx = master.headerMap["Participant Name"];
  const capDateIdx = master.headerMap["Capitation Date"];


  const idtIdx = master.headerMap["IDT Meeting Date"];
  const cpsIdx = master.headerMap["Care Plan Start Date"];
  const nextDueIdx = master.headerMap["Next Care Plan Due"];
  const cpTypeIdx = master.headerMap["CP Type"];


  const unlockedSheet = getNewestFormattedMonthlySheetByPrefix_(ss, "Unlocked CP");
  const dueSheet = getNewestFormattedMonthlySheetByPrefix_(ss, "CP Due");


  const unlockedSet = new Set();
  if (unlockedSheet) {
    const unlocked = getDataValues_(unlockedSheet, HEADER_ROW, DATA_START_ROW);
    const sourcePMRIdx = findHeaderIndex_(unlocked.headerMap, ["PMR #", "Participant PMR#"]);
    if (sourcePMRIdx !== -1) {
      unlocked.values.forEach(function(row) {
        const pmr = normalizePMR_(row[sourcePMRIdx]);
        if (pmr) unlockedSet.add(pmr);
      });
    }
  }


  const dueSet = new Set();
  if (dueSheet) {
    const due = getDataValues_(dueSheet, HEADER_ROW, DATA_START_ROW);
    const nameIdx = due.headerMap["Participant Name"];
    const enrollmentIdx = due.headerMap["Enrollment Date"];
    if (nameIdx !== undefined && enrollmentIdx !== undefined) {
      due.values.forEach(function(row) {
        const key = normalizeKeyPart_(row[nameIdx]) + "|||" + normalizeKeyPart_(row[enrollmentIdx]);
        if (key.replace(/\|/g, "")) dueSet.add(key);
      });
    }
  }


  master.values.forEach(function(row) {
    const pmr = masterPMRIdx === -1 ? "" : normalizePMR_(row[masterPMRIdx]);
    if (pmr && unlockedSet.size) {
      if (unlockedSet.has(pmr)) metrics.unlockedMatches++;
      else metrics.unlockedUnmatched++;
    }


    const dueKey = (masterNameIdx === undefined || capDateIdx === undefined)
      ? ""
      : normalizeKeyPart_(row[masterNameIdx]) + "|||" + normalizeKeyPart_(row[capDateIdx]);
    if (dueKey && dueKey.replace(/\|/g, "") && dueSet.size) {
      if (dueSet.has(dueKey)) metrics.dueMatches++;
      else metrics.dueUnmatched++;
    }


    const hasIDT = idtIdx !== undefined && normalizeCompareValue_(row[idtIdx]) !== "";
    const hasCPS = cpsIdx !== undefined && normalizeCompareValue_(row[cpsIdx]) !== "";
    const hasNextDue = nextDueIdx !== undefined && normalizeCompareValue_(row[nextDueIdx]) !== "";
    const hasCPType = cpTypeIdx !== undefined && normalizeCompareValue_(row[cpTypeIdx]) !== "";


    if (!hasIDT) metrics.missingIDT++;
    if (!hasCPS) metrics.missingCarePlanStart++;
    if (!hasNextDue) metrics.missingNextDue++;
    if (!hasCPType) metrics.missingCPType++;


    if (hasIDT || hasCPS || hasNextDue || hasCPType) metrics.rowsWithCPData++;
    else metrics.rowsMissingCPData++;
  });


  return metrics;
}

function verifyDemoPPrimaryPMRAssignment_(values, headerMap) {
  const rows = [];
  const pmrIdx = getPMRIndex_(headerMap);
  const primaryIdx = headerMap["Primary PMR Row"];

  if (pmrIdx === -1) {
    return [["Primary PMR Assignment", "FAIL", "Missing PMR header", "Demo P is missing Participant PMR# / PMR #."]];
  }

  if (primaryIdx === undefined) {
    return [["Primary PMR Assignment", "FAIL", "Missing Primary PMR Row", "Demo P is missing Primary PMR Row."]];
  }

  const byPMR = {};
  (values || []).forEach(function(row) {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr) return;
    if (!byPMR[pmr]) byPMR[pmr] = { total: 0, primary: 0 };
    byPMR[pmr].total++;
    if (String(row[primaryIdx] || "").trim().toLowerCase() === "yes") byPMR[pmr].primary++;
  });

  const keys = Object.keys(byPMR);
  let missing = 0;
  let duplicate = 0;

  keys.forEach(function(pmr) {
    if (byPMR[pmr].primary === 0) missing++;
    if (byPMR[pmr].primary > 1) duplicate++;
  });

  rows.push(["Primary PMR Assignment - Unique PMRs", keys.length > 0 ? "PASS" : "FAIL", keys.length, "Demo P PMR groups."]);
  rows.push(["Primary PMR Assignment - Missing Primary", missing === 0 ? "PASS" : "FAIL", missing, "Every PMR must have exactly one Primary PMR Row = Yes."]);
  rows.push(["Primary PMR Assignment - Duplicate Primary", duplicate === 0 ? "PASS" : "FAIL", duplicate, "No PMR may have multiple Primary PMR Row = Yes values."]);

  return rows;
}

function verifyDemoPContactProcessing_(values, headerMap) {
  const rows = [];
  const summaryIdx = headerMap["Contact - Summary"];
  const contactIndexes = [];

  for (let i = 1; i <= 8; i++) {
    const idx = headerMap["Contact - " + i];
    if (idx !== undefined) contactIndexes.push(idx);
  }

  if (summaryIdx === undefined && contactIndexes.length === 0) {
    return [["Contact Processing", "FAIL", "Missing columns", "Contact - Summary and Contact - 1 through Contact - 8 were not found."]];
  }

  let rowsWithContact = 0;
  let rowsWithSummary = 0;
  let missingSummaryWhenContactExists = 0;

  (values || []).forEach(function(row) {
    const hasContact = contactIndexes.some(function(idx) {
      return String(row[idx] || "").trim() !== "";
    });
    const hasSummary = summaryIdx !== undefined && String(row[summaryIdx] || "").trim() !== "";

    if (hasContact) rowsWithContact++;
    if (hasSummary) rowsWithSummary++;
    if (hasContact && !hasSummary && summaryIdx !== undefined) missingSummaryWhenContactExists++;
  });

  rows.push(["Contact Processing - Columns", "PASS", contactIndexes.length + (summaryIdx !== undefined ? 1 : 0), "Contact output columns found."]);
  rows.push(["Contact Processing - Summary", missingSummaryWhenContactExists === 0 ? "PASS" : "WARNING", rowsWithSummary, "Rows with Contact Summary; missing where contact exists: " + missingSummaryWhenContactExists]);

  return rows;
}

function verifyDemoPPhoneSplit_(values, headerMap) {
  const rows = [];
  const valueIndexes = [];
  const labelIndexes = [];

  for (let i = 1; i <= 4; i++) {
    const valueIdx = headerMap["Phone " + i + " - Value"];
    const labelIdx = headerMap["Phone " + i + " - Label"];
    if (valueIdx !== undefined) valueIndexes.push(valueIdx);
    if (labelIdx !== undefined) labelIndexes.push(labelIdx);
  }

  if (!valueIndexes.length) {
    return [["Phone Split", "FAIL", "Missing Phone Value columns", "Expected Phone 1 - Value through Phone 4 - Value."]];
  }

  let totalPhones = 0;
  let rowsWithPhone = 0;
  let gapRows = 0;

  (values || []).forEach(function(row) {
    let rowPhoneCount = 0;
    let seenBlankBeforeValue = false;
    let blankSeen = false;

    valueIndexes.forEach(function(idx) {
      const hasValue = String(row[idx] || "").trim() !== "";
      if (hasValue) {
        rowPhoneCount++;
        totalPhones++;
        if (blankSeen) seenBlankBeforeValue = true;
      } else {
        blankSeen = true;
      }
    });

    if (rowPhoneCount > 0) rowsWithPhone++;
    if (seenBlankBeforeValue) gapRows++;
  });

  rows.push(["Phone Split - Columns", valueIndexes.length >= 4 ? "PASS" : "WARNING", valueIndexes.length, "Phone value columns found; label columns found: " + labelIndexes.length]);
  rows.push(["Phone Split - Values", totalPhones > 0 ? "PASS" : "WARNING", totalPhones, "Rows with at least one phone: " + rowsWithPhone]);
  rows.push(["Phone Split - Sequence Gaps", gapRows === 0 ? "PASS" : "WARNING", gapRows, "Rows where a later phone is populated after an earlier blank phone."]);

  return rows;
}
