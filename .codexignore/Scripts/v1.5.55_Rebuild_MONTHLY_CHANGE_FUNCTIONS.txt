function getMonthlyChangeReportHeaders_(sourceHeaders) {
  const headers = sourceHeaders.slice();


  if (headers.indexOf("Columns With Change") === -1) {
    headers.push("Columns With Change");
  }


  return headers;
}

function convertMonthlyChangeReportDateValues_(rowValues, reportHeaders) {
  const output = rowValues.slice();


  GLOBAL_DATE_FORMAT_HEADERS.forEach(headerName => {
    const idx = reportHeaders.indexOf(headerName);
    if (idx === -1 || idx >= output.length) return;


    const value = output[idx];


    if (value instanceof Date && !isNaN(value.getTime())) return;


    if (typeof value === "number" && !isNaN(value) && value > 20000) {
      const convertedDate = spreadsheetSerialDateToLocalDate_(value);
      if (convertedDate) output[idx] = convertedDate;
      return;
    }


    const parsed = coerceToValidDate_(value);
    if (parsed) output[idx] = parsed;
  });


  return output;
}

function buildMonthlyChangeReportRow_(sourceRow, sourceHeaders, reportHeaders, changedColumns) {
  const output = sourceRow.slice(0, sourceHeaders.length);


  while (output.length < reportHeaders.length) {
    output.push("");
  }


  const changeIdx = reportHeaders.indexOf("Columns With Change");
  if (changeIdx !== -1) {
    output[changeIdx] = changedColumns && changedColumns.size > 0
      ? Array.from(changedColumns).sort().join(", ")
      : "";
  }


  return convertMonthlyChangeReportDateValues_(output, reportHeaders);
}

function buildMonthlyChangeReportFromDemoP() {
  const monthParts = promptForLockedYearReportMonth_("Build Monthly Change Report");
  if (!monthParts) return;


  buildMonthlyChangeReportFromDemoPForMonth_(monthParts);
}

function buildMonthlyChangeReportFromDemoPForMonth_(monthParts) {
  if (!monthParts) return null;


  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const currentDemo = getCurrentDemoPSheet_(monthParts);
  const previousDemo = getPreviousDemoPSheet_(monthParts);


  if (!currentDemo) {
    notify_(
      "Current month Demo P was not found.\n\n" +
      `Expected current month start date: ${formatDateForSheetName_(monthParts.firstDay)}\n` +
      "Example accepted name:\n" +
      `Demo P ${formatDateForSheetName_(monthParts.firstDay)} - ${formatDateForSheetName_(monthParts.lastDay)}`
    );
    return null;
  }


  if (!previousDemo) {
    notify_(
      "Previous month Demo P was not found.\n\n" +
      "The Monthly Change Report compares previous month Demo P to current month Demo P.\n\n" +
      `Current source found: ${currentDemo.getName()}\n` +
      `Expected previous month start date: ${formatDateForSheetName_(monthParts.previousMonthFirstDay)}\n` +
      "Example accepted name:\n" +
      `Demo P ${formatDateForSheetName_(monthParts.previousMonthFirstDay)} - ${formatDateForSheetName_(monthParts.previousMonthLastDay)}`
    );
    return null;
  }


  formatDateColumnsByHeader_(currentDemo, HEADER_ROW, DATA_START_ROW);
  formatDateColumnsByHeader_(previousDemo, HEADER_ROW, DATA_START_ROW);


  const sectionData = compareRawDemoPForSectionReport_(previousDemo, currentDemo, monthParts);
  const totalPMRs =
    sectionData.enrollmentPMRs.size +
    sectionData.disenrollmentPMRs.size +
    sectionData.demographicPMRs.size +
    sectionData.caseloadPMRs.size +
    sectionData.contactPMRs.size;


  if (totalPMRs === 0) {
    notify_(
      "No raw Demo P changes found. Monthly Change Report was not created.\n\n" +
      `Current source: ${currentDemo.getName()}\n` +
      `Previous source: ${previousDemo.getName()}\n\n` +
      `Current Demo P rows read: ${sectionData.currentData.values.length}\n` +
      `Previous Demo P rows read: ${sectionData.previousData.values.length}\n` +
      `Current PMRs found: ${sectionData.currentData.rowsByPMR.size}\n` +
      `Previous PMRs found: ${sectionData.previousData.rowsByPMR.size}\n\n` +
      "For enrollments, Capitation Date must equal the report date. " +
      "For disenrollments, Disenrollment Effective Date must be between the previous month first day and the report date."
    );
    return null;
  }


  const reportName = buildMonthlySheetName_(MONTHLY_CHANGE_REPORT_PREFIX, monthParts.firstDay, monthParts.lastDay);
  let reportSheet = ss.getSheetByName(reportName);


  if (reportSheet) ss.deleteSheet(reportSheet);


  reportSheet = ss.insertSheet(reportName);
  reportSheet;


  buildMonthlyChangeReportSectionLayout_(reportSheet, currentDemo, sectionData.currentData.headers, monthParts);
  populateMonthlyChangeReportSections_(reportSheet, sectionData, monthParts);
  formatMonthlyChangeReportSectionSheet_(reportSheet, sectionData.currentData.headers);


  setUniqueSheetName_(reportSheet, reportName);
  formatDateColumnsOnAllSheets_();
  createIndexSheet();


    applyStandardRowHeights_(reportSheet);


notify_(
    "Monthly Change Report created.\n\n" +
    `Current source: ${currentDemo.getName()}\n` +
    `Previous source: ${previousDemo.getName()}\n\n` +
    `Enrollments: ${sectionData.enrollmentPMRs.size}\n` +
    `Disenrollments: ${sectionData.disenrollmentPMRs.size}\n` +
    `Demographic Changes: ${sectionData.demographicPMRs.size}\n` +
    `Caseload Changes: ${sectionData.caseloadPMRs.size}\n` +
    `Contact Changes: ${sectionData.contactPMRs.size}`
  );


  return reportSheet;
}

function getOrBuildMonthlyChangeReport_(ss, monthParts, currentDemo, previousDemo) {
  let reportSheet = getMonthlySheetByPrefixAndDate_(ss, MONTHLY_CHANGE_REPORT_PREFIX, monthParts.firstDay, monthParts.lastDay);


  if (reportSheet) return reportSheet;


  if (!currentDemo) {
    throw new Error("Current month Demo P was not found. Format Demo P first.");
  }


  if (!previousDemo) {
    throw new Error("Previous month Demo P was not found. Format the previous Demo P first.");
  }


  formatDateColumnsByHeader_(currentDemo, HEADER_ROW, DATA_START_ROW);
  formatDateColumnsByHeader_(previousDemo, HEADER_ROW, DATA_START_ROW);


  const sectionData = compareRawDemoPForSectionReport_(previousDemo, currentDemo, monthParts);
  const totalPMRs =
    sectionData.enrollmentPMRs.size +
    sectionData.disenrollmentPMRs.size +
    sectionData.demographicPMRs.size +
    sectionData.caseloadPMRs.size +
    sectionData.contactPMRs.size;


  if (totalPMRs === 0) return null;


  const reportName = buildMonthlySheetName_(MONTHLY_CHANGE_REPORT_PREFIX, monthParts.firstDay, monthParts.lastDay);
  reportSheet = ss.insertSheet(reportName);
  reportSheet;


  buildMonthlyChangeReportSectionLayout_(reportSheet, currentDemo, sectionData.currentData.headers, monthParts);
  populateMonthlyChangeReportSections_(reportSheet, sectionData, monthParts);
  formatMonthlyChangeReportSectionSheet_(reportSheet, sectionData.currentData.headers);
  setUniqueSheetName_(reportSheet, reportName);


  return reportSheet;
}

function compareRawDemoPForSectionReport_(previousDemo, currentDemo, monthParts) {
  const previousData = getRawDemoPDataForCompare_(previousDemo);
  const currentData = getRawDemoPDataForCompare_(currentDemo);


  const enrollmentPMRs = new Set();
  const disenrollmentPMRs = new Set();
  const demographicPMRs = new Set();
  const caseloadPMRs = new Set();
  const contactPMRs = new Set();


  const demographicChangedColumnsByPMR = new Map();
  const caseloadChangedColumnsByPMR = new Map();
  const contactChangedColumnsByPMR = new Map();


  currentData.rowsByPMR.forEach((currentRows, pmr) => {
    const previousRows = previousData.rowsByPMR.get(pmr) || [];


    const capitationHeader = "Capitation Date";
    const disenrollEffectiveHeader = "Disenrollment Effective Date";


    const mostRecentCapitationDate = getMostRecentDateFromRowsByHeader_(
      currentRows,
      currentData.headerMap,
      capitationHeader
    );


    const isEnrollmentPMR = isSameDate_(mostRecentCapitationDate, monthParts.firstDay);


    const isDisenrollmentPMR =
      currentRows.some(item => {
        const effectiveDate = getFieldValueFromRow_(
          item.values,
          currentData.headerMap,
          disenrollEffectiveHeader
        );


        return isDateInStrictLocalRangeInclusive_(
          effectiveDate,
          monthParts.previousMonthFirstDay,
          monthParts.firstDay
        );
      });


    if (isEnrollmentPMR) enrollmentPMRs.add(pmr);
    if (isDisenrollmentPMR) disenrollmentPMRs.add(pmr);


    if (isEnrollmentPMR || isDisenrollmentPMR) return;


    if (isParticipantEnrollmentStatusDisenrolled_(currentRows, currentData.headerMap)) return;


    const currentDobRows = rowsWithDOBOnlyForSection_(currentRows, currentData.headerMap);
    const previousDobRows = rowsWithDOBOnlyForSection_(previousRows, previousData.headerMap);


    const demographicChangedColumns = getChangedColumnsForSectionRows_(
      currentDobRows,
      previousDobRows,
      currentData.headers,
      previousData.headers,
      RAW_DEMO_P_DEMOGRAPHIC_FIELDS
    );


    if (demographicChangedColumns.size > 0) {
      demographicPMRs.add(pmr);
      demographicChangedColumnsByPMR.set(pmr, demographicChangedColumns);
    }


    const caseloadChangedColumns = getChangedColumnsForSectionRows_(
      currentDobRows,
      previousDobRows,
      currentData.headers,
      previousData.headers,
      RAW_DEMO_P_CASELOAD_FIELDS
    );


    if (caseloadChangedColumns.size > 0) {
      caseloadPMRs.add(pmr);
      caseloadChangedColumnsByPMR.set(pmr, caseloadChangedColumns);
    }


    const contactChangedColumns = getChangedColumnsForSectionRows_(
      currentRows,
      previousRows,
      currentData.headers,
      previousData.headers,
      RAW_DEMO_P_CONTACT_FIELDS
    );


    if (contactChangedColumns.size > 0) {
      contactPMRs.add(pmr);
      contactChangedColumnsByPMR.set(pmr, contactChangedColumns);
    }
  });


  enrollmentPMRs.forEach(pmr => {
    demographicPMRs.delete(pmr);
    caseloadPMRs.delete(pmr);
    contactPMRs.delete(pmr);
  });


  disenrollmentPMRs.forEach(pmr => {
    demographicPMRs.delete(pmr);
    caseloadPMRs.delete(pmr);
    contactPMRs.delete(pmr);
  });


  return {
    previousData: previousData,
    currentData: currentData,
    enrollmentPMRs: enrollmentPMRs,
    disenrollmentPMRs: disenrollmentPMRs,
    demographicPMRs: demographicPMRs,
    caseloadPMRs: caseloadPMRs,
    contactPMRs: contactPMRs,
    demographicChangedColumnsByPMR: demographicChangedColumnsByPMR,
    caseloadChangedColumnsByPMR: caseloadChangedColumnsByPMR,
    contactChangedColumnsByPMR: contactChangedColumnsByPMR
  };
}

function rowsWithDOBOnlyForSection_(items, headerMap) {
  const dobIdx = getDOBIndex_(headerMap);
  if (dobIdx === -1) return [];
  return (items || []).filter(item => normalizeCompareValue_(item.values[dobIdx]) !== "");
}

function getChangedColumnsForSectionRows_(currentItems, previousItems, currentHeaders, previousHeaders, columnsToCompare) {
  const changed = new Set();


  if (!currentItems || currentItems.length === 0) return changed;


  if (!previousItems || previousItems.length === 0) return changed;


  columnsToCompare.forEach(header => {


    if (currentHeaders.indexOf(header) === -1 || previousHeaders.indexOf(header) === -1) return;


    const currentSig = buildColumnSignatureForSection_(currentItems, currentHeaders, [header]);
    const previousSig = buildColumnSignatureForSection_(previousItems, previousHeaders, [header]);


    if (currentSig !== previousSig) changed.add(header);
  });


  return changed;
}

function buildColumnSignatureForSection_(items, headers, columnsToCompare) {
  const headerMap = buildHeaderIndexMap_(headers);


  return (items || [])
    .map(item => {
      return columnsToCompare.map(header => {
        const idx = headerMap[header];
        if (idx === undefined || idx === -1) return "";
        return normalizeCompareValue_(item.values[idx]);
      }).join("~");
    })
    .sort()
    .join("||");
}

function compareRawDemoPForChanges_(previousDemo, currentDemo, monthParts) {
  const previousData = getRawDemoPDataForCompare_(previousDemo);
  const currentData = getRawDemoPDataForCompare_(currentDemo);


  const outputRows = [];
  const changedPMRs = new Map();


  currentData.participantRows.forEach((currentMain, pmr) => {
    const previousMain = previousData.participantRows.get(pmr);


    const capitationIdx = currentData.headerMap["Capitation Date"];
    const currentCapitation = capitationIdx !== undefined ? currentMain.values[capitationIdx] : "";


    const isNewPMR = !previousData.allPMRs.has(pmr);
    const isReportMonthCapitation = isSameMonth_(currentCapitation, monthParts.firstDay);


    if (isNewPMR || isReportMonthCapitation) {
      RAW_DEMO_P_ENROLLMENT_FIELDS.forEach(field => {
        const currentValue = getFieldValueFromRow_(currentMain.values, currentData.headerMap, field);
        if (normalizeCompareValue_(currentValue) !== "") {
          addMCRRow_(
            outputRows,
            changedPMRs,
            CHANGE_SECTION_ENROLLMENTS,
            pmr,
            currentMain,
            null,
            currentData,
            previousData,
            field,
            "",
            currentValue,
            monthParts
          );
        }
      });
    }
  });


  currentData.participantRows.forEach((currentMain, pmr) => {
    const effective = getFieldValueFromRow_(currentMain.values, currentData.headerMap, "Disenrollment Effective Date");
    const disenrollDate = getFieldValueFromRow_(currentMain.values, currentData.headerMap, "Disenrollment Date");
    const deathDate = getFieldValueFromRow_(currentMain.values, currentData.headerMap, "Date of Death");


    const isCurrentEffective = isSameDate_(effective, monthParts.firstDay);
    const isPriorMonthDisenroll = isSameMonth_(disenrollDate, monthParts.previousMonthFirstDay);
    const isPriorMonthDeath = isSameMonth_(deathDate, monthParts.previousMonthFirstDay);


    if (isCurrentEffective || isPriorMonthDisenroll || isPriorMonthDeath) {
      RAW_DEMO_P_DISENROLLMENT_FIELDS.forEach(field => {
        const currentValue = getFieldValueFromRow_(currentMain.values, currentData.headerMap, field);
        if (normalizeCompareValue_(currentValue) !== "") {
          addMCRRow_(
            outputRows,
            changedPMRs,
            CHANGE_SECTION_DISENROLLMENTS,
            pmr,
            currentMain,
            null,
            currentData,
            previousData,
            field,
            "",
            currentValue,
            monthParts
          );
        }
      });
    }
  });


  currentData.participantRows.forEach((currentMain, pmr) => {
    const previousMain = previousData.participantRows.get(pmr);
    if (!previousMain) return;


    RAW_DEMO_P_DEMOGRAPHIC_FIELDS.forEach(field => {
      compareSingleFieldAndAdd_(
        outputRows,
        changedPMRs,
        CHANGE_SECTION_DEMOGRAPHIC,
        pmr,
        previousMain,
        currentMain,
        previousData,
        currentData,
        field,
        monthParts,
        true
      );
    });


    RAW_DEMO_P_CASELOAD_FIELDS.forEach(field => {
      compareSingleFieldAndAdd_(
        outputRows,
        changedPMRs,
        CHANGE_SECTION_CASELOAD,
        pmr,
        previousMain,
        currentMain,
        previousData,
        currentData,
        field,
        monthParts,
        true
      );
    });
  });


  const allCurrentPMRs = Array.from(currentData.allPMRs);
  allCurrentPMRs.forEach(pmr => {
    if (!previousData.allPMRs.has(pmr)) return;


    const previousRows = previousData.rowsByPMR.get(pmr) || [];
    const currentRows = currentData.rowsByPMR.get(pmr) || [];


    const maxLength = Math.max(previousRows.length, currentRows.length);


    for (let i = 0; i < maxLength; i++) {
      const previousItem = previousRows[i] || null;
      const currentItem = currentRows[i] || null;


      RAW_DEMO_P_CONTACT_FIELDS.forEach(field => {
        const previousValue = previousItem ? getFieldValueFromRow_(previousItem.values, previousData.headerMap, field) : "";
        const currentValue = currentItem ? getFieldValueFromRow_(currentItem.values, currentData.headerMap, field) : "";


        if (!valuesAreEqual_(previousValue, currentValue)) {
          addMCRRow_(
            outputRows,
            changedPMRs,
            CHANGE_SECTION_CONTACT,
            pmr,
            currentItem,
            previousItem,
            currentData,
            previousData,
            field,
            previousValue,
            currentValue,
            monthParts
          );
        }
      });
    }
  });


  return {
    rows: outputRows,
    changedPMRs: changedPMRs
  };
}

function getRawDemoPDataForCompare_(sheet) {
  const data = getDataValues_(sheet, HEADER_ROW, DATA_START_ROW);


  const headers = data.headers;
  const headerMap = data.headerMap;
  const pmrIdx = getPMRIndex_(headerMap);
  const dobIdx = getDOBIndex_(headerMap);


  if (pmrIdx === -1) {
    throw new Error(`${sheet.getName()} is missing Participant PMR# / PMR # column.`);
  }


  const values = data.values;


  const rowsByPMR = new Map();
  const participantRows = new Map();
  const allPMRs = new Set();


  values.forEach((row, idx) => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr) return;


    allPMRs.add(pmr);


    const item = {
      rowNumber: DATA_START_ROW + idx,
      values: row
    };


    if (!rowsByPMR.has(pmr)) rowsByPMR.set(pmr, []);
    rowsByPMR.get(pmr).push(item);


    if (!participantRows.has(pmr)) {
      if (dobIdx === -1 || normalizeCompareValue_(row[dobIdx]) !== "") {
        participantRows.set(pmr, item);
      }
    }
  });


  rowsByPMR.forEach((items, pmr) => {
    if (!participantRows.has(pmr) && items.length > 0) {
      participantRows.set(pmr, items[0]);
    }
  });


  return {
    sheet: sheet,
    sheetName: sheet.getName(),
    headers: headers,
    headerMap: headerMap,
    values: values,
    rowsByPMR: rowsByPMR,
    participantRows: participantRows,
    allPMRs: allPMRs
  };
}

function compareSingleFieldAndAdd_(
  outputRows,
  changedPMRs,
  section,
  pmr,
  previousItem,
  currentItem,
  previousData,
  currentData,
  field,
  monthParts,
  skipIfCurrentDOBBlank
) {
  if (!previousItem || !currentItem) return;


  if (skipIfCurrentDOBBlank) {
    const currentDOBIdx = getDOBIndex_(currentData.headerMap);
    if (currentDOBIdx !== -1 && normalizeCompareValue_(currentItem.values[currentDOBIdx]) === "") return;
  }


  if (field !== "Participant PMR#" && currentData.headerMap[field] === undefined && previousData.headerMap[field] === undefined) {
    return;
  }


  const previousValue = getFieldValueFromRow_(previousItem.values, previousData.headerMap, field);
  const currentValue = getFieldValueFromRow_(currentItem.values, currentData.headerMap, field);


  if (!valuesAreEqual_(previousValue, currentValue)) {
    addMCRRow_(
      outputRows,
      changedPMRs,
      section,
      pmr,
      currentItem,
      previousItem,
      currentData,
      previousData,
      field,
      previousValue,
      currentValue,
      monthParts
    );
  }
}

function addMCRRow_(
  outputRows,
  changedPMRs,
  section,
  pmr,
  currentItem,
  previousItem,
  currentData,
  previousData,
  field,
  previousValue,
  currentValue,
  monthParts
) {
  const mainItem = currentItem || previousItem;
  const currentValues = currentItem ? currentItem.values : (mainItem ? mainItem.values : []);
  const currentMap = currentData.headerMap;


  const lastName = getFieldValueFromRow_(currentValues, currentMap, "Last Name");
  const firstName = getFieldValueFromRow_(currentValues, currentMap, "First Name");
  const participantName = buildParticipantName_(currentValues, currentMap);


  outputRows.push([
    section,
    pmr,
    participantName,
    lastName,
    firstName,
    field,
    displayValueForReport_(previousValue),
    displayValueForReport_(currentValue),
    previousItem ? previousItem.rowNumber : "",
    currentItem ? currentItem.rowNumber : "",
    monthParts.firstDay
  ]);


  if (!changedPMRs.has(pmr)) changedPMRs.set(pmr, new Set());
  changedPMRs.get(pmr).add(section);
}

function getFieldValueFromRow_(row, headerMap, field) {
  const idx = headerMap[field];
  if (idx === undefined || idx < 0) return "";
  return row[idx];
}

function buildParticipantName_(row, headerMap) {
  const participantName = getFieldValueFromRow_(row, headerMap, "Participant Name");
  if (normalizeCompareValue_(participantName) !== "") return participantName;


  const last = getFieldValueFromRow_(row, headerMap, "Last Name");
  const first = getFieldValueFromRow_(row, headerMap, "First Name");


  const lastText = String(last || "").trim();
  const firstText = String(first || "").trim();


  if (lastText && firstText) return `${lastText}, ${firstText}`;
  return lastText || firstText || "";
}

function displayValueForReport_(value) {
  if (value instanceof Date && !isNaN(value.getTime())) {
    return formatDateDisplay_(value);
  }
  return value === null || value === undefined ? "" : String(value);
}

function insertMonthlyDisenrollmentRowsStrict_(reportSheet, currentData, pmrSet, monthParts) {


  if (!monthParts) return;


  const sectionTitle = "Disenrollments";
  const titleRow = findMonthlyChangeSectionTitleRow_(reportSheet, sectionTitle);
  if (titleRow === -1) return;


  const headerRow = titleRow + 2;
  const insertAt = headerRow + 1;
  const reportHeaders = getMonthlyChangeReportHeaders_(currentData.headers);
  const lastCol = reportHeaders.length;


  const effectiveIdx = currentData.headerMap["Disenrollment Effective Date"];
  const pmrIdx = getPMRIndex_(currentData.headerMap);


  if (effectiveIdx === undefined || effectiveIdx === -1 || pmrIdx === -1) return;


  const rowsToInsert = [];
  const usedPMRs = new Set();


  currentData.values.forEach(row => {
    const pmr = normalizePMR_(row[pmrIdx]);
    if (!pmr || usedPMRs.has(pmr)) return;


    const effectiveDate = row[effectiveIdx];


    if (!isDateDisplayInReportWindow_(
      effectiveDate,
      monthParts.previousMonthFirstDay,
      monthParts.firstDay
    )) {
      return;
    }


    rowsToInsert.push(buildMonthlyChangeReportRow_(
      row,
      currentData.headers,
      reportHeaders,
      new Set()
    ));


    usedPMRs.add(pmr);
  });


  if (rowsToInsert.length === 0) return;


  reportSheet.insertRowsBefore(insertAt, rowsToInsert.length);


  const dataRange = reportSheet.getRange(insertAt, 1, rowsToInsert.length, lastCol);
  dataRange.setValues(rowsToInsert);
  dataRange;


  try {
    reportSheet.getRange(headerRow, 1, 1, lastCol)
      .copyTo(dataRange, SpreadsheetApp.CopyPasteType.PASTE_FORMAT, false);
  } catch (err) {
    Logger.log("Monthly Change disenrollment data format copy skipped: " + err.message);
  }


  dataRange;


  reportSheet;
}

function insertMonthlyChangeSectionRows_(reportSheet, currentData, pmrSet, sectionTitle, rowMode, changedColumnsByPMR, monthParts) {
  if (!pmrSet || pmrSet.size === 0) return;


  const titleRow = findMonthlyChangeSectionTitleRow_(reportSheet, sectionTitle);
  if (titleRow === -1) return;


  const headerRow = titleRow + 2;
  const insertAt = headerRow + 1;
  const reportHeaders = getMonthlyChangeReportHeaders_(currentData.headers);
  const lastCol = reportHeaders.length;
  const dobIdx = getDOBIndex_(currentData.headerMap);
  const disenrollEffectiveIdx = currentData.headerMap["Disenrollment Effective Date"];


  const allowDuplicateRowsForPMR = sectionTitle === "Contact Changes";


  const rowsToInsert = [];
  const pmrForEachInsertedRow = [];


  Array.from(pmrSet).sort().forEach(pmr => {
    const items = currentData.rowsByPMR.get(pmr) || [];
    let insertedForThisPMR = false;


    items.forEach(item => {
      if (!allowDuplicateRowsForPMR && insertedForThisPMR) return;


      if (rowMode === "dobOnly" && (dobIdx === -1 || normalizeCompareValue_(item.values[dobIdx]) === "")) return;


      if (rowMode === "disenrollmentEffectiveRange") {
        if (disenrollEffectiveIdx === undefined || disenrollEffectiveIdx === -1 || !monthParts) return;


        const effectiveDate = item.values[disenrollEffectiveIdx];


        if (!isDateInStrictLocalRangeInclusive_(
          effectiveDate,
          monthParts.previousMonthFirstDay,
          monthParts.firstDay
        )) {
          return;
        }
      }


      const changedColumns = changedColumnsByPMR && changedColumnsByPMR.has(pmr)
        ? changedColumnsByPMR.get(pmr)
        : new Set();


      rowsToInsert.push(buildMonthlyChangeReportRow_(item.values, currentData.headers, reportHeaders, changedColumns));
      pmrForEachInsertedRow.push(pmr);


      if (!allowDuplicateRowsForPMR) insertedForThisPMR = true;
    });
  });


  if (rowsToInsert.length === 0) return;


  reportSheet.insertRowsBefore(insertAt, rowsToInsert.length);


  const dataRange = reportSheet.getRange(insertAt, 1, rowsToInsert.length, lastCol);
  dataRange.setValues(rowsToInsert);
  dataRange;


  const headerFormatRange = reportSheet.getRange(headerRow, 1, 1, lastCol);
  try {
    headerFormatRange.copyTo(dataRange, SpreadsheetApp.CopyPasteType.PASTE_FORMAT, false);
  } catch (err) {
    Logger.log("Monthly Change data format copy skipped: " + err.message);
  }


  dataRange;


  dataRange;


  if (changedColumnsByPMR && changedColumnsByPMR.size > 0) {
    const backgrounds = dataRange.getBackgrounds();


    for (let r = 0; r < backgrounds.length; r++) {
      const pmr = pmrForEachInsertedRow[r];
      const changedColumns = changedColumnsByPMR.get(pmr);


      if (!changedColumns || changedColumns.size === 0) continue;


      reportHeaders.forEach((header, colIndex) => {
        if (changedColumns.has(header)) {
          backgrounds[r][colIndex] = "#f3ffc7";
        }
      });
    }


    dataRange;
  }


  reportSheet;
}

function findMonthlyChangeSectionTitleRow_(sheet, title) {
  const lastRow = sheet.getLastRow();
  if (lastRow < 1) return -1;


  const values = sheet.getRange(1, 1, lastRow, 1).getValues();


  for (let i = 0; i < values.length; i++) {
    if (String(values[i][0] || "").trim() === title) return i + 1;
  }


  return -1;
}

function findNextMonthlyChangeSectionTitleRow_(sheet, afterRow) {
  const titles = new Set(["Enrollments", "Disenrollments", "Demographic Changes", "Caseload Changes", "Contact Changes"]);
  const lastRow = sheet.getLastRow();


  if (afterRow >= lastRow) return -1;


  const values = sheet.getRange(afterRow + 1, 1, lastRow - afterRow, 1).getValues();


  for (let i = 0; i < values.length; i++) {
    const value = String(values[i][0] || "").trim();
    if (titles.has(value)) return afterRow + 1 + i;
  }


  return -1;
}

function getChangedPMRsFromMonthlyChangeReport_(reportSheet) {
  const result = new Map();
  const lastRow = reportSheet.getLastRow();
  const lastCol = reportSheet.getLastColumn();


  if (lastRow < 1 || lastCol < 1) return result;


  const sectionTitles = ["Enrollments", "Disenrollments", "Demographic Changes", "Caseload Changes", "Banner Changes", "Contact Changes"];


  let foundSectionLayout = false;


  sectionTitles.forEach(section => {
    const titleRow = findMonthlyChangeSectionTitleRow_(reportSheet, section);
    if (titleRow === -1) return;


    foundSectionLayout = true;


    const headerRow = titleRow + 2;
    const dataStartRow = headerRow + 1;
    const nextTitleRow = findNextMonthlyChangeSectionTitleRow_(reportSheet, titleRow);
    const dataEndRow = nextTitleRow === -1 ? lastRow : nextTitleRow - 1;


    if (dataEndRow < dataStartRow) return;


    const headers = reportSheet.getRange(headerRow, 1, 1, lastCol)
      .getValues()[0]
      .map(h => String(h || "").trim());
    const headerMap = buildHeaderIndexMap_(headers);
    const pmrIdx = getPMRIndex_(headerMap);


    if (pmrIdx === -1) return;


    const values = reportSheet.getRange(dataStartRow, 1, dataEndRow - dataStartRow + 1, lastCol).getValues();


    values.forEach(row => {
      const pmr = normalizePMR_(row[pmrIdx]);
      if (!pmr) return;


      if (!result.has(pmr)) result.set(pmr, new Set());
      result.get(pmr).add(section);
    });
  });


  if (foundSectionLayout) return result;


  if (lastRow < 5) return result;


  const headers = reportSheet.getRange(4, 1, 1, lastCol)
    .getValues()[0]
    .map(h => String(h || "").trim());
  const headerMap = buildHeaderIndexMap_(headers);


  const sectionIdx = headerMap["Section"];
  const pmrIdx = headerMap["Participant PMR#"];


  if (sectionIdx === undefined || pmrIdx === undefined) return result;


  const values = reportSheet.getRange(5, 1, lastRow - 4, lastCol).getValues();


  values.forEach(row => {
    const pmr = normalizePMR_(row[pmrIdx]);
    const section = String(row[sectionIdx] || "").trim();


    if (!pmr) return;


    if (!result.has(pmr)) result.set(pmr, new Set());
    if (section) result.get(pmr).add(section);
  });


  return result;
}

function runMonthlyChangeValidation() {
  const monthParts = promptForLockedYearReportMonth_("Monthly Change Validation");
  if (!monthParts) return;


  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const reportSheet = getMonthlySheetByPrefixAndDate_(ss, MONTHLY_CHANGE_REPORT_PREFIX, monthParts.firstDay, monthParts.lastDay);


  if (!reportSheet) {
    notify_("Monthly Change Report was not found for the selected month.");
    return;
  }


  const sections = ["Enrollments", "Disenrollments", "Demographic Changes", "Caseload Changes", "Contact Changes"];
  const summaryRows = [];
  const detailRows = [];


  sections.forEach(section => {
    const titleRow = findMonthlyChangeSectionTitleRow_(reportSheet, section);
    if (titleRow === -1) {
      summaryRows.push([section, "FAIL", 0, "Section title not found."]);
      detailRows.push(["FAIL", section, "Missing section title."]);
      return;
    }


    const headerRow = titleRow + 2;
    const dataStartRow = headerRow + 1;
    const nextTitleRow = findNextMonthlyChangeSectionTitleRow_(reportSheet, titleRow);
    const dataEndRow = nextTitleRow === -1 ? reportSheet.getLastRow() : nextTitleRow - 1;


    if (dataEndRow < dataStartRow) {
      summaryRows.push([section, "PASS", 0, "No rows in section."]);
      return;
    }


    const lastCol = Math.max(reportSheet.getLastColumn(), 1);
    const headers = reportSheet.getRange(headerRow, 1, 1, lastCol).getValues()[0].map(h => String(h || "").trim());
    const headerMap = buildHeaderIndexMap_(headers);
    const pmrIdx = getPMRIndex_(headerMap);


    const values = reportSheet.getRange(dataStartRow, 1, dataEndRow - dataStartRow + 1, lastCol).getValues();
    const nonBlankRows = values.filter(row => row.some(value => String(value || "").trim() !== ""));
    const uniquePMRs = new Set();
    let missingPMR = 0;


    nonBlankRows.forEach((row, idx) => {
      const rowNumber = dataStartRow + idx;
      const pmr = pmrIdx === -1 ? "" : normalizePMR_(row[pmrIdx]);
      if (!pmr) {
        missingPMR++;
        detailRows.push(["WARNING", section, "Missing PMR on row " + rowNumber]);
      } else {
        uniquePMRs.add(pmr);
      }
    });


    summaryRows.push([section, missingPMR === 0 ? "PASS" : "WARNING", nonBlankRows.length, "Unique PMRs: " + uniquePMRs.size + "; Missing PMRs: " + missingPMR]);
  });


  writeDiagnosticReport_(
    "Monthly Change Validation",
    "Monthly Change Validation",
    monthParts,
    ["Section", "Status", "Rows", "Details"],
    summaryRows,
    ["Status", "Section", "Details"],
    detailRows
  );


  notify_("Monthly Change Validation complete.");
}

function writeDiagnosticReport_(sheetName, title, monthParts, summaryHeaders, summaryRows, detailHeaders, detailRows) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(sheetName);


  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
  } else {
    sheet.clear();
  }


  sheet;


  sheet.getRange("A1").setValue(title);
  sheet.getRange("A2").setValue("Date");
  sheet.getRange("B2").setValue(monthParts.firstDay);
  sheet.getRange("C2").setValue("to");
  sheet.getRange("D2").setValue(monthParts.lastDay);


  let row = 4;
  sheet.getRange(row, 1, 1, summaryHeaders.length).setValues([summaryHeaders]);
  sheet.getRange(row, 1, 1, summaryHeaders.length);
  row++;


  if (summaryRows && summaryRows.length > 0) {
    sheet.getRange(row, 1, summaryRows.length, summaryHeaders.length).setValues(summaryRows);
    row += summaryRows.length;
  }


  row += 2;
  sheet.getRange(row, 1).setValue("Details");
  row++;
  sheet.getRange(row, 1, 1, detailHeaders.length).setValues([detailHeaders]);
  sheet.getRange(row, 1, 1, detailHeaders.length);
  row++;


  if (detailRows && detailRows.length > 0) {
    sheet.getRange(row, 1, detailRows.length, detailHeaders.length).setValues(detailRows);
  } else {
    sheet.getRange(row, 1).setValue("No issues found.");
  }


  const lastRow = Math.max(sheet.getLastRow(), 1);
  const lastCol = Math.max(sheet.getLastColumn(), 1);


  sheet.getRange(1, 1, lastRow, lastCol);


  sheet;
  applyStandardRowHeights_(sheet);


  for (let col = 1; col <= lastCol; col++) {
    sheet;
  }


  if (sheet.getFilter()) sheet;
  sheet.getRange(4, 1, Math.max(lastRow - 3, 1), lastCol);
}
